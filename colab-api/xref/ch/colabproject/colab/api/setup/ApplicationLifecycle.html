7   package ch.colabproject.colab.api.setup;
8   
9   import ch.colabproject.colab.api.ejb.UserManagement;
10  import ch.colabproject.colab.api.model.user.User;
11  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
12  import javax.inject.Inject;
13  import javax.servlet.ServletConfig;
14  import javax.servlet.ServletException;
15  import javax.servlet.annotation.WebServlet;
16  import javax.servlet.http.HttpServlet;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  
20  /**
21   *
22   * @author maxence
23   */
24  @WebServlet(loadOnStartup = 2)
25  public class ApplicationLifecycle extends HttpServlet {
26  
27      private static final long serialVersionUID = 1L;
28  
29      /**
30       * Logger.Default level, set in logback.xml, is INFO
31       */
32      private static final Logger logger = LoggerFactory.getLogger(ApplicationLifecycle.class);
33  
34      /**
35       * User related methods
36       */
37      @Inject
38      private UserManagement userManagement;
39  
40      /**
41       * Create a default admin user if there is no admin at all
42       */
43      private void createDefaultAdminIfNone() {
44          if (userManagement.findAllAdmin().isEmpty()) {
45              try {
46                  logger.info("Create default admin user");
47                  User admin = userManagement.createUser(
48                      ColabConfiguration.getDefaultAdminUsername(),
49                      ColabConfiguration.getDefaultAdminEmail(),
50                      ColabConfiguration.getDefaultAdminPassword()
51                  );
52                  userManagement.grantAdminRight(admin.getId());
53              } catch (HttpErrorMessage ex) {
54                  logger.error("Fails to create default amdin user. "
55                      + "Does non-admin user exists with same username or email address");
56              } catch (RuntimeException ex) {
57                  logger.error("Fails to create default amdin user for some unknown reason."
58                      + " Please check config");
59              }
60          }
61      }
62  
63      @Override
64      public void init(ServletConfig config) throws ServletException {
65          super.init(config);
66          logger.trace("LIFE CYCLE : init");
67          createDefaultAdminIfNone();
68      }
69  
70      @Override
71      public void destroy() {
72          super.destroy();
73          logger.trace("LIFE CYCLE : destroy");
74      }
75  }
