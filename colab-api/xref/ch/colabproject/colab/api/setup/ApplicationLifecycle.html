7   package ch.colabproject.colab.api.setup;
8   
9   import ch.colabproject.colab.api.ejb.ApplicationLifecycleFacade;
10  import ch.colabproject.colab.generator.model.tools.PolymorphicDeserializer;
11  import javax.inject.Inject;
12  import javax.servlet.ServletConfig;
13  import javax.servlet.ServletException;
14  import javax.servlet.annotation.WebServlet;
15  import javax.servlet.http.HttpServlet;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  /**
20   *
21   * @author maxence
22   */
23  @WebServlet(loadOnStartup = 2)
24  public class ApplicationLifecycle extends HttpServlet {
25  
26      private static final long serialVersionUID = 1L;
27  
28      /**
29       * Logger. Default level, set in logback.xml, is INFO
30       */
31      private static final Logger logger = LoggerFactory.getLogger(ApplicationLifecycle.class);
32  
33      /**
34       * User related methods
35       */
36      @Inject
37      private ApplicationLifecycleFacade applicationLifecycleFacade;
38  
39      @Override
40      public void init(ServletConfig config) throws ServletException {
41          super.init(config);
42          logger.trace("LIFE CYCLE : init");
43  
44          PolymorphicDeserializer.includePackage("ch.colabproject.colab.api");
45  
46          logger.trace("Create admin");
47          applicationLifecycleFacade.createDefaultAdminIfNone();
48  
49      }
50  
51      @Override
52      public void destroy() {
53          super.destroy();
54          logger.trace("LIFE CYCLE : destroy");
55      }
56  }
