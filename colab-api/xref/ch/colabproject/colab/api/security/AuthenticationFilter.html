7   package ch.colabproject.colab.api.security;
8   
9   import ch.colabproject.colab.api.controller.RequestManager;
10  import ch.colabproject.colab.api.model.user.User;
11  import ch.colabproject.colab.generator.model.annotations.AdminResource;
12  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
13  import ch.colabproject.colab.generator.model.annotations.ConsentNotRequired;
14  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  import javax.annotation.Priority;
19  import javax.inject.Inject;
20  import javax.ws.rs.container.ContainerRequestContext;
21  import javax.ws.rs.container.ContainerRequestFilter;
22  import javax.ws.rs.container.ResourceInfo;
23  import javax.ws.rs.core.Context;
24  import javax.ws.rs.ext.ExceptionMapper;
25  import javax.ws.rs.ext.Provider;
26  import java.io.IOException;
27  import java.lang.annotation.Annotation;
28  import java.lang.reflect.Method;
29  import java.util.ArrayList;
30  import java.util.List;
31  
32  /**
33   * Intercept all request to the API and check user has required permission.
34   * <p>
35   * This filter has a priority of 10, which means it is executed after {@link CookieFilter } and
36   * {@link ch.colabproject.colab.api.rest.utils.filter.RequestFilter RequestFilter}
37   *
38   * @author maxence
39   */
40  @Provider
41  @Priority(10)
42  public class AuthenticationFilter implements ContainerRequestFilter {
43  
44      /**
45       * Logger
46       */
47      private static final Logger logger = LoggerFactory.getLogger(AuthenticationFilter.class);
48  
49      /**
50       * Request related logic
51       */
52      @Inject
53      private RequestManager requestManager;
54  
55      /**
56       * Post-matching filter knows the targeted resource by injecting such a ResourceInfo
57       */
58      @Context
59      private ResourceInfo resourceInfo;
60  
61      /**
62       * To re-use exception to response mapper
63       */
64      @Inject
65      private ExceptionMapper<Exception> exceptionMapper;
66  
67      /**
68       * Cluster-wide session cache.
69       */
70      @Inject
71      private SessionManager sessionManager;
72  
73      /**
74       * To get the last timestamp when the Terms of Use and Data Policy were updated
75       */
76      @Inject
77      private TermsOfUseManager termsOfUseManager;
78  
79      /**
80       * Get all method or class annotations matching the given type.
81       *
82       * @param <T>        type of annotation to search
83       * @param annotation type of annotation to search
84       * @param klass      targeted class
85       * @param method     targeted method
86       * @return the list of all matching annotations found on class and method
87       */
88      private <T extends Annotation> List<T> getAnnotations(Class<T> annotation,
89                                                            Class<?> klass, Method method) {
90  
91          List<T> list = new ArrayList<>();
92  
93          T a = method.getAnnotation(annotation);
94          if (a != null) {
95              list.add(a);
96          }
97          a = klass.getAnnotation(annotation);
98  
99          if (a != null) {
100             list.add(a);
101         }
102 
103         return list;
104     }
105 
106     /**
107      * Intercept request and make sure current user has access to targeted class and method
108      * {@inheritDoc }
109      */
110     @Override
111     public void filter(ContainerRequestContext requestContext) throws IOException {
112         // Targeted Class & method
113         final Class<?> targetClass = resourceInfo.getResourceClass();
114         Method targetMethod = resourceInfo.getResourceMethod();
115 
116         User currentUser = requestManager.getCurrentUser();
117         HttpErrorMessage abortWith = null;
118 
119         List<AuthenticationRequired> authAnnotations = getAnnotations(
120                 AuthenticationRequired.class,
121                 targetClass, targetMethod);
122 
123         if (!authAnnotations.isEmpty()) {
124             if (currentUser == null) {
125                 // current user not authenticated: make sure the targeted method is accessible to
126                 // unauthenticated user
127                 // No current user but annotation required to be authenticated
128                 // abort with 401 code
129                 logger.trace("Request aborted:user is not authenticated");
130                 abortWith = HttpErrorMessage.authenticationRequired();
131             } else {
132                 List<ConsentNotRequired> consentAnnotations = getAnnotations(
133                         ConsentNotRequired.class,
134                         targetClass, targetMethod);
135 
136                 if (consentAnnotations.isEmpty() && (currentUser.getAgreedTime() == null
137                         || currentUser.getAgreedTime().isBefore(termsOfUseManager.getTimestamp()))) {
138                     // current user is authenticated but need to accept new TermsOfUse
139                     logger.trace("Request aborted:user has not agreed to new TermsOfUse");
140                     abortWith = HttpErrorMessage.forbidden();
141                 }
142             }
143         }
144 
145 
146         List<AdminResource> adminAnnotations = getAnnotations(
147                 AdminResource.class,
148                 targetClass, targetMethod);
149         if (!adminAnnotations.isEmpty()) {
150             if (currentUser == null) {
151                 // no current user : unauthorized asks for user to authenticate
152                 logger.trace("Request aborted:user is not authenticated");
153                 abortWith = HttpErrorMessage.authenticationRequired();
154             } else {
155                 if (!currentUser.isAdmin()) {
156                     // current user is authenticated but lack admin right: forbidden
157                     logger.trace("Request aborted:user tries to access admin resource");
158                     abortWith = HttpErrorMessage.forbidden();
159                 }
160             }
161         }
162 
163         if (abortWith != null) {
164             requestContext.abortWith(exceptionMapper.toResponse(abortWith));
165         } else {
166             sessionManager.touchUserActivityDate();
167         }
168     }
169 }
