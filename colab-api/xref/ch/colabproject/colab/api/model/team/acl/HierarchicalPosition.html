7   package ch.colabproject.colab.api.model.team.acl;

9   /**
10   * Describe the hierarchical position of a team member
11   */
12  //WARNING ! DO NOT CHANGE THE ENUM NAMES, THEY ARE USED AS KEYS IN DB !!
13  public enum HierarchicalPosition {
14      /**
15       * Guests do not have access to anything unless explicit authorization
16       */
17      GUEST(InvolvementLevel.OUT_OF_THE_LOOP),
18      /**
19       * Internals have read/write access to everything by default
20       */
21      INTERNAL(InvolvementLevel.INFORMED_READWRITE),
22      /**
23       * owners has full read/write access to the whole project, bypassing any access control
24       */
25      OWNER(InvolvementLevel.INFORMED_READWRITE);
26  
27      /**
28       * give read-write access ?
29       */
30      private final InvolvementLevel defaultLevel;
31  
32      /**
33       * Build a Hierarchical position
34       *
35       * @param defaultLevel default level of involvements
36       */
37      /* private */ HierarchicalPosition(InvolvementLevel defaultLevel) {
38          this.defaultLevel = defaultLevel;
39      }
40  
41      /**
42       * Get the default involvement level.
43       *
44       * @return the level
45       */
46      public InvolvementLevel getDefaultInvolvement() {
47          return this.defaultLevel;
48      }
49  }
