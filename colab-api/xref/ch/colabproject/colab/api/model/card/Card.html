638     @Override
639     public void merge(ColabEntity other) throws ColabMergeException {
640         if (other instanceof Card) {
641             Card o = (Card) other;
642             this.setTitle(o.getTitle());
643             this.setColor(o.getColor());
644             this.setDefaultInvolvementLevel(o.getDefaultInvolvementLevel());
645             // do not update position
646         } else {
647             throw new ColabMergeException(this, other);
648         }
649     }
650 
651     @Override
652     public void duplicate(ColabEntity other) throws ColabMergeException {
653         // same as merge but copy position too
654         this.merge(other);
655         if (other instanceof Card) {
656             Card o = (Card) other;
657             this.setX(o.getX());
658             this.setY(o.getY());
659             this.setWidth(o.getWidth());
660             this.setHeight(o.getHeight());
661         }
662     }

664     /**
665      * Get the project this card belongs to.
666      *
667      * @return card owner
668      */
669     @JsonbTransient
670     public Project getProject() {
671         if (this.rootCardProject != null) {
672             // this card is a root card
673             return this.rootCardProject;
674         } else if (this.cardType != null) {
675             // the card type has a direct access to project
676             return this.cardType.getProject();
677         } else if (this.parent != null) {
678             // nothing easier, so get the one of its parent
679             return this.parent.getProject();
680         }
681         return null;
682     }
683 
684     @Override
685     public ChannelsBuilder getChannelsBuilder() {
686         if (this.rootCardProject != null) {
687             // this card is a root card, propagate through the project content channel
688             return new ProjectContentChannelBuilder(this.rootCardProject);
689         } else if (this.parent != null) {
690             // this card is a sub-card, propagate through its parent channels
691             return this.parent.getChannelsBuilder();
692         } else if (this.cardType != null) {
693             // such a card shouldn't exist...
694             // Lorem-ipsum cards for global cardTypes ???
695             return this.cardType.getChannelsBuilder();
696         } else {
697             // such an orphan shouldn't exist...
698             return new EmptyChannelBuilder();
699         }
700     }

702     @Override
703     @JsonbTransient
704     public Conditions.Condition getReadCondition() {
705         // genuine hack inside
706         // any member can read any card of the project
707         // if a member lacks the read right on a card, it will not be able to read the deliverable,
708         // resources and so on, but it will still be able to view the card "from the outside"
709         return new Conditions.IsCurrentUserMemberOfProject(getProject());
710     }
