156     /**
157      * Build a HTML body to send a message for the invitation token
158      *
159      * @param token the invitation token
160      * @param link  the link in order to consume the token
161      *
162      * @return the HTML body of the message to send
163      */
164     public static String build(InvitationToken token, String link) {
165         String subHeading = MessageFormat.format(MESSAGE_SUBHEADING,
166             token.getSender() != null
167                 ? token.getSender()
168                 : "Someone",
169             token.getProject() != null
170                 ? "<i>" + token.getProject().getName() + "</i>"
171                 : "a co.LAB project");
172 
173         return new TokenMessageBuilder(token)
174             .headTitle(MESSAGE_HEAD_TITLE)
175             .picture(MESSAGE_PICTURE)
176             .heading(MESSAGE_HEADING)
177             .subheading(subHeading)
178             .info(MESSAGE_INFO)
179             .linkHref(link)
180             .linkLabel(MESSAGE_LINK_LABEL)
181             // let default footer
182             .build();
183     }
