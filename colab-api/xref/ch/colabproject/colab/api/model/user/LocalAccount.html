7   package ch.colabproject.colab.api.model.user;
8   
9   import ch.colabproject.colab.api.Helper;
10  import ch.colabproject.colab.api.exceptions.ColabMergeException;
11  import ch.colabproject.colab.api.model.ColabEntity;
12  import javax.json.bind.annotation.JsonbTransient;
13  import javax.persistence.Column;
14  import javax.persistence.Entity;
15  import javax.persistence.EnumType;
16  import javax.persistence.Enumerated;
17  import javax.persistence.Index;
18  import javax.persistence.NamedQuery;
19  import javax.persistence.Table;
20  import javax.validation.constraints.NotNull;
21  
22  /**
23   * Password based authentication.
24   *
25   * @author maxence
26   */
27  @Entity
28  @NamedQuery(name = "LocalAccount.findByEmail",
29      query = "SELECT a from LocalAccount a where a.email = :email")
30  @Table(
31      // make sure to have JOINED inheritance, otherwise indexes and constraints will be ignored!
32      indexes = {
33          @Index(columnList = "email", unique = true)
34      })
35  public class LocalAccount extends Account {
36  
37      private static final long serialVersionUID = 1L;
38  
39      /**
40       * username-like email address
41       */
42      @NotNull
43      private String email;
44  
45      /**
46       * salt+password hash. Hashed with currentDbHashMethod and dbSalt.
47       */
48      @JsonbTransient
49      @NotNull
50      private byte[] hashedPassword;
51  
52      /**
53       * Has the email address been verified with a VerificationToken ?
54       */
55      @NotNull
56      private Boolean verified;
57  
58      /**
59       * Salt to used before hashing the password
60       */
61      @JsonbTransient
62      @NotNull
63      private byte[] dbSalt;
64  
65      /**
66       * Hash method to use to hashedPassword the salt+password
67       */
68      @Column(length = 100)
69      @Enumerated(value = EnumType.STRING)
70      @JsonbTransient
71      @NotNull
72      private HashMethod currentDbHashMethod;
73  
74      /**
75       * New hash method to use to hash the salt+password. If not null, rotate hash method on next
76       * successful authentication
77       */
78      @Column(length = 100)
79      @Enumerated(value = EnumType.STRING)
80      @JsonbTransient
81      private HashMethod nextDbHashMethod;
82  
83      /**
84       * Salt the client shall use to before hashing its password. Salt is hex-encoded byte array
85       */
86      @NotNull
87      @JsonbTransient
88      private String clientSalt;
89  
90      /**
91       * New salt the client shall use to prefix its password before hashing it.
92       * <p>
93       * In case this is not null, client shall send two hashes. First one is the plain_password
94       * prefixed with clientSalt and hashed with currentClientHashMethod (to authenticate), second is
95       * its password prefixed with this new salt and hashed with nextClientHashMethod if set, current
96       * otherwise. successful authentication (to rotate salt and/or method)
97       */
98      @JsonbTransient
99      private String newClientSalt;
100 
101     /**
102      * Hash method the client shall use to hashedPassword the clientSalt+plain_password
103      */
104     @Column(length = 100)
105     @Enumerated(value = EnumType.STRING)
106     @NotNull
107     @JsonbTransient
108     private HashMethod currentClientHashMethod;
109 
110     /**
111      * New hash method the client shall use to hash its clientSalt+plain_password.
112      * <p>
113      * In case this is not null, client shall send two hashes. First one is its plain_password
114      * prefixed clientSalt and hashed with currentClientHashMethod (to authenticate), second is its
115      * password prefixed with the new_salt (if set)or the current salt and hashed with this method
116      */
117     @Column(length = 100)
118     @Enumerated(value = EnumType.STRING)
119     @JsonbTransient
120     private HashMethod nextClientHashMethod;
121 
122     /**
123      *
124      * @return email associated with this account
125      */
126     public String getEmail() {
127         return email;
128     }
129 
130     /**
131      * update email. If the email is not the same, this account will not be verified any longer
132      *
133      * @param email new email to use.
134      */
135     public void setEmail(String email) {
136         if (Helper.isEmailAddress(email)) {
137 
138             if (!email.equals(this.email)) {
139                 this.verified = false;
140             }
141             this.email = email;
142         }
143     }
144 
145     /**
146      * get the stored hashedPassword to challenge authentication against
147      *
148      * @return hashedPassword hashedPassword to challenge authentication against
149      */
150     public byte[] getHashedPassword() {
151         return hashedPassword;
152     }
153 
154     /**
155      * Update hashedPassword
156      *
157      * @param hashedPassword new hashedPassword
158      */
159     public void setHashedPassword(byte[] hashedPassword) {
160         this.hashedPassword = hashedPassword;
161     }
162 
163     /**
164      * has the email address been verified ?
165      *
166      * @return true if the account is verified
167      */
168     public Boolean isVerified() {
169         return verified;
170     }
171 
172     /**
173      * Set if the account has been verified or not
174      *
175      * @param verified yes or no ?
176      */
177     public void setVerified(Boolean verified) {
178         this.verified = verified;
179     }
180 
181     /**
182      * @return the salt to used server-side
183      */
184     public byte[] getDbSalt() {
185         return dbSalt;
186     }
187 
188     /**
189      * Update the server-side hashedPassword
190      *
191      * @param dbSalt new server-side salt
192      */
193     public void setDbSalt(byte[] dbSalt) {
194         this.dbSalt = dbSalt;
195     }
196 
197     /**
198      *
199      * @return the current hashedPassword method to used to hashedPassword provided password
200      */
201     public HashMethod getCurrentDbHashMethod() {
202         return currentDbHashMethod;
203     }
204 
205     /**
206      * Set the method to use to hashedPassword provided password
207      *
208      * @param currentDbHashMethod hashedPassword method
209      */
210     public void setCurrentDbHashMethod(HashMethod currentDbHashMethod) {
211         this.currentDbHashMethod = currentDbHashMethod;
212     }
213 
214     /**
215      * @return the next hashedPassword method to use. If not null, hashedPassword methods will be
216      *         rotated on next authentication
217      */
218     public HashMethod getNextDbHashMethod() {
219         return nextDbHashMethod;
220     }
221 
222     /**
223      * change the next hashedPassword method to used
224      *
225      * @param nextDbHashMethod next hashedPassword method
226      */
227     public void setNextDbHashMethod(HashMethod nextDbHashMethod) {
228         this.nextDbHashMethod = nextDbHashMethod;
229     }
230 
231     /**
232      * @return the salt the user shall use before client-side plain_password hashedPassword
233      */
234     public String getClientSalt() {
235         return clientSalt;
236     }
237 
238     /**
239      * Update the salt the client shall use
240      *
241      * @param clientSalt client salt
242      */
243     public void setClientSalt(String clientSalt) {
244         this.clientSalt = clientSalt;
245     }
246 
247     /**
248      * @return the salt to use to rotate authentication
249      */
250     public String getNewClientSalt() {
251         return newClientSalt;
252     }
253 
254     /**
255      * set a next client-side salt
256      *
257      * @param newClientSalt new salt we want the client to use
258      */
259     public void setNewClientSalt(String newClientSalt) {
260         this.newClientSalt = newClientSalt;
261     }
262 
263     /**
264      * @return the hashedPassword method the client shall use to hashedPassword its
265      *         salt+plain_password
266      */
267     public HashMethod getCurrentClientHashMethod() {
268         return currentClientHashMethod;
269     }
270 
271     /**
272      * CHange hashedPassword method the client shall use
273      *
274      * @param currentClientHashMethod hashedPassword method the client shall use
275      */
276     public void setCurrentClientHashMethod(HashMethod currentClientHashMethod) {
277         this.currentClientHashMethod = currentClientHashMethod;
278     }
279 
280     /**
281      * @return the next hashedPassword method the client shall use
282      */
283     public HashMethod getNextClientHashMethod() {
284         return nextClientHashMethod;
285     }
286 
287     /**
288      * Update the next client-side hashedPassword method
289      *
290      * @param nextClientHashMethod new next client hashedPassword method
291      */
292     public void setNextClientHashMethod(HashMethod nextClientHashMethod) {
293         this.nextClientHashMethod = nextClientHashMethod;
294     }
295 
296     @Override
297     public void merge(ColabEntity other) throws ColabMergeException {
298         // TODO
299     }
300 
301     @Override
302     public String toString() {
303         return "LocalAccount{" + "id=" + this.getId() + ", email=" + email
304             + ", verified=" + verified + '}';
305     }
306 
307 }
