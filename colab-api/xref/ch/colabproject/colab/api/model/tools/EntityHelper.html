7   package ch.colabproject.colab.api.model.tools;
8   
9   import ch.colabproject.colab.api.model.WithId;
10  import java.util.Objects;
11  
12  /**
13   *
14   * @author maxence
15   */
16  public class EntityHelper {
17  
18      /**
19       * never-called private constructor
20       */
21      private EntityHelper() {
22          throw new UnsupportedOperationException(
23              "This is a utility class and cannot be instantiated");
24      }
25  
26      /**
27       * {@link Object#hashCode} implementation for all WithId implementation
28       *
29       * @param object object to hash
30       *
31       * @return a hash code value for the given object.
32       */
33      public static int hashCode(WithId object) {
34          int hash = 7;
35          hash = 31 * hash + Objects.hashCode(object.getId());
36          return hash;
37      }
38  
39      /**
40       * {@link Object#equals} implementation for all WithId implementation. Objects equal if
41       * <ul>
42       * <li>it's the same reference (strict == equality)
43       * <li>both are null
44       * <li>they are instance of the same class and have the same id
45       * </ul>
46       *
47       * @param a first object
48       * @param b second object
49       *
50       * @return {@code true} if both objects represent the same persisted entity
51       */
52      public static boolean equals(WithId a, Object b) {
53          if (a == b) {
54              // same object or both are null
55              return true;
56          }
57          if (a == null || b == null) {
58              // only one is null
59              return false;
60          }
61  
62          // at this point, both are not null
63          if (a.getClass() != b.getClass()) {
64              // class mismatch
65              return false;
66          } else {
67              // same class
68              WithId otherB = (WithId) b;
69              if (a.getId() == null || otherB.getId() == null) {
70                  // if at least one object has no id, object are not equals
71                  // if both id are null, objects may equal, but this case is handled by the strict
72                  // (a == b) equality check
73                  return false;
74              }
75              // same class, ids must equals
76              return Objects.equals(a.getId(), otherB.getId());
77          }
78      }
79  }
