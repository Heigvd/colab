7   package ch.colabproject.colab.api.ejb;
8   
9   import ch.colabproject.colab.api.model.card.CardContent;
10  import ch.colabproject.colab.api.model.document.Block;
11  import ch.colabproject.colab.api.model.document.BlockDocument;
12  import ch.colabproject.colab.api.model.document.Document;
13  import ch.colabproject.colab.api.persistence.card.CardContentDao;
14  import ch.colabproject.colab.api.persistence.document.DocumentDao;
15  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
16  import java.util.List;
17  import javax.ejb.LocalBean;
18  import javax.ejb.Stateless;
19  import javax.inject.Inject;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  /**
24   * Document specific logic
25   *
26   * @author sandra
27   */
28  @Stateless
29  @LocalBean
30  public class DocumentFacade {
31  
32      /** logger */
33      private static final Logger logger = LoggerFactory.getLogger(DocumentFacade.class);
34  
35      // *********************************************************************************************
36      // injections
37      // *********************************************************************************************
38  
39      /**
40       * Document persistence handler
41       */
42      @Inject
43      private DocumentDao documentDao;
44  
45      /**
46       * Card content persistence handler
47       */
48      @Inject
49      private CardContentDao cardContentDao;
50  
51      // *********************************************************************************************
52      // general documents
53      // *********************************************************************************************
54  
55      /**
56       * Complete and persist the given new document
57       *
58       * @param document the document to persist
59       *
60       * @return the new persisted document
61       */
62      public Document createDocument(Document document) {
63          logger.debug("create document : {}", document);
64  
65          if (document.getDeliverableCardContentId() != null) {
66              CardContent cardContent = cardContentDao
67                  .getCardContent(document.getDeliverableCardContentId());
68              if (cardContent == null) {
69                  throw HttpErrorMessage.relatedObjectNotFoundError();
70              }
71  
72              cardContent.setDeliverable(document);
73              document.setDeliverableCardContent(cardContent);
74          }
75  
76          if (document.getResourceId() != null) {
77              throw HttpErrorMessage.dataIntegrityFailure();
78          }
79  
80          return documentDao.persistDocument(document);
81      }
82  
83      /**
84       * Delete the given document
85       *
86       * @param documentId the id of the document to delete
87       *
88       * @return the freshly deleted document
89       */
90      public Document deleteDocument(Long documentId) {
91          logger.debug("delete document #{}", documentId);
92  
93          Document document = documentDao.findDocument(documentId);
94          if (document == null) {
95              throw HttpErrorMessage.relatedObjectNotFoundError();
96          }
97  
98          CardContent cardContent = document.getDeliverableCardContent();
99          if (cardContent != null) {
100             cardContent.setDeliverable(null);
101         }
102 
103         // to check : resource handling
104 
105         return documentDao.deleteDocument(documentId);
106     }
107 
108     // *********************************************************************************************
109     // block documents
110     // *********************************************************************************************
111 
112     /**
113      * Get all blocks that make up the document
114      *
115      * @param documentId the id of the document
116      *
117      * @return all blocks of the document
118      */
119     public List<Block> getBlocksOfDocument(Long documentId) {
120         logger.debug("get blocks composing the document #{}", documentId);
121 
122         Document document = documentDao.findDocument(documentId);
123         if (!(document instanceof BlockDocument)) {
124             throw HttpErrorMessage.relatedObjectNotFoundError();
125         }
126 
127         return ((BlockDocument) document).getBlocks();
128     }
129 
130     // *********************************************************************************************
131     //
132     // *********************************************************************************************
133 
134 }
