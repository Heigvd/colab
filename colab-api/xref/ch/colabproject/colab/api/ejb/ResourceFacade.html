167     /**
168      * Find the resources at the end of the references chains.
169      * <p>
170      * Only the active and available resources are taken into account.
171      *
172      * @param baseAbstractResources The list of abstract resources to search from
173      *
174      * @return the resources that match
175      */
176     private List<Resource> findAvailableAndActiveTargetResource(
177         List<AbstractResource> baseAbstractResources) {
178         List<Resource> result = new ArrayList<>();
179 
180         for (AbstractResource abstractResource : baseAbstractResources) {
181             Resource resource = getAvailableActiveResource(abstractResource);
182             if (resource != null) {
183                 result.add(resource);
184             }
185         }
186 
187         return result;
188     }

190     /**
191      * Recursive method to get the resource at the end of each references chain.
192      * <p>
193      * Only the active and available resources are taken into account.
194      *
195      * @param abstractResource An abstract resource
196      *
197      * @return the resource at the end of each references chain as long as it is available and
198      *         active
199      */
200     private Resource getAvailableActiveResource(AbstractResource abstractResource) {
201         if (abstractResource instanceof Resource) {
202             return (Resource) abstractResource;
203         } else if (abstractResource instanceof ResourceRef) {
204             if (isResourceAvailableAndActive((ResourceRef) abstractResource)) {
205                 return getAvailableActiveResource(
206                     ((ResourceRef) abstractResource).getTarget());
207             }
208         } else {
209             throw HttpErrorMessage.dataIntegrityFailure();
210         }
211 
212         return null;
213     }

215     /**
216      * Filter to only take into account resources that are available and active
217      * <p>
218      * <ul>
219      * <li>a resource must be published to be passed the sub cards</li>
220      * <li>the reference must not be refused</li>
221      * </ul>
222      *
223      * @param resourceRef the resource reference
224      *
225      * @return whether the resource can be taken into account or not
226      */
227     private boolean isResourceAvailableAndActive(ResourceRef resourceRef) {
228         AbstractResource target = resourceRef.getTarget();
229         if (resourceRef.hasCard() && target.hasCardContent()
230             && !resourceRef.resolve().isPublished()) {
231             return false;
232         }
233 
234         return !resourceRef.isRefused();
235     }

241     /**
242      * Create a new resource for the document linked to the card type (or card type reference).
243      * <p>
244      * Every child of the card type (recursively) acquires a reference to that resource or, for the
245      * grandchildren, to a reference to that resource.
246      *
247      * @param document   the document the resource represents
248      * @param cardTypeId the id of the card type the resource must be linked to
249      *
250      * @return the brand new resource
251      */
252     public Resource createResourceForAbstractCardType(Document document, Long cardTypeId) {
253         logger.debug("create a resource for document {} and an abstract card type #{}", document,
254             cardTypeId);
255 
256         if (document == null) {
257             throw HttpErrorMessage.relatedObjectNotFoundError();
258         }
259 
260         // A document can be related at max to one resource
261         if (document.hasResource()) {
262             throw HttpErrorMessage.dataIntegrityFailure();
263         }
264 
265         AbstractCardType abstractCardType = cardTypeDao.getAbstractCardType(cardTypeId);
266         if (abstractCardType == null) {
267             throw HttpErrorMessage.relatedObjectNotFoundError();
268         }
269 
270         securityFacade.assertCanCreateResource(document, abstractCardType);
271 
272         Resource resource = Resource.initNewResource();
273 
274         resource.setAbstractCardType(abstractCardType);
275         abstractCardType.getDirectAbstractResources().add(resource);
276 
277         resource.setDocument(document);
278         document.setResource(resource);
279 
280         resource = resourceDao.persistResource(resource);
281 
282         createResourceRefForChildren(abstractCardType, resource);
283 
284         return resource;
285     }

378     /**
379      * Each child of the card type acquires a reference to the resource.
380      *
381      * @param cardTypeOrRef the card type or card type reference
382      * @param resource      the resource we are linking
383      */
384     private void createResourceRefForChildren(AbstractCardType cardTypeOrRef,
385         AbstractResource resource) {
386         for (AbstractCardType cardTypeRef : cardTypeOrRef.getReferences()) {
387             ResourceRef resourceRef = ResourceRef.initNewResourceRef();
388 
389             resourceRef.setTarget(resource);
390             resource.getDirectReferences().add(resourceRef);
391 
392             resourceRef.setAbstractCardType(cardTypeRef);
393             cardTypeRef.getDirectAbstractResources().add(resourceRef);
394 
395             resourceRefDao.persistResourceRef(resourceRef);
396 
397             createResourceRefForChildren(cardTypeRef, resourceRef);
398         }
399 
400         for (Card card : cardTypeOrRef.getImplementingCards()) {
401             ResourceRef resourceRef = ResourceRef.initNewResourceRef();
402 
403             resourceRef.setTarget(resource);
404             resource.getDirectReferences().add(resourceRef);
405 
406             resourceRef.setCard(card);
407             card.getDirectAbstractResources().add(resourceRef);
408 
409             resourceRefDao.persistResourceRef(resourceRef);
410 
411             createResourceRefForChildren(card, resourceRef);
412         }
413     }

463     /**
464      * Delete a resource
465      *
466      * @param resourceId the id of the resource to delete
467      */
468     public void deleteResource(Long resourceId) {
469         logger.debug("delete resource #{}", resourceId);
470 
471         AbstractResource resource = resourceOrRefDao.findResourceOrRef(resourceId);
472         if (resource == null) {
473             throw HttpErrorMessage.relatedObjectNotFoundError();
474             // or just return. see what is best
475         }
476 
477         if (!(resource instanceof Resource)) {
478             throw HttpErrorMessage.dataIntegrityFailure();
479         }
480 
481         deleteResourceRefs(resource);
482 
483         Document document = ((Resource) resource).getDocument();
484 
485         resourceDao.deleteResource(resource.getId());
486 
487         // TODO see if we keep the document
488         documentFacade.deleteDocument(document.getId());
489     }

491     /**
492      * Delete each reference pointing at the given resourceOrRef
493      *
494      * @param resourceOrRef The initial abstract resource to delete
495      */
496     private void deleteResourceRefs(AbstractResource resourceOrRef) {
497         List<ResourceRef> references = resourceOrRef.getDirectReferences();
498         if (references != null) {
499             references.stream().forEach(ref -> deleteResourceRefs(ref));
500         }
501 
502         if (resourceOrRef.hasAbstractCardType()) {
503             AbstractCardType cardTypeOrRef = resourceOrRef.getAbstractCardType();
504             cardTypeOrRef.getDirectAbstractResources().remove(resourceOrRef);
505         }
506 
507         if (resourceOrRef.hasCard()) {
508             Card card = resourceOrRef.getCard();
509             card.getDirectAbstractResources().remove(resourceOrRef);
510         }
511 
512         if (resourceOrRef.hasCardContent()) {
513             CardContent cardContent = resourceOrRef.getCardContent();
514             cardContent.getDirectAbstractResources().remove(resourceOrRef);
515         }
516 
517         resourceRefDao.deleteResourceRef(resourceOrRef.getId());
518     }

663     /**
664      * Rename the category in a card type / card type reference<br>
665      * And do it also for the implementing cards as long as they are of the same project
666      *
667      * @param cardTypeOrRef the card type / card type reference (scope of the renaming)
668      * @param projectId     the id of the project concerned (scope of the renaming)
669      * @param oldName       the old name of the category
670      * @param newName       the new name of the category
671      */
672     private void renameCategory(AbstractCardType cardTypeOrRef, Long projectId, String oldName,
673         String newName) {
674         cardTypeOrRef.getDirectAbstractResources().stream()
675             .forEach(resourceOrRef -> renameCategoryIfMatch(resourceOrRef, oldName, newName));
676 
677         cardTypeOrRef.getImplementingCards().stream()
678             .filter(card -> Objects.equals(projectId, card.getProject().getId()))
679             .forEach(card -> renameCategory(card, oldName, newName));
680 
681         cardTypeOrRef.getReferences().stream()
682             .forEach(cardRef -> renameCategory(cardRef, projectId, oldName, newName));
683     }

750         AbstractResource resource = resourceOrRefDao.findResourceOrRef(resourceOrRefId);
751         if (resource == null) {
752             throw HttpErrorMessage.relatedObjectNotFoundError();
753         }
754         return resource.getStickyNoteLinksAsSrc();
755     }
