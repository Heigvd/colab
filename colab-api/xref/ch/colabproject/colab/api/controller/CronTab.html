7   package ch.colabproject.colab.api.controller;
8   
9   import ch.colabproject.colab.api.controller.document.ExternalDataManager;
10  import ch.colabproject.colab.api.controller.monitoring.CronJobLogManager;
11  import ch.colabproject.colab.api.model.monitoring.CronJobLogName;
12  import ch.colabproject.colab.api.security.SessionManager;
13  import javax.ejb.Schedule;
14  import javax.ejb.Singleton;
15  import javax.ejb.Startup;
16  import javax.inject.Inject;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  
20  /**
21   * Do periodic tasks
22   *
23   * @author maxence
24   */
25  @Singleton
26  @Startup
27  public class CronTab {
28  
29      /** logger */
30      private static final Logger logger = LoggerFactory.getLogger(CronTab.class);
31  
32      /** session manager */
33      @Inject
34      private SessionManager sessionManager;
35  
36      /** To manager OpenGraph cache */
37      @Inject
38      private ExternalDataManager externalDataManager;
39  
40      /** To manage CronJobLogs */
41      @Inject
42      private CronJobLogManager cronJobLogManager;
43  
44      /**
45       * Each minute
46       */
47      @Schedule(hour = "*", minute = "*")
48      public void saveActivityDates() {
49          logger.trace("CRON: Persist activity dates to database");
50          sessionManager.writeActivityDatesToDatabase();
51          cronJobLogManager.updateCronJobLogLastRunTime(CronJobLogName.SAVE_ACTIVITIES_DATE);
52      }
53  
54      /**
55       * each midnight, clear expired sessions
56       */
57      @Schedule(hour = "0", minute = "0")
58      public void dropOldHttpSession() {
59          logger.info("CRON: drop expired http session");
60          sessionManager.clearExpiredHttpSessions();
61          cronJobLogManager.updateCronJobLogLastRunTime(CronJobLogName.DROP_OLD_HTTP_SESSIONS);
62      }
63  
64      /**
65       * each 00:30, clean outdated UrlMetadata
66       */
67      @Schedule(hour = "0", minute = "30")
68      public void dropOldUrlMetadata() {
69          logger.info("CRON: clean url metadata cache");
70          externalDataManager.clearOutdated();
71          cronJobLogManager.updateCronJobLogLastRunTime(CronJobLogName.DROP_OLD_URL_METADATA);
72      }
73  }
