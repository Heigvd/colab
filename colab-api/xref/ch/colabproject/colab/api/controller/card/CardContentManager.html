40      /**
41       * Card content persistence handler
42       */
43      @Inject
44      private CardContentDao cardContentDao;
45  
46      // *********************************************************************************************
47      // find card contents
48      // *********************************************************************************************
49  
50      /**
51       * Retrieve the card content. If not found, throw a {@link HttpErrorMessage}.
52       *
53       * @param cardContentId the id of the card content
54       *
55       * @return the card content if found
56       *
57       * @throws HttpErrorMessage if the card content was not found
58       */
59      public CardContent assertAndGetCardContent(Long cardContentId) {
60          CardContent cardContent = cardContentDao.getCardContent(cardContentId);
61  
62          if (cardContent == null) {
63              throw HttpErrorMessage.relatedObjectNotFoundError();
64          }
65  
66          return cardContent;
67      }
