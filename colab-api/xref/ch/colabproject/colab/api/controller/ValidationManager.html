7   package ch.colabproject.colab.api.controller;
8   
9   import ch.colabproject.colab.api.model.ColabEntity;
10  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
11  import ch.colabproject.colab.generator.model.exceptions.MessageI18nKey;
12  import java.util.Set;
13  import javax.ejb.LocalBean;
14  import javax.ejb.Stateless;
15  import javax.validation.ConstraintViolation;
16  import javax.validation.Validation;
17  import javax.validation.Validator;
18  import javax.validation.ValidatorFactory;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  /**
23   * Entity validation management
24   *
25   * @author sandra
26   */
27  @Stateless
28  @LocalBean
29  public class ValidationManager {
30  
31      /** logger */
32      private static final Logger logger = LoggerFactory.getLogger(ValidationManager.class);
33  
34      /**
35       * the validator factory
36       */
37      private ValidatorFactory factory = Validation.buildDefaultValidatorFactory();
38  
39      /**
40       * Ascertain that the given entity is valid according to its JPA constraints.
41       *
42       * @param <T>    type of the entity
43       * @param entity What we want to validate
44       *
45       * @throws HttpErrorMessage if the given entity is not valid
46       */
47      public <T extends ColabEntity> void assertValid(T entity) {
48          Validator validator = factory.getValidator();
49  
50          Set<ConstraintViolation<T>> validationErrors = validator.validate(entity);
51  
52          if (!validationErrors.isEmpty()) {
53              logger.info("{} not valid : {}", entity, validationErrors);
54              throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
55          }
56      }
57  
58  }
