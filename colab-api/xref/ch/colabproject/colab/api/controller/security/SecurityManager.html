83      /**
84       * Mark the current request as {@link RequestManager#setInSecurityTx(boolean) in-security-tx}
85       * and run the given action.
86       *
87       * @param action action to run
88       */
89      private void runInSecurityTx(Runnable action) {
90          requestManager.setInSecurityTx(true);
91          action.run();
92          requestManager.setInSecurityTx(false);
93      }

95      /**
96       * Assert the given condition is true
97       *
98       * @param condition the condition to check
99       * @param message   message to log in case the assertion failed
100      *
101      * @throws HttpErrorMessage
102      *                          <ul>
103      *                          <li>with authenticationRequired if assertion fails and current user
104      *                          is not authenticated;
105      *                          <li>with forbidden if the authenticated user does not have enough
106      *                          permission
107      *                          </ul>
108      */
109     @TransactionAttribute(TransactionAttributeType.REQUIRES_NEW)
110     public void assertConditionTx(Condition condition, String message) {
111         runInSecurityTx(() -> this.assertCondition(condition, message, null));
112     }
113 
114     /**
115      * Assert the given condition is true. If the current user is an admin, this assertion will
116      * never fail
117      *
118      * @param condition the condition to evaluate
119      * @param message   message to display if the assertion failed
120      * @param o         related object to log, may be null
121      *
122      * @throws HttpErrorMessage
123      *                          <ul>
124      *                          <li>with authenticationRequired if assertion fails and current user
125      *                          is not authenticated;
126      *                          <li>with forbidden if the authenticated user does not have enough
127      *                          permission
128      *                          </ul>
129      */
130     private void assertCondition(Condition condition, String message, WithPermission o) {
131         if (!requestManager.isAdmin()) {
132             requestManager.sudo(() -> {
133                 if (!condition.eval(requestManager, this)) {
134                     if (logger.isErrorEnabled()) {
135                         if (o != null) {
136                             logger.error("{} Permission denied: {} ({}) currentUser: {}",
137                                 message, o, condition, requestManager.getCurrentUser());
138                         } else {
139                             logger.error("{} Permission denied: ({}) currentUser: {}",
140                                 message, condition, requestManager.getCurrentUser());
141                         }
142                     }
143                     if (requestManager.isAuthenticated()) {
144                         throw HttpErrorMessage.forbidden();
145                     } else {
146                         throw HttpErrorMessage.authenticationRequired();
147                     }
148                 }
149             });
150         }
151     }

253     /**
254      * Has the current user read/write access to the given card
255      *
256      * @param card the card
257      *
258      * @return true if current user can write the card
259      */
260     public boolean hasReadWriteAccess(Card card) {
261         User currentUser = requestManager.getCurrentUser();
262         if (card == null || currentUser == null) {
263             return false;
264         }
265         TeamMember member = teamManager.findMemberByProjectAndUser(card.getProject(), currentUser);
266         return member != null && (member.getPosition() == HierarchicalPosition.OWNER
267             || teamManager.getEffectiveInvolvementLevel(card, member).isRw());
268     }

354     /**
355      * Has the current user the right to read the card type (/ reference) ?
356      * <p>
357      * A user can read
358      * <ul>
359      * <li>any global published card type</li>
360      * <li>any card type (/ reference) defined in a project he is member of</li>
361      * <li>and all the chain of targets of those card types references</li>
362      * </ul>
363      *
364      * @param cardTypeOrRefId the id of the card type or reference
365      *
366      * @return true if the current user can read the card type or reference
367      */
368     public boolean isCardTypeOrRefReadableByCurrentUser(Long cardTypeOrRefId) {
369         User currentUser = requestManager.getCurrentUser();
370         if (cardTypeOrRefId == null || currentUser == null) {
371             return false;
372         }
373 
374         List<Long> globalPublished = cardTypeManager.findGlobalPublishedCardTypeIds();
375         if (globalPublished.contains(cardTypeOrRefId)) {
376             return true;
377         }
378 
379         List<Long> accessibleFromProjects = cardTypeManager
380             .findCurrentUserReadableProjectsCardTypesIds();
381         if (accessibleFromProjects.contains(cardTypeOrRefId)) {
382             return true;
383         }
384 
385         return false;
386     }

388     /**
389      * Has the current user the right to read the project ?
390      * <p>
391      * A user can read any project he is a member of, has instance maker for or any project which
392      * contains a card type or reference the current user has a read access.
393      *
394      * @param projectId the id of the project
395      *
396      * @return True if the current user can read the project
397      */
398     public boolean isProjectReadableByCurrentUser(Long projectId) {
399         User currentUser = requestManager.getCurrentUser();
400 
401         if (projectId == null || currentUser == null) {
402             return false;
403         }
404 
405         List<Long> projectsWhereMemberOf = projectManager
406             .findIdsOfProjectsCurrentUserIsMemberOf();
407         if (projectsWhereMemberOf.contains(projectId)) {
408             return true;
409         }
410 
411         List<Long> projectsWhereInstanceMakerFor = projectManager
412             .findIdsOfProjectsCurrentUserIsInstanceMakerFor();
413         if (projectsWhereInstanceMakerFor.contains(projectId)) {
414             return true;
415         }
416 
417         List<Long> accessibleByCardTypes = projectManager
418             .findIdsOfProjectsReadableThroughCardTypes();
419         if (accessibleByCardTypes.contains(projectId)) {
420             return true;
421         }
422 
423         return false;
424     }
