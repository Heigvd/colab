84      /**
85       * Duplicate the given project. No database action is provided.
86       *
87       * @param originalProject  the project we want to duplicate
88       * @param resourceSpreader helper for resource references
89       *
90       * @return the duplicated project
91       */
92      public Project duplicateProject(Project originalProject,
93          ResourceReferenceSpreadingHelper resourceSpreader) {
94          try {
95              Project newProject = new Project();
96              newProject.duplicate(originalProject);
97  
98              ////////////////////////////////////////////////////////////////////////////////////////
99              // Team roles
100             if (params.isWithRoles()) {
101                 List<TeamRole> teamRoles = originalProject.getRoles();
102                 teamRoles.sort(ID_COMPARATOR);
103 
104                 for (TeamRole original : teamRoles) {
105                     TeamRole newTeamRole = duplicateTeamRole(original);
106 
107                     newTeamRole.setProject(newProject);
108                     newProject.getRoles().add(newTeamRole);
109                 }
110             } else {
111                 logger.info("param do not duplicate project's team roles");
112             }
113 
114             ////////////////////////////////////////////////////////////////////////////////////////
115             // Team members
116             if (params.isWithTeamMembers()) {
117                 List<TeamMember> teamMembers = originalProject.getTeamMembers();
118                 teamMembers.sort(ID_COMPARATOR);
119 
120                 for (TeamMember original : teamMembers) {
121                     TeamMember newTeamMember = duplicateTeamMember(original);
122 
123                     newTeamMember.setProject(newProject);
124                     newProject.getTeamMembers().add(newTeamMember);
125                 }
126             } else {
127                 logger.info("param do not duplicate project's team members");
128             }
129 
130             ////////////////////////////////////////////////////////////////////////////////////////
131             // Card types
132             if (params.isWithCardTypes()) {
133                 List<AbstractCardType> cardTypes = originalProject.getElementsToBeDefined();
134                 cardTypes.sort(ID_COMPARATOR);
135 
136                 for (AbstractCardType original : cardTypes) {
137                     AbstractCardType newCardType = duplicateCardType(original, resourceSpreader);
138 
139                     newCardType.setProject(newProject);
140                     newProject.getElementsToBeDefined().add(newCardType);
141                 }
142             }
143 
144             ////////////////////////////////////////////////////////////////////////////////////////
145             // Cards
146             if (params.isWithCardsStructure()) {
147                 Card original = originalProject.getRootCard();
148                 if (original != null) {
149                     Card newRootCard = duplicateCard(original);
150 
151                     newRootCard.setRootCardProject(newProject);
152                     newProject.setRootCard(newRootCard);
153                 }
154 
155             }
156 
157             ////////////////////////////////////////////////////////////////////////////////////////
158             // TODO sticky notes
159 //            if (params.isWithStickyNotes()) {
160 //
161 //            }
162 
163             ////////////////////////////////////////////////////////////////////////////////////////
164             // TODO activity flow
165 //            if (params.isWithActivityFlow()) {
166 //
167 //            }
168 
169             return newProject;
170         } catch (ColabMergeException e) {
171             throw HttpErrorMessage.dataIntegrityFailure();
172         }
173         // TODO sandra work in progress : handle exceptions
174     }

176     private TeamRole duplicateTeamRole(TeamRole original) throws ColabMergeException {
177         TeamRole newTeamRole = new TeamRole();
178         newTeamRole.duplicate(original);
179 
180         teamRoleMatching.put(original.getId(), newTeamRole);
181 
182         return newTeamRole;
183     }

185     private TeamMember duplicateTeamMember(TeamMember original) throws ColabMergeException {
186         TeamMember newTeamMember = new TeamMember();
187         newTeamMember.duplicate(original);
188 
189         teamMemberMatching.put(original.getId(), newTeamMember);
190 
191         ////////////////////////////////////////////////////////////////////////////////////////////
192         // Team members's user
193         User user = original.getUser();
194         if (user != null) {
195             newTeamMember.setUser(user);
196         }
197 
198         ////////////////////////////////////////////////////////////////////////////////////////////
199         // Team members's roles
200         if (params.isWithRoles()) {
201             List<TeamRole> linkedRoles = original.getRoles();
202             linkedRoles.sort(ID_COMPARATOR);
203 
204             for (TeamRole linkedRole : linkedRoles) {
205                 TeamRole newRole = teamRoleMatching.get(linkedRole.getId());
206 
207                 if (newRole == null) {
208                     throw HttpErrorMessage.dataIntegrityFailure();
209                 }
210 
211                 newRole.getMembers().add(newTeamMember);
212                 newTeamMember.getRoles().add(newRole);
213             }
214         }
215 
216         return newTeamMember;
217     }

219     private AbstractCardType duplicateCardType(AbstractCardType original,
220         ResourceReferenceSpreadingHelper resourceSpreader) throws ColabMergeException {
221 
222         if (params.isMakeOnlyCardTypeReferences()) {
223             CardTypeRef newCardTypeRef = new CardTypeRef();
224             newCardTypeRef.setTarget(original);
225             newCardTypeRef.setDeprecated(original.isDeprecated());
226             newCardTypeRef.setPublished(false);
227 
228             cardTypeMatching.put(original.getId(), newCardTypeRef);
229 
230             resourceSpreader.extractReferencesFromUp(newCardTypeRef);
231 
232             return newCardTypeRef;
233         } else {
234             AbstractCardType newAbstractCardType;
235             if (original instanceof CardType) {
236                 CardType originalCardType = (CardType) original;
237 
238                 CardType newCardType = new CardType();
239                 newCardType.duplicate(originalCardType);
240 
241                 cardTypeMatching.put(originalCardType.getId(), newCardType);
242 
243                 TextDataBlock purpose = originalCardType.getPurpose();
244                 if (purpose != null) {
245                     TextDataBlock newPurpose = (TextDataBlock) duplicateDocument(purpose);
246                     newPurpose.setPurposingCardType(newCardType);
247                     newCardType.setPurpose(newPurpose);
248                 }
249 
250                 newAbstractCardType = newCardType;
251             } else if (original instanceof CardTypeRef) {
252                 CardTypeRef originalCardTypeRef = (CardTypeRef) original;
253 
254                 CardTypeRef newCardTypeRef = new CardTypeRef();
255                 newCardTypeRef.duplicate(originalCardTypeRef);
256 
257                 cardTypeMatching.put(originalCardTypeRef.getId(), newCardTypeRef);
258 
259                 AbstractCardType originalTarget = originalCardTypeRef.getTarget();
260                 if (originalTarget != null) {
261                     if (originalTarget.getProjectId() != originalCardTypeRef.getProjectId()) {
262                         newCardTypeRef.setTarget(originalTarget);
263                     } else {
264                         throw new IllegalStateException(
265                             "the target of a card type reference must be outside the project");
266                         // Note for an hypothetical future evolution :
267                         // if we break the condition that, in a project,
268                         // there is only one reference per target type outside the project
269                         // we must process the card types in the appropriate order
270                     }
271                 }
272 
273                 newAbstractCardType = newCardTypeRef;
274             } else {
275                 throw new IllegalStateException("abstract card type implementation not handled");
276             }
277 
278             List<AbstractResource> originalResources = original.getDirectAbstractResources();
279             originalResources.sort(ID_COMPARATOR);
280 
281             for (AbstractResource originalResource : originalResources) {
282                 AbstractResource newResource = duplicateResource(originalResource);
283 
284                 newResource.setAbstractCardType(newAbstractCardType);
285                 newAbstractCardType.getDirectAbstractResources().add(newResource);
286             }
287 
288             return newAbstractCardType;
289         }
290     }

292     private Card duplicateCard(Card original) throws ColabMergeException {
293         Card newCard = new Card();
294         newCard.duplicate(original);
295 
296         cardMatching.put(original.getId(), newCard);
297 
298         AbstractCardType originalCardType = original.getCardType();
299         if (originalCardType != null) {
300             AbstractCardType newCardType = cardTypeMatching.get(originalCardType.getId());
301 
302             if (newCardType == null) {
303                 throw HttpErrorMessage.dataIntegrityFailure();
304             }
305 
306             newCard.setCardType(newCardType);
307         }
308 
309         List<AbstractResource> originalResources = original.getDirectAbstractResources();
310         originalResources.sort(ID_COMPARATOR);
311 
312         for (AbstractResource originalResource : originalResources) {
313             AbstractResource newResource = duplicateResource(originalResource);
314 
315             newResource.setCard(newCard);
316             newCard.getDirectAbstractResources().add(newResource);
317         }
318 
319         List<CardContent> originalContents = original.getContentVariants();
320         originalContents.sort(ID_COMPARATOR);
321 
322         for (CardContent originalCardContent : originalContents) {
323             CardContent newCardContent = duplicateCardContent(originalCardContent);
324 
325             newCardContent.setCard(newCard);
326             newCard.getContentVariants().add(newCardContent);
327         }
328 
329         ////////////////////////////////////////////////////////////////////////////////////////////
330         // Access control
331 
332         List<AccessControl> originalACL = original.getAccessControlList();
333         originalACL.sort(ID_COMPARATOR);
334 
335         for (AccessControl originalAccessControl : originalACL) {
336             if ((params.isWithTeamMembers() || originalAccessControl.getMember() == null)
337                 && (params.isWithRoles() || originalAccessControl.getRole() == null)) {
338 
339                 AccessControl newAccessControl = duplicateAccessControl(originalAccessControl);
340 
341                 if (newAccessControl != null) { // if we got a null, no need to duplicate
342                     newAccessControl.setCard(newCard);
343                     newCard.getAccessControlList().add(newAccessControl);
344                 }
345             }
346 
347         }
348 
349         return newCard;
350     }

352     private CardContent duplicateCardContent(CardContent original) throws ColabMergeException {
353         CardContent newCardContent = new CardContent();
354         newCardContent.duplicate(original);
355 
356         cardContentMatching.put(original.getId(), newCardContent);
357 
358         List<Card> originalSubCards = original.getSubCards();
359         originalSubCards.sort(ID_COMPARATOR);
360 
361         List<AbstractResource> originalResources = original.getDirectAbstractResources();
362         originalResources.sort(ID_COMPARATOR);
363 
364         for (AbstractResource originalResource : originalResources) {
365             AbstractResource newResource = duplicateResource(originalResource);
366 
367             newResource.setCardContent(newCardContent);
368             newCardContent.getDirectAbstractResources().add(newResource);
369         }
370 
371         for (Card originalSubCard : originalSubCards) {
372             Card newSubCard = duplicateCard(originalSubCard);
373 
374             newSubCard.setParent(newCardContent);
375             newCardContent.getSubCards().add(newSubCard);
376         }
377 
378         if (params.isWithDeliverables()) {
379             List<Document> originalDeliverables = original.getDeliverables();
380             originalDeliverables.sort(ID_COMPARATOR);
381 
382             for (Document originalDoc : originalDeliverables) {
383                 Document newDoc = duplicateDocument(originalDoc);
384 
385                 newDoc.setOwningCardContent(newCardContent);
386                 newCardContent.getDeliverables().add(newDoc);
387             }
388         }
389 
390         return newCardContent;
391     }

393     private Document duplicateDocument(Document original) throws ColabMergeException {
394         if (original instanceof DocumentFile) {
395             // TODO sandra work in progress : copy also the file
396             DocumentFile originalDocumentFile = (DocumentFile) original;
397             DocumentFile newDocumentFile = new DocumentFile();
398             newDocumentFile.duplicate(originalDocumentFile);
399             return newDocumentFile;
400         } else if (original instanceof ExternalLink) {
401             ExternalLink originalExternalLink = (ExternalLink) original;
402             ExternalLink newExternalLink = new ExternalLink();
403             newExternalLink.duplicate(originalExternalLink);
404             return newExternalLink;
405         } else if (original instanceof TextDataBlock) {
406             TextDataBlock originalTextDataBlock = (TextDataBlock) original;
407             TextDataBlock newTextDataBlock = new TextDataBlock();
408             newTextDataBlock.duplicate(originalTextDataBlock);
409             return newTextDataBlock;
410         } else {
411             throw new IllegalStateException("abstract card type implementation not handled");
412         }
413     }

415     private AbstractResource duplicateResource(AbstractResource original)
416         throws ColabMergeException {
417         if (original instanceof Resource) {
418             Resource originalResource = (Resource) original;
419 
420             Resource newResource = new Resource();
421             newResource.duplicate(originalResource);
422 
423             resourceMatching.put(originalResource.getId(), newResource);
424 
425             List<Document> originalDocuments = originalResource.getDocuments();
426             originalDocuments.sort(ID_COMPARATOR);
427 
428             for (Document originalDocument : originalDocuments) {
429                 Document newDocument = duplicateDocument(originalDocument);
430 
431                 newDocument.setOwningResource(newResource);
432                 newResource.getDocuments().add(newDocument);
433             }
434 
435             TextDataBlock teaser = originalResource.getTeaser();
436             if (teaser != null) {
437                 TextDataBlock newTeaser = (TextDataBlock) duplicateDocument(teaser);
438 
439                 newTeaser.setTeasingResource(newResource);
440                 newResource.setTeaser(newTeaser);
441             }
442 
443             return newResource;
444         } else if (original instanceof ResourceRef) {
445             ResourceRef originalResourceRef = (ResourceRef) original;
446 
447             ResourceRef newResourceRef = new ResourceRef();
448             newResourceRef.duplicate(originalResourceRef);
449 
450             resourceMatching.put(originalResourceRef.getId(), newResourceRef);
451 
452             AbstractResource originalTarget = originalResourceRef.getTarget();
453             if (originalTarget != null) {
454                 AbstractResource newTarget = originalTarget;
455 
456                 newTarget = resourceMatching.get(originalTarget.getId());
457 
458                 if (newTarget == null) {
459                     throw HttpErrorMessage.dataIntegrityFailure();
460                 }
461 
462                 newResourceRef.setTarget(newTarget);
463             }
464 
465             return newResourceRef;
466 
467         } else {
468             throw new IllegalStateException("abstract card type implementation not handled");
469         }
470     }

472     private AccessControl duplicateAccessControl(AccessControl original)
473         throws ColabMergeException {
474         AccessControl newAccessControl = new AccessControl();
475         newAccessControl.duplicate(original);
476 
477         if (original.getMember() != null) {
478             TeamMember member = teamMemberMatching.get(original.getMember().getId());
479 
480             if (member == null) {
481                 throw HttpErrorMessage.dataIntegrityFailure();
482             }
483 
484             member.getAccessControlList().add(newAccessControl);
485             newAccessControl.setMember(member);
486         }
487 
488         if (original.getRole() != null) {
489             TeamRole role = teamRoleMatching.get(original.getRole().getId());
490 
491             if (role == null) {
492                 throw HttpErrorMessage.dataIntegrityFailure();
493             }
494 
495             role.getAccessControl().add(newAccessControl);
496             newAccessControl.setRole(role);
497         }
498 
499         return newAccessControl;
500     }
