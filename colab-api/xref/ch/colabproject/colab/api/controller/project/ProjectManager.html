101     /**
102      * Retrieve the project. If not found, throw a {@link HttpErrorMessage}.
103      *
104      * @param projectId the id of the project
105      *
106      * @return the project if found
107      *
108      * @throws HttpErrorMessage if the project was not found
109      */
110     public Project assertAndGetProject(Long projectId) {
111         Project project = projectDao.findProject(projectId);
112 
113         if (project == null) {
114             logger.error("project #{} not found", projectId);
115             throw HttpErrorMessage.relatedObjectNotFoundError();
116         }
117 
118         return project;
119     }

140     /**
141      * Complete and persist the given project and add the current user to the project team
142      *
143      * @param project project to persist
144      *
145      * @return the new persisted project
146      */
147     public Project createProject(Project project) {
148         try {
149             return requestManager.sudo(() -> {
150                 logger.debug("Create project: {}", project);
151 
152                 initProject(project);
153 
154                 return projectDao.persistProject(project);
155             });
156         } catch (RuntimeException ex) {
157             throw ex;
158         } catch (Exception ex) {
159             return null;
160         }
161     }
162 
163     /**
164      * Initialize the project with a root card (if it does not have already one) and set the current
165      * user as a team member owner of the project
166      *
167      * @param project the project to fill
168      */
169     private void initProject(Project project) {
170         if (project.getRootCard() == null) {
171             Card rootCard = cardManager.initNewRootCard();
172 
173             project.setRootCard(rootCard);
174             rootCard.setRootCardProject(project);
175         }
176 
177         User user = securityManager.assertAndGetCurrentUser();
178         Optional<TeamMember> currentUserTeamMember = project.getTeamMembers().stream()
179             .filter(tm -> tm.getUserId() == user.getId()).findFirst();
180         if (currentUserTeamMember.isPresent()) {
181             currentUserTeamMember.get().setPosition(HierarchicalPosition.OWNER);
182         } else {
183             teamManager.addMember(project, user, HierarchicalPosition.OWNER);
184         }
185     }

187     /**
188      * Create a new project based on a model
189      *
190      * @param name        the name of the new project
191      * @param description the description of the new project
192      * @param modelId     the id of the model the new project is based on
193      *
194      * @return the new project
195      */
196     public Project createProjectFromModel(String name, String description, Long modelId) {
197         try {
198             return requestManager.sudo(() -> {
199                 Project project = duplicateProject(modelId,
200                     DuplicationParam.buildForCreationFromModel());
201 
202                 project.setName(name);
203                 project.setDescription(description);
204 
205                 return project;
206             });
207         } catch (RuntimeException ex) {
208             throw ex;
209         } catch (Exception ex) {
210             return null;
211         }
212     }

218     /**
219      * Duplicate the given project with the given parameters to fine tune the duplication
220      *
221      * @param projectId the id of the project to duplicate
222      * @param params    the parameters to fine tune the duplication
223      *
224      * @return the new project
225      */
226     public Project duplicateProject(Long projectId, DuplicationParam params) {
227         Project originalProject = assertAndGetProject(projectId);
228 
229         Project newProject = new DuplicationManager(params).duplicateProject(originalProject,
230             resourceReferenceSpreadingHelper);
231 
232         return createProject(newProject);
233     }

365     /**
366      * Check the integrity of the project
367      *
368      * @param project the project to check
369      *
370      * @return true iff the project is complete and safe
371      */
372     public boolean checkIntegrity(Project project) {
373         if (project == null) {
374             return false;
375         }
376 
377         if (project.getRootCard() == null) {
378             return false;
379         }
380 
381         List<TeamMember> teamMembers = new ArrayList<>(project.getTeamMembers());
382         if (teamMembers.stream().noneMatch(m -> m.getPosition() == HierarchicalPosition.OWNER)) {
383             return false;
384         }
385 
386         return true;
387     }
