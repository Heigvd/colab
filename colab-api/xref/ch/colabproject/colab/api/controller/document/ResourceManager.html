119     /**
120      * Retrieve the resource (or reference). If not found, throw a {@link HttpErrorMessage}.
121      *
122      * @param resourceOrRefId the id of the resource or reference
123      *
124      * @return the resource or reference if found
125      *
126      * @throws HttpErrorMessage if the resource or reference was not found
127      */
128     public AbstractResource assertAndGetResourceOrRef(Long resourceOrRefId) {
129         AbstractResource resourceOrRef = resourceAndRefDao.findResourceOrRef(resourceOrRefId);
130 
131         if (resourceOrRef == null) {
132             logger.error("resource or reference #{} not found", resourceOrRefId);
133             throw HttpErrorMessage.relatedObjectNotFoundError();
134         }
135 
136         return resourceOrRef;
137     }

139     /**
140      * Retrieve the resource. If not found, throw a {@link HttpErrorMessage}.
141      *
142      * @param resourceId the id of the resource
143      *
144      * @return the resource if found
145      *
146      * @throws HttpErrorMessage if the resource was not found
147      */
148     public Resource assertAndGetResource(Long resourceId) {
149         AbstractResource abstractResource = assertAndGetResourceOrRef(resourceId);
150 
151         if (!(abstractResource instanceof Resource)) {
152             throw HttpErrorMessage.relatedObjectNotFoundError();
153         }
154 
155         return (Resource) abstractResource;
156     }

163      * Retrieve the abstract resources directly linked to an abstract card type and all the chain of

217     /**
218      * @param directResourcesAndRefs List of resources or references
219      *
220      * @return for each resource / resource reference, retrieve the all chain from it until a
221      *         concrete resource
222      */
223     private List<List<AbstractResource>> expandCompleteChains(
224         List<AbstractResource> directResourcesAndRefs) {
225         return directResourcesAndRefs
226             .stream()
227             .map(resourceOrRef -> resourceOrRef.expand())
228             .collect(Collectors.toList());
229     }

235     /**
236      * Get the resources directly linked to the given project.
237      * <p>
238      * Does not fetch all chain references.
239      *
240      * @param projectId the id of the project
241      *
242      * @return resources directly linked to the given project
243      */
244     public List<AbstractResource> getDirectResourcesOfProject(Long projectId) {
245         logger.debug("get all the resources directly linked to the project #{}", projectId);
246 
247         Project project = projectManager.assertAndGetProject(projectId);
248 
249         List<Resourceable> allResourceables = new ArrayList<>();
250 
251         allResourceables.addAll(project.getElementsToBeDefined());
252         allResourceables.addAll(projectManager.getCards(projectId));
253         allResourceables.addAll(projectManager.getCardContents(projectId));
254 
255         List<AbstractResource> allDirectResources = new ArrayList<>();
256 
257         allDirectResources.addAll(allResourceables.stream().flatMap(card -> {
258             return card.getDirectAbstractResources().stream();
259         }).collect(Collectors.toList()));
260 
261         return allDirectResources;
262     }

268     /**
269      * Complete and persist the given resource and create all the references needed.
270      * <p>
271      * Every child of the entity acquires a reference to that resource.<br>
272      * And recursively the grand children acquires a reference to the reference of their parent.
273      *
274      * @param resource the resource to create
275      *
276      * @return the brand new resource
277      */
278     public Resource createResource(Resource resource) {
279         logger.debug("create resource {}", resource);
280 
281         if (resource.getTeaser() == null) {
282             TextDataBlock teaserTextDataBlock = blockManager.makeNewTextDataBlock();
283 
284             resource.setTeaser(teaserTextDataBlock);
285             teaserTextDataBlock.setTeasingResource(resource);
286         }
287 
288         // implicitly resource.setPublished(false);
289         // implicitly resource.setRequestingForGlory(false);
290         // implicitly resource.setDeprecated(false);
291 
292         Resourceable owner;
293         if (resource.getAbstractCardTypeId() != null) {
294             owner = cardTypeManager.assertAndGetCardTypeOrRef(resource.getAbstractCardTypeId());
295 
296         } else if (resource.getCardId() != null) {
297             owner = cardManager.assertAndGetCard(resource.getCardId());
298 
299         } else if (resource.getCardContentId() != null) {
300             owner = cardContentManager.assertAndGetCardContent(resource.getCardContentId());
301 
302         } else {
303             throw HttpErrorMessage.dataIntegrityFailure();
304         }
305 
306         resource.setOwner(owner);
307         owner.getDirectAbstractResources().add(resource);
308 
309         resourceReferenceSpreadingHelper.spreadNewResourceDown(resource);
310 
311         return resourceAndRefDao.persistResource(resource);
312     }

314     /**
315      * Delete the given resource
316      *
317      * @param resourceId the id of the resource to delete
318      */
319     public void deleteResource(Long resourceId) {
320         logger.debug("delete resource #{}", resourceId);
321 
322         AbstractResource resource = assertAndGetResourceOrRef(resourceId);
323 
324         if (!checkDeletionAcceptability(resource)) {
325             throw HttpErrorMessage.dataIntegrityFailure();
326         }
327 
328         deleteResourceAndRefs(resource);
329     }

331     /**
332      * Ascertain that the resource can be deleted.
333      *
334      * @param resourceOrRef the resource (or reference) to check for deletion
335      *
336      * @return True iff it can be safely deleted
337      */
338     private boolean checkDeletionAcceptability(AbstractResource resourceOrRef) {
339         // the only resources we delete manually are the concrete resources
340         if (!(resourceOrRef instanceof Resource)) {
341             return false;
342         }
343 
344         return true;
345     }

347     /**
348      * Delete each reference pointing at the given resourceOrRef and remove the resource from the
349      * card type / card / card content.
350      *
351      * @param resourceOrRef The initial abstract resource to delete
352      */
353     private void deleteResourceAndRefs(AbstractResource resourceOrRef) {
354         List<ResourceRef> references = resourceAndRefDao.findDirectReferences(resourceOrRef);
355         if (references != null) {
356             references.stream().forEach(ref -> deleteResourceAndRefs(ref));
357         }
358 
359         if (resourceOrRef.getOwner() != null) {
360             Resourceable owner = resourceOrRef.getOwner();
361             owner.getDirectAbstractResources().remove(resourceOrRef);
362         }
363 
364         resourceAndRefDao.deleteResourceOrRef(resourceOrRef.getId());
365 
366         // Note : the document is deleted by cascade
367     }

403     /**
404      * Add the document to the resource.
405      *
406      * @param resourceId      the id of the resource
407      * @param document        the document to use in the resource. It must be a new document
408      * @param relatedPosition to define the place where the document will be added in the resource
409      * @param neighbourDocId  the existing document which defines where the new document will be
410      *                        set. If relatedPosition is BEFOR or AFTER, it must be not null
411      *
412      * @return the newly created document
413      */
414     public Document addDocument(Long resourceId, Document document, RelatedPosition relatedPosition,
415         Long neighbourDocId) {
416         logger.debug("add document {} to resource #{} {} doc #{}", document, resourceId,
417             relatedPosition, neighbourDocId);
418 
419         Resource resource = assertAndGetResource(resourceId);
420 
421         if (document == null) {
422             throw HttpErrorMessage.dataIntegrityFailure();
423         }
424 
425         if (document.hasOwningResource() || document.hasOwningCardContent()) {
426             throw HttpErrorMessage.dataIntegrityFailure();
427         }
428 
429         if (resource.getDocuments().contains(document)) {
430             throw HttpErrorMessage.dataIntegrityFailure();
431         }
432 
433         switch (relatedPosition) {
434             case BEFORE:
435                 Document neighbourBDocument = documentManager.assertAndGetDocument(neighbourDocId);
436                 indexGenerator.moveItemBefore(document, neighbourBDocument,
437                     resource.getDocuments());
438                 break;
439             case AFTER:
440                 Document neighbourADocument = documentManager.assertAndGetDocument(neighbourDocId);
441                 indexGenerator.moveItemAfter(document, neighbourADocument,
442                     resource.getDocuments());
443                 break;
444             case AT_BEGINNING:
445                 indexGenerator.moveItemToBeginning(document, resource.getDocuments());
446                 break;
447             case AT_END:
448             default:
449                 indexGenerator.moveItemToEnd(document, resource.getDocuments());
450                 break;
451         }
452 
453         resource.getDocuments().add(document);
454         document.setOwningResource(resource);
455 
456         return documentDao.persistDocument(document);
457     }

459     /**
460      * Remove the document of the resource
461      *
462      * @param resourceId the id of the resource
463      * @param documentId the id of the document to remove
464      */
465     public void removeDocument(Long resourceId, Long documentId) {
466         logger.debug("remove document #{} of resource #{}", documentId, resourceId);
467 
468         Resource resource = assertAndGetResource(resourceId);
469 
470         Document document = documentManager.assertAndGetDocument(documentId);
471 
472         if (!(resource.getDocuments().contains(document))) {
473             throw HttpErrorMessage.dataIntegrityFailure();
474         }
475 
476         resource.getDocuments().remove(document);
477 
478         documentDao.deleteDocument(document.getId());
479     }

524     /**
525      * Check the integrity of the card type (or reference)
526      *
527      * @param resourceOrRef the resource (or reference) to check
528      *
529      * @return true iff the project is complete and safe
530      */
531     public boolean checkIntegrity(AbstractResource resourceOrRef) {
532         if (resourceOrRef == null) {
533             return false;
534         }
535 
536         if (resourceOrRef instanceof Resource) {
537             Resource resource = (Resource) resourceOrRef;
538             if (CollectionUtils.isEmpty(resource.getDocuments())) {
539                 return false;
540             }
541         }
542 
543         if (resourceOrRef instanceof ResourceRef) {
544             ResourceRef reference = (ResourceRef) resourceOrRef;
545             Resource finalTarget = reference.resolve();
546             if (finalTarget == null) {
547                 return false;
548             }
549         }
550 
551         int nbCardType = resourceOrRef.getAbstractCardType() == null ? 0 : 1;
552         int nbCard = resourceOrRef.getCard() == null ? 0 : 1;
553         int nbCardContent = resourceOrRef.getCardContent() == null ? 0 : 1;
554         if ((nbCardType + nbCard + nbCardContent) != 1) {
555             return false;
556         }
557 
558         return true;
559     }
