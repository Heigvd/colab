9   import ch.colabproject.colab.api.model.document.LexicalDataOwnershipKind;
10  import ch.colabproject.colab.api.setup.ColabConfiguration;
11  
12  import org.apache.hc.client5.http.classic.HttpClient;
13  import org.apache.hc.client5.http.classic.methods.HttpPost;
14  import org.apache.hc.client5.http.classic.methods.HttpGet;
15  import org.apache.hc.client5.http.impl.classic.HttpClientBuilder;
16  import org.apache.hc.core5.http.HttpMessage;
17  import org.apache.hc.core5.http.HttpResponse;
18  import org.apache.hc.core5.http.HttpStatus;
19  import org.apache.hc.core5.net.URIBuilder;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  /**
24   * To manage the data in the colab-yjs server (that is used to store the lexical
25   * text editor data).
26   *
27   * @author sandra
28   */
29  public class YjsLexicalCaller {
30      /** Logger */
31      private static final Logger logger = LoggerFactory.getLogger(YjsLexicalCaller.class);
32  
33      /** Duplicate URL */
34      private static final String DUPLICATE_URL = "duplicate";
35  
36      /** OwnerId parameter */
37      private static final String PARAM_OWNER_ID = "ownerId";
38  
39      /** Kind parameter */
40      private static final String PARAM_KIND = "kind";
41  
42      /** ToDuplicateId parameter */
43      private static final String PARAM_DUPLICATE_ID = "toDuplicateId";
44  
45      /** ToDuplicatedKind parameter */
46      private static final String PARAM_DUPLICATE_KIND = "toDuplicateKind";
47  
48      /** HTTP client */
49      private final HttpClient client;
50  
51      /** Internal YJS URL */
52      private final String internalUrl;
53  
54      // private String cookie;
55  
56      /**
57       * Constructor.
58       * <p>
59       * Initialize the client and base URL
60       */
61      public YjsLexicalCaller(/* String cookie */) {
62          this.client = HttpClientBuilder.create().build();
63          this.internalUrl = ColabConfiguration.getYjsInternalUrl();
64          // this.cookie = cookie;
65      }
66  
67      /**
68       * Send a request to duplicate a lexical data of a source owner to a destination
69       * owner
70       *
71       * @param srcOwnerId    the id of the original owner
72       * @param srcOwnerKind  the kind of the original owner
73       * @param destOwnerId   the id of the new owner
74       * @param destOwnerkind the kind of the new owner
75       */
76      public void sendDuplicationRequest(Long srcOwnerId, LexicalDataOwnershipKind srcOwnerKind,
77              Long destOwnerId, LexicalDataOwnershipKind destOwnerkind) {
78  
79          logger.trace("internal url : " + internalUrl);
80  
81          HttpResponse response = null;
82          try {
83              URIBuilder builder = new URIBuilder(internalUrl + "/" + DUPLICATE_URL);
84              builder.addParameter(PARAM_DUPLICATE_ID, Long.toString(srcOwnerId));
85              builder.addParameter(PARAM_DUPLICATE_KIND, srcOwnerKind.getKeyword());
86              builder.addParameter(PARAM_OWNER_ID, Long.toString(destOwnerId));
87              builder.addParameter(PARAM_KIND, destOwnerkind.getKeyword());
88  
89              HttpPost request = new HttpPost(builder.build());
90  
91              setHeaders(request);
92  
93              logger.debug("duplicate : " + request.getRequestUri());
94  
95              response = client.execute(request);
96  
97          } catch (Throwable cause) {
98              throw new YjsException(cause);
99          }
100 
101         if (response == null) {
102             throw new YjsException("No response");
103         }
104 
105         logger.debug("duplication return code " + response.getCode());
106 
107         if (response.getCode() >= HttpStatus.SC_CLIENT_ERROR) {
108             throw new YjsException(
109                     "code " + response.getCode() + " : " + response.getReasonPhrase());
110         }
111     }
112 
113     /**
114      * Set HTTP message headers
115      *
116      * @param msg the HTTP message
117      */
118     private void setHeaders(HttpMessage msg) {
119         msg.setHeader("Content-Type", "application/json");
120         msg.setHeader("Accept", "*/*");
121         // msg.setHeader("Cookie", cookie);
122         msg.setHeader("Managed-Mode", "false");
123         msg.setHeader("User-Agent", "coLAB");
124         // msg.setHeader("Authorization", auth);
125     }
126 }
