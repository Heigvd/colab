7   package ch.colabproject.colab.api.persistence.jpa.project;
8   
9   import ch.colabproject.colab.api.exceptions.ColabMergeException;
10  import ch.colabproject.colab.api.model.project.Project;
11  import ch.colabproject.colab.api.model.user.User;
12  import java.util.List;
13  import javax.ejb.LocalBean;
14  import javax.ejb.Stateless;
15  import javax.persistence.EntityManager;
16  import javax.persistence.PersistenceContext;
17  import javax.persistence.TypedQuery;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  /**
22   * Project persistence
23   * <p>
24   * Note : Most of database operations are handled by managed entities and cascade.
25   *
26   * @author maxence
27   */
28  @Stateless
29  @LocalBean
30  public class ProjectDao {
31  
32      /** logger */
33      private static final Logger logger = LoggerFactory.getLogger(ProjectDao.class);
34  
35      /**
36       * Access to the persistence unit
37       */
38      @PersistenceContext(unitName = "COLAB_PU")
39      private EntityManager em;
40  
41      /**
42       * Find a project by id
43       *
44       * @param id the id of the project to fetch
45       *
46       * @return the project with the given id or null is such a project does not exist
47       */
48      public Project findProject(Long id) {
49          logger.trace("find project #{}", id);
50  
51          return em.find(Project.class, id);
52      }
53  
54      /**
55       * Get the list of all project
56       *
57       * @return list of all projects
58       */
59      public List<Project> findAllProject() {
60          logger.trace("find all projects");
61  
62          TypedQuery<Project> query = em.createNamedQuery("Project.findAll", Project.class);
63  
64          return query.getResultList();
65      }
66  
67      /**
68       * Get all projects the user is a team member of
69       *
70       * @param userId the id of the user
71       *
72       * @return list of projects
73       */
74      public List<Project> findProjectsByTeamMember(Long userId) {
75          logger.trace("find the projects user #{} is team member of", userId);
76  
77          TypedQuery<Project> query = em.createNamedQuery("Project.findByTeamMemberUser",
78              Project.class);
79  
80          query.setParameter("userId", userId);
81  
82          return query.getResultList();
83      }
84  
85      /**
86       * Get the ids of the projects the user is a team member of
87       *
88       * @param userId the id of the user
89       *
90       * @return list of ids of projects
91       */
92      public List<Long> findProjectsIdsByTeamMember(Long userId) {
93          logger.trace("find the ids of the projects user #{} is team member of", userId);
94  
95          TypedQuery<Long> query = em.createNamedQuery("Project.findIdsByTeamMemberUser",
96              Long.class);
97  
98          query.setParameter("userId", userId);
99  
100         return query.getResultList();
101     }
102 
103     /**
104      * Get all projects the user is an instance maker for
105      *
106      * @param userId the id of the user
107      *
108      * @return list of projects
109      */
110     public List<Project> findProjectsByInstanceMaker(Long userId) {
111         logger.trace("find the projects user #{} is an instance maker for", userId);
112 
113         TypedQuery<Project> query = em.createNamedQuery("Project.findByInstanceMakerUser",
114             Project.class);
115 
116         query.setParameter("userId", userId);
117 
118         return query.getResultList();
119     }
120 
121     /**
122      * Get the ids of the models the user is an instance maker for
123      *
124      * @param userId the id of the user
125      *
126      * @return list of ids of models
127      */
128     public List<Long> findProjectsIdsByInstanceMaker(Long userId) {
129         logger.trace("find the ids of the projects user #{} is an instance maker for", userId);
130 
131         TypedQuery<Long> query = em.createNamedQuery("Project.findIdsByInstanceMakerUser",
132             Long.class);
133 
134         query.setParameter("userId", userId);
135 
136         return query.getResultList();
137     }
138 
139     /**
140      * Do two users work with a common project ?
141      *
142      * @param a a user
143      * @param b another user
144      *
145      * @return true if both user are both member or instance maker of the same project
146      */
147     public boolean findIfUsersHaveCommonProject(User a, User b) {
148         logger.trace("find if users {} and {} have comme project", a, b);
149 
150         TypedQuery<Boolean> query = em.createNamedQuery(
151             "Project.doUsersHaveACommonProject",
152             Boolean.class);
153 
154         query.setParameter("aUserId", a.getId());
155         query.setParameter("bUserId", b.getId());
156 
157         return !query.getResultList().isEmpty();
158     }
159 
160     /**
161      * Update project. Only fields which are editable by users will be impacted.
162      *
163      * @param project the project as supplied by clients (ie not managed by JPA)
164      *
165      * @return return updated managed project
166      *
167      * @throws ColabMergeException if the update failed
168      */
169     public Project updateProject(Project project) throws ColabMergeException {
170         logger.trace("update project {}", project);
171 
172         Project managedProject = this.findProject(project.getId());
173 
174         managedProject.merge(project);
175 
176         return managedProject;
177     }
178 
179     /**
180      * Persist a brand new project to database
181      *
182      * @param project the new project to persist
183      *
184      * @return the new persisted and managed project
185      */
186     public Project persistProject(Project project) {
187         logger.trace("persist project {}", project);
188 
189         em.persist(project);
190 
191         return project;
192     }
193 
194     /**
195      * Delete the project from database. This can't be undone
196      *
197      * @param project the project to delete
198      */
199     public void deleteProject(Project project) {
200         logger.trace("delete project {}", project);
201 
202         // TODO: move to recycle bin first
203 
204         em.remove(project);
205     }
206 
207 }
