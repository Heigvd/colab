7   package ch.colabproject.colab.api.persistence.jcr;
8   
9   import ch.colabproject.colab.api.model.project.Project;
10  import java.io.Serializable;
11  import java.util.HashMap;
12  import java.util.Map;
13  import javax.annotation.Resource;
14  import javax.inject.Inject;
15  import javax.jcr.Repository;
16  import javax.jcr.RepositoryException;
17  import javax.transaction.TransactionScoped;
18  import javax.transaction.TransactionSynchronizationRegistry;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  /**
23   * @author maxence
24   */
25  @TransactionScoped
26  public class JcrSessionManager implements Serializable {
27  
28      private static final long serialVersionUID = 1L;
29  
30      /** Logger */
31      private static final Logger logger = LoggerFactory.getLogger(JcrSessionManager.class);
32  
33      /**
34       * Opened session mapped by project id
35       */
36      private final Map<Long, JcrSession> sessions = new HashMap<>();
37  
38      /** Access to the repository */
39      @Inject
40      private JcrRepository jcrRepository;
41  
42      /**
43       * Tx sync registry
44       */
45      @Resource
46      private transient TransactionSynchronizationRegistry jtaSyncRegistry;
47  
48      /**
49       * Get a session to the project workspace. If the current transaction has already opened a
50       * session to this project, existing session is returned.
51       *
52       * @param project the project
53       *
54       * @return the session
55       *
56       * @throws javax.jcr.RepositoryException if something went wring
57       */
58      public JcrSession getSession(Project project) throws RepositoryException {
59          Long id = -1l; // hack: id -1 means global data
60          if (project != null){
61              id = project.getId();
62          }
63  
64          if (sessions.containsKey(id)) {
65              logger.trace("GetExisting session: project #{}", id);
66              return sessions.get(id);
67          } else {
68              logger.trace("Create new session: project #{}", id);
69              Repository repo = jcrRepository.getRepository();
70              JcrSession session = new JcrSession(repo, project);
71              sessions.put(id, session);
72  
73              JcrSynchronizer jcrSynchronizer = new JcrSynchronizer(session);
74              jtaSyncRegistry.registerInterposedSynchronization(jcrSynchronizer);
75  
76              return session;
77          }
78      }
79  }
