7   package ch.colabproject.colab.api.rest.config;
8   
9   import ch.colabproject.colab.api.rest.utils.ColabDeserializer;
10  import ch.colabproject.colab.generator.model.annotations.JsonbMapperProvider;
11  import javax.json.bind.Jsonb;
12  import javax.json.bind.JsonbBuilder;
13  import javax.json.bind.JsonbConfig;
14  import javax.ws.rs.ext.ContextResolver;
15  import javax.ws.rs.ext.Provider;
16  
17  /**
18   * convenient way to customize the JSON-B mapper
19   *
20   * @author Maxence
21   */
22  @Provider
23  public class JsonbProvider implements JsonbMapperProvider, ContextResolver<Jsonb> {
24  
25      /**
26       * {@inheritDoc}
27       */
28      @Override
29      public Jsonb getContext(Class<?> theClass) {
30          return JsonbProvider.getJsonb();
31      }
32  
33      /**
34       * Get a customized mapper
35       *
36       * @return an Jsonb object
37       */
38      public static Jsonb getJsonb() {
39          JsonbConfig config = new JsonbConfig()
40              .withFormatting(false)
41              //.withSerializers(new DateSerializer())
42              //.withDeserializers(new DateDeserializer())
43              .withDeserializers(new ColabDeserializer());
44  
45          return JsonbBuilder.create(config);
46      }
47  
48      @Override
49      public Jsonb getJsonbMapper() {
50          return JsonbProvider.getJsonb();
51      }
52  }
