7   package ch.colabproject.colab.api.rest.config;
8   
9   import ch.colabproject.colab.api.controller.config.ConfigurationManager;
10  import ch.colabproject.colab.api.rest.config.bean.ColabConfig;
11  import javax.inject.Inject;
12  import javax.ws.rs.Consumes;
13  import javax.ws.rs.GET;
14  import javax.ws.rs.Path;
15  import javax.ws.rs.Produces;
16  import javax.ws.rs.core.MediaType;
17  
18  /**
19   * REST MonitoringRestEndpoint to monitor the system
20   *
21   * @author maxence
22   */
23  @Path("config")
24  @Consumes(MediaType.APPLICATION_JSON)
25  @Produces(MediaType.APPLICATION_JSON)
26  public class ConfigRestEndpoint {
27  
28      /**
29       * Monitoring business logic
30       */
31      @Inject
32      private ConfigurationManager configManager;
33  
34      /**
35       * Get account-related configuration
36       *
37       * @return configuration
38       */
39      @GET
40      public ColabConfig getConfig() {
41          return configManager.getConfig();
42      }
43  
44  }
