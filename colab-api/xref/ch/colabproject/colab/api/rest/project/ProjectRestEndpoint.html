7   package ch.colabproject.colab.api.rest.project;
8   
9   import ch.colabproject.colab.api.controller.project.ProjectManager;
10  import ch.colabproject.colab.api.controller.team.TeamManager;
11  import ch.colabproject.colab.api.exceptions.ColabMergeException;
12  import ch.colabproject.colab.api.model.card.AbstractCardType;
13  import ch.colabproject.colab.api.model.card.Card;
14  import ch.colabproject.colab.api.model.card.CardContent;
15  import ch.colabproject.colab.api.model.link.ActivityFlowLink;
16  import ch.colabproject.colab.api.model.project.Project;
17  import ch.colabproject.colab.api.model.team.TeamMember;
18  import ch.colabproject.colab.api.model.team.TeamRole;
19  import ch.colabproject.colab.api.persistence.project.ProjectDao;
20  import ch.colabproject.colab.generator.model.annotations.AdminResource;
21  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
22  import java.util.List;
23  import java.util.Set;
24  import javax.inject.Inject;
25  import javax.ws.rs.Consumes;
26  import javax.ws.rs.DELETE;
27  import javax.ws.rs.GET;
28  import javax.ws.rs.POST;
29  import javax.ws.rs.PUT;
30  import javax.ws.rs.Path;
31  import javax.ws.rs.PathParam;
32  import javax.ws.rs.Produces;
33  import javax.ws.rs.core.MediaType;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  /**
38   * REST Project controller
39   *
40   * @author maxence
41   */
42  @Path("projects")
43  @Consumes(MediaType.APPLICATION_JSON)
44  @Produces(MediaType.APPLICATION_JSON)
45  @AuthenticationRequired
46  public class ProjectRestEndpoint {
47  
48      /** logger */
49      private static final Logger logger = LoggerFactory.getLogger(ProjectRestEndpoint.class);
50  
51      /**
52       * The Project business logic
53       */
54      @Inject
55      private ProjectManager projectManager;
56  
57      /** TeamMembers and roles management */
58      @Inject
59      private TeamManager teamManager;
60  
61      /**
62       * The Project DAO
63       */
64      @Inject
65      private ProjectDao projectDao;
66  
67      /**
68       * Retrieve the list of all projects. This is available to admin only
69       *
70       * @return all known project
71       */
72      @GET
73      @AdminResource
74      public List<Project> getAllProjects() {
75          logger.debug("Get all projects");
76          return projectDao.getAllProject();
77      }
78  
79      /**
80       * Get all projects the current user is member of
81       *
82       * @return list of projects
83       */
84      @GET
85      @Path("MyOwn")
86      public List<Project> getUserProjects() {
87          logger.debug("Get user projects");
88          return projectManager.findProjectsOfCurrentUser();
89      }
90  
91      /**
92       * Get project identified by the given id.
93       *
94       * @param id id of the project to fetch
95       *
96       * @return the project or null
97       */
98      @GET
99      @Path("/{id}")
100     public Project getProject(@PathParam("id") Long id) {
101         logger.debug("Get project #{}", id);
102         return projectDao.getProject(id);
103     }
104 
105     /**
106      * Create a new project and register current user as a member.
107      *
108      * @param project the project to persist
109      *
110      * @return id of the persisted new project
111      */
112     @POST
113     public Long createProject(Project project) {
114         logger.debug("Create project {}", project);
115         return projectManager.createProject(project).getId();
116     }
117 
118     /**
119      * Save changes to database.
120      *
121      * @param project project to update
122      *
123      * @throws ColabMergeException if the merge is not possible
124      */
125     @PUT
126     public void updateProject(Project project) throws ColabMergeException {
127         logger.debug("Update project {}", project);
128         projectDao.updateProject(project);
129     }
130 
131     /**
132      * Permanently delete a project.
133      *
134      * @param id id of the project to delete
135      */
136     @DELETE
137     @Path("/{id}")
138     public void deleteProject(@PathParam("id") Long id) {
139         logger.debug("Delete project #{}", id);
140         projectDao.deleteProject(id);
141     }
142 
143     /**
144      * Get the root card of the project
145      *
146      * @param projectId the id of the project
147      *
148      * @return the matching card
149      */
150     @GET
151     @Path("{projectId}/RootCard")
152     public Card getRootCardOfProject(@PathParam("projectId") Long projectId) {
153         logger.debug("get the root card of the project #{}", projectId);
154         return projectManager.getRootCard(projectId);
155     }
156 
157     /**
158      * Get all members of the project teams.
159      *
160      * @param id id of the project
161      *
162      * @return list of members
163      */
164     @GET
165     @Path("{id}/Members")
166     public List<TeamMember> getMembers(@PathParam("id") Long id) {
167         logger.debug("Get project #{} members", id);
168         return teamManager.getTeamMembers(id);
169     }
170 
171     /**
172      * Get all roles defined in a project
173      *
174      * @param projectId projectId of the project
175      *
176      * @return list of roles
177      */
178     @GET
179     @Path("{projectId: [0-9]+}/roles")
180     public List<TeamRole> getRoles(@PathParam("projectId") Long projectId) {
181         logger.debug("Get project #{} members", projectId);
182         return teamManager.getProjectRoles(projectId);
183     }
184 
185     /**
186      * Get all card types belonging to a project
187      *
188      * @param id ID of the project the card types belong to
189      *
190      * @return the card types linked to the project
191      */
192     @GET
193     @Path("{id}/CardTypes")
194     public Set<AbstractCardType> getCardTypesOfProject(@PathParam("id") Long id) {
195         logger.debug("Get project #{} card types", id);
196         return projectManager.getCardTypes(id);
197     }
198 
199     /**
200      * Get all cards belonging to a project
201      *
202      * @param id ID of the project the cards belong to
203      *
204      * @return the cards linked to the project
205      */
206     @GET
207     @Path("{id}/Cards")
208     public Set<Card> getCardsOfProject(@PathParam("id") Long id) {
209         logger.debug("Get project #{} cards", id);
210         return projectManager.getCards(id);
211     }
212 
213     /**
214      * Get all cardContents belonging to a project
215      *
216      * @param id ID of the project the cardContents belong to
217      *
218      * @return the cardContents linked to the project
219      */
220     @GET
221     @Path("{id}/CardContents")
222     public Set<CardContent> getCardContentsOfProject(@PathParam("id") Long id) {
223         logger.debug("Get project #{} cardContents", id);
224         return projectManager.getCardContents(id);
225     }
226 
227     /**
228      * Get all activityflowlinks belonging to a project
229      *
230      * @param id ID of the project activityflowlinks belong to
231      *
232      * @return all activityFlowLinks linked to the project
233      */
234     @GET
235     @Path("{id}/ActivityFlowLink")
236     public Set<ActivityFlowLink> getActivityFlowLinks(@PathParam("id") Long id) {
237         logger.debug("Get project #{} activityflowlinks", id);
238         return projectManager.getActivityFlowLinks(id);
239     }
240 }
