7   package ch.colabproject.colab.api.rest;
8   
9   import ch.colabproject.colab.api.ejb.ProjectFacade;
10  import ch.colabproject.colab.api.exceptions.ColabMergeException;
11  import ch.colabproject.colab.api.model.card.CardDef;
12  import ch.colabproject.colab.api.model.project.Project;
13  import ch.colabproject.colab.api.model.team.TeamMember;
14  import ch.colabproject.colab.api.persistence.project.ProjectDao;
15  import ch.colabproject.colab.generator.model.annotations.AdminResource;
16  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
17  import java.util.List;
18  import javax.inject.Inject;
19  import javax.ws.rs.Consumes;
20  import javax.ws.rs.DELETE;
21  import javax.ws.rs.GET;
22  import javax.ws.rs.POST;
23  import javax.ws.rs.PUT;
24  import javax.ws.rs.Path;
25  import javax.ws.rs.PathParam;
26  import javax.ws.rs.Produces;
27  import javax.ws.rs.core.MediaType;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  /**
32   * REST Project controller
33   *
34   * @author maxence
35   */
36  @Path("projects")
37  @Consumes(MediaType.APPLICATION_JSON)
38  @Produces(MediaType.APPLICATION_JSON)
39  @AuthenticationRequired
40  public class ProjectController {
41  
42      /** logger */
43      private static final Logger logger = LoggerFactory.getLogger(ProjectController.class);
44  
45      /**
46       * The Project business logic
47       */
48      @Inject
49      private ProjectFacade projectFacade;
50  
51      /**
52       * The Project DAO
53       */
54      @Inject
55      private ProjectDao projectDao;
56  
57      /**
58       * Retrieve the list of all projects. This is available to admin only
59       *
60       * @return all known project
61       */
62      @GET
63      @AdminResource
64      public List<Project> getAllProjects() {
65          logger.debug("Get all projects");
66          return projectDao.getAllProject();
67      }
68  
69      /**
70       * Get all projects the current user is member of
71       *
72       * @return list of projects
73       */
74      @GET
75      @Path("MyOwn")
76      public List<Project> getUserProjects() {
77          logger.debug("Get user projects");
78          return projectFacade.getCurrentUserProject();
79      }
80  
81      /**
82       * Get project identified by the given id.
83       *
84       * @param id id of the project to fetch
85       *
86       * @return the project or null
87       */
88      @GET
89      @Path("/{id}")
90      public Project getProject(@PathParam("id") Long id) {
91          logger.debug("Get project #{}", id);
92          return projectDao.getProject(id);
93      }
94  
95      /**
96       * Create a new project and register current user as a member.
97       *
98       * @param project the project to persist
99       *
100      * @return id of the persisted new project
101      */
102     @POST
103     public Long createProject(Project project) {
104         logger.debug("Create project {}", project);
105         return projectFacade.createNewProject(project).getId();
106     }
107 
108     /**
109      * Save changes to database.
110      *
111      * @param project project to update
112      *
113      * @throws ColabMergeException if the merge is not possible
114      */
115     @PUT
116     public void updateProject(Project project) throws ColabMergeException {
117         logger.debug("Update project {}", project);
118         projectDao.updateProject(project);
119     }
120 
121     /**
122      * Permanently delete a project.
123      *
124      * @param id id of the project to delete
125      */
126     @DELETE
127     @Path("/{id}")
128     public void deleteProject(@PathParam("id") Long id) {
129         logger.debug("Delete project #{}", id);
130         projectDao.deleteProject(id);
131     }
132 
133     /**
134      * Get all members of the project teams.
135      *
136      * @param id id of the project
137      *
138      * @return list of members
139      */
140     @GET
141     @Path("{id}/Members")
142     public List<TeamMember> getMembers(@PathParam("id") Long id) {
143         logger.debug("Get project #{} members", id);
144         return projectFacade.getTeamMembers(id);
145     }
146 
147     /**
148      * Send invitation to someone.
149      *
150      * @param projectId id of the project
151      * @param email recipient address
152      */
153     @POST
154     @Path("Invite/{projectId: [0-9]+}/{email}")
155     public void inviteSomeone(@PathParam("projectId") Long projectId,
156             @PathParam("email") String email) {
157         logger.debug("Invite {} to joint project #{}", email, projectId);
158         projectFacade.invite(projectId, email);
159     }
160 
161     /**
162      * Get all card definitions belonging to a project
163      *
164      * @param id ID of the project the card definitions belong to
165      *
166      * @return the card definitions linked to the project
167      */
168     @GET
169     @Path("{id}/CardDefs")
170     public List<CardDef> getCardDefsOfProject(@PathParam("id") Long id) {
171         logger.debug("Get project #{} card definitions", id);
172         return projectFacade.getCardDefs(id);
173     }
174 }
