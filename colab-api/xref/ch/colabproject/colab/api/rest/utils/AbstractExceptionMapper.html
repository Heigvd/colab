7   package ch.colabproject.colab.api.rest.utils;
8   
9   import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
10  import ch.colabproject.colab.generator.model.exceptions.HttpException;
11  import javax.ejb.EJBException;
12  import javax.validation.ConstraintViolationException;
13  import javax.ws.rs.NotFoundException;
14  import javax.ws.rs.core.MediaType;
15  import javax.ws.rs.core.Response;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  /**
20   * Intercept exception and send them back to client
21   *
22   * @author maxence
23   */
24  public class AbstractExceptionMapper {
25  
26      /** logger */
27      private static final Logger logger = LoggerFactory.getLogger(AbstractExceptionMapper.class);
28  
29      /**
30       *
31       * @param exception exception to process
32       *
33       * @return a HTTP response which wrap the exception
34       */
35      protected Response processException(Exception exception) {
36          if (exception == null) {
37              return Response.status(Response.Status.INTERNAL_SERVER_ERROR).build();
38          } else if (exception instanceof EJBException) {
39              return processException(((EJBException) exception).getCausedByException());
40          } else if (exception instanceof HttpException) {
41              return Response.status(((HttpException) exception).getHttpStatus())
42                  .type(MediaType.APPLICATION_JSON)
43                  .entity(exception)
44                  .build();
45          } else if (exception instanceof NotFoundException) {
46              return processException(HttpErrorMessage.notFound());
47          } else if (exception instanceof ConstraintViolationException) {
48              ConstraintViolationException constraintViolation
49                  = (ConstraintViolationException) exception;
50              logger.error("Constraint Violation Error");
51              constraintViolation.getConstraintViolations().forEach(violation -> {
52                  logger.error(" Violation: {}", violation);
53              });
54              return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
55                  .entity(constraintViolation.toString())
56                  .build();
57          } else {
58              logger.error("Unknown Internal Error", exception);
59              return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
60                  .entity(exception.toString())
61                  .build();
62          }
63      }
64  
65  }
