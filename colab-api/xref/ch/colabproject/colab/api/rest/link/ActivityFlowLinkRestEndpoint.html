26  /**
27   * REST activity flow link controller
28   *
29   * @author sandra
30   */
31  @Path("activityFlowLinks")
32  @Consumes(MediaType.APPLICATION_JSON)
33  @Produces(MediaType.APPLICATION_JSON)
34  @AuthenticationRequired
35  public class ActivityFlowLinkRestEndpoint {
36  
37      /** logger */
38      private static final Logger logger = LoggerFactory
39          .getLogger(ActivityFlowLinkRestEndpoint.class);
40  
41      // *********************************************************************************************
42      // injections
43      // *********************************************************************************************
44  
45      /**
46       * The activity flow link persistence manager
47       */
48      @Inject
49      private ActivityFlowLinkDao linkDao;
50  
51      /**
52       * The activity flow link related logic
53       */
54      @Inject
55      private ActivityFlowLinkManager linkManager;
56  
57      // *********************************************************************************************
58      //
59      // *********************************************************************************************
60  
61      /**
62       * Get link identified by the given id
63       *
64       * @param id id of the link to fetch
65       *
66       * @return the link or null
67       */
68      @GET
69      @Path("{id: [0-9]+}")
70      public ActivityFlowLink getLink(@PathParam("id") Long id) {
71          logger.debug("get activity flow link #{}", id);
72          return linkDao.findActivityFlowLink(id);
73      }

88      /**
89       * Persist the link
90       *
91       * @param link the link to persist
92       *
93       * @return id of the persisted new link
94       */
95      @POST
96      public Long createLink(ActivityFlowLink link) {
97          logger.debug("create activity flow link {}", link);
98          return linkManager.createActivityFlowLink(link).getId();
99      }
100 
101     /**
102      * Permanently delete a link
103      *
104      * @param id id of the link to delete
105      */
106     @DELETE
107     @Path("{id: [0-9]+}")
108     public void deleteLink(@PathParam("id") Long id) {
109         logger.debug("delete activity flow link #{}", id);
110         linkManager.deleteActivityFlowLink(id);
111     }
112 
113     /**
114      * Change the previous card
115      *
116      * @param linkId the id of the link to update
117      * @param cardId the id of the new previous card
118      */
119     @PUT
120     @Path("changePrevious/{linkId: [0-9]+}")
121     public void changePreviousCard(@PathParam("linkId") Long linkId, Long cardId) {
122         logger.debug("change activity flow link #{} previous card to #{}", linkId, cardId);
123         linkManager.changeActivityFlowLinkPrevious(linkId, cardId);
124     }
125 
126     /**
127      * Change the next card
128      *
129      * @param linkId the id of the link to update
130      * @param cardId the id of the new next card
131      */
132     @PUT
133     @Path("changeNext/{linkId: [0-9]+}")
134     public void changeNextCard(@PathParam("linkId") Long linkId, Long cardId) {
135         logger.debug("change activity flow link #{} next card to #{}", linkId, cardId);
136         linkManager.changeActivityFlowLinkNext(linkId, cardId);
137     }
