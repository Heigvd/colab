7   package ch.colabproject.colab.api.rest.document;
8   
9   import ch.colabproject.colab.api.microchanges.live.LiveManager;
10  import ch.colabproject.colab.api.microchanges.live.monitoring.BlockMonitoring;
11  import ch.colabproject.colab.api.microchanges.model.Change;
12  import ch.colabproject.colab.generator.model.annotations.AdminResource;
13  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
14  import java.util.List;
15  import javax.inject.Inject;
16  import javax.ws.rs.Consumes;
17  import javax.ws.rs.GET;
18  import javax.ws.rs.PUT;
19  import javax.ws.rs.Path;
20  import javax.ws.rs.PathParam;
21  import javax.ws.rs.Produces;
22  import javax.ws.rs.core.MediaType;
23  
24  /**
25   * REST micro changes controller
26   *
27   * @author maxence
28   */
29  @Path("changes")
30  @Consumes(MediaType.APPLICATION_JSON)
31  @Produces(MediaType.APPLICATION_JSON)
32  @AuthenticationRequired
33  public class ChangeRestEndpoint {
34  
35      /** logger */
36      // private static final Logger logger = LoggerFactory.getLogger(ChangesRestEndpoint.class);
37  
38      /**
39       * Live changes manager
40       */
41      @Inject
42      private LiveManager liveManager;
43  
44      /**
45       * Get all pending changes for given block.
46       *
47       * @param id id of the block
48       *
49       * @return list of changes.
50       */
51      @GET
52      @Path("{id: [0-9]+}/changes")
53      public List<Change> getChanges(@PathParam("id") Long id) {
54          return liveManager.getPendingChanges(id);
55      }
56  
57      /**
58       * Patch a block with a change
59       *
60       * @param id     id of the block
61       * @param change change
62       */
63      @PUT
64      @Path("{id: [0-9]+}")
65      public void patchBlock(@PathParam("id") Long id, Change change) {
66          liveManager.patchBlock(id, change);
67      }
68  
69      /**
70       * Patch a block with a change
71       *
72       * @param id id of the block
73       */
74      @PUT
75      @Path("{id: [0-9]+}/dropChanges")
76      public void deletePendingChanges(@PathParam("id") Long id) {
77          liveManager.deletePendingChangesAndPropagate(id);
78      }
79  
80      /**
81       * get data to monitor block with pending changes
82       *
83       * @return list of block being edited and their status
84       */
85      @GET
86      @Path("Monitoring")
87      @AdminResource
88      public List<BlockMonitoring> getMonitoringData() {
89          return liveManager.getMonitoringData();
90      }
91  
92  }
