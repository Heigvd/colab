29  /**
30   * REST Teams controller. Allow to manage roles and teams members
31   *
32   * @author maxence
33   * @author sandra
34   */
35  @Path("teams")
36  @Consumes(MediaType.APPLICATION_JSON)
37  @Produces(MediaType.APPLICATION_JSON)
38  @AuthenticationRequired
39  public class TeamRestEndpoint {
40  
41      /** logger */
42      private static final Logger logger = LoggerFactory.getLogger(TeamRestEndpoint.class);
43  
44      /** Team specific logic */
45      @Inject
46      private TeamManager teamManager;
47  
48      /** Assignment specific logic */
49      @Inject
50      private AssignmentManager assignmentManager;
51  
52      /** Team member persistence handling */
53      @Inject
54      private TeamMemberDao teamMemberDao;
55  
56      /** Team role persistence handling */
57      @Inject
58      private TeamRoleDao teamRoleDao;
59  
60      // *********************************************************************************************
61      // TeamMembers

63  
64      /**
65       * Get the members of the project team
66       *
67       * @param projectId id of the project
68       *
69       * @return list of team members
70       */
71      @GET
72      @Path("members/byproject/{projectId: [0-9]+}")
73      public List<TeamMember> getTeamMembersForProject(@PathParam("projectId") Long projectId) {
74          logger.debug("Get project #{} members", projectId);
75          return teamManager.getTeamMembersForProject(projectId);
76      }
77  
78      /**
79       * Get a TeamMember
80       *
81       * @param memberId id of the team member
82       *
83       * @return the team member
84       */
85      @GET
86      @Path("member/{memberId: [0-9]+}")
87      public TeamMember getTeamMember(@PathParam("memberId") Long memberId) {
88          logger.debug("Get member #{}", memberId);
89          return teamMemberDao.findTeamMember(memberId);
90      }
91  
92      /**
93       * Update a TeamMember. Only fields which are editable by users will be impacted.
94       *
95       * @param member new value
96       *
97       * @throws ColabMergeException if update failed
98       */
99      @PUT
100     @Path("member")
101     public void updateTeamMember(TeamMember member) throws ColabMergeException {
102         logger.debug("Update member {}", member);
103         teamMemberDao.updateTeamMember(member);
104     }
105 
106     /**
107      * Update hierarchical position of a member
108      *
109      * @param memberId id of the member
110      * @param position new hierarchical position
111      */
112     @PUT
113     @Path("member/{memberId: [0-9]+}/{position}")
114     public void changeMemberPosition(
115         @PathParam("memberId") Long memberId,
116         @PathParam("position") HierarchicalPosition position
117     ) {
118         teamManager.updatePosition(memberId, position);
119     }

121     /**
122      * Delete a team member
123      *
124      * @param memberId id of the member
125      */
126     @DELETE
127     @Path("member/{memberId: [0-9]+}")
128     public void deleteTeamMember(@PathParam("memberId") Long memberId) {
129         logger.debug("Delete team member #{}", memberId);
130         teamManager.deleteTeamMember(memberId);
131     }

137     /**
138      * Send invitation to someone.
139      *
140      * @param projectId id of the project
141      * @param email     recipient address
142      *
143      * @return the pending team member
144      */
145     @POST
146     @Path("Invite/{projectId: [0-9]+}/{email}")
147     public TeamMember inviteSomeone(@PathParam("projectId") Long projectId,
148                                     @PathParam("email") String email) {
149         logger.debug("Invite {} to joint project #{}", email, projectId);
150         return teamManager.invite(projectId, email);
151     }
152 
153     /**
154      * Create a token to share the project.
155      *
156      * @param projectId The id of the project that will become visible (mandatory)
157      * @param cardId    The id of the card that will become editable (optional)
158      *
159      * @return the URL to use to consume the token
160      */
161     @POST
162     @Path("sharingLink/generate/{projectId: [0-9]+}/{cardId: [0-9]+}")
163     public SerializationStringWrapper generateSharingLinkToken(@PathParam("projectId") Long projectId,
164                                                        @PathParam("cardId") Long cardId) {
165         String url = teamManager.generateSharingLinkToken(projectId, cardId);
166         return SerializationStringWrapper.build(url);
167     }
168 
169     /**
170      * Delete all sharing link tokens for the given project.
171      *
172      * @param projectId the id of the project
173      */
174     @DELETE
175     @Path("sharingLink/deleteByProject/{projectId: [0-9]+}")
176     public void deleteSharingLinkTokensByProject(@PathParam("projectId") Long projectId) {
177         teamManager.deleteSharingLinkTokensByProject(projectId);
178     }
179 
180     /**
181      * Delete all sharing link tokens for the given card.
182      *
183      * @param cardId the id of the card
184      */
185     @DELETE
186     @Path("sharingLink/deleteByCard/{cardId: [0-9]+}")
187     public void deleteSharingLinkTokensByCard(@PathParam("cardId") Long cardId) {
188         teamManager.deleteSharingLinkTokensByCard(cardId);
189     }

209     /**
210      * Create a role. The role must have a projectId set.
211      *
212      * @param role the role to create
213      *
214      * @return id of the new role
215      */
216     @POST
217     @Path("role")
218     public Long createRole(TeamRole role) {
219         logger.debug("Create role {}", role);

221         return role.getId();
222     }

230      */
231     @GET
232     @Path("role/{roleId: [0-9]+}")
233     public TeamRole getRole(@PathParam("roleId") Long roleId) {
234         logger.debug("Get Role #{}", roleId);

266     /**
267      * Give a role to a member. Member and role must belong to the same project. CurrentUser must
268      * have the right to edit the role
269      *
270      * @param roleId   id of the role
271      * @param memberId id of the team member
272      */
273     @PUT
274     @Path("role/{roleId: [0-9]+}/giveto/{memberId: [0-9]+}")
275     public void giveRoleTo(
276         @PathParam("roleId") Long roleId,
277         @PathParam("memberId") Long memberId
278     ) {
279         logger.debug("Give role #{} to member#{}", roleId, memberId);
280         teamManager.giveRole(roleId, memberId);
281     }
282 
283     /**
284      * Remove a role from some team member. CurrentUser must have the right to edit the role.
285      *
286      * @param roleId   id of the role
287      * @param memberId id of the team member
288      */
289     @PUT
290     @Path("role/{roleId: [0-9]+}/removeto/{memberId : [0-9]+}")
291     public void removeRoleFrom(
292         @PathParam("roleId") Long roleId,
293         @PathParam("memberId") Long memberId
294     ) {
295         logger.debug("Remove role #{} to member#{}", roleId, memberId);
296         teamManager.removeRole(roleId, memberId);
297     }

330     /**
331      * Add an assignment for a card and a member without involvement level

353     @Path("assignment/card/{cardId: [0-9]+}/member/{memberId: [0-9]+}/{level}")
354     public void setAssignment(
355         @PathParam("cardId") Long cardId,

369     @Path("assignment/card/{cardId: [0-9]+}/member/{memberId: [0-9]+}")
