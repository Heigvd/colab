7   package ch.colabproject.colab.api.ws.message;
8   
9   import javax.validation.constraints.NotNull;
10  import javax.websocket.Session;
11  
12  /**
13   * Message sent to each client as soon as they initialize the websocket connection.
14   *
15   * @author maxence
16   */
17  public class WsInitMessage extends WsMessage {
18  
19      private static final long serialVersionUID = 1L;
20  
21      /**
22       * Unique session ID TODO: check uniqueness through the cluster
23       */
24      @NotNull
25      private String sessionId;
26  
27      /**
28       * Default constructor
29       */
30      public WsInitMessage() {
31          /* no-op */
32      }
33  
34      /**
35       * create a message based on a websocket session
36       *
37       * @param session client session
38       */
39      public WsInitMessage(Session session) {
40          this.sessionId = session.getId();
41      }
42  
43      /**
44       * @return the socket ID to send to the client
45       */
46      public String getSessionId() {
47          return sessionId;
48      }
49  
50      /**
51       * Set session id
52       *
53       * @param sessionId session id
54       */
55      public void setSessionId(String sessionId) {
56          this.sessionId = sessionId;
57      }
58  
59  }
