150     /**
151      * Guess a Class based on its simple name.
152      *
153      * @param simpleName  simple name of the class
154      * @param reflections reflections stores
155      *
156      * @return class the class if found, null otherwise
157      */
158     private Class<? extends Serializable> getClassFromSimpleName(
159         String simpleName, Reflections reflections
160     ) {
161         Optional<Class<? extends Serializable>> any
162             = reflections.getSubTypesOf(Serializable.class).stream()
163                 .filter(type -> {
164                     return type.getSimpleName().equals(simpleName);
165                 }).findAny();
166         if (any.isPresent()) {
167             return any.get();
168         } else {
169             return null;
170         }
171     }

173     /**
174      * Process java doc and make it exportable
175      *
176      * @param javadoc     javadoc text
177      * @param reflections reflections store
178      *
179      * @return processed javadoc
180      */
181     private String processJavaDoc(String javadoc, boolean keepJava, Reflections reflections) {
182         Pattern atLink = Pattern.compile("\\{@link ([a-zA-Z]+)\\}");
183         Matcher atLinkMatcher = atLink.matcher(javadoc);
184         String atLinkProcessed = atLinkMatcher.replaceAll(match -> {
185             var klass = this.getClassFromSimpleName(match.group(1), reflections);
186             if (keepJava && klass != null) {
187                 return "{@link " + klass.getName() + " " + match.group(1) + "}";
188             } else {
189                 return match.group(1);
190             }
191         });
192 
193         Pattern throwsPattern = Pattern.compile("@throws ([a-zA-Z]+)");
194         Matcher throwsMatcher = throwsPattern.matcher(atLinkProcessed);
195 
196         return throwsMatcher.replaceAll(match -> {
197             Class<? extends Serializable> klass
198                 = this.getClassFromSimpleName(match.group(1), reflections);
199             if (klass != null && HttpException.class.isAssignableFrom(klass)) {
200                 return "@throws " + klass.getName();
201             } else {
202                 return "";
203             }
204         });
205     }

238     /**
239      * Try to imports given class. If the import is possible (ie no simple name collision), register
240      * the import in {@code imports} and return the simpleName. If importing the class is not
241      * possible, return the class fullname.
242      * <p>
243      * This method works for simple types, not for parametrized
244      *
245      * @param name    class full name
246      * @param imports map of simplename to fullname to be imported
247      *
248      * @return the name to use
249      */
250     private String resolveSimpleImport(String name, Map<String, String> imports) {
251         String[] split = name.split("\\.");
252         String simpleName = split[split.length - 1];
253 
254         if (name.startsWith("java.lang.")) {
255             imports.put(simpleName, null);
256             return simpleName;
257         }
258 
259         if (imports.containsKey(simpleName)) {
260             if (name.equals(imports.get(simpleName))) {
261                 return simpleName;
262             } else {
263                 // Same simple name, different packages -> use full name
264                 return name;
265             }
266         } else {
267             // first simpleName usage, register it
268             imports.put(simpleName, name);
269             return simpleName;
270         }
271 
272     }

274     /**
275      * Try to imports given class. If the import is possible (ie no simple name collision), register
276      * the import in {@code imports} and return the simpleName. If importing the class is not
277      * possible, return the class fullname.
278      * <p>
279      * This method works for simple types and for parametrized ones.
280      *
281      * @param name    class full name
282      * @param imports map of simplename to fullname to be imported
283      *
284      * @return the name to use
285      */
286     private String resolveImport(String name, Map<String, String> imports) {
287         // case 1: standard type eg java.lang.Long
288         // case 2: generic type eg. java.lang.List<java.lang.Long>
289 
290         if (name.contains("<")) {
291             int templateStart = name.indexOf('<');
292             int templateEnd = name.lastIndexOf('>');
293 
294             String type = name.substring(0, templateStart);
295             String leftPart = resolveSimpleImport(type, imports);
296 
297             String template = name.substring(templateStart + 1, templateEnd);
298             if (template.contains(",") && template.contains("<")) {
299                 // very complex template
300                 // eg <java.lang.Long, java.lang.List<java.lang.String>>
301                 // @TODO  not yet implemented
302                 Logger.warn("Very Complex generic type " + template);
303                 return leftPart + "<" + template + ">";
304             } else if (template.contains(",")) {
305                 // multiple simple parameters
306                 return leftPart + "<"
307                     + Arrays.stream(template.split(","))
308                         .map(item -> resolveImport(item, imports))
309                         .collect(Collectors.joining(","))
310                     + ">";
311             } else {
312                 // simple template or generic template -> simple recursive call
313                 return leftPart + "<" + resolveImport(template, imports) + ">";
314             }
315 
316         } else {
317             return resolveSimpleImport(name, imports);
318         }
319     }
