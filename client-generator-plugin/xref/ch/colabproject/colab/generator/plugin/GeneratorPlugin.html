7   package ch.colabproject.colab.generator.plugin;
8   
9   import org.apache.maven.plugin.AbstractMojo;
10  import org.apache.maven.plugin.MojoExecutionException;
11  import org.apache.maven.plugin.MojoFailureException;
12  import org.apache.maven.plugins.annotations.LifecyclePhase;
13  import org.apache.maven.plugins.annotations.Mojo;
14  import org.apache.maven.plugins.annotations.Parameter;
15  import org.apache.maven.plugins.annotations.ResolutionScope;
16  import org.apache.maven.project.MavenProject;
17  
18  /**
19   * Maven plugin to generate REST client. It allows to scan packages by reflections. It looks for
20   * Jakarta EE {@link javax.ws.rs.Path} annotations.
21   *
22   * @author maxence
23   */
24  @Mojo(
25      name = "generate",
26      defaultPhase = LifecyclePhase.GENERATE_SOURCES,
27      requiresDependencyResolution = ResolutionScope.COMPILE
28  )
29  public class GeneratorPlugin extends AbstractMojo {
30  
31      /**
32       * Package to extract REST method from
33       */
34      @Parameter(property = "generate.restPackages", required = true)
35      private String[] restPackages;
36  
37      /**
38       * Generate java client in this package
39       */
40      @Parameter(property = "generate.packageName", required = true)
41      private String packageName;
42  
43      /**
44       * Class name of the client to generate
45       */
46      @Parameter(property = "generate.clientName", required = true)
47      private String clientName;
48  
49      /**
50       * the maven project that will receive the generated clients
51       */
52      @Parameter(readonly = true, defaultValue = "${project}")
53      private MavenProject project;
54  
55      /**
56       * path of generated java client
57       */
58      @Parameter(
59          property = "generate.javaOutputDir",
60          defaultValue = "${project.build.directory}/generated-sources/javaClient"
61      )
62      private String javaOutputDir;
63  
64      /**
65       * path of generated typescript client
66       */
67      @Parameter(
68          property = "generate.tsOutputDir",
69          defaultValue = "${project.build.directory}/generated-sources/tsClient"
70      )
71      private String tsOutputDir;
72  
73      /**
74       * if true, do not generate any file but print classes to console
75       */
76      private boolean dryRun;
77  
78      /**
79       * Default constructor.
80       */
81      public GeneratorPlugin() {
82          this(false);
83      }
84  
85      /**
86       * DryRun constructor
87       *
88       * @param dryRun set dryRun value
89       */
90      private GeneratorPlugin(boolean dryRun) {
91          this.dryRun = dryRun;
92      }
93  
94      /**
95       * {@inheritDoc }
96       */
97      @Override
98      public void execute() throws MojoExecutionException, MojoFailureException {
99          Logger.initLogger(getLog());
100         Generator generator = new Generator(restPackages, packageName, clientName);
101 
102         generator.processPackages();
103 
104         // generate package in the output dir
105         generator.generateJavaClient(javaOutputDir, this.dryRun);
106 
107         generator.generateTypescriptClient(tsOutputDir, this.dryRun);
108 
109         if (!dryRun) {
110             // and add it to the current project to make sure the client
111             // can be use during this very run
112             this.project.addCompileSourceRoot(javaOutputDir);
113         }
114 
115         // @TODO generate typescript client
116         // generator.generateTypescriptClient();
117     }
118 
119     /**
120      * Simple app to test the generator.it will not generate any files but will print output
121      *
122      * @param args unused yet
123      *
124      * @throws org.apache.maven.plugin.MojoExecutionException if ...
125      * @throws org.apache.maven.plugin.MojoFailureException   if ...
126      */
127     public static void main(String... args) throws MojoExecutionException, MojoFailureException {
128         GeneratorPlugin generator = new GeneratorPlugin(true);
129         generator.packageName = "ch.colabproject.colab.generator";
130         generator.clientName = "TestClient";
131         String[] p = { "ch.colabproject.colab.api" };
132 
133         generator.restPackages = p;
134 
135         generator.execute();
136     }
137 }
