7   package ch.colabproject.colab.api.model;
8   
9   import ch.colabproject.colab.api.ws.channel.WebsocketChannel;
10  import ch.colabproject.colab.generator.model.interfaces.WithId;
11  import ch.colabproject.colab.generator.model.interfaces.WithJsonDiscriminator;
12  import ch.colabproject.colab.generator.model.tools.PolymorphicDeserializer;
13  import java.util.Set;
14  import javax.json.bind.annotation.JsonbTransient;
15  import javax.json.bind.annotation.JsonbTypeDeserializer;
16  
17  /**
18   * Depict an entity which can be propagated through websocket.
19   *
20   * @author maxence
21   */
22  @JsonbTypeDeserializer(PolymorphicDeserializer.class)
23  public interface WithWebsocketChannels extends WithId, WithJsonDiscriminator {
24  
25      /**
26       * Get the channels this entity shall be sent through.
27       *
28       * @return list of channels
29       */
30      @JsonbTransient
31      Set<WebsocketChannel> getChannels();
32  
33      /**
34       * Get the payload to embed within an IndexEntry
35       *
36       * @return the payload
37       */
38      @JsonbTransient
39      default Object getIndexEntryPayload() {
40          return null;
41      }
42  }
