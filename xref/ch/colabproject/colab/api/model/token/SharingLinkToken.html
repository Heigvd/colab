7   package ch.colabproject.colab.api.model.token;
8   
9   import ch.colabproject.colab.api.controller.token.TokenManager;
10  
11  import javax.json.bind.annotation.JsonbTransient;
12  import javax.persistence.Entity;
13  import javax.persistence.Index;
14  import javax.persistence.NamedQuery;
15  import javax.persistence.Table;
16  import javax.validation.constraints.NotNull;
17  
18  /**
19   * A token to share a project. The goal is to have a URL to send to people in
20   * order to allow them to view the project and take part of the edition of a specific card.
21   * <p>
22   * The user must be authenticated.
23   * <p>
24   * If the user is not yet a member of the project, he becomes a guest team member.
25   * <p>
26   * If a card is specified, the user will gain a support task allowing him to
27   * write in the card.
28   * <p>
29   * The link has no expiration. But it can be manually deleted.
30   *
31   * @author sandra
32   */
33  @Entity
34  @Table(indexes = {
35          @Index(columnList = "project_id"),
36          @Index(columnList = "card_id"),
37  })
38  @NamedQuery(name = "SharingLinkToken.findByProject", query = "SELECT t from SharingLinkToken t "
39          + "WHERE t.projectId = :projectId")
40  @NamedQuery(name = "SharingLinkToken.findByCard", query = "SELECT t from SharingLinkToken t "
41          + "WHERE t.cardId = :cardId")
42  public class SharingLinkToken extends Token {
43  
44      private static final long serialVersionUID = 1L;
45  
46      // ---------------------------------------------------------------------------------------------
47      // fields
48      // ---------------------------------------------------------------------------------------------
49  
50      /**
51       * The id of the project that becomes visible when the token is consumed
52       */
53      @NotNull
54      @JsonbTransient
55      private Long projectId;
56  
57      /**
58       * The id of the card that becomes writable when the token is consumed.
59       * <p>
60       * This is optional
61       */
62      @JsonbTransient
63      private Long cardId;
64  
65      // ---------------------------------------------------------------------------------------------
66      // getters and setters
67      // ---------------------------------------------------------------------------------------------
68  
69      /**
70       * @return the id of the project that becomes visible when the token is consumed
71       */
72      public Long getProjectId() {
73          return projectId;
74      }
75  
76      /**
77       * @param projectId the id of the project that becomes visible when the token is
78       *                  consumed
79       */
80      public void setProjectId(Long projectId) {
81          this.projectId = projectId;
82      }
83  
84      /**
85       * @return the id of the card that becomes writable when the token is consumed
86       */
87      public Long getCardId() {
88          return cardId;
89      }
90  
91      /**
92       * @param cardId the id of the card that becomes writable when the token is
93       *               consumed
94       */
95      public void setCardId(Long cardId) {
96          this.cardId = cardId;
97      }
98  
99      // ---------------------------------------------------------------------------------------------
100     // helpers
101     // ---------------------------------------------------------------------------------------------
102 
103     @Override
104     public String getRedirectTo() {
105         if (this.projectId != null) {
106             if (this.cardId != null) {
107                 // WARNING this path must be handled by front-end
108                 return "/project/" + projectId + "/card/" + cardId;
109             } else {
110                 // WARNING this path must be handled by front-end
111                 return "/project/" + projectId;
112             }
113         }
114         return "";
115     }
116 
117     @Override
118     public boolean consume(TokenManager tokenManager) {
119         return tokenManager.consumeSharingLinkToken(projectId, cardId);
120     }
121 
122     @Override
123     public ExpirationPolicy getExpirationPolicy() {
124         return ExpirationPolicy.NEVER;
125     }
126 
127     // ---------------------------------------------------------------------------------------------
128     // concerning the whole class
129     // ---------------------------------------------------------------------------------------------
130 
131     @Override
132     public String toString() {
133         return "SharingLinkToken{" + "id=" + getId()
134                 + ", deletion=" + getDeletionStatus() + '}';
135     }
136 }
