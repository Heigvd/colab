57      /** Header 1 */
58      private static final String MESSAGE_HEADING = "Please verify your email address.";

66      private VerifyLocalAccountMessageBuilder() {

70      /**
71       * Build a HTML body to send a message to verify the local account.
72       *
73       * @param token the verify local account token
74       * @param link  the ink in order to consume the token
75       *
76       * @return the HTML body of the message to send
77       */
78      public static String build(VerifyLocalAccountToken token, String link) {
79          String subHeading = MessageFormat.format(MESSAGE_SUBHEADING,
80              (token.getLocalAccount() != null && token.getLocalAccount().getUser() != null)
81                  ? token.getLocalAccount().getUser().getDisplayName()
82                  : "");
83  
84          return new TokenMessageBuilder(token)
85              .headTitle(MESSAGE_HEAD_TITLE)
86              .picture(MESSAGE_PICTURE)
87              .heading(MESSAGE_HEADING)
88              .subheading(subHeading)
89              // no info
90              .linkHref(link)
91              .linkLabel(MESSAGE_LINK_LABEL)
92              // let default footer
93              .build();
94      }
