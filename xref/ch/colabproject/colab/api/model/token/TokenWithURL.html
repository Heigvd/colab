11  /**
12   * Interface to extract the data needed to generate a URL.
13   * <p>
14   * When a URL is generated, the token hash data are changed.
15   *
16   * @author sandra
17   */
18  
19  public interface TokenWithURL {
20  
21      /**
22       * @return the token ID
23       */
24      Long getId();
25  
26      /**
27       * When the token is sent, its hash is changed
28       *
29       * @param hashMethod new value of hash method
30       */
31      void setHashMethod(HashMethod hashMethod);
32  
33      /**
34       * When the token is sent, its hash is changed
35       *
36       * @param hashedToken new value of hashed token
37       */
38      void setHashedToken(byte[] hashedToken);
39  }
