7   package ch.colabproject.colab.api.model.user;
8   
9   import static ch.colabproject.colab.api.model.user.User.USER_SEQUENCE_NAME;
10  import ch.colabproject.colab.api.model.WithPermission;
11  import ch.colabproject.colab.api.model.WithWebsocketChannels;
12  import ch.colabproject.colab.api.model.tools.EntityHelper;
13  import ch.colabproject.colab.api.security.permissions.Conditions;
14  import ch.colabproject.colab.api.ws.channel.tool.ChannelsBuilders.AboutAccountChannelsBuilder;
15  import ch.colabproject.colab.api.ws.channel.tool.ChannelsBuilders.ChannelsBuilder;
16  import ch.colabproject.colab.api.ws.channel.tool.ChannelsBuilders.ForAdminChannelsBuilder;
17  import ch.colabproject.colab.generator.model.interfaces.WithId;
18  import ch.colabproject.colab.generator.model.interfaces.WithJsonDiscriminator;
19  import java.time.OffsetDateTime;
20  import javax.json.bind.annotation.JsonbTransient;
21  import javax.persistence.Entity;
22  import javax.persistence.FetchType;
23  import javax.persistence.GeneratedValue;
24  import javax.persistence.GenerationType;
25  import javax.persistence.Id;
26  import javax.persistence.Index;
27  import javax.persistence.ManyToOne;
28  import javax.persistence.NamedQuery;
29  import javax.persistence.Table;
30  import javax.persistence.Transient;
31  import javax.validation.constraints.NotEmpty;
32  import javax.validation.constraints.Size;
33  
34  /**
35   * store session related information
36   *
37   * @author maxence
38   */
39  @Entity
40  @Table(
41      indexes = {
42          @Index(columnList = "account_id"),
43          @Index(columnList = "session_id", unique = true),}
44  )
45  @NamedQuery(
46      name = "HttpSession.getOlderThan",
47      query = "SELECT session FROM HttpSession session WHERE session.lastSeen < :time"
48  )
49  public class HttpSession implements WithId, WithJsonDiscriminator, WithPermission, WithWebsocketChannels {
50  
51      private static final long serialVersionUID = 1L;
52  
53      // ---------------------------------------------------------------------------------------------
54      // fields
55      // ---------------------------------------------------------------------------------------------
56  
57      /**
58       * Not so secret id
59       */
60      @Id
61      @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = USER_SEQUENCE_NAME)
62      private Long id;
63  
64      /**
65       * raw secret, never persisted, never serialized to client. This is just a temporary field to
66       * store the raw value to put in SET-COOKIE
67       */
68      @Size(max = 255)
69      @JsonbTransient
70      @Transient
71      private String rawSessionSecret;
72  
73      /**
74       * Session secret id. This value is hashed and is persisted in db.
75       * <p>
76       * DO NOT SERIALIZE IN JSON EVER
77       */
78      @JsonbTransient
79      @NotEmpty
80      private byte[] sessionSecret;
81  
82      /**
83       * Last activity date
84       */
85      private OffsetDateTime lastSeen;
86  
87      /**
88       * User Agent who create the session
89       */
90      @Size(max = 255)
91      private String userAgent;
92  
93      /**
94       * A HttpSession belongs to an account
95       */
96      @ManyToOne(optional = false, fetch = FetchType.LAZY)
97      @JsonbTransient
98      private Account account;
99  
100     /**
101      * id of the account used for authentication
102      */
103     @Transient
104     private Long accountId;
105 
106     // ---------------------------------------------------------------------------------------------
107     // getters and setters
108     // ---------------------------------------------------------------------------------------------
109 
110     /**
111      * @return account id
112      */
113     @Override
114     public Long getId() {
115         return id;
116     }
117 
118     /**
119      * set id
120      *
121      * @param id id
122      */
123     public void setId(Long id) {
124         this.id = id;
125     }
126 
127     /**
128      * Get the raw secret. This will almost always return null
129      *
130      * @return the raw secret or null
131      */
132     public String getRawSessionSecret() {
133         return rawSessionSecret;
134     }
135 
136     /**
137      * Set raw secret value
138      *
139      * @param rawSessionSecret the raw secret
140      */
141     public void setRawSessionSecret(String rawSessionSecret) {
142         this.rawSessionSecret = rawSessionSecret;
143     }
144 
145     /**
146      * @return Get the session id
147      */
148     public byte[] getSessionSecret() {
149         return sessionSecret;
150     }
151 
152     /**
153      * Set session id
154      *
155      * @param sessionSecret id of the session
156      */
157     public void setSessionSecret(byte[] sessionSecret) {
158         this.sessionSecret = sessionSecret;
159     }
160 
161     /**
162      * @return Get last activity date
163      */
164     public OffsetDateTime getLastSeen() {
165         return lastSeen;
166     }
167 
168     /**
169      * Set last activity date
170      *
171      * @param lastSeen lastSeen
172      */
173     public void setLastSeen(OffsetDateTime lastSeen) {
174         this.lastSeen = lastSeen;
175     }
176 
177     /**
178      * Get the value of userAgent
179      *
180      * @return the value of userAgent
181      */
182     public String getUserAgent() {
183         return userAgent;
184     }
185 
186     /**
187      * Set the value of userAgent
188      *
189      * @param userAgent new value of userAgent
190      */
191     public void setUserAgent(String userAgent) {
192         this.userAgent = userAgent;
193     }
194 
195     /**
196      * Get the account linked to this session. The only case the account may be null is on logout or
197      * when the session is going to be deleted
198      *
199      * @return authenticated account
200      */
201     public Account getAccount() {
202         return account;
203     }
204 
205     /**
206      * Set the account linked to this HttpSession
207      *
208      * @param account authenticated account
209      */
210     public void setAccount(Account account) {
211         this.account = account;
212     }
213 
214     /**
215      * @return authenticated account, null if not authenticated
216      */
217     public Long getAccountId() {
218         if (account != null) {
219             return account.getId();
220         } else {
221             return accountId;
222         }
223     }
224 
225     /**
226      * Set authenticated account id
227      *
228      * @param accountId id of the account account
229      */
230     public void setAccountId(Long accountId) {
231         this.accountId = accountId;
232     }
233 
234     // ---------------------------------------------------------------------------------------------
235     // concerning the whole class
236     // ---------------------------------------------------------------------------------------------
237 
238     @Override
239     public ChannelsBuilder getChannelsBuilder() {
240         if (this.account != null) {
241             return new AboutAccountChannelsBuilder(this.account);
242         } else {
243             return new ForAdminChannelsBuilder();
244         }
245     }
246 
247     @Override
248     @JsonbTransient
249     public Conditions.Condition getReadCondition() {
250         if (this.account != null) {
251             // same
252             return this.account.getReadCondition();
253         } else {
254             // not linked to any account, nothing to hide
255             // This case may only exist when the session is about to be destroyed
256             return Conditions.alwaysTrue;
257         }
258     }
259 
260     @Override
261     @JsonbTransient
262     public Conditions.Condition getUpdateCondition() {
263         if (this.account != null) {
264             // same
265             return this.account.getUpdateCondition();
266         } else {
267             // not linked to any account, nothing to hide
268             // This case may only exist when the session is about to be destroyed
269             return Conditions.alwaysTrue;
270         }
271     }
272 
273     @Override
274     @JsonbTransient
275     public Conditions.Condition getCreateCondition() {
276         // anyone can create a session
277         return Conditions.alwaysTrue;
278     }
279 
280     @Override
281     public int hashCode() {
282         return EntityHelper.hashCode(this);
283     }
284 
285     @Override
286     @SuppressWarnings("EqualsWhichDoesntCheckParameterClass")
287     public boolean equals(Object obj) {
288         return EntityHelper.equals(this, obj);
289     }
290 
291     @Override
292     public String toString() {
293         return "HttpSession{" + "id=" + id + ", account=" + account + ", lastSeen=" + lastSeen + '}';
294     }
295 
296 }
