7   package ch.colabproject.colab.api.model.user;
8   
9   import ch.colabproject.colab.api.Helper;
10  import java.security.NoSuchAlgorithmException;
11  import java.security.spec.InvalidKeySpecException;
12  import java.security.spec.KeySpec;
13  import javax.crypto.SecretKeyFactory;
14  import javax.crypto.spec.PBEKeySpec;
15  
16  /**
17   * Hash method supported by the coLAB platform
18   *
19   * @author maxence
20   */
21  public enum HashMethod {
22      /**
23       * PBKDF2 with HMAC SHA-512, 65536 iterations generates 64 bytes (ie. 512 bits as SHA-512 does)
24       */
25      PBKDF2WithHmacSHA512_65536_64 {
26  
27          /**
28           * {@inheritDoc }
29           */
30          @Override
31          public byte[] hash(String value, byte[] salt) {
32              try {
33                  KeySpec spec = new PBEKeySpec(value.toCharArray(), salt, 65536, 512);
34                  SecretKeyFactory factory = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA512");
35  
36                  return factory.generateSecret(spec).getEncoded();
37              } catch (NoSuchAlgorithmException | InvalidKeySpecException ex) {
38                  return null;
39              }
40          }
41      };
42  
43      /**
44       * compute digest from value.If given, the value is prefixed with the salt.
45       *
46       * @param value the value to hash
47       * @param salt  optional salt
48       *
49       * @return digested salted value
50       */
51      public abstract byte[] hash(String value, byte[] salt);
52  
53      /**
54       * @see #hash(java.lang.String, byte[])
55       * 
56       * @param value   value to hash
57       * @param hexSalt hexEncoded salt
58       *
59       * @return digested salted value
60       */
61      public byte[] hash(String value, String hexSalt) {
62          return this.hash(value, Helper.hextToBytes(hexSalt));
63      }
64  }
