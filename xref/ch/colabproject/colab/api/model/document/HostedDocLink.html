7   package ch.colabproject.colab.api.model.document;
8   
9   import ch.colabproject.colab.api.exceptions.ColabMergeException;
10  import ch.colabproject.colab.api.model.ColabEntity;
11  import ch.colabproject.colab.api.model.tools.EntityHelper;
12  import javax.persistence.DiscriminatorValue;
13  import javax.persistence.Entity;
14  
15  /**
16   * Document referenced by a link to a record of the co.LAB internal document-based database
17   *
18   * @author sandra
19   */
20  //TODO adjust the constraints / indexes
21  @Entity
22  @DiscriminatorValue("HOSTED_DOC_LINK")
23  public class HostedDocLink extends Document {
24  
25      private static final long serialVersionUID = 1L;
26  
27      // ---------------------------------------------------------------------------------------------
28      // fields
29      // ---------------------------------------------------------------------------------------------
30  
31      /**
32       * The path to access the document
33       */
34      private String path;
35  
36      // ---------------------------------------------------------------------------------------------
37      // getters and setters
38      // ---------------------------------------------------------------------------------------------
39  
40      /**
41       * @return the path to access the document
42       */
43      public String getPath() {
44          return path;
45      }
46  
47      /**
48       * @param path the path to access the document to set
49       */
50      public void setPath(String path) {
51          this.path = path;
52      }
53  
54      // ---------------------------------------------------------------------------------------------
55      // concerning the whole class
56      // ---------------------------------------------------------------------------------------------
57  
58      @Override
59      public void merge(ColabEntity other) throws ColabMergeException {
60          if (other instanceof HostedDocLink) {
61              HostedDocLink o = (HostedDocLink) other;
62              super.merge(o);
63              this.setPath(o.getPath());
64          } else {
65              throw new ColabMergeException(this, other);
66          }
67      }
68  
69      @Override
70      public int hashCode() {
71          return EntityHelper.hashCode(this);
72      }
73  
74      @Override
75      @SuppressWarnings("EqualsWhichDoesntCheckParameterClass")
76      public boolean equals(Object obj) {
77          return EntityHelper.equals(this, obj);
78      }
79  
80      @Override
81      public String toString() {
82          return "HostedDocLink{" + super.toPartialString() + ", path=" + path + "}";
83      }
84  
85  }
