7   package ch.colabproject.colab.api.model.tracking;
8   
9   import ch.colabproject.colab.generator.model.interfaces.WithJsonDiscriminator;
10  import java.time.OffsetDateTime;
11  import javax.json.bind.annotation.JsonbDateFormat;
12  import javax.persistence.Embeddable;
13  
14  /**
15   * To store the creation and modification dates as well as the user who initiate the action
16   *
17   * @author maxence
18   */
19  @Embeddable
20  public class Tracking implements WithJsonDiscriminator {
21  
22      private static final long serialVersionUID = 1L;
23  
24      /**
25       * Creation date
26       */
27      @JsonbDateFormat(value = JsonbDateFormat.TIME_IN_MILLIS)
28      private OffsetDateTime creationDate;
29  
30      /**
31       * Create by "username"
32       */
33      private String createdBy;
34  
35      /**
36       * Modification date
37       */
38      @JsonbDateFormat(value = JsonbDateFormat.TIME_IN_MILLIS)
39      private OffsetDateTime modificationDate;
40  
41      /**
42       * Modified by
43       */
44      private String modifiedBy;
45  
46      /**
47       * Get the creation date
48       *
49       * @return creation date
50       */
51      public OffsetDateTime getCreationDate() {
52          return creationDate;
53      }
54  
55      /**
56       * Set the creation date
57       *
58       * @param creationDate creation date
59       */
60      public void setCreationDate(OffsetDateTime creationDate) {
61          this.creationDate = creationDate;
62      }
63  
64      /**
65       * get created by
66       *
67       * @return created by
68       */
69      public String getCreatedBy() {
70          return createdBy;
71      }
72  
73      /**
74       * Set create by
75       *
76       * @param createdBy created by value
77       */
78      public void setCreatedBy(String createdBy) {
79          this.createdBy = createdBy;
80      }
81  
82      /**
83       * get modification date
84       *
85       * @return date of most recent modification
86       */
87      public OffsetDateTime getModificationDate() {
88          return modificationDate;
89      }
90  
91      /**
92       * set the date of most recent modification
93       *
94       * @param modificationDate the date
95       */
96      public void setModificationDate(OffsetDateTime modificationDate) {
97          this.modificationDate = modificationDate;
98      }
99  
100     /**
101      * Get name of last user who touch the entity
102      *
103      * @return username
104      */
105     public String getModifiedBy() {
106         return modifiedBy;
107     }
108 
109     /**
110      * Set the name of user who touch the entity
111      *
112      * @param modifiedBy username
113      */
114     public void setModifiedBy(String modifiedBy) {
115         this.modifiedBy = modifiedBy;
116     }
117 }
