7   package ch.colabproject.colab.api.model.card;
8   
9   import java.util.ArrayList;
10  import java.util.List;
11  import javax.json.bind.annotation.JsonbTransient;
12  import javax.persistence.Entity;
13  import javax.persistence.FetchType;
14  import javax.persistence.ManyToOne;
15  import javax.persistence.Transient;
16  
17  /**
18   * Reference to another existing abstract card type.
19   *
20   * @author maxence
21   */
22  @Entity
23  public class CardTypeRef extends AbstractCardType {
24  
25      /**
26       * Serial version UID
27       */
28      private static final long serialVersionUID = 1L;
29  
30      // ---------------------------------------------------------------------------------------------
31      // fields
32      // ---------------------------------------------------------------------------------------------
33      /**
34       * The abstract card type this reference aims at
35       */
36      @ManyToOne(fetch = FetchType.LAZY)
37      @JsonbTransient
38      private AbstractCardType target;
39  
40      /**
41       * The id of the abstract card type this reference aims at (serialization sugar)
42       */
43      @Transient
44      private Long targetId;
45  
46      // ---------------------------------------------------------------------------------------------
47      // getters and setters
48      // ---------------------------------------------------------------------------------------------
49      /**
50       * get the card type (or card type reference) this reference aims at
51       *
52       * @return the referenced type
53       */
54      public AbstractCardType getTarget() {
55          return this.target;
56      }
57  
58      /**
59       * @param target the card type (or card type reference) this reference aims at
60       */
61      public void setTarget(AbstractCardType target) {
62          this.target = target;
63      }
64  
65      /**
66       * get the id of the card type (or card type reference) this reference aims at. To be sent to client
67       *
68       * @return id of the abstractCardType or null
69       */
70      public Long getTargetId() {
71          if (this.target != null) {
72              return this.target.getId();
73          } else {
74              return targetId;
75          }
76      }
77  
78      /**
79       * set the id of the card type (or card type reference) this reference aims at. For serialization only
80       *
81       * @param id the id of the abstractCardType
82       */
83      public void setTargetId(Long id) {
84          this.targetId = id;
85      }
86  
87      // ---------------------------------------------------------------------------------------------
88      // concerning the whole class
89      // ---------------------------------------------------------------------------------------------
90      /**
91       * Resolve the reference.
92       *
93       * @return the abstractCardType
94       */
95      @Override
96      public CardType resolve() {
97          if (this.target instanceof CardType) {
98              return (CardType) this.target;
99          } else if (this.target instanceof CardTypeRef) {
100             return ((CardTypeRef) target).resolve();
101         }
102         return null;
103     }
104 
105     @Override
106     public List<AbstractCardType> expand() {
107         List<AbstractCardType> list = new ArrayList<>();
108 
109         list.add(this);
110         if (this.target != null) {
111             list.addAll(this.target.expand());
112         }
113         return list;
114     }
115 
116     @Override
117     public String toString() {
118         return "CardTypeRef{"
119             + "id=" + getId()
120             + ", targetId=" + targetId + '}';
121     }
122 }
