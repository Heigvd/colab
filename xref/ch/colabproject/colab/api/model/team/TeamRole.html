7   package ch.colabproject.colab.api.model.team;
8   
9   import static ch.colabproject.colab.api.model.team.TeamMember.TEAM_SEQUENCE_NAME;
10  import ch.colabproject.colab.api.exceptions.ColabMergeException;
11  import ch.colabproject.colab.api.model.ColabEntity;
12  import ch.colabproject.colab.api.model.WithWebsocketChannels;
13  import ch.colabproject.colab.api.model.common.Tracking;
14  import ch.colabproject.colab.api.model.project.Project;
15  import ch.colabproject.colab.api.model.team.acl.Assignment;
16  import ch.colabproject.colab.api.model.tools.EntityHelper;
17  import ch.colabproject.colab.api.security.permissions.Conditions;
18  import ch.colabproject.colab.api.ws.channel.tool.ChannelsBuilders.ChannelsBuilder;
19  import ch.colabproject.colab.api.ws.channel.tool.ChannelsBuilders.EmptyChannelBuilder;
20  import ch.colabproject.colab.api.ws.channel.tool.ChannelsBuilders.ProjectContentChannelBuilder;
21  import java.util.ArrayList;
22  import java.util.List;
23  import javax.json.bind.annotation.JsonbTransient;
24  import javax.persistence.CascadeType;
25  import javax.persistence.Embedded;
26  import javax.persistence.Entity;
27  import javax.persistence.FetchType;
28  import javax.persistence.GeneratedValue;
29  import javax.persistence.GenerationType;
30  import javax.persistence.Id;
31  import javax.persistence.Index;
32  import javax.persistence.ManyToMany;
33  import javax.persistence.ManyToOne;
34  import javax.persistence.OneToMany;
35  import javax.persistence.Table;
36  import javax.persistence.Transient;
37  import javax.validation.constraints.NotBlank;
38  import javax.validation.constraints.NotNull;
39  import javax.validation.constraints.Size;

41  /**
42   * A role within the development team. A role is used to group several member sharing same skills or
43   * objective within the project.
44   *
45   * @author maxence
46   */
47  @Entity
48  @Table(
49      indexes = {
50          @Index(columnList = "project_id"), }
51  )
52  public class TeamRole implements ColabEntity, WithWebsocketChannels {
53  
54      private static final long serialVersionUID = 1L;
55  
56      // ---------------------------------------------------------------------------------------------
57      // fields
58      // ---------------------------------------------------------------------------------------------
59      /**
60       * TeamRole ID.
61       */
62      @Id
63      @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = TEAM_SEQUENCE_NAME)
64      private Long id;
65  
66      /**
67       * creation + modification tracking data
68       */
69      @Embedded
70      private Tracking trackingData;
71  
72      /**
73       * Name of the role. Can not be null or blank
74       */
75      @Size(max = 255)
76      @NotBlank
77      private String name;
78  
79      /**
80       * The project
81       */
82      @ManyToOne(fetch = FetchType.LAZY)
83      @NotNull
84      @JsonbTransient
85      private Project project;
86  
87      /**
88       * The project ID (serialization sugar)
89       */
90      @Transient
91      private Long projectId;
92  
93      /**
94       * List of members who are part of this role
95       */
96      @ManyToMany(mappedBy = "roles")
97      @JsonbTransient
98      private List<TeamMember> members = new ArrayList<>();
99  
100     /**
101      * List of assignments relative to this role
102      */
103     @OneToMany(mappedBy = "role", cascade = CascadeType.ALL)
104     @JsonbTransient
105     private List<Assignment> assignments = new ArrayList<>();
106 
107     // ---------------------------------------------------------------------------------------------
108     // getters and setters
109     // ---------------------------------------------------------------------------------------------
110     /**
111      * @return the project ID
112      */
113     @Override
114     public Long getId() {
115         return id;
116     }
117 
118     /**
119      * Set id
120      *
121      * @param id id
122      */
123     public void setId(Long id) {
124         this.id = id;
125     }
126 
127     /**
128      * Get the tracking data
129      *
130      * @return tracking data
131      */
132     @Override
133     public Tracking getTrackingData() {
134         return trackingData;
135     }
136 
137     /**
138      * Set tracking data
139      *
140      * @param trackingData new tracking data
141      */
142     @Override
143     public void setTrackingData(Tracking trackingData) {
144         this.trackingData = trackingData;
145     }
146 
147     /**
148      * Get role name
149      *
150      * @return the name
151      */
152     public String getName() {
153         return name;
154     }
155 
156     /**
157      * Set the name
158      *
159      * @param name new role name
160      */
161     public void setName(String name) {
162         this.name = name;
163     }
164 
165     /**
166      * @return the project
167      */
168     public Project getProject() {
169         return project;
170     }
171 
172     /**
173      * @param project the project
174      */
175     public void setProject(Project project) {
176         this.project = project;
177     }
178 
179     /**
180      * get the project id. To be sent to client
181      *
182      * @return id of the project or null
183      */
184     public Long getProjectId() {
185         if (this.project != null) {
186             return this.project.getId();
187         } else {
188             return projectId;
189         }
190     }
191 
192     /**
193      * set the project id. For serialization only
194      *
195      * @param id the id of the project
196      */
197     public void setProjectId(Long id) {
198         this.projectId = id;
199     }
200 
201     /**
202      * Get members
203      *
204      * @return members
205      */
206     public List<TeamMember> getMembers() {
207         return members;
208     }
209 
210     /**
211      * Set the list of members
212      *
213      * @param members list of members
214      */
215     public void setMembers(List<TeamMember> members) {
216         this.members = members;
217     }
218 
219     /**
220      * Get the list of assignments
221      *
222      * @return assignments list
223      */
224     public List<Assignment> getAssignments() {
225         return assignments;
226     }
227 
228     /**
229      * Set the list of assignments
230      *
231      * @param assignments new list of assignments
232      */
233     public void setAssignments(List<Assignment> assignments) {
234         this.assignments = assignments;
235     }
236 
237     // ---------------------------------------------------------------------------------------------
238     // concerning the whole class
239     // ---------------------------------------------------------------------------------------------
240     @Override
241     public void merge(ColabEntity other) throws ColabMergeException {
242         if (other instanceof TeamRole) {
243             TeamRole o = (TeamRole) other;
244             this.setName(o.getName());
245         } else {
246             throw new ColabMergeException(this, other);
247         }
248     }
249 
250     @Override
251     public ChannelsBuilder getChannelsBuilder() {
252         if (this.getProject() != null) {
253             return new ProjectContentChannelBuilder(project);
254         } else {
255             return new EmptyChannelBuilder();
256         }
257     }
258 
259     @Override
260     @JsonbTransient
261     public Conditions.Condition getReadCondition() {
262         if (this.project != null) {
263             return new Conditions.IsCurrentUserMemberOfProject(this.project);
264         } else {
265             // should not exist
266             return Conditions.alwaysTrue;
267         }
268     }
269 
270     @Override
271     @JsonbTransient
272     public Conditions.Condition getUpdateCondition() {
273         if (this.project != null) {
274             return new Conditions.IsCurrentUserInternalToProject(this.project);
275         } else {
276             // should not exist
277             return Conditions.alwaysTrue;
278         }
279     }
280 
281     @Override
282     public int hashCode() {
283         return EntityHelper.hashCode(this);
284     }
285 
286     @Override
287     @SuppressWarnings("EqualsWhichDoesntCheckParameterClass")
288     public boolean equals(Object obj) {
289         return EntityHelper.equals(this, obj);
290     }
291 
292     @Override
293     public String toString() {
294         return "Role{" + "id=" + id + ", name=" + name + ", projectId=" + projectId + '}';
295     }
296 
297 }
