7   package ch.colabproject.colab.api.model.common;
8   
9   import ch.colabproject.colab.generator.model.annotations.ExtractJavaDoc;
10  import ch.colabproject.colab.generator.model.interfaces.WithJsonDiscriminator;
11  import javax.persistence.Column;
12  import javax.persistence.Embeddable;
13  import javax.persistence.EnumType;
14  import javax.persistence.Enumerated;
15  import javax.validation.constraints.NotNull;
16  import javax.validation.constraints.Size;
17  import org.apache.commons.lang3.builder.EqualsBuilder;
18  import org.apache.commons.lang3.builder.HashCodeBuilder;
19  
20  /**
21   * To store data of the illustration of some item.
22   * <p>
23   * It is a simple picture that illustrate a project or a card type.
24   *
25   * @author sandra
26   */
27  @Embeddable
28  @ExtractJavaDoc
29  public class Illustration implements WithJsonDiscriminator {
30  
31      private static final long serialVersionUID = 1L;
32  
33      // ---------------------------------------------------------------------------------------------
34      // fields
35      // ---------------------------------------------------------------------------------------------
36  
37      /**
38       * Which library of icons contains the key
39       */
40      @Column(length = 128)
41      @Enumerated(value = EnumType.STRING)
42      @NotNull
43      private IconLibrary iconLibrary = IconLibrary.FONT_AWESOME_SOLID;
44  
45      /**
46       * The key of the icon within the library of icons
47       */
48      @Size(max = 128)
49      @NotNull
50      private String iconKey;
51  
52      /**
53       * The background color to set behind the icon
54       */
55      @Size(max = 64)
56      @NotNull
57      private String iconBkgdColor;
58  
59      // ---------------------------------------------------------------------------------------------
60      // getters and setters
61      // ---------------------------------------------------------------------------------------------
62  
63      /**
64       * @return which library of icons contains the key
65       */
66      public IconLibrary getIconLibrary() {
67          return iconLibrary;
68      }
69  
70      /**
71       * @param iconLibrary which library of icons contains the key
72       */
73      public void setIconLibrary(IconLibrary iconLibrary) {
74          this.iconLibrary = iconLibrary;
75      }
76  
77      /**
78       * @return the key of the icon within the library of icons
79       */
80      public String getIconKey() {
81          return iconKey;
82      }
83  
84      /**
85       * @param iconKey the key of the icon within the library of icons
86       */
87      public void setIconKey(String iconKey) {
88          this.iconKey = iconKey;
89      }
90  
91      /**
92       * @return the background color to set behind the icon
93       */
94      public String getIconBkgdColor() {
95          return iconBkgdColor;
96      }
97  
98      /**
99       * @param iconBkgdColor the background color to set behind the icon
100      */
101     public void setIconBkgdColor(String iconBkgdColor) {
102         this.iconBkgdColor = iconBkgdColor;
103     }
104 
105     // ---------------------------------------------------------------------------------------------
106     // concerning the whole class
107     // ---------------------------------------------------------------------------------------------
108 
109     @Override
110     public int hashCode() {
111         return new HashCodeBuilder()
112             .append(this.iconLibrary)
113             .append(this.iconKey)
114             .append(this.iconBkgdColor)
115             .toHashCode();
116     }
117 
118     @Override
119     public boolean equals(Object obj) {
120         if (this == obj) {
121             return true;
122         }
123         if (obj == null) {
124             return false;
125         }
126         if (getClass() != obj.getClass()) {
127             return false;
128         }
129         final Illustration other = (Illustration) obj;
130         return new EqualsBuilder()
131             .append(this.iconLibrary, other.iconLibrary)
132             .append(this.iconKey, other.iconKey)
133             .append(this.iconBkgdColor, other.iconBkgdColor)
134             .isEquals();
135     }
136 
137     @Override
138     public String toString() {
139         return "Illustration{" + " iconLibrary=" + iconLibrary + ", iconKey=" + iconKey
140             + ", iconBkgdColor=" + iconBkgdColor + " }";
141     }
142 
143 }
