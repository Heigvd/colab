7   package ch.colabproject.colab.api.model.common;
8   
9   import ch.colabproject.colab.generator.model.annotations.ExtractJavaDoc;
10  import ch.colabproject.colab.generator.model.interfaces.WithJsonDiscriminator;
11  import ch.colabproject.colab.generator.model.tools.DateSerDe;
12  import java.time.OffsetDateTime;
13  import javax.json.bind.annotation.JsonbTypeDeserializer;
14  import javax.json.bind.annotation.JsonbTypeSerializer;
15  import javax.persistence.Embeddable;
16  import javax.validation.constraints.Size;
17  
18  /**
19   * To store the creation and modification dates as well as the user who initiate the action
20   *
21   * @author maxence
22   * @author sandra
23   */
24  @Embeddable
25  @ExtractJavaDoc
26  public class Tracking implements WithJsonDiscriminator {
27  
28      private static final long serialVersionUID = 1L;
29  
30      /**
31       * Creation timestamp
32       */
33      @JsonbTypeDeserializer(DateSerDe.class)
34      @JsonbTypeSerializer(DateSerDe.class)
35      private OffsetDateTime creationTime;
36  
37      /**
38       * Created by "username"
39       */
40      @Size(max = 255)
41      private String createdBy;
42  
43      /**
44       * Modification timestamp
45       */
46      @JsonbTypeDeserializer(DateSerDe.class)
47      @JsonbTypeSerializer(DateSerDe.class)
48      private OffsetDateTime modificationTime;
49  
50      /**
51       * Modified by "username"
52       */
53      @Size(max = 255)
54      private String modifiedBy;
55  
56      /**
57       * Erasure timestamp (when it was moved to bin)
58       */
59      @JsonbTypeDeserializer(DateSerDe.class)
60      @JsonbTypeSerializer(DateSerDe.class)
61      private OffsetDateTime erasureTime;
62  
63      /**
64       * Erased by "username" (who sent it to bin)
65       */
66      @Size(max = 255)
67      private String erasedBy;
68  
69      /**
70       * Get the timestamp of the creation
71       *
72       * @return the timestamp of the creation
73       */
74      public OffsetDateTime getCreationTime() {
75          return creationTime;
76      }
77  
78      /**
79       * Set the timestamp of the creation
80       *
81       * @param creationTime the timestamp of the creation
82       */
83      public void setCreationTime(OffsetDateTime creationTime) {
84          this.creationTime = creationTime;
85      }
86  
87      /**
88       * Get the name of the user who created the entity
89       *
90       * @return username
91       */
92      public String getCreatedBy() {
93          return createdBy;
94      }
95  
96      /**
97       * Set the name of the user who created the entity
98       *
99       * @param createdBy username
100      */
101     public void setCreatedBy(String createdBy) {
102         this.createdBy = createdBy;
103     }
104 
105     /**
106      * Get the timestamp of the most recent modification
107      *
108      * @return timestamp of the last change
109      */
110     public OffsetDateTime getModificationTime() {
111         return modificationTime;
112     }
113 
114     /**
115      * Set the timestamp of the most recent modification
116      *
117      * @param modificationTime timestamp of the last change
118      */
119     public void setModificationTime(OffsetDateTime modificationTime) {
120         this.modificationTime = modificationTime;
121     }
122 
123     /**
124      * Get the name of the last user who changed the entity
125      *
126      * @return username
127      */
128     public String getModifiedBy() {
129         return modifiedBy;
130     }
131 
132     /**
133      * Set the name of the last user who changed the entity
134      *
135      * @param modifiedBy username
136      */
137     public void setModifiedBy(String modifiedBy) {
138         this.modifiedBy = modifiedBy;
139     }
140 
141     /**
142      * Get the timestamp of the erasure of the entity (when it was moved to bin)
143      *
144      * @return the timestamp of the erasure
145      */
146     public OffsetDateTime getErasureTime() {
147         return erasureTime;
148     }
149 
150     /**
151      * Set the timestamp of the erasure of the entity (when it was moved to bin)
152      *
153      * @param erasureTime the timestamp of the erasure
154      */
155     public void setErasureTime(OffsetDateTime erasureTime) {
156         this.erasureTime = erasureTime;
157     }
158 
159     /**
160      * Get the name of the user who erased the entity (moved it to bin)
161      *
162      * @return username
163      */
164     public String getErasedBy() {
165         return erasedBy;
166     }
167 
168     /**
169      * Set the name of the user who erased the entity (moved it to bin)
170      *
171      * @param erasedBy username
172      */
173     public void setErasedBy(String erasedBy) {
174         this.erasedBy = erasedBy;
175     }
176 
177 }
