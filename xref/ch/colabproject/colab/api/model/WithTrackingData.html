7   package ch.colabproject.colab.api.model;
8   
9   import ch.colabproject.colab.api.model.common.Tracking;
10  import ch.colabproject.colab.api.model.user.User;
11  import java.time.OffsetDateTime;
12  
13  /**
14   * Simple interface which depicts objects having tracking data
15   *
16   * @author sandra
17   */
18  public interface WithTrackingData {
19  
20      /**
21       * Get the tracking data
22       *
23       * @return the tracking data
24       */
25      Tracking getTrackingData();
26  
27      /**
28       * Set the tracking data
29       *
30       * @param trackingData new tracking data
31       */
32      void setTrackingData(Tracking trackingData);
33  
34      /**
35       * Update tracking data
36       *
37       * @param user the current user
38       */
39      default void touch(User user) {
40          String username = user != null ? user.getUsername() : null;
41          OffsetDateTime now = OffsetDateTime.now();
42  
43          Tracking trackingData = getTrackingData();
44          if (trackingData == null) {
45              setTrackingData(new Tracking());
46              trackingData = getTrackingData();
47  
48              trackingData.setCreatedBy(username);
49              trackingData.setCreationTime(now);
50          }
51  
52          trackingData.setModifiedBy(username);
53          trackingData.setModificationTime(now);
54      }
55  
56      /**
57       * Update the erasure tracking data.
58       *
59       * @param user the current user
60       */
61      default void initErasureTrackingData(User user) {
62          String username = user != null ? user.getUsername() : null;
63          OffsetDateTime now = OffsetDateTime.now();
64  
65          Tracking trackingData = getTrackingData();
66  
67          if (trackingData != null) {
68              trackingData.setErasedBy(username);
69              trackingData.setErasureTime(now);
70          }
71      }
72  
73      /**
74       * Remove the erasure tracking data.
75       */
76      default void resetErasureTrackingData() {
77          Tracking trackingData = getTrackingData();
78  
79          if (trackingData != null) {
80              trackingData.setErasedBy(null);
81              trackingData.setErasureTime(null);
82          }
83      }
84  
85  }
