7   package ch.colabproject.colab.api.model;
8   
9   import ch.colabproject.colab.api.model.common.Tracking;
10  import ch.colabproject.colab.api.model.user.User;
11  import java.time.OffsetDateTime;
12  
13  /**
14   * Simple interface which depict objects having tracking data
15   *
16   * @author sandra
17   */
18  public interface WithTrackingData {
19  
20      /**
21       * Get the tracking data
22       *
23       * @return the tracking data
24       */
25      Tracking getTrackingData();
26  
27      /**
28       * Set the tracking data
29       *
30       * @param trackingData new tracking data
31       */
32      void setTrackingData(Tracking trackingData);
33  
34      /**
35       * Update tracking data
36       *
37       * @param user current user
38       */
39      default void touch(User user) {
40          Tracking trackingData = getTrackingData();
41          String username = user != null ? user.getUsername() : null;
42          OffsetDateTime now = OffsetDateTime.now();
43          if (trackingData == null) {
44              setTrackingData(new Tracking());
45              trackingData = getTrackingData();
46              trackingData.setCreatedBy(username);
47              trackingData.setCreationDate(now);
48          }
49          trackingData.setModifiedBy(username);
50          trackingData.setModificationDate(now);
51      }
52  
53  }
