11  /**
12   * Thrown when a merge is not possible
13   *
14   * @author maxence
15   */
16  public class ColabMergeException extends Exception {
17  
18      private static final long serialVersionUID = 1L;
19  
20      /**
21       * The entity to update
22       */
23      private final ColabEntity supplier;
24  
25      /**
26       * The entity to take new values from
27       */
28      private final ColabEntity receiver;
29  
30      /**
31       * Create a merge exception
32       *
33       * @param receiver the object to update
34       * @param supplier the object which contains new values
35       */
36      public ColabMergeException(ColabEntity/../../../ch/colabproject/colab/api/model/ColabEntity.html#ColabEntity">ColabEntity receiver, ColabEntity supplier) {
37          this.receiver = receiver;
38          this.supplier = supplier;
39      }
40  
41      /**
42       * {@inheritDoc }
43       */
44      @Override
45      public String getMessage() {
46          return "Impossible to patch " + this.receiver + " with " + this.supplier;
47      }
48  }
