7   package ch.colabproject.colab.api.controller;
8   
9   import javax.transaction.Status;
10  import javax.transaction.Synchronization;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  /**
15   * To synchronize websockets with JTA session
16   *
17   * @author maxence
18   */
19  public class WebsocketTxSync implements Synchronization {
20  
21      /** logger */
22      private static final Logger logger = LoggerFactory.getLogger(WebsocketTxSync.class);
23  
24      /**
25       * the TransactionManaget bound to this synchronizer
26       */
27      private final EntityGatheringBagForPropagation context;
28  
29      /**
30       * Make sure to "beforeCompletion" only once
31       */
32      private boolean beforeDone = false;
33  
34      /**
35       * Make sure to "afterCompletion" only once
36       */
37      private boolean afterDone = false;
38  
39      /**
40       * New Synchronizer to link to the current transaction.
41       *
42       * @param context TranssactionManager which is in charge for the current transaction
43       */
44      public WebsocketTxSync(EntityGatheringBagForPropagation context) {
45          this.context = context;
46      }
47  
48      /**
49       * At this point, the final state of the transaction in known.
50       * <p>
51       * From this point, any committed changes will be available to other transactions, and any
52       * rolled back changes will be definitely loosed.
53       *
54       * @param status status of the current transaction, either STATUS_COMMITTED or STATUS_ROLLEDBACK
55       */
56      @Override
57      public void afterCompletion(int status) {
58          if (!afterDone) {
59              afterDone = true;
60              logger.trace("afterCompletion with status: {}", status);
61              context.afterCompletion(status);
62          }
63      }
64  
65      /**
66       * This is called before just before the commit occurs.
67       * <p>
68       * A RuntimeException may be thrown if there is any problem with the underling Xapi repository.
69       * If so, the transaction will be rolled back
70       */
71      @Override
72      public void beforeCompletion() {
73          if (!beforeDone) {
74              beforeDone = true;
75              logger.trace("beforeCompletion");
76              // make sure the commit will success or throw some Runtimeexception
77              context.prepare();
78          }
79      }
80  
81      /**
82       * Force sending websocket messages now
83       */
84      public void flush() {
85          this.beforeCompletion();
86          this.afterCompletion(Status.STATUS_COMMITTED);
87      }
88  }
