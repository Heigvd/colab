7   package ch.colabproject.colab.api.controller.document;
8   
9   /**
10   * Exception when trying to call the colab-yjs server (that is used to store the lexical text editor
11   * data)
12   *
13   * @author sandra
14   */
15  public class YjsException extends RuntimeException {
16  
17      private static final long serialVersionUID = 1L;
18  
19      /**
20       * Exception when trying to call the colab-yjs server.
21       *
22       * @param message the detail message (which is saved for later retrieval by the
23       *                {@link #getMessage()} method).
24       */
25      public YjsException(String message) {
26          super(message);
27      }
28  
29      /**
30       * Exception when trying to call the colab-yjs server.
31       *
32       * @param message the detail message (which is saved for later retrieval by the
33       *                {@link #getMessage()} method).
34       * @param cause   the cause (which is saved for later retrieval by the {@link #getCause()}
35       *                method). (A {@code null} value is permitted, and indicates that the cause is
36       *                nonexistent or unknown.)
37       */
38      public YjsException(String message, Throwable cause) {
39          super(message, cause);
40      }
41  
42      /**
43       * Exception when trying to call the colab-yjs server.
44       *
45       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()}
46       *              method). (A {@code null} value is permitted, and indicates that the cause is
47       *              nonexistent or unknown.)
48       */
49      public YjsException(Throwable cause) {
50          super(cause);
51      }
52  }
