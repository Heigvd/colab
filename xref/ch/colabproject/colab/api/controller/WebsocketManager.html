203     /**
204      * Get list of all occupied channels.
205      * <p>
206      * This method is cluster-aware. In short, {@link #getSubscriptionsCount() } will be called for
207      * each instance of the cluster.
208      *
209      * @return list of all occupied channels URN mapped to the number of subscribers
210      */
211     public Map<String, Integer> getExistingChannels() {
212         IExecutorService executorService = hzInstance.getExecutorService("COLAB_WS");
213         Map<Member, Future<Map<String, Integer>>> results = executorService
214             .submitToAllMembers(new CallableGetChannel());
215 
216         Map<String, Integer> map = new HashMap<>();
217 
218         results.values().stream()
219             .flatMap((result) -> {
220                 try {
221                     return result.get(5, TimeUnit.SECONDS).entrySet().stream();
222                 } catch (InterruptedException | ExecutionException | TimeoutException ex) {
223                     return null;
224                 }
225             })
226             .forEach(entry -> {
227                 String key = entry.getKey();
228                 var currentCount = map.get(entry.getKey());
229                 if (currentCount != null) {
230                     map.put(key, currentCount + entry.getValue());
231                 } else {
232                     map.put(key, entry.getValue());
233                 }
234             });
235 
236         return map;
237     }

424     /**
425      * Process subscription request
426      *
427      * @param request the subscription request
428      */
429     public void processSubscription(
430         @Observes @Inbound(eventName = WS_SUBSCRIPTION_EVENT_CHANNEL) SubscriptionRequest request) {
431         // all security checks have been done before firing the subscription event
432         requestManager.sudo(() -> {
433             logger.debug("Channel subscription request: {}", request);
434             Session session = WebsocketEndpoint.getSession(request.getWsSessionId());
435             if (session != null) {
436 
437                 logger.debug("Process channel subscription request: {}", request);
438 
439                 // first determine the effective channel
440                 WebsocketChannel channel = getChannel(request);
441                 if (channel != null) {
442                     synchronized (this) {
443                         // make sure the http session has its own set of wsSessions
444                         // and make sure the websocket session is linked to the http session
445                         httpSessionToWsSessions
446                             .computeIfAbsent(request.getColabSessionId(), (key) -> {
447                                 return new HashSet<>();
448                             })
449                             // and make sure the websocket session is linked to the http session
450                             .add(session);
451 
452                         // make sure to link wsSession to its Http session
453                         // wsSessionToHttpSession.put(session, request.getColabSessionId()); //
454                         // TODO: is it even used ?
455                         if (request.getType() == SubscriptionRequest.SubscriptionType.SUBSCRIBE) {
456                             // make sure the http session has its own list of channels
457                             wsSessionMap.computeIfAbsent(session, (key) -> {
458                                 return new HashSet<>();
459                             })
460                                 // keep wsSession to channel registry up-to date
461                                 .add(channel);
462 
463                             // subscribe to channel
464                             subscribe(channel, session);
465                         } else {
466                             // Remove the channel from the set of channel linked to the wsSession
467                             if (wsSessionMap.containsKey(session)) {
468                                 Set<WebsocketChannel> channels = wsSessionMap.get(session);
469                                 channels.remove(channel);
470                                 if (channels.isEmpty()) {
471                                     wsSessionMap.remove(session);
472                                 }
473                             }
474                             unsubscribe(channel, Set.of(session));
475                         }
476                     }
477                 } else {
478                     logger.debug("Failed to resolve {} to an effective channel", request);
479                 }
480             } else {
481                 logger.debug("Ignore channel subscription: {}", request);
482             }
483         });
484     }

486     /**
487      * Add the given session to the set identified by the given channel, in the given map.
488      *
489      * @param channel websocket channel to which we subscribe
490      * @param session session to remove from the set
491      */
492     private void subscribe(WebsocketChannel channel, Session session) {
493         if (logger.isDebugEnabled()) {
494             String sessionId = WebsocketEndpoint.getSessionId(session);
495             logger.debug("Session {} subscribes to {}", sessionId, channel);
496         }
497         subscriptions.computeIfAbsent(channel, (key -> {
498             return new HashSet<>();
499         })).add(session);
500 
501         // make sure to propagate channelCHange after subscription
502         // (the ChannelChange event may be send through this very subscription, eg if an admin
503         // is subscribing to its own userChannel)
504         this.propagateChannelChange(channel, 1);
505     }

507     /**
508      * Unsubscribe all sessions from given channel. If the channel is empty after the operation, it
509      * will be destroyed.
510      *
511      * @param channel  channel to update
512      * @param sessions session to remove from channel
513      */
514     private void unsubscribe(WebsocketChannel channel, Set<Session> sessions) {
515         Set<Session> chSessions = subscriptions.get(channel);
516         if (logger.isDebugEnabled()) {
517             logger.debug("Sessions {} unsubscribes from {}",
518                 sessions.stream().map(session -> WebsocketEndpoint.getSessionId(session)
519                 ), channel);
520         }
521         if (chSessions != null) {
522             int size = chSessions.size();
523             chSessions.removeAll(sessions);
524 
525             // make sure to propagate change before the unsubscription is effective
526             this.propagateChannelChange(channel, chSessions.size() - size);
527 
528             if (chSessions.isEmpty()) {
529                 subscriptions.remove(channel);
530             }
531         }
532     }

553     /**
554      * Determine the EffectiveChannel
555      *
556      * @param request the request
557      *
558      * @return the channel which match the request
559      */
560     private WebsocketChannel getChannel(SubscriptionRequest request) {
561         if (request.getChannelType() == SubscriptionRequest.ChannelType.PROJECT) {
562             Project project = projectManager.assertAndGetProject(request.getChannelId());
563             if (project != null) {
564                 return ProjectContentChannel.build(project.getId());
565             }
566         } else if (request.getChannelType() == SubscriptionRequest.ChannelType.BLOCK) {
567             TextDataBlock block = blockManager.findBlock(request.getChannelId());
568             if (block != null) {
569                 return BlockChannel.build(block.getId());
570             }
571         } else if (request.getChannelType() == SubscriptionRequest.ChannelType.USER) {
572             User user = userDao.findUser(request.getChannelId());
573             if (user != null) {
574                 return UserChannel.build(user);
575             }
576         } else if (request.getChannelType() == SubscriptionRequest.ChannelType.BROADCAST) {
577             return BroadcastChannel.build();
578         }
579 
580         // not found...
581         return null;
582     }

585      * Propagate precomputed message to clients. Actually, this methods, will ask all instances of
586      * the hazelcast cluster to propagate the message to session they're in charge. This will call
587      * {@link #onMessagePropagation(PrecomputedWsMessages) onMessagePropagation}cluster-wide.
588      *
589      * @param message precomputed message to propagate.
590      */
591     public void propagate(PrecomputedWsMessages message) {
592         this.messagePropagation.fire(message);
593     }

595     /**
596      * On Hazelcast event. Each instance receive precomputed message.
597      *
598      * @param payload the messagesByChannels to send to clients though relevant websocket channels
599      */
600     public void onMessagePropagation(
601         @Observes @Inbound(eventName = WS_PROPAGATION_CHANNEL) PrecomputedWsMessages payload) {
602 
603         Map<WebsocketChannel, List<String>> messagesByChannels = payload.getMessages();
604         if (messagesByChannels != null) {
605 
606             Map<Session, List<String>> mappedMessages = new HashMap<>();
607 
608             // Group messages by effective websocket session
609             messagesByChannels.forEach((channel, messages) -> {
610                 Set<Session> subscribers = this.subscriptions.get(channel);
611                 if (subscribers != null) {
612                     subscribers.forEach(session -> {
613                         List<String> list = mappedMessages.computeIfAbsent(session,
614                             (k) -> new LinkedList<>());
615                         list.addAll(messages);
616                     });
617                 }
618             });
619 
620             // send one big message to each session
621             mappedMessages.entrySet().forEach(entry -> {
622                 Session session = entry.getKey();
623                 String jsonArray = entry.getValue().stream()
624                     .collect(Collectors.joining(", ", "[", "]"));
625                 try {
626                     logger.debug("Send {} to {} ({})", jsonArray, session.getId());
627                     if (session.isOpen()) {
628                         session.getBasicRemote().sendText(jsonArray);
629                     }
630                 } catch (IOException ex) {
631                     logger.error("Failed to send websocket message {} to {}",
632                         jsonArray, session);
633                 }
634             });
635         }
636     }

638     /**
639      * Propagate the logout and unsubscribe from all channels linked to the given http session.
640      * Clear the space.
641      *
642      * @param session the http session which just logged out
643      */
644     public void signoutAndUnsubscribeFromAll(HttpSession session) {
645         synchronized (this) {
646             // propagate before removing the channels
647             propagateSignOut(session);
648 
649             Set<Session> wsSessions = this.httpSessionToWsSessions.get(session.getId());
650             if (wsSessions != null) {
651                 // the http session is linked to one or more websocket session, let's cancel all
652                 // their
653                 // subscriptions
654                 wsSessions.stream()
655                     // get channels from each wsSession
656                     .map(wsSession -> this.wsSessionMap.get(wsSession))
657                     // filter out null channels set
658                     .filter(channels -> channels != null)
659                     // convert "stream of set of channels" to "stream of channels"
660                     .flatMap(Collection::stream)
661                     // no need to list same channel twice
662                     .distinct()
663                     // clean subscriptions
664                     .forEach(channel -> {
665                         this.unsubscribe(channel, wsSessions);
666                     });
667             }
668 
669             this.httpSessionToWsSessions.remove(session.getId());
670         }
671     }

673     /**
674      * Propagate a logout
675      *
676      * @param httpSession The http session that is now closed
677      */
678     private void propagateSignOut(HttpSession httpSession) {
679         try {
680             PrecomputedWsMessages prepareWsMessage = WebsocketMessagePreparer
681                 .prepareWsMessage(
682                     userDao,
683                     teamMemberDao,
684                     cardTypeDao,
685                     projectDao,
686                     httpSession.getChannelsBuilder(),
687                     new WsSignOutMessage(httpSession));
688             this.propagate(prepareWsMessage);
689         } catch (EncodeException ex) {
690             logger.error("Faild to propagate sign out : {}", httpSession);
691         }
692     }

694     /**
695      * Clean subscription on session close
696      *
697      * @param session   websocket session to clean subscription for
698      * @param sessionId public websocket identifier
699      */
700     public void unsubscribeFromAll(Session session, String sessionId) {
701         synchronized (this) {
702             Set<WebsocketChannel> set = this.wsSessionMap.get(session);
703             if (set != null) {
704                 Set<Session> setOfSession = Set.of(session);
705                 set.forEach(channel -> {
706                     unsubscribe(channel, setOfSession);
707                     if (channel instanceof ProjectContentChannel) {
708                         ProjectContentChannel pChannel = (ProjectContentChannel) channel;
709                         presenceManager.clearWsSession(pChannel.getProjectId(), sessionId);
710                     }
711                 });
712                 this.wsSessionMap.remove(session);
713                 // flush changes ASAP
714                 WebsocketTxSync synchronizer = bag.getSynchronizer();
715                 if (synchronizer != null) {
716                     synchronizer.flush();
717                 }
718             }
719         }
720     }
