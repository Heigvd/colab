135     /**
136      * Get the current authenticated account
137      *
138      * @return the current account or null if none
139      */
140     public Account getCurrentAccount() {
141         if (this.httpSession != null) {
142             return userDao.findAccount(this.httpSession.getAccountId());
143         } else if (this.currentAccountId != null) {
144             return userDao.findAccount(this.currentAccountId);
145         }
146         return null;
147     }

150      * Get the current authenticated user
151      *
152      * @return the current user or null if none
153      */
154     public User getCurrentUser() {
155         Account account = this.getCurrentAccount();
156 
157         if (account != null) {
158             return account.getUser();
159         } else {
160             return null;
161         }
162     }

192     /**
193      * Set the current account.
194      *
195      * @param account new current account
196      */
197     public void login(Account account) {
198         HttpSession session = this.getHttpSession();
199         if (session != null) {
200             session.setAccountId(account.getId());
201         }
202         this.currentAccountId = account.getId();
203     }
204 
205     /**
206      * Clear current account and unsubscribe from all websocket channels.
207      */
208     public void logout() {
209         HttpSession session = this.getHttpSession();
210         this.currentAccountId = null;
211         if (session != null) {
212             session.setAccountId(null);
213             conditionCache.clear();
214             websocketManager.signoutAndUnsubscribeFromAll(this.getHttpSession().getSessionId());
215         }
216     }

218     /**
219      * Is current transaction still alive ?
220      *
221      * @return true if current tx is not dead
222      */
223     private boolean txExists() {
224         return !this.txDone;
225     }
226 
227     /**
228      * Synchronize the persistence context to the underlying database.
229      */
230     public void flush() {
231         if (txExists()) {
232             em.flush();
233         }
234     }
235 
236     /**
237      * Execute some piece of code with admin privileges.
238      *
239      * @param action code to execute with admin privileges
240      */
241     public void sudo(Runnable action) {
242         if (txExists()) {
243             // make sure to flush to check every pending changes before granting admin rights
244             em.flush();
245         }
246 
247         this.sudoAsAdmin++;
248         logger.trace("Sudo #{}", this.sudoAsAdmin);
249         action.run();
250         if (txExists()) {
251             // make sure to flush to apply all pending changes with admin rights
252             em.flush();
253         }
254 
255         logger.trace("EndOfSudo #{}", this.sudoAsAdmin);
256 
257         this.sudoAsAdmin--;
258     }
259 
260     /**
261      * Execute some piece of code with admin privileges and return something.
262      *
263      * @param <R>    return type
264      * @param action code to execute with admin privileges
265      *
266      * @return action result
267      *
268      * @throws java.lang.Exception if something is thrown during the call
269      */
270     public <R> R sudo(Callable<R> action) throws Exception {
271         if (txExists()) {
272             // make sure to flush to check every pending changes before granting admin rights
273             em.flush();
274         }
275         this.sudoAsAdmin++;
276         logger.trace("Sudo #{}", this.sudoAsAdmin);
277         R result = action.call();
278         if (txExists()) {
279             // make sure to flush to apply all pending changes with admin rights
280             em.flush();
281         }
282         logger.trace("EndOfSudo #{}", this.sudoAsAdmin);
283         this.sudoAsAdmin--;
284 
285         return result;
286     }

288     /**
289      * Is the currentUser is an admin or sudo as an admin ?
290      *
291      * @return true if current user can act as an admin
292      */
293     public boolean isAdmin() {
294         // the sudoAsAdmin is done at first,
295         // so we do not need to get the current user if this condition is fulfilled
296         if (sudoAsAdmin > 0) {
297             return true;
298         }
299 
300         User currentUser = this.getCurrentUser();
301         return currentUser != null && currentUser.isAdmin();
302     }
