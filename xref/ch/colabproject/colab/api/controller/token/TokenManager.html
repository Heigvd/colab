92      /**
93       * Persist the token
94       *
95       * @param token token to persist
96       *
97       * @return the new persisted token
98       */
99      private Token persistToken(Token token) {
100         logger.debug("persist token {}", token);
101 
102         // set something to respect notNull constraints
103         // otherwise persist will fail
104         // These values will be reset when the e-mail is sent.
105         if (token.getHashMethod() == null) {
106             token.setHashMethod(Helper.getDefaultHashMethod());
107         }
108         if (token.getHashedToken() == null) {
109             token.setHashedToken(new byte[0]);
110         }
111 
112         return tokenDao.persistToken(token);
113     }

115     /**
116      * Finalize initialization of the token and send it to the recipient.
117      * <p>
118      * As the plain token is not stored in the database, the token is regenerated in this method.
119      *
120      * @param token     the token to send
121      * @param recipient recipient email address
122      *
123      * @throws javax.mail.MessagingException if sending the message fails
124      */
125     public void sendTokenByEmail(Token token, String recipient) throws MessagingException {
126         logger.debug("Send token {} to {}", token, recipient);
127         String plainToken = Helper.generateHexSalt(64);
128         HashMethod hashMethod = Helper.getDefaultHashMethod();
129         byte[] hashedToken = hashMethod.hash(plainToken, Token.SALT);
130 
131         token.setHashMethod(hashMethod);
132         token.setHashedToken(hashedToken);
133 
134         String baseUrl = requestManager.getBaseUrl();
135 
136         // todo generate URL
137         String body = token.getEmailBody(baseUrl + "/#/token/" + token.getId()
138             + "/" + plainToken);
139 
140         // this log message contains sensitive information (body contains the plain-text token)
141         logger.trace("Send token {} to {} with body {}", token, recipient, body);
142         Sendmail.send(
143             Message.create()
144                 .from("noreply@" + ColabConfiguration.getSmtpDomain())
145                 .to(recipient)
146                 .subject(token.getSubject())
147                 .htmlBody(body)
148                 .build()
149         );
150     }

152     /**
153      * Consume the token
154      *
155      * @param id         the id of the token to consume
156      * @param plainToken the plain secret token as sent by e-mail
157      *
158      * @return the consumed token
159      *
160      * @throws HttpErrorMessage notFound if the token does not exist;<br>
161      *                          badRequest if token does not match;<br>
162      *                          authenticationRequired if token requires authentication but current
163      *                          user is not
164      */
165     public Token consume(Long id, String plainToken) {
166         logger.debug("Consume token #{}", id);
167 
168         Token token = tokenDao.findToken(id);
169 
170         if (token != null) {
171             if (token.isAuthenticationRequired() && !requestManager.isAuthenticated()) {
172                 logger.debug("Token requires an authenticated user");
173                 throw HttpErrorMessage.authenticationRequired();
174             } else {
175                 if (token.checkHash(plainToken)) {
176                     requestManager.sudo(() -> {
177                         boolean isConsumed = token.consume(this);
178 
179                         if (isConsumed
180                             && token.getExpirationPolicy() == ExpirationPolicy.ONE_SHOT) {
181                             tokenDao.deleteToken(token);
182                         }
183                     });
184                     return token;
185                 } else {
186                     logger.debug("Provided plain-token does not match");
187                     throw HttpErrorMessage.badRequest();
188                 }
189             }
190 
191         } else {
192             logger.debug("There is no token #{}", id);
193             throw HttpErrorMessage.notFound();
194         }
195     }

201     /**
202      * Create or update a validation token.
203      * <p>
204      * If a validate token already exists for the given account, it will be update so there is never
205      * more than one validation token per localAccount.
206      *
207      * @param account token owner
208      *
209      * @return a brand new token or a refresh
210      */
211     private VerifyLocalAccountToken getOrCreateVerifyAccountToken(LocalAccount account) {
212         logger.debug("getOrCreate VerifyToken for {}", account);
213         VerifyLocalAccountToken token = tokenDao.findVerifyTokenByAccount(account);
214 
215         if (token == null) {
216             logger.debug("no token, create one");
217             token = new VerifyLocalAccountToken();
218             token.setAuthenticationRequired(false);
219             token.setLocalAccount(account);
220             persistToken(token);
221         }
222         // token.setExpirationDate(OffsetDateTime.now().plus(1, ChronoUnit.WEEKS));
223         token.setExpirationDate(null);
224 
225         return token;
226     }

228     /**
229      * Send a "Please verify your email address" message.
230      *
231      * @param account      account to verify
232      * @param failsOnError if false, silent SMTP error
233      *
234      * @throws HttpErrorMessage smtpError if there is a SMPT error AND failsOnError is set to true
235      *                          messageError if the message contains errors (eg. malformed
236      *                          addresses)
237      */
238     public void requestEmailAddressVerification(LocalAccount account, boolean failsOnError) {
239         try {
240             VerifyLocalAccountToken token = this.getOrCreateVerifyAccountToken(account);
241             sendTokenByEmail(token, account.getEmail());
242         } catch (MessagingException ex) {
243             logger.error("Fails to send email address verification email", ex);
244             if (failsOnError) {
245                 throw HttpErrorMessage.smtpError();
246             }
247         }
248     }

267     /**
268      * get existing reset password token if it exists or create new one otherwise.
269      *
270      * @param account token owner
271      *
272      * @return the token to user
273      */
274     private ResetLocalAccountPasswordToken getOrCreateResetToken(LocalAccount account) {
275         logger.debug("getOrCreate Reset for {}", account);
276         ResetLocalAccountPasswordToken token = tokenDao.findResetTokenByAccount(account);
277 
278         if (token == null) {
279             token = new ResetLocalAccountPasswordToken();
280             logger.debug("no token, create one");
281             token.setAuthenticationRequired(false);
282             token.setLocalAccount(account);
283             persistToken(token);
284         }
285         token.setExpirationDate(OffsetDateTime.now().plus(1, ChronoUnit.HOURS));
286 
287         return token;
288     }

290     /**
291      * Send a "Click here the reset your password" message.
292      *
293      * @param account      The account whose password is to be reset
294      * @param failsOnError if false, silent SMTP error
295      *
296      * @throws HttpErrorMessage smtpError if there is a SMPT error AND failsOnError is set to true
297      *                          messageError if the message contains errors (eg. malformed
298      *                          addresses)
299      */
300     public void sendResetPasswordToken(LocalAccount account, boolean failsOnError) {
301         try {
302             logger.debug("Send reset password token to {}", account);
303             ResetLocalAccountPasswordToken token = this.getOrCreateResetToken(account);
304             sendTokenByEmail(token, account.getEmail());
305         } catch (MessagingException ex) {
306             logger.error("Failed to send password reset email", ex);
307             if (failsOnError) {
308                 throw HttpErrorMessage.smtpError();
309             }
310         }
311     }

330     /**
331      * Send invitation to join the project team to the recipient.
332      *
333      * @param project   the project to join
334      * @param recipient email address to send invitation to
335      *
336      * @return the pending teamMember of null if none was sent
337      */
338     public TeamMember sendMembershipInvitation(Project project, String recipient) {
339         User currentUser = securityManager.assertAndGetCurrentUser();
340 
341         InvitationToken token = tokenDao.findInvitationByProjectAndRecipient(project, recipient);
342         if (token == null) {
343             // create a member and link it to the project, but do not link it to any user
344             // this link will be set during token consumption
345             TeamMember newMember = teamManager.addMember(project, null,
346                 HierarchicalPosition.INTERNAL);
347             token = new InvitationToken();
348 
349             token.setTeamMember(newMember);
350             // never expire
351             token.setExpirationDate(null);
352             token.setAuthenticationRequired(Boolean.TRUE);
353             token.setRecipient(recipient);
354 
355             newMember.setDisplayName(recipient);
356 
357             persistToken(token);
358         }
359 
360         token.setSender(currentUser.getDisplayName());
361 
362         try {
363             sendTokenByEmail(token, recipient);
364         } catch (MessagingException ex) {
365             logger.error("Failed to send membership invitation email", ex);
366             throw HttpErrorMessage.smtpError();
367         }
368 
369         return token.getTeamMember();
370     }

375      * @param teamMember the team member for which we delete all invitations

377     public void deleteInvitationsByTeamMember(TeamMember teamMember) {

382     /**
383      * Consume the invitation token
384      *
385      * @param teamMember the team member related to the token
386      *
387      * @return true if the token can be consumed
388      */
389     public boolean consumeInvitationToken(TeamMember teamMember) {
390         User user = requestManager.getCurrentUser();
391 
392         if (user == null) {
393             throw HttpErrorMessage.authenticationRequired();
394         }
395 
396         Project project = teamMember.getProject();
397 
398         TeamMember existingTeamMember = teamManager.findMemberByProjectAndUser(project, user);
399         if (existingTeamMember != null) {
400             throw HttpErrorMessage
401                 .tokenProcessingFailure(MessageI18nKey.USER_IS_ALREADY_A_TEAM_MEMBER);
402         }
403 
404         teamMember.setUser(user);
405         teamMember.setDisplayName(null);
406 
407         return true;
408     }

414     /**
415      * Send a model sharing token to register the project as a model to use
416      * <p>
417      * If the token does not exist yet, create it and link to it a new pending instance maker.
418      *
419      * @param model     the id of the model
420      * @param recipient the address to send the sharing token to
421      *
422      * @return the pending instance maker
423      */
424     public InstanceMaker sendModelSharingToken(Project model, String recipient) {
425         User currentUser = securityManager.assertAndGetCurrentUser();
426 
427         ModelSharingToken token = tokenDao.findModelSharingByProjectAndRecipient(model, recipient);
428 
429         if (token == null) {
430             // create an instance maker and link it to the project, but do not link it to any user
431             // this link will be set during token consumption
432             InstanceMaker newInstanceMaker = projectManager.addAndPersistInstanceMaker(model, null);
433 
434             token = new ModelSharingToken();
435 
436             token.setInstanceMaker(newInstanceMaker);
437             token.setExpirationDate(null);
438             token.setAuthenticationRequired(Boolean.TRUE);
439             token.setRecipient(recipient);
440 
441             newInstanceMaker.setDisplayName(recipient);
442 
443             persistToken(token);
444         }
445 
446         token.setSender(currentUser.getDisplayName());
447 
448         try {
449             sendTokenByEmail(token, recipient);
450         } catch (MessagingException ex) {
451             logger.error("Failed to send model sharing email", ex);
452             throw HttpErrorMessage.smtpError();
453         }
454 
455         return token.getInstanceMaker();
456     }

511     /**
512      * Fetch token with given id from DAO. If it's outdated, it will be destroyed and null will be
513      * returned
514      *
515      * @param id id of the token
516      *
517      * @return token if it exists and is not outdated, null otherwise
518      */
519     public Token getNotExpiredToken(Long id) {
520         Token token = tokenDao.findToken(id);
521 
522         if (token != null && token.isOutdated()) {
523             requestManager.sudo(() -> tokenDao.deleteToken(token));
524             return null;
525         }
526 
527         return token;
528     }
