120     /**
121      * Retrieve the card content. If not found, throw a {@link HttpErrorMessage}.
122      *
123      * @param cardContentId the id of the card content
124      *
125      * @return the card content if found
126      *
127      * @throws HttpErrorMessage if the card content was not found
128      */
129     public CardContent assertAndGetCardContent(Long cardContentId) {
130         CardContent cardContent = cardContentDao.findCardContent(cardContentId);
131 
132         if (cardContent == null) {
133             logger.error("card content #{} not found", cardContentId);
134             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_NOT_FOUND);
135         }
136 
137         return cardContent;
138     }

140     /**
141      * Retrieve the grand parent card content of the given card.
142      *
143      * @param card the card
144      *
145      * @return the grand parent card content of the given card
146      *
147      * @throws HttpErrorMessage if any of the level was not found
148      */
149     public CardContent assertAndGetGrandParentCardContent(Card card) {
150         if (card == null) {
151             logger.error("card {} is null", card);
152             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_NOT_FOUND);
153         }
154 
155         CardContent parentCardContent = card.getParent();
156         if (parentCardContent == null) {
157             logger.error("parent card content of card {} is null", card);
158             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
159         }
160 
161         Card parentCard = parentCardContent.getCard();
162         if (parentCard == null) {
163             logger.error("parent card of card {} is null", card);
164             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
165         }
166 
167         CardContent grandParentCardContent = parentCard.getParent();
168         if (grandParentCardContent == null) {
169             logger.error("grand parent card content of card {} is null", card);
170             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
171         }
172 
173         return grandParentCardContent;
174 
175     }

197     /**
198      * Complete and persist a new card content variant for the given card.
199      * <p>
200      * Also create its default resource references.
201      *
202      * @param cardId the id of the card needing a new card content variant
203      *
204      * @return a new, initialized and persisted card content
205      */
206     public CardContent createNewCardContent(Long cardId) {
207         logger.debug("create a new card content for the card #{}", cardId);
208 
209         Card card = cardManager.assertAndGetCard(cardId);
210 
211         CardContent cardContent = initNewCardContentForCard(card);
212 
213         resourceReferenceSpreadingHelper.extractReferencesFromUp(cardContent);
214 
215         return cardContentDao.persistCardContent(cardContent);
216     }

218     /**

305     /**
306      * Delete the given card content
307      *
308      * @param cardContentId the id of the card content to delete
309      */
310     public void deleteCardContent(Long cardContentId) {
311         CardContent cardContent = assertAndGetCardContent(cardContentId);
312 
313         if (!checkDeletionAcceptability(cardContent)) {
314             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
315         }
316 
317         cardContent.getCard().getContentVariants().remove(cardContent);
318 
319         cardContentDao.deleteCardContent(cardContent);
320     }

322     /**
323      * Ascertain that the card content can be deleted.
324      *
325      * @param cardContent the card content to check for deletion
326      *
327      * @return True iff it can be safely deleted
328      */
329     private boolean checkDeletionAcceptability(CardContent cardContent) {
330         // A card must have at least one card content
331         if (cardContent.getCard().getContentVariants().size() == 1) {
332             return false;
333         }
334 
335         return true;
336     }

374     /**
375      * Add the deliverable to the card content.
376      *
377      * @param cardContentId   the id of the card content
378      * @param document        the document to use as deliverable. It must be a new document
379      * @param relatedPosition to define the place where the deliverable will be added in the card
380      *                        content
381      * @param neighbourDocId  the existing document which defines where the new document will be
382      *                        set. If relatedPosition is BEFOR or AFTER, it must be not null
383      *
384      * @return the newly created document
385      */
386     public Document addDeliverable(Long cardContentId, Document document,
387         RelatedPosition relatedPosition, Long neighbourDocId) {
388         logger.debug("add deliverable {} to card content #{} {} doc #{}", document, cardContentId,
389             relatedPosition, neighbourDocId);
390 
391         CardContent cardContent = assertAndGetCardContent(cardContentId);
392 
393         if (document == null) {
394             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
395         }
396 
397         if (document.hasOwningResource() || document.hasOwningCardContent()) {
398             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
399         }
400 
401         if (cardContent.getDeliverables().contains(document)) {
402             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
403         }
404 
405         switch (relatedPosition) {
406             case BEFORE:
407                 Document neighbourBDocument = documentManager.assertAndGetDocument(neighbourDocId);
408                 indexGenerator.moveItemBefore(document, neighbourBDocument,
409                     cardContent.getDeliverables());
410                 break;
411             case AFTER:
412                 Document neighbourADocument = documentManager.assertAndGetDocument(neighbourDocId);
413                 indexGenerator.moveItemAfter(document, neighbourADocument,
414                     cardContent.getDeliverables());
415                 break;
416             case AT_BEGINNING:
417                 indexGenerator.moveItemToBeginning(document, cardContent.getDeliverables());
418                 break;
419             case AT_END:
420             default:
421                 indexGenerator.moveItemToEnd(document, cardContent.getDeliverables());
422                 break;
423         }
424 
425         cardContent.getDeliverables().add(document);
426         document.setOwningCardContent(cardContent);
427 
428         return documentDao.persistDocument(document);
429     }

431     /**
432      * Remove the deliverable of the card content.
433      *
434      * @param cardContentId the id of the card content
435      * @param documentId    the id of the document to remove from the card content
436      */
437     public void removeDeliverable(Long cardContentId, Long documentId) {
438         logger.debug("remove deliverable #{} of card content #{}", documentId, cardContentId);
439 
440         CardContent cardContent = assertAndGetCardContent(cardContentId);
441 
442         Document document = documentManager.assertAndGetDocument(documentId);
443 
444         if (!(cardContent.getDeliverables().contains(document))) {
445             throw HttpErrorMessage.dataError(MessageI18nKey.DATA_INTEGRITY_FAILURE);
446         }
447 
448         cardContent.getDeliverables().remove(document);
449 
450         documentDao.deleteDocument(document);
451     }

511     /**
512      * Check the integrity of the project
513      *
514      * @param cardContent the card content to check
515      *
516      * @return true iff the card content is complete and safe
517      */
518     public boolean checkIntegrity(CardContent cardContent) {
519         if (cardContent == null) {
520             return false;
521         }
522 
523         if (cardContent.getCard() == null) {
524             return false;
525         }
526 
527         if (cardContent.getCompletionLevel() < MIN_COMPLETION_LEVEL) {
528             return false;
529         }
530 
531         if (cardContent.getCompletionLevel() > MAX_COMPLETION_LEVEL) {
532             return false;
533         }
534 
535         return true;
536     }
