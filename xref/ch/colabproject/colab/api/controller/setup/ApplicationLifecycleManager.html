9   import ch.colabproject.colab.api.controller.RequestManager;
10  import ch.colabproject.colab.api.controller.user.UserManager;
11  import ch.colabproject.colab.api.model.user.User;
12  import ch.colabproject.colab.api.persistence.jpa.user.UserDao;

14  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
15  import javax.ejb.LocalBean;
16  import javax.ejb.Stateless;
17  import javax.inject.Inject;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  /**
22   * Some business method required by {@link ApplicationLifecycleManager}.
23   * <p>
24   * This is a required step. Without this step, beans injection and CCI context are quite messed up.
25   * <p>
26   * This beans shall only call methods with annotation {@link javax.ejb.TransactionAttribute} set to
27   * {@link javax.ejb.TransactionAttributeType#REQUIRES_NEW}.
28   * <p>
29   * @author maxence
30   */
31  @Stateless
32  @LocalBean
33  public class ApplicationLifecycleManager {
34  
35      /**
36       * Logger. Default level, set in logback.xml, is INFO
37       */
38      private static final Logger logger = LoggerFactory.getLogger(ApplicationLifecycleManager.class);
39  
40      /**
41       * User persistence
42       */
43      @Inject
44      private UserDao userDao;
45  
46      /**
47       * User management
48       */
49      @Inject
50      private UserManager userManager;
51  
52      /**
53       * request manager
54       */
55      @Inject
56      private RequestManager requestManager;
57  
58      /**
59       * Create a default admin user if there is no admin at all.
60       */
61      public void createDefaultAdminIfNone() {
62          requestManager.sudo(() -> {
63              if (userDao.findAllAdmin().isEmpty()) {
64                  try {
65                      logger.info("No admin exists, create one");
66                      //make sure to create the user within a brand new transaction
67                      User admin = userManager.createAdminUserTx(
68                          ColabConfiguration.getDefaultAdminUsername(),
69                          ColabConfiguration.getDefaultAdminEmail(),
70                          ColabConfiguration.getDefaultAdminPassword()
71                      );
72                      logger.info("New admin user: {}", admin);
73                  } catch (HttpErrorMessage ex) {
74                      logger.error(
75                          "Fails to create default admin user. Does non-admin user exist with same username or email address");
76                  } catch (RuntimeException ex) {
77                      logger.error(
78                          "Fails to create default amdin user for some unknown reason. Please check config",
79                          ex);
80                  }
81              }
82          });
83      }
84  }
