7   package ch.colabproject.colab.api.microchanges.tools;
8   
9   import ch.colabproject.colab.api.Helper;
10  import ch.colabproject.colab.api.ejb.CdiBridgeSingleton;
11  import java.io.Serializable;
12  import java.util.concurrent.Callable;
13  
14  /**
15   * Cancel any {@link Debouncer}
16   *
17   * @author maxence
18   */
19  public class CancelDebounce implements Callable<Boolean>, Serializable {
20  
21      private static final long serialVersionUID = 1L;
22  
23      /**
24       * If of the block
25       */
26      private final Long blockId;
27  
28      /**
29       * Create the callable canceler
30       *
31       * @param blockId id of the block
32       */
33      public CancelDebounce(Long blockId) {
34          this.blockId = blockId;
35      }
36  
37      @Override
38      public Boolean call() throws Exception {
39          CdiBridgeSingleton cdiBridge = Helper.lookup(CdiBridgeSingleton.class);
40          return cdiBridge.cancelDebounce(blockId);
41      }
42  }
