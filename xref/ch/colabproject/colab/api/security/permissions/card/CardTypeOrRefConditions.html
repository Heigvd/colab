7   package ch.colabproject.colab.api.security.permissions.card;
8   
9   import ch.colabproject.colab.api.controller.RequestManager;
10  import ch.colabproject.colab.api.controller.security.SecurityManager;
11  import ch.colabproject.colab.api.security.permissions.Conditions.Condition;
12  import java.util.Objects;
13  
14  /**
15   * Card type and reference access control conditions
16   *
17   * @author sandra
18   */
19  public final class CardTypeOrRefConditions {
20  
21      /**
22       * Private constructor prevents instantiation
23       */
24      private CardTypeOrRefConditions() {
25          throw new UnsupportedOperationException("This is a utility class");
26      }
27  
28      /**
29       * Has the current user access to the card type or reference ?
30       *
31       * @author sandra
32       */
33      public static class IsCardTypeOrRefReadable extends Condition {
34  
35          /** the card type or reference id */
36          private final Long cardTypeOrRefId;
37  
38          /**
39           * Create a has card type or reference read access statement
40           *
41           * @param cardTypeOrRefId the card type or reference id
42           */
43          public IsCardTypeOrRefReadable(Long cardTypeOrRefId) {
44              this.cardTypeOrRefId = cardTypeOrRefId;
45          }
46  
47          @Override
48          protected boolean internalEval(RequestManager requestManager,
49              SecurityManager securityManager) {
50              return securityManager.isCardTypeOrRefReadableByCurrentUser(this.cardTypeOrRefId);
51          }
52  
53          @Override
54          public String toString() {
55              return "IsCardTypeOrRefReadable(" + cardTypeOrRefId + ")";
56          }
57  
58          @Override
59          public int hashCode() {
60              int hash = 3;
61              hash = 43 * hash + Objects.hashCode(this.cardTypeOrRefId);
62              return hash;
63          }
64  
65          @Override
66          public boolean equals(Object obj) {
67              if (this == obj) {
68                  return true;
69              }
70              if (obj == null) {
71                  return false;
72              }
73              if (getClass() != obj.getClass()) {
74                  return false;
75              }
76              final IsCardTypeOrRefReadable other = (IsCardTypeOrRefReadable) obj;
77              if (!Objects.equals(this.cardTypeOrRefId, other.cardTypeOrRefId)) {
78                  return false;
79              }
80              return true;
81          }
82      }
83  
84  }
