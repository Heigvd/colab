7   package ch.colabproject.colab.api.ws.utils;
8   
9   import ch.colabproject.colab.api.Helper;
10  import ch.colabproject.colab.api.controller.CdiBridgeSingleton;
11  import java.io.Serializable;
12  import java.util.Map;
13  import java.util.concurrent.Callable;
14  
15  /**
16   * Serializable callable to be submitted to all instances of the cluster.
17   *
18   * @author maxence
19   */
20  public class CallableGetChannel
21      implements
22      Serializable,
23      Callable<Map<String, Integer>> {
24  
25      private static final long serialVersionUID = 1L;
26  
27      @Override
28      public Map<String, Integer> call() throws Exception {
29          CdiBridgeSingleton cdiBridge = Helper.lookup(CdiBridgeSingleton.class);
30          if (cdiBridge != null) {
31              return cdiBridge.getSubscriptionsCount();
32          } else {
33              return Map.of();
34          }
35      }
36  
37  }
