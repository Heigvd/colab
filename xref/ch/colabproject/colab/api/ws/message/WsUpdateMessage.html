7   package ch.colabproject.colab.api.ws.message;
8   
9   import ch.colabproject.colab.api.model.WithWebsocketChannels;
10  import java.util.HashSet;
11  import java.util.Set;
12  import javax.validation.constraints.NotNull;
13  
14  /**
15   * Websocket update message. Contains a list of new or updated objects and a list of destroyed objects.
16   *
17   * @author maxence
18   */
19  public class WsUpdateMessage extends WsMessage {
20  
21      private static final long serialVersionUID = 1L;
22  
23      /**
24       * Set of new or updated objects
25       */
26      @NotNull
27      private Set<WithWebsocketChannels> updated;
28  
29      /**
30       * Set of destroyed objects
31       */
32      @NotNull
33      private Set<IndexEntry> deleted;
34  
35      /**
36       * Default constructor
37       */
38      public WsUpdateMessage() {
39          this.updated = new HashSet<>();
40          this.deleted = new HashSet<>();
41      }
42  
43      /**
44       * @return the list of new or updated objects to send to clients
45       */
46      public Set<WithWebsocketChannels> getUpdated() {
47          return updated;
48      }
49  
50      /**
51       * Set payload
52       *
53       * @param updated list of updated elements
54       */
55      public void setUpdated(Set<WithWebsocketChannels> updated) {
56          this.updated = updated;
57      }
58  
59      /**
60       * Get the list of deleted objects
61       *
62       * @return all deleted objects
63       */
64      public Set<IndexEntry> getDeleted() {
65          return deleted;
66      }
67  
68      /**
69       * Register index of deleted objects
70       *
71       * @param deleted index
72       */
73      public void setDeleted(Set<IndexEntry> deleted) {
74          this.deleted = deleted;
75      }
76  
77      @Override
78      public String toString() {
79          return "WsUpdateMessage{" + "updated=" + updated + ", deleted=" + deleted + '}';
80      }
81  }
