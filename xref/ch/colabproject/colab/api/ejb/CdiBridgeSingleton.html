7   package ch.colabproject.colab.api.ejb;
8   
9   import ch.colabproject.colab.api.ws.channel.WebsocketEffectiveChannel;
10  import java.util.Map;
11  import javax.enterprise.context.ApplicationScoped;
12  import javax.inject.Inject;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  /**
17   * Single which give access to the CDI world.
18   *
19   * @author maxence
20   */
21  @ApplicationScoped
22  public class CdiBridgeSingleton {
23  
24      /**
25       * Logger. Default level, set in logback.xml, is INFO
26       */
27      private static final Logger logger = LoggerFactory.getLogger(CdiBridgeSingleton.class);
28  
29      /**
30       * Expose application-scoped instance to non-CDI world
31       */
32      private static CdiBridgeSingleton self;
33  
34      /**
35       * Websockets
36       */
37      @Inject
38      private WebsocketFacade websocketFacade;
39  
40      /**
41       * Initialize the non-CDI bean
42       */
43      public void init() {
44          CdiBridgeSingleton.self = this;
45      }
46  
47      /**
48       * Give non-CDI world access to this bean
49       *
50       * @return singleton instance
51       */
52      public static CdiBridgeSingleton getInstance() {
53          return CdiBridgeSingleton.self;
54      }
55  
56      /**
57       * Bridge to {@link WebsocketFacade#getSubscrciptionsCount() }
58       *
59       * @return the list of channels and the number of sessions subscribed to each of them
60       */
61      public Map<WebsocketEffectiveChannel, Integer> getSubscrciptionsCount() {
62          logger.debug("Get bridged Subscriptions");
63          return websocketFacade.getSubscrciptionsCount();
64      }
65  }
