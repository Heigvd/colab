154     /**
155      * Find the resources at the end of the references chains.
156      * <p>
157      * Only the active and available resources are taken into account.
158      *
159      * @param baseAbstractResources The list of abstract resources to search from
160      *
161      * @return the resources that match
162      */
163     private List<Resource> findAvailableAndActiveTargetResource(
164         List<AbstractResource> baseAbstractResources) {
165         List<Resource> result = new ArrayList<>();
166 
167         for (AbstractResource abstractResource : baseAbstractResources) {
168             Resource resource = getAvailableActiveResource(abstractResource);
169             if (resource != null) {
170                 result.add(resource);
171             }
172         }
173 
174         return result;
175     }

177     /**
178      * Recursive method to get the resource at the end of each references chain.
179      * <p>
180      * Only the active and available resources are taken into account.
181      *
182      * @param abstractResource An abstract resource
183      *
184      * @return the resource at the end of each references chain as long as it is available and
185      *         active
186      */
187     private Resource getAvailableActiveResource(AbstractResource abstractResource) {
188         if (abstractResource instanceof Resource) {
189             return (Resource) abstractResource;
190         } else if (abstractResource instanceof ResourceRef) {
191             if (isResourceAvailableAndActive((ResourceRef) abstractResource)) {
192                 return getAvailableActiveResource(
193                     ((ResourceRef) abstractResource).getTarget());
194             }
195         } else {
196             throw HttpErrorMessage.dataIntegrityFailure();
197         }
198 
199         return null;
200     }

202     /**
203      * Filter to only take into account resources that are available and active
204      * <p>
205      * <ul>
206      * <li>a resource must be published to be passed the sub cards</li>
207      * <li>the reference must not be refused</li>
208      * </ul>
209      *
210      * @param resourceRef the resource reference
211      *
212      * @return whether the resource can be taken into account or not
213      */
214     private boolean isResourceAvailableAndActive(ResourceRef resourceRef) {
215         AbstractResource target = resourceRef.getTarget();
216         if (resourceRef.hasCard() && target.hasCardContent()
217             && !resourceRef.resolve().isPublished()) {
218             return false;
219         }
220 
221         return !resourceRef.isRefused();
222     }

351     /**
352      * Create a new resource.
353      * <p>
354      * Every child of the card type acquires a reference to that resource.<br>
355      * And recursively the grandchildren acquires a reference to the reference of their parent.
356      *
357      * @param resource the resource to create

359      * @return the brand new resource

362         logger.debug("create a resource {}", resource);
363 
364         if (resource.getAbstractCardTypeId() != null) {
365             return createResourceForAbstractCardType(resource);
366         }
367

369             return createResourceForCard(resource);
370         }
371 
372         if (resource.getCardContentId() != null) {
373             return createResourceForCardContent(resource);
374         }
375 
376        throw HttpErrorMessage.dataIntegrityFailure();
377     }

383     /**
384      * Create a new resource linked to a card type (or a card type reference).
385      * <p>
386      * Every child of the card type acquires a reference to that resource.<br>
387      * And recursively the grandchildren acquires a reference to the reference of their parent.
388      *
389      * @param resource the resource to create
390      *
391      * @return the brand new resource
392      */
393     private Resource createResourceForAbstractCardType(Resource resource) {
394         logger.debug("create the resource {} for abstract card type", resource);
395 
396         if (resource.getDocument() == null) {
397             throw HttpErrorMessage.dataIntegrityFailure();
398         }
399 
400         AbstractCardType abstractCardType = cardTypeDao
401             .getAbstractCardType(resource.getAbstractCardTypeId());
402         if (abstractCardType == null) {
403             throw HttpErrorMessage.relatedObjectNotFoundError();
404         }
405 
406         if (resource.getTeaser() == null) {
407             initTeaser(resource);
408         }
409 
410         resource.setAbstractCardType(abstractCardType);
411         abstractCardType.getDirectAbstractResources().add(resource);
412 
413         resource = resourceDao.persistResource(resource);
414 
415         createResourceRefForChildren(abstractCardType, resource);
416 
417         return resource;
418     }

491     /**
492      * Each child of the card type acquires a reference to the resource.
493      *
494      * @param cardTypeOrRef the card type or card type reference
495      * @param resource      the resource we are linking
496      */
497     private void createResourceRefForChildren(AbstractCardType cardTypeOrRef,
498         AbstractResource resource) {
499         for (AbstractCardType cardTypeRef : cardTypeOrRef.getDirectReferences()) {
500             ResourceRef resourceRef = ResourceRef.initNewResourceRef();
501 
502             resourceRef.setTarget(resource);
503             resource.getDirectReferences().add(resourceRef);
504 
505             resourceRef.setAbstractCardType(cardTypeRef);
506             cardTypeRef.getDirectAbstractResources().add(resourceRef);
507 
508             resourceRefDao.persistResourceRef(resourceRef);
509 
510             createResourceRefForChildren(cardTypeRef, resourceRef);
511         }
512 
513         for (Card card : cardTypeOrRef.getImplementingCards()) {
514             ResourceRef cardResourceRef = ResourceRef.initNewResourceRef();
515 
516             cardResourceRef.setTarget(resource);
517             resource.getDirectReferences().add(cardResourceRef);
518 
519             cardResourceRef.setCard(card);
520             card.getDirectAbstractResources().add(cardResourceRef);
521 
522             resourceRefDao.persistResourceRef(cardResourceRef);
523 
524             for (CardContent cardContent : card.getContentVariants()) {
525                 ResourceRef cardContentResourceRef = ResourceRef.initNewResourceRef();
526 
527                 cardContentResourceRef.setTarget(cardResourceRef);
528                 cardResourceRef.getDirectReferences().add(cardContentResourceRef);
529 
530                 cardContentResourceRef.setCardContent(cardContent);
531                 cardContent.getDirectAbstractResources().add(cardContentResourceRef);
532 
533                 resourceRefDao.persistResourceRef(cardContentResourceRef);
534             }
535         }
536     }

586     /**
587      * Delete a resource
588      *
589      * @param resourceId the id of the resource to delete
590      */
591     public void deleteResource(Long resourceId) {
592         logger.debug("delete resource #{}", resourceId);
593 
594         AbstractResource resource = resourceOrRefDao.findResourceOrRef(resourceId);
595         if (resource == null) {
596             throw HttpErrorMessage.relatedObjectNotFoundError();
597             // or just return. see what is best
598         }
599 
600         if (!(resource instanceof Resource)) {
601             throw HttpErrorMessage.dataIntegrityFailure();
602         }
603 
604         deleteResourceAndRefs(resource);
605 
606         // Note : the document is deleted by cascade
607     }
608 
609     /**
610      * Delete each reference pointing at the given resourceOrRef
611      *
612      * @param resourceOrRef The initial abstract resource to delete
613      */
614     private void deleteResourceAndRefs(AbstractResource resourceOrRef) {
615         List<ResourceRef> references = resourceOrRef.getDirectReferences();
616         if (references != null) {
617             references.stream().forEach(ref -> deleteResourceAndRefs(ref));
618         }
619 
620         if (resourceOrRef.hasAbstractCardType()) {
621             AbstractCardType cardTypeOrRef = resourceOrRef.getAbstractCardType();
622             cardTypeOrRef.getDirectAbstractResources().remove(resourceOrRef);
623         }
624 
625         if (resourceOrRef.hasCard()) {
626             Card card = resourceOrRef.getCard();
627             card.getDirectAbstractResources().remove(resourceOrRef);
628         }
629 
630         if (resourceOrRef.hasCardContent()) {
631             CardContent cardContent = resourceOrRef.getCardContent();
632             cardContent.getDirectAbstractResources().remove(resourceOrRef);
633         }
634 
635         resourceOrRefDao.deleteResourceOrRef(resourceOrRef.getId());
636     }

647      *                        reference

649     public void setCategory(Long resourceOrRefId, String categoryName) {
650         logger.debug("set category {} to abstract resource #{}", categoryName, resourceOrRefId);
651 
652         if (StringUtils.isBlank(categoryName)) {
653             removeCategory(resourceOrRefId);
654         } else {
655             AbstractResource resourceOrRef = resourceOrRefDao.findResourceOrRef(resourceOrRefId);
656             if (resourceOrRef == null) {
657                 throw HttpErrorMessage.relatedObjectNotFoundError();
658             }
659 
660             resourceOrRef.setCategory(categoryName);
661         }

781     /**
782      * Rename the category in a card type / card type reference<br>
783      * And do it also for the implementing cards as long as they are of the same project
784      *
785      * @param cardTypeOrRef the card type / card type reference (scope of the renaming)
786      * @param projectId     the id of the project concerned (scope of the renaming)
787      * @param oldName       the old name of the category
788      * @param newName       the new name of the category
789      */
790     private void renameCategory(AbstractCardType cardTypeOrRef, Long projectId, String oldName,
791         String newName) {
792         cardTypeOrRef.getDirectAbstractResources().stream()
793             .forEach(resourceOrRef -> renameCategoryIfMatch(resourceOrRef, oldName, newName));
794 
795         cardTypeOrRef.getImplementingCards().stream()
796             .filter(card -> Objects.equals(projectId, card.getProject().getId()))
797             .forEach(card -> renameCategory(card, oldName, newName));
798 
799         cardTypeOrRef.getDirectReferences().stream()
800             .forEach(cardRef -> renameCategory(cardRef, projectId, oldName, newName));
801     }

836     /**
837      * Replace the category of the resource if it matches the oldName
838      *
839      * @param resourceOrRef the resource or resource reference
840      * @param oldName       the old name of the category
841      * @param newName       the new name of the category
842      */
843     private void renameCategoryIfMatch(AbstractResource resourceOrRef, String oldName,
844         String newName) {
845         if (Objects.equals(resourceOrRef.getCategory(), oldName)) {
846             if (StringUtils.isBlank(newName)) {
847                 resourceOrRef.setCategory(null);
848             } else {
849                 resourceOrRef.setCategory(StringUtils.trim(newName));
850             }
851         }
