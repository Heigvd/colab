156     /**
157      * Find the resources at the end of the references chains.
158      * <p>
159      * Only the active and available resources are taken into account.
160      *
161      * @param baseAbstractResources The list of abstract resources to search from
162      *
163      * @return the resources that match
164      */
165     private List<Resource> findAvailableAndActiveTargetResource(
166         List<AbstractResource> baseAbstractResources) {
167         List<Resource> result = new ArrayList<>();
168 
169         for (AbstractResource abstractResource : baseAbstractResources) {
170             Resource resource = getAvailableActiveResource(abstractResource);
171             if (resource != null) {
172                 result.add(resource);
173             }
174         }
175 
176         return result;
177     }

179     /**
180      * Recursive method to get the resource at the end of each references chain.
181      * <p>
182      * Only the active and available resources are taken into account.
183      *
184      * @param abstractResource An abstract resource
185      *
186      * @return the resource at the end of each references chain as long as it is available and
187      *         active
188      */
189     private Resource getAvailableActiveResource(AbstractResource abstractResource) {
190         if (abstractResource instanceof Resource) {
191             return (Resource) abstractResource;
192         } else if (abstractResource instanceof ResourceRef) {
193             if (isResourceAvailableAndActive((ResourceRef) abstractResource)) {
194                 return getAvailableActiveResource(
195                     ((ResourceRef) abstractResource).getTarget());
196             }
197         } else {
198             throw HttpErrorMessage.dataIntegrityFailure();
199         }
200 
201         return null;
202     }

204     /**
205      * Filter to only take into account resources that are available and active
206      * <p>
207      * <ul>
208      * <li>a resource must be published to be passed the sub cards</li>
209      * <li>the reference must not be refused</li>
210      * </ul>
211      *
212      * @param resourceRef the resource reference
213      *
214      * @return whether the resource can be taken into account or not
215      */
216     private boolean isResourceAvailableAndActive(ResourceRef resourceRef) {
217         AbstractResource target = resourceRef.getTarget();
218         if (resourceRef.hasCard() && target.hasCardContent()
219             && !resourceRef.resolve().isPublished()) {
220             return false;
221         }
222 
223         return !resourceRef.isRefused();
224     }

353     /**
354      * Create a new resource for the document linked to the card type (or card type reference).
355      * <p>
356      * Every child of the card type (recursively) acquires a reference to that resource or, for the
357      * grandchildren, to a reference to that resource.
358      *
359      * @param document   the document the resource represents
360      * @param cardTypeId the id of the card type the resource must be linked to
361      * @param category   the category of the resource (for organization purpose)
362      *
363      * @return the brand new resource
364      */
365     public Resource createResourceForAbstractCardType(Document document, Long cardTypeId,
366         String category) {
367         logger.debug(
368             "create a resource for document {} and an abstract card type #{} with category {}",
369             document, cardTypeId, category);
370 
371         if (document == null) {
372             throw HttpErrorMessage.relatedObjectNotFoundError();
373         }
374 
375         // A document can be related at max to one resource
376         if (document.hasResource()) {
377             throw HttpErrorMessage.dataIntegrityFailure();
378         }
379 
380         AbstractCardType abstractCardType = cardTypeDao.getAbstractCardType(cardTypeId);
381         if (abstractCardType == null) {
382             throw HttpErrorMessage.relatedObjectNotFoundError();
383         }
384 
385         Resource resource = Resource.initNewResource();
386 
387         resource.setAbstractCardType(abstractCardType);
388         abstractCardType.getDirectAbstractResources().add(resource);
389 
390         resource.setDocument(document);
391         document.setResource(resource);
392 
393         if (!Strings.isNullOrEmpty(category)) {
394             resource.setCategory(category);
395         }
396 
397         resource = resourceDao.persistResource(resource);
398 
399         createResourceRefForChildren(abstractCardType, resource);
400 
401         return resource;
402     }

503     /**
504      * Each child of the card type acquires a reference to the resource.
505      *
506      * @param cardTypeOrRef the card type or card type reference
507      * @param resource the resource we are linking
508      */
509     private void createResourceRefForChildren(AbstractCardType cardTypeOrRef,
510         AbstractResource resource) {
511         for (AbstractCardType cardTypeRef : cardTypeOrRef.getDirectReferences()) {
512             ResourceRef resourceRef = ResourceRef.initNewResourceRef();
513 
514             resourceRef.setTarget(resource);
515             resource.getDirectReferences().add(resourceRef);
516 
517             resourceRef.setAbstractCardType(cardTypeRef);
518             cardTypeRef.getDirectAbstractResources().add(resourceRef);
519 
520             resourceRefDao.persistResourceRef(resourceRef);
521 
522             createResourceRefForChildren(cardTypeRef, resourceRef);
523         }
524 
525         for (Card card : cardTypeOrRef.getImplementingCards()) {
526             ResourceRef cardResourceRef = ResourceRef.initNewResourceRef();
527 
528             cardResourceRef.setTarget(resource);
529             resource.getDirectReferences().add(cardResourceRef);
530 
531             cardResourceRef.setCard(card);
532             card.getDirectAbstractResources().add(cardResourceRef);
533 
534             resourceRefDao.persistResourceRef(cardResourceRef);
535 
536             for (CardContent cardContent : card.getContentVariants()) {
537                 ResourceRef cardContentResourceRef = ResourceRef.initNewResourceRef();
538 
539                 cardContentResourceRef.setTarget(cardResourceRef);
540                 cardResourceRef.getDirectReferences().add(cardContentResourceRef);
541 
542                 cardContentResourceRef.setCardContent(cardContent);
543                 cardContent.getDirectAbstractResources().add(cardContentResourceRef);
544 
545                 resourceRefDao.persistResourceRef(cardContentResourceRef);
546             }
547         }
548     }

598     /**
599      * Delete a resource
600      *
601      * @param resourceId the id of the resource to delete
602      */
603     public void deleteResource(Long resourceId) {
604         logger.debug("delete resource #{}", resourceId);
605 
606         AbstractResource resource = resourceOrRefDao.findResourceOrRef(resourceId);
607         if (resource == null) {
608             throw HttpErrorMessage.relatedObjectNotFoundError();
609             // or just return. see what is best
610         }
611 
612         if (!(resource instanceof Resource)) {
613             throw HttpErrorMessage.dataIntegrityFailure();
614         }
615 
616         deleteResourceAndRefs(resource);
617 
618         // Note : the document is deleted by cascade
619     }
620 
621     /**
622      * Delete each reference pointing at the given resourceOrRef
623      *
624      * @param resourceOrRef The initial abstract resource to delete
625      */
626     private void deleteResourceAndRefs(AbstractResource resourceOrRef) {
627         List<ResourceRef> references = resourceOrRef.getDirectReferences();
628         if (references != null) {
629             references.stream().forEach(ref -> deleteResourceAndRefs(ref));
630         }
631 
632         if (resourceOrRef.hasAbstractCardType()) {
633             AbstractCardType cardTypeOrRef = resourceOrRef.getAbstractCardType();
634             cardTypeOrRef.getDirectAbstractResources().remove(resourceOrRef);
635         }
636 
637         if (resourceOrRef.hasCard()) {
638             Card card = resourceOrRef.getCard();
639             card.getDirectAbstractResources().remove(resourceOrRef);
640         }
641 
642         if (resourceOrRef.hasCardContent()) {
643             CardContent cardContent = resourceOrRef.getCardContent();
644             cardContent.getDirectAbstractResources().remove(resourceOrRef);
645         }
646 
647         resourceOrRefDao.deleteResourceOrRef(resourceOrRef.getId());
648     }

793     /**
794      * Rename the category in a card type / card type reference<br>
795      * And do it also for the implementing cards as long as they are of the same project
796      *
797      * @param cardTypeOrRef the card type / card type reference (scope of the renaming)
798      * @param projectId     the id of the project concerned (scope of the renaming)
799      * @param oldName       the old name of the category
800      * @param newName       the new name of the category
801      */
802     private void renameCategory(AbstractCardType cardTypeOrRef, Long projectId, String oldName,
803         String newName) {
804         cardTypeOrRef.getDirectAbstractResources().stream()
805             .forEach(resourceOrRef -> renameCategoryIfMatch(resourceOrRef, oldName, newName));
806 
807         cardTypeOrRef.getImplementingCards().stream()
808             .filter(card -> Objects.equals(projectId, card.getProject().getId()))
809             .forEach(card -> renameCategory(card, oldName, newName));
810 
811         cardTypeOrRef.getDirectReferences().stream()
812             .forEach(cardRef -> renameCategory(cardRef, projectId, oldName, newName));
813     }

848     /**
849      * Replace the category of the resource if it matches the oldName
850      *
851      * @param resourceOrRef the resource or resource reference
852      * @param oldName       the old name of the category
853      * @param newName       the new name of the category
854      */
855     private void renameCategoryIfMatch(AbstractResource resourceOrRef, String oldName,
856         String newName) {
857         if (Objects.equals(resourceOrRef.getCategory(), oldName)) {
858             if (StringUtils.isBlank(newName)) {
859                 resourceOrRef.setCategory(null);
860             } else {
861                 resourceOrRef.setCategory(StringUtils.trim(newName));
862             }
863         }
864     }
