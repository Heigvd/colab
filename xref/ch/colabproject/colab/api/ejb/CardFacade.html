112     /**
113      * Create a new card type. The new type will be a global type if the type is not bound to any
114      * project.
115      *
116      * @param cardType the type to create
117      *
118      * @return a new, persisted card type
119      */
120     public CardType createNewCardType(CardType cardType) {
121         Long projectId = cardType.getProjectId();
122         if (projectId != null) {
123             logger.debug("create a new card type in the project #{}", projectId);
124             Project project = projectDao.getProject(projectId);
125             if (project == null) {
126                 throw HttpErrorMessage.relatedObjectNotFoundError();
127             }
128             project.getElementsToBeDefined().add(cardType);
129             cardType.setProject(project);
130         } else {
131             logger.debug("create a new global card type");
132             cardType.setProject(null);
133         }
134 
135         if (cardType.getPurpose() == null) {
136             cardType.setPurpose(Block.initNewDefaultBlock());
137         }
138 
139         cardTypeDao.createCardType(cardType);
140 
141         return cardType;
142     }

213     /**
214      * Get a card and all cards within in one set.
215      *
216      * @param rootCard the first card
217      *
218      * @return the rootCard + all cards within
219      */
220     public Set<Card> getAllCards(Card rootCard) {
221         Set<Card> cards = new HashSet<>();
222         List<Card> queue = new LinkedList<>();
223         queue.add(rootCard);
224 
225         while (!queue.isEmpty()) {
226             Card card = queue.remove(0);
227             cards.add(card);
228             card.getContentVariants().forEach(content -> queue.addAll(content.getSubCards()));
229         }
230         return cards;
231     }

270     public Card createNewCard(Long parentId, Long cardTypeId) {
271         logger.debug("create a new sub card of #{} with the type of #{}", parentId,
272             cardTypeId);
273 
274         CardContent parent = cardContentDao.getCardContent(parentId);
275         if (parent == null) {
276             throw HttpErrorMessage.relatedObjectNotFoundError();
277         }
278 
279         AbstractCardType cardType = cardTypeDao.getAbstractCardType(cardTypeId);
280         if (cardType == null) {

282         }
283 
284         Card card = initNewCard(parent, cardType);

286         ResourceReferenceSpreadingHelper.spreadResourceFromUp(card);
287 
288         cardDao.createCard(card);
289 
290         return card;

293     /**
294      * Initialize card. Card will be bound to the given type. If the type does not belongs to the
295      * same project as the card do, a type ref is created.
296      *
297      * @param parent   Parent of the new card
298      * @param cardType Related card type
299      *
300      * @return a new card containing a new card content with cardType
301      */
302     private Card initNewCard(CardContent parent, AbstractCardType cardType) {
303         Card card = initNewCard();
304 
305         card.setParent(parent);
306         parent.getSubCards().add(card);
307 
308         Project project = parent.getProject();
309 
310         if (project != null) {
311             AbstractCardType effectiveType = null;
312             if (project.equals(cardType.getProject())) {
313                 // Given type belongs to the project
314                 // it can be used as-is
315                 effectiveType = cardType;
316             } else {
317                 // second case: cardType belongs to another project
318 
319                 // shall we check something to prevent extending same type several times?
320                 //
321                 // Check if the project already got a direct reference the super-type
322                 Optional<AbstractCardType> findFirst = project.getElementsToBeDefined().stream()
323                     .filter(type -> {
324                         return this.isDirectRef(type, cardType);
325                     }).findFirst();
326 
327                 if (findFirst.isPresent()) {
328                     // direct ref found, reuse it
329                     effectiveType = findFirst.get();
330                 } else {
331                     // no direct ref. Create one.
332                     effectiveType = createReference(cardType, project);
333                 }
334             }
335 
336             if (effectiveType != null) {
337                 card.setCardType(effectiveType);
338                 effectiveType.getImplementingCards().add(card);
339 
340             } else {
341                 logger.error("Unable to find effective type for {}", cardType);
342                 throw HttpErrorMessage.relatedObjectNotFoundError();
343             }
344         }
345 
346         return card;
347     }

360     /**
361      * Create a type reference to the given type. The ref will belongs to the given project.
362      *
363      * @param cardType type to reference
364      * @param project  reference owner
365      *
366      * @return the reference
367      */
368     private CardTypeRef createReference(AbstractCardType cardType, Project project) {
369         CardTypeRef ref = new CardTypeRef();
370         ref.setProject(project);
371 
372         ref.setTarget(cardType);
373         cardType.getDirectReferences().add(ref);
374 
375         // TODO: copy deprecated state or do never deprecate just created types?
376         // ref.setDeprecated(cardType.isDeprecated());
377         ref.setDeprecated(false);
378         ref.setPublished(false);
379 
380         ResourceReferenceSpreadingHelper.spreadResourceFromUp(ref);
381 
382         cardTypeDao.createCardType(ref);
383         project.getElementsToBeDefined().add(ref);
384 
385         return ref;
386     }

400     /**
401      * Move a card to a new parent
402      *
403      * @param card      the card to move
404      * @param newParent the new parent
405      *
406      * @throws HttpErrorMessage if card or parent does not exist or if parent if a child of the card
407      */
408     public void moveCard(Card card, CardContent newParent) {
409         if (card != null && newParent != null) {
410             if (card.getRootCardProject() != null) {
411                 // Do never move root card
412                 throw HttpErrorMessage.dataIntegrityFailure();
413             }
414 
415             CardContent previousParent = card.getParent();
416             if (previousParent != null) {
417                 // check if newParent is a child of the card
418                 Card c = newParent.getCard();
419                 while (c != null) {
420                     if (c.equals(card)) {
421                         throw HttpErrorMessage.dataIntegrityFailure();
422                     }
423                     CardContent parent = c.getParent();
424                     if (parent != null) {
425                         c = parent.getCard();
426                     } else {
427                         c = null;
428                     }
429                 }
430                 previousParent.getSubCards().remove(card);
431                 newParent.getSubCards().add(card);
432                 card.setParent(newParent);
433             }
434         }
435     }

459     /**
460      * Is the given child a direct reference to the given parent
461      *
462      * @param child  the child
463      * @param parent the parent
464      *
465      * @return true if child is a direct reference to the parent
466      */
467     public boolean isDirectRef(AbstractCardType child, AbstractCardType parent) {
468         if (child instanceof CardTypeRef) {
469             CardTypeRef ref = (CardTypeRef) child;
470             AbstractCardType refTarget = ref.getTarget();
471             return parent != null && parent.equals(refTarget);
472         }
473         return false;
474     }
475 
476     /**
477      * Is the given ancestor an ancestor of the given child
478      *
479      * @param child    the child
480      * @param ancestor the ancestor
481      *
482      * @return true if child is a ref and if ancestor is an ancestor of the ref
483      */
484     public boolean isTransitiveRef(AbstractCardType child, AbstractCardType ancestor) {
485         if (isDirectRef(child, ancestor)) {
486             return true;
487         } else if (child instanceof CardTypeRef) {
488             AbstractCardType parent = ((CardTypeRef) child).getTarget();
489             return this.isTransitiveRef(parent, ancestor);
490         }
491 
492         return false;
493     }

632     /**
633      * Delete the card content
634      *
635      * @param cardContentId the id of the card content to delete
636      *
637      * @return the freshly deleted card content
638      */
639     public CardContent deleteCardContent(Long cardContentId) {
640         CardContent cardContent = cardContentDao.getCardContent(cardContentId);
641 
642         // A card must have at least one card content
643         if (cardContent.getCard().getContentVariants().size() == 1) {
644             throw HttpErrorMessage.dataIntegrityFailure();
645         }
646 
647         if (cardContent != null) {
648             cardContent.getCard().getContentVariants().remove(cardContent);
649 
650             Long deliverableId = cardContent.getDeliverableId();
651             if (deliverableId != null) {
652                 documentDao.deleteDocument(deliverableId);
653             }
654         }
655 
656         return cardContentDao.deleteCardContent(cardContentId);
657     }

710     /**
711      * Set the deliverable to the card content
712      *
713      * @param cardContentId the id of the card content
714      * @param document      the document to use as deliverable. It must be a new document
715      *
716      * @return the newly created document
717      */
718     public Document assignDeliverable(Long cardContentId, Document document) {
719         logger.debug("set deliverable {} to card content #{}", document, cardContentId);
720 
721         if (document == null) {
722             throw HttpErrorMessage.relatedObjectNotFoundError();
723         }
724 
725         // A document can be related at max to one card content
726         if (document.hasDeliverableCardContent()) {
727             throw HttpErrorMessage.dataIntegrityFailure();
728         }
729 
730         CardContent cardContent = cardContentDao.getCardContent(cardContentId);
731         if (cardContent == null) {
732             throw HttpErrorMessage.relatedObjectNotFoundError();
733         }
734 
735         cardContent.setDeliverable(document);
736         document.setDeliverableCardContent(cardContent);
737 
738         Document persistedDocument = documentDao.persistDocument(document);
739 
740         return persistedDocument;
741     }
