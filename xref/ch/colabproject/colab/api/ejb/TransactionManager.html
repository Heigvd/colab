7   package ch.colabproject.colab.api.ejb;
8   
9   import ch.colabproject.colab.api.exceptions.ColabRollbackException;
10  import ch.colabproject.colab.api.model.WithWebsocketChannels;
11  import ch.colabproject.colab.api.persistence.user.UserDao;
12  import ch.colabproject.colab.api.ws.WebsocketHelper;
13  import ch.colabproject.colab.api.ws.message.IndexEntry;
14  import ch.colabproject.colab.api.ws.message.PrecomputedWsMessages;
15  import java.io.Serializable;
16  import java.util.HashSet;
17  import java.util.Set;
18  import java.util.stream.Collectors;
19  import javax.ejb.AfterCompletion;
20  import javax.ejb.BeforeCompletion;
21  import javax.ejb.Stateful;
22  import javax.enterprise.context.RequestScoped;
23  import javax.inject.Inject;
24  import javax.persistence.EntityManager;
25  import javax.persistence.PersistenceContext;
26  import javax.websocket.EncodeException;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  /**
31   * Transaction sidekick.
32   *
33   * @author maxence
34   */
35  @RequestScoped // one would use a TransactionScoped bean but such a scope is not compatible with
36  //                the @AfterCompletion annotation
37  @Stateful // Stateful beans allows to use Before and AfterCompletion annotations
38  public class TransactionManager implements Serializable {
39  
40      private static final long serialVersionUID = 1L;
41  
42      /** logger */
43      private static final Logger logger = LoggerFactory.getLogger(TransactionManager.class);
44  
45      /**
46       * Access to the persistence unit
47       */
48      @PersistenceContext(unitName = "COLAB_PU")
49      private EntityManager em;
50  
51      /**
52       * To send message to clients
53       */
54      @Inject
55      private WebsocketFacade websocketFacade;
56  
57      /**
58       * To resolve meta channels
59       */
60      @Inject
61      private UserDao userDao;
62  
63      /**
64       * set of updated entities to be propagated
65       */
66      private Set<WithWebsocketChannels> updated = new HashSet<>();
67  
68      /**
69       * set of entities which have been deleted during the transaction
70       */
71      private Set<IndexEntry> deleted = new HashSet<>();
72  
73      /**
74       * store the prepared messages
75       */
76      private PrecomputedWsMessages message;
77  
78      /**
79       * Is the message has been precomputed ?
80       */
81      private boolean precomputed = false;
82  
83      /**
84       * Register updated object within the WS JTA synchronizer.
85       *
86       * @param o object to register
87       */
88      public void registerUpdate(WithWebsocketChannels o) {
89          updated.add(o);
90          logger.trace("UpdatedSet: {}", updated);
91      }
92  
93      /**
94       * Register deleted object within the WS JTA synchronizer
95       *
96       * @param o just deleted object
97       */
98      public void registerDelete(WithWebsocketChannels o) {
99          deleted.add(IndexEntry.build(o));
100         logger.trace("Deleted set: {}", deleted);
101     }
102 
103     /**
104      * Pre compute the message.
105      */
106     private void precomputeMessage() throws EncodeException {
107         logger.debug("Precompute messages; Update:{}, Deletes:{}", updated, deleted);
108         Set<WithWebsocketChannels> filtered = updated.stream()
109             .filter(
110                 (u) -> !deleted.stream()
111                     .anyMatch(
112                         (d) -> d.getId().equals(u.getId())
113                         && d.getType().equals(u.getJsonDiscriminator())
114                     )
115             ).collect(Collectors.toSet());
116 
117         this.precomputed = true;
118         this.message = WebsocketHelper.prepareWsMessage(userDao, filtered, deleted);
119         logger.trace("Precomputed: {}", message);
120     }
121 
122     /**
123      * Prepare websocket messages
124      */
125     @BeforeCompletion
126     public void beforeCompletion() {
127         //make sure to flush everything to database
128         em.flush();
129         logger.info("Before transactionCompletion: "
130             + "This method is not called for each transaction, why ???");
131         try {
132             this.precomputeMessage();
133         } catch (EncodeException ex) {
134             //throw runtime error to rollback
135             throw new ColabRollbackException(ex);
136         }
137     }
138 
139     /**
140      * After commit, if the transaction was successful, send pre-computed messages.
141      *
142      * @param successful true of the transaction was successfully committed
143      */
144     @AfterCompletion
145     public void afterCompletion(boolean successful) {
146         logger.debug("After transaction completion : {}; messages: {}", successful, message);
147         if (successful) {
148             if (!precomputed) {
149                 logger.info("Messages were not precomputed @BeforeCompletion!!!");
150                 // ,essage shall be precomputed during the "before completion" phase, but the
151                 // dedicated method is never called, and I do not understand the reason...
152                 try {
153                     this.precomputeMessage();
154                 } catch (EncodeException ex) {
155                     logger.error("Failed to precompute websocket messages");
156                 }
157             }
158             if (message != null && !message.getMessages().isEmpty()) {
159                 logger.info("Send messages");
160                 websocketFacade.propagate(message);
161             }
162         }
163     }
164 }
