7   package ch.colabproject.colab.api.ejb;
8   
9   import ch.colabproject.colab.api.model.WithId;
10  import java.io.Serializable;
11  import javax.annotation.PostConstruct;
12  import javax.annotation.Resource;
13  import javax.transaction.TransactionScoped;
14  import javax.transaction.TransactionSynchronizationRegistry;
15  
16  /**
17   * Transaction sidekick.
18   *
19   * @author maxence
20   */
21  @TransactionScoped
22  public class TransactionManager implements Serializable {
23  
24      /**
25       * JTA registry used to attach websocket synchronizer
26       */
27      @Resource
28      private transient TransactionSynchronizationRegistry jtaSyncRegistry;
29  
30      /**
31       * Websocket synchronizer
32       */
33      private WebsocketSynchronizer sync;
34  
35      /**
36       * Some postConstruct logic. Attach a WS synchronizer to each request
37       */
38      @PostConstruct
39      public void construct() {
40          if (jtaSyncRegistry != null && sync == null) {
41              sync = new WebsocketSynchronizer();
42              jtaSyncRegistry.registerInterposedSynchronization(sync);
43          }
44      }
45  
46      /**
47       * Register updated object within the WS JTA synchronizer.
48       *
49       * @param o object to register
50       */
51      public void registerUpdate(WithId o) {
52          sync.registerUpdate(o);
53      }
54  
55      /**
56       * Register deleted object within the WS JTA synchronizer
57       *
58       * @param o just deleted object
59       */
60      public void registerDelete(WithId o) {
61          sync.registerDelete(o);
62      }
63  }
