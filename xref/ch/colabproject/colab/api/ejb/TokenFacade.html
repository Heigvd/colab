70      /**
71       * Finalize initialization of the token and send it to the recipient.
72       * <p>
73       * As the plain token is not stored in the database, the togen is regenerated in this method.
74       *
75       * @param token     the token to send
76       * @param recipient recipient email address
77       *
78       * @throws javax.mail.MessagingException if sending the message fails
79       */
80      public void sendTokenByEmail(Token token, String recipient) throws MessagingException {
81          logger.debug("Send token {} to {}", token, recipient);
82          String plainToken = Helper.generateHexSalt(64);
83          HashMethod hashMethod = Helper.getDefaultHashMethod();
84          byte[] hashedToken = hashMethod.hash(plainToken, Token.SALT);
85  
86          token.setHashMethod(hashMethod);
87          token.setHashedToken(hashedToken);
88  
89          String baseUrl = requestManager.getBaseUrl();
90  
91          // todo generate URL
92          String body = token.getEmailBody(baseUrl + "/#/token/" + token.getId()
93              + "/" + plainToken);
94  
95          String footer = "";
96          if (token.getExpirationDate() != null) {
97              footer = "<br /><br /><i>This link can be used until "
98                  + token.getExpirationDate().format(DateTimeFormatter.RFC_1123_DATE_TIME)
99                  + " GMT</i>";
100         }
101 
102         // this log message contains sensitive information (body contains the plain-text token)
103         logger.trace("Send token {} to {} with body {}", token, recipient, body);
104         Sendmail.send(
105             Message.create()
106                 .from("noreply@" + ColabConfiguration.getSmtpDomain())
107                 .to(recipient)
108                 .subject(token.getSubject())
109                 .htmlBody(body + footer)
110                 .build()
111         );
112     }

114     /**
115      * Consume the token
116      *
117      * @param id         id of the token to consume
118      * @param plainToken the plain secret token as sent by e-mail
119      *
120      * @return the consumed token
121      *
122      * @throws HttpErrorMessage notFound if the token does not exists; badRequest if token does not
123      *                          match; authenticationRequired if token requires authentication but
124      *                          current user id not
125      */
126     public Token consume(Long id, String plainToken) {
127         logger.debug("Consume token #{}", id);
128         Token token = tokenDao.getToken(id);
129         if (token != null) {
130             if (token.isAuthenticationRequired() && !requestManager.isAuthenticated()) {
131                 logger.debug("Token requires an authenticated user");
132                 throw HttpErrorMessage.authenticationRequired();
133             } else {
134                 if (token.checkHash(plainToken)) {
135                     requestManager.sudo(() -> {
136                         token.consume(requestManager);
137                         tokenDao.deleteToken(token);
138                     });
139                     return token;
140                 } else {
141                     logger.debug("Provided plain-token does not match");
142                     throw HttpErrorMessage.badRequest();
143                 }
144             }
145         } else {
146             logger.debug("There is no token #{}", id);
147             throw HttpErrorMessage.notFound();
148         }
149     }

154     /**
155      * Create or update a validation token.
156      * <p>
157      * If a validate token already exists for the given account, it will be update so there is never
158      * more than one validation token per localAccount.
159      *
160      * @param account token owner
161      *
162      * @return a brand new token or a refresh
163      */
164     public VerifyLocalAccountToken getOrCreateVerifyAccountToken(LocalAccount account) {
165         logger.debug("getOrCreate VerifyToken for {}", account);
166         VerifyLocalAccountToken token = tokenDao.findVerifyTokenByAccount(account);
167 
168         if (token == null) {
169             logger.debug("no token, create one");
170             token = new VerifyLocalAccountToken();
171             token.setAuthenticationRequired(false);
172             token.setLocalAccount(account);
173             tokenDao.persistToken(token);
174         }
175         //token.setExpirationDate(OffsetDateTime.now().plus(1, ChronoUnit.WEEKS));
176         token.setExpirationDate(null);
177 
178         return token;
179     }

181     /**
182      * Send a "Please verify your email address" message.
183      *
184      * @param account      account to verify
185      * @param failsOnError if false, silent SMTP error
186      *
187      * @throws HttpErrorMessage smtpError if there is a SMPT error AND failsOnError is set to true
188      *                          messageError if the message contains errors (eg. malformed
189      *                          addresses)
190      */
191     public void requestEmailAddressVerification(LocalAccount account, boolean failsOnError) {
192         try {
193             VerifyLocalAccountToken token = this.getOrCreateVerifyAccountToken(account);
194             sendTokenByEmail(token, account.getEmail());
195         } catch (MessagingException ex) {
196             logger.error("Fails to send email address verification email", ex);
197             if (failsOnError) {
198                 throw HttpErrorMessage.smtpError();
199             }
200         }
201     }

206     /**
207      * get existing reset password token if it exists or create new one otherwise.
208      *
209      * @param account token owner
210      *
211      * @return the token to user
212      */
213     private ResetLocalAccountPasswordToken getOrCreateResetToken(LocalAccount account) {
214         logger.debug("getOrCreate Reset for {}", account);
215         ResetLocalAccountPasswordToken token = tokenDao.findResetTokenByAccount(account);
216 
217         if (token == null) {
218             token = new ResetLocalAccountPasswordToken();
219             logger.debug("no token, create one");
220             token.setAuthenticationRequired(false);
221             token.setLocalAccount(account);
222             tokenDao.persistToken(token);
223         }
224         token.setExpirationDate(OffsetDateTime.now().plus(1, ChronoUnit.HOURS));
225 
226         return token;
227     }

229     /**
230      * Send a "Click here the reset your password" message.
231      *
232      * @param account      The account whose password is to be reset
233      * @param failsOnError if false, silent SMTP error
234      *
235      * @throws HttpErrorMessage smtpError if there is a SMPT error AND failsOnError is set to true
236      *                          messageError if the message contains errors (eg. malformed
237      *                          addresses)
238      */
239     public void sendResetPasswordToken(LocalAccount account, boolean failsOnError) {
240         try {
241             logger.debug("Send reset password token to {}", account);
242             ResetLocalAccountPasswordToken token = this.getOrCreateResetToken(account);
243             sendTokenByEmail(token, account.getEmail());
244         } catch (MessagingException ex) {
245             logger.error("Failed to send pssword reset email", ex);
246             if (failsOnError) {
247                 throw HttpErrorMessage.smtpError();
248             }
249         }
250     }

255     /**
256      * Send invitation to join the project team to the recipient.
257      *
258      * @param project   the project to join
259      * @param recipient email address to send invitation to
260      *
261      * @return the pending teamMember of null if none was sent
262      */
263     public TeamMember sendMembershipInvitation(Project project, String recipient) {
264         User currentUser = securityFacade.assertAndGetCurrentUser();
265 
266         InvitationToken token = tokenDao.findInvitationByProjectAndRecipient(project, recipient);
267         if (token == null) {
268             // create a member and link it to the project, but do not link it to any user
269             // this link will be set during token consumtion
270             TeamMember newMember = teamFacade.addMember(project, null,
271                 HierarchicalPosition.INTERN);
272             token = new InvitationToken();
273 
274             token.setTeamMember(newMember);
275             // never expire
276             token.setExpirationDate(null);
277             token.setAuthenticationRequired(Boolean.TRUE);
278             token.setRecipient(recipient);
279 
280             tokenDao.persistToken(token);
281         }
282 
283         token.setSender(currentUser.getDisplayName());
284         try {
285             sendTokenByEmail(token, recipient);
286         } catch (MessagingException ex) {
287             logger.error("Failed to send pssword reset email", ex);
288             throw HttpErrorMessage.smtpError();
289         }
290         return token.getTeamMember();
291     }

293     /**
294      * Fetch token with given id from DAO. If it's outdated, it will be destroyed and null will be
295      * returned
296      *
297      * @param id id of the token
298      *
299      * @return token if it exists and is not outdated, null otherwise
300      */
301     public Token getNotExpiredToken(Long id) {
302         Token token = tokenDao.getToken(id);
303         if (token != null && token.isOutdated()) {
304             requestManager.sudo(() -> tokenDao.deleteToken(token));
305             return null;
306         } else {
307             return token;
308         }
