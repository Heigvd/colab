109     /**
110      * Create a new project with the provided data and register current user as a member.
111      *
112      * @param creationData the data needed to fill the project
113      *
114      * @return id of the persisted new project
115      */
116     @POST
117     @Path("createWithModel")
118     public Long createProject(ProjectCreationData creationData) {
119         logger.debug("Create a project with {}", creationData);
120 
121         Project project;
122 
123         if (creationData.getModelId() == null) {
124             project = new Project();
125             project.setType(creationData.getType());
126             project.setName(creationData.getName());
127             project.setDescription(creationData.getDescription());
128             project.setIllustration(creationData.getIllustration());
129 
130             projectManager.createProject(project);
131         } else {
132             project = projectManager.createProjectFromModel(creationData.getName(),
133                 creationData.getDescription(), creationData.getIllustration(),
134                 creationData.getModelId());
135         }
136 
137         creationData.getGuestsEmail().stream().forEach(email -> {
138             teamManager.invite(project.getId(), email);
139         });
140 
141         return project.getId();
142     }

170     /**
171      * Duplicate the given project
172      *
173      * @param projectId the id of the project we want to duplicate
174      * @param params    the parameters to fine tune
175      *
176      * @return the id of the duplicated project
177      */
178     @PUT
179     @Path("copyProject/{id}")
180     public Long duplicateProject(@PathParam("id") Long projectId, DuplicationParam params) {
181         logger.debug("duplicate the project #{} with params {}", projectId, params);
182 
183         DuplicationParam effectiveParams = params;
184         if (effectiveParams == null) {
185             effectiveParams = DuplicationParam.buildDefaultForCopyOfProject();
186         }
187 
188         return projectManager.duplicateProject(projectId, effectiveParams).getId();
189     }
