7   package ch.colabproject.colab.api.rest.utils;
8   
9   import ch.colabproject.colab.generator.model.annotations.ExtractJavaDoc;
10  
11  import java.io.Serializable;
12  import java.util.Objects;
13  
14  /**
15   * To use when we want to return a simple string as a result of a REST method.
16   * <p>
17   * If we try to use just String as return type, we have a problem with the json. It may be easy to
18   * fix, but to be as quick as possible, I made this class.
19   *
20   * @author sandra
21   */
22  @ExtractJavaDoc
23  public class SerializationStringWrapper implements Serializable {
24  
25      private static final long serialVersionUID = 1L;
26  
27      /**
28       * the string value
29       */
30      private String value;
31  
32      /**
33       * constructor
34       *
35       * @param value the string value
36       *
37       * @return a new SerializationStringWrapper
38       */
39      public static SerializationStringWrapper build(String value) {
40          SerializationStringWrapper ssw = new SerializationStringWrapper();
41          ssw.setValue(value);
42          return ssw;
43      }
44  
45      /**
46       * @return the string value
47       */
48      public String getValue() {
49          return value;
50      }
51  
52      /**
53       * @param value the string value
54       */
55      public void setValue(String value) {
56          this.value = value;
57      }
58  
59      @Override
60      public int hashCode() {
61          int hash = 7;
62          hash = 37 * hash + Objects.hashCode(this.getValue());
63          return hash;
64      }
65  
66      @Override
67      public boolean equals(Object obj) {
68          if (this == obj) {
69              // same object or both are null
70              return true;
71          }
72          if (obj == null) {
73              // only one is null
74              return false;
75          }
76  
77          // at this point, both are not null
78          if (this.getClass() != obj.getClass()) {
79              // class mismatch
80              return false;
81          } else {
82              // same class
83              SerializationStringWrapper otherB = (SerializationStringWrapper) obj;
84              if (this.getValue() == null || otherB.getValue() == null) {
85                  // if at least one object has no id, object are not equals
86                  // if both id are null, objects may equal, but this case is handled by the strict
87                  // (a == b) equality check
88                  return false;
89              }
90              // same class, values must be equal
91              return Objects.equals(this.getValue(), otherB.getValue());
92          }
93      }
94  }
