7   package ch.colabproject.colab.api.rest;
8   
9   import ch.colabproject.colab.api.presence.model.TouchUserPresence;
10  import ch.colabproject.colab.api.presence.model.UserPresence;
11  import ch.colabproject.colab.api.presence.PresenceManager;
12  import ch.colabproject.colab.generator.model.annotations.AdminResource;
13  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
14  import java.util.Collection;
15  import javax.inject.Inject;
16  import javax.ws.rs.Consumes;
17  import javax.ws.rs.DELETE;
18  import javax.ws.rs.GET;
19  import javax.ws.rs.PUT;
20  import javax.ws.rs.Path;
21  import javax.ws.rs.PathParam;
22  import javax.ws.rs.Produces;
23  import javax.ws.rs.core.MediaType;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  /**
28   * API to manage presence
29   *
30   * @author maxence
31   */
32  @Path("presence")
33  @Consumes(MediaType.APPLICATION_JSON)
34  @Produces(MediaType.APPLICATION_JSON)
35  @AuthenticationRequired
36  public class PresenceRestEndpoint {
37  
38      /** logger */
39      private static final Logger logger = LoggerFactory.getLogger(PresenceRestEndpoint.class);
40  
41      /**
42       * Presence business logic
43       */
44      @Inject
45      private PresenceManager presenceManager;
46  
47      /**
48       * get the presence list for the current project.
49       *
50       * @param projectId id of the project
51       *
52       * @return current presence list
53       */
54      @GET
55      @Path("{projectId: [0-9]*}")
56      public Collection<UserPresence> getProjectPresence(
57          @PathParam("projectId") Long projectId
58      ) {
59          logger.debug("Get Project Presence List #{}", projectId);
60          return presenceManager.getPresenceList(projectId);
61      }
62  
63      /**
64       * Make presence of a user known
65       *
66       * @param presence presence data
67       */
68      @PUT
69      public void updateUserPresence(TouchUserPresence presence) {
70          presenceManager.updateUserPresence(presence);
71      }
72  
73      /**
74       * Clear presence list for the given project
75       *
76       * @param projectId id of the project
77       */
78      @DELETE
79      @Path("{projectId: [0-9]*}")
80      public void clearProjectPresenceList(@PathParam("projectId") Long projectId) {
81          presenceManager.clearProjectPresenceList(projectId);
82      }
83  
84      /**
85       * Clear all presence lists
86       */
87      @DELETE
88      @AdminResource
89      public void clearAllPresenceList() {
90          presenceManager.clearAllPresenceLists();
91      }
92  }
