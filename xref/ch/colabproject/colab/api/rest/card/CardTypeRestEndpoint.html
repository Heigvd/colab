7   package ch.colabproject.colab.api.rest.card;
8   
9   import ch.colabproject.colab.api.controller.card.CardTypeManager;
10  import ch.colabproject.colab.api.exceptions.ColabMergeException;
11  import ch.colabproject.colab.api.model.card.AbstractCardType;
12  import ch.colabproject.colab.api.model.card.CardType;
13  import ch.colabproject.colab.api.persistence.jpa.card.CardTypeDao;
14  import ch.colabproject.colab.api.rest.card.bean.CardTypeCreationBean;
15  import ch.colabproject.colab.generator.model.annotations.AdminResource;
16  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
17  import java.util.List;
18  import java.util.Set;
19  import javax.inject.Inject;
20  import javax.ws.rs.Consumes;
21  import javax.ws.rs.DELETE;
22  import javax.ws.rs.GET;
23  import javax.ws.rs.POST;
24  import javax.ws.rs.PUT;
25  import javax.ws.rs.Path;
26  import javax.ws.rs.PathParam;
27  import javax.ws.rs.Produces;
28  import javax.ws.rs.core.MediaType;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  /**
33   * REST card type controller
34   *
35   * @author sandra
36   */
37  @Path("cardTypes")
38  @Consumes(MediaType.APPLICATION_JSON)
39  @Produces(MediaType.APPLICATION_JSON)
40  @AuthenticationRequired
41  public class CardTypeRestEndpoint {
42  
43      /** logger */
44      private static final Logger logger = LoggerFactory.getLogger(CardTypeRestEndpoint.class);
45  
46      /**
47       * The card type persistence manager
48       */
49      @Inject
50      private CardTypeDao cardTypeDao;
51  
52      /**
53       * Card type specific logic management
54       */
55      @Inject
56      private CardTypeManager cardTypeManager;
57  
58      /**
59       * Retrieve the list of all global card types. This is available to admin only
60       *
61       * @return all known global card types
62       */
63      @GET
64      @Path("allGlobals")
65      @AdminResource
66      public List<CardType> getAllGlobalCardTypes() {
67          logger.debug("get all global card types");
68          return cardTypeDao.findGlobalCardTypes();
69      }
70  
71      /**
72       * Retrieve the list of published global card types.
73       *
74       * @return all published global types
75       */
76      @GET
77      @Path("allPublishedGlobals")
78      public List<CardType> getPublishedGlobalsCardTypes() {
79          logger.debug("get published global card types");
80          return cardTypeDao.findPublishedGlobalCardTypes();
81      }
82  
83      /**
84       * Retrieve the list of published card types accessible to current user. It means all the
85       * published types of project the current user is member of.
86       *
87       * @return all published card types defined in a project the current user has access to
88       */
89      @GET
90      @Path("allProjectsPublished")
91      public Set<AbstractCardType> getPublishedCardTypesOfReachableProjects() {
92          logger.debug("get published projects card types");
93          return cardTypeManager.getCurrentUserExpandedPublishedProjectTypes();
94      }
95  
96      /**
97       * Retrieve the abstract card type and its chain of targets until the card type.
98       *
99       * @param id the id of the wanted abstract card type
100      *
101      * @return the corresponding abstract card type and its target if it is a reference (recursively
102      *         until the card type)
103      */
104     @GET
105     @Path("expanded/{id}")
106     public List<AbstractCardType> getExpandedCardType(@PathParam("id") Long id) {
107         logger.debug("get abstract card type #{} and its target chain", id);
108         return cardTypeManager.getExpandedCardType(id);
109     }
110 
111     // TODO sandra work in progress remove when the tests are changed to getExpandedCardType
112     /**
113      * Get card type identified by the given id
114      *
115      * @param id id of the card type to fetch
116      *
117      * @return the card type or null
118      */
119     @GET
120     @Path("{id}")
121     public AbstractCardType getCardType(@PathParam("id") Long id) {
122         logger.debug("get card type #{}", id);
123         return cardTypeDao.findAbstractCardType(id);
124     }
125 
126     /**
127      * Persist a card type.
128      *
129      * @param cardTypeCreationBean Everything need to create the card type
130      *
131      * @return id of the persisted new card type
132      */
133     @POST
134     public Long createCardType(CardTypeCreationBean cardTypeCreationBean) {
135         logger.debug("create card type {}", cardTypeCreationBean);
136 
137         CardType cardType = new CardType();
138         cardType.setProjectId(cardTypeCreationBean.getProjectId());
139         cardType.setTitle(cardTypeCreationBean.getTitle());
140         cardType.setPurpose(cardTypeCreationBean.getPurpose());
141         cardType.setTags(cardTypeCreationBean.getTags());
142 
143         if (cardTypeCreationBean.getPurpose() != null) {
144             cardTypeCreationBean.getPurpose().setPurposingCardType(cardType);
145         }
146 
147         return cardTypeManager.createCardType(cardType).getId();
148     }
149 
150     /**
151      * Save changes to database
152      *
153      * @param cardType the card type to update
154      *
155      * @throws ColabMergeException if the merge is not possible
156      */
157     @PUT
158     public void updateCardType(AbstractCardType cardType) throws ColabMergeException {
159         logger.debug("update abstract card type {}", cardType);
160 
161         cardTypeDao.updateAbstractCardType(cardType);
162     }
163 
164     /**
165      * Permanently delete a card type
166      *
167      * @param id the id of the card type to delete
168      */
169     @DELETE
170     @Path("{id}")
171     public void deleteCardType(@PathParam("id") Long id) {
172         logger.debug("delete card type #{}", id);
173         cardTypeManager.deleteCardType(id);
174     }
175 
176     /**
177      * If the card type is not already in the project, create a reference to it. Else simply return
178      * the card type.
179      *
180      * @param cardTypeId the id of the target card type (or reference)
181      * @param projectId  the id of the project in which we want to use the card type
182      *
183      * @return The card type
184      */
185     @PUT
186     @Path("useCardTypeInProject/{cardTypeId}/{projectId}")
187     public AbstractCardType useCardTypeInProject(@PathParam("cardTypeId") Long cardTypeId,
188         @PathParam("projectId") Long projectId) {
189         logger.debug("use card type #{} in project #{}", cardTypeId, projectId);
190 
191         return cardTypeManager.useCardTypeInProject(cardTypeId, projectId);
192     }
193 
194     /**
195      * Remove the card type use of the project. That means delete the reference in the project if it
196      * has no use. If the abstract card type is used, throws an error.
197      *
198      * @param cardTypeRefId the id of the card type reference no more useful for the project
199      * @param projectId  the id of the project in which we don't want to use the card type anymore
200      */
201     @PUT
202     @Path("removeCardTypeFromProject/{cardTypeId}/{projectId}")
203     public void removeCardTypeRefFromProject(@PathParam("cardTypeId") Long cardTypeRefId,
204         @PathParam("projectId") Long projectId) {
205         logger.debug("remove card type reference #{} from project #{}", cardTypeRefId, projectId);
206 
207         cardTypeManager.removeCardTypeRefFromProject(cardTypeRefId, projectId);
208     }
209 
210 }
