7   package ch.colabproject.colab.api.rest.card;
8   
9   import ch.colabproject.colab.api.ejb.CardFacade;
10  import ch.colabproject.colab.api.exceptions.ColabMergeException;
11  import ch.colabproject.colab.api.model.card.CardType;
12  import ch.colabproject.colab.api.persistence.card.CardTypeDao;
13  import ch.colabproject.colab.generator.model.annotations.AdminResource;
14  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
15  import java.util.List;
16  import javax.inject.Inject;
17  import javax.ws.rs.Consumes;
18  import javax.ws.rs.DELETE;
19  import javax.ws.rs.GET;
20  import javax.ws.rs.POST;
21  import javax.ws.rs.PUT;
22  import javax.ws.rs.Path;
23  import javax.ws.rs.PathParam;
24  import javax.ws.rs.Produces;
25  import javax.ws.rs.core.MediaType;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  /**
30   * REST card def controller
31   *
32   * @author sandra
33   */
34  // TODO will need access to a unique id generator
35  @Path("cardTypes")
36  @Consumes(MediaType.APPLICATION_JSON)
37  @Produces(MediaType.APPLICATION_JSON)
38  @AuthenticationRequired
39  public class CardTypeRestEndpoint {
40  
41      /** logger */
42      private static final Logger logger = LoggerFactory.getLogger(CardTypeRestEndpoint.class);
43  
44      /**
45       * The card definition persistence manager
46       */
47      @Inject
48      private CardTypeDao cardTypeDao;
49  
50      /**
51       * The card-related logic
52       */
53      @Inject
54      private CardFacade facade;
55  
56      /**
57       * Retrieve the list of all card defs. This is available to admin only
58       *
59       * @return all known card defs
60       */
61      @GET
62      @AdminResource
63      public List<CardType> getAllCardTypes() {
64          logger.debug("get all card defs");
65          return cardTypeDao.getAllCardType();
66      }
67  
68      /**
69       * Retrieve the list of all global card types. This is available to admin only
70       *
71       * @return all known global card types
72       */
73      @GET
74      @Path("allGlobals")
75      @AdminResource
76      public List<CardType> getAllGlobalCardTypes() {
77          logger.debug("get all global card defs");
78          return cardTypeDao.getGlobalCardType();
79      }
80  
81      /**
82       * Retrieve the list of public global card types.
83       *
84       * @return all published global types
85       */
86      @GET
87      @Path("publishedGlobals")
88      public List<CardType> getPublishedGlobalsCardTypes() {
89          logger.debug("get published global card defs");
90          return cardTypeDao.getPublishedGlobalCardType();
91      }
92  
93  
94      /**
95       * Retrieve the list of published card types accessible to current user.
96       * It means all the published types of project the current user is member of.
97       *
98       * @return all published card types
99       */
100     @GET
101     @Path("allProjectsPublished")
102     public List<CardType> getPublishedCardTypes() {
103         logger.debug("get published projects card defs");
104         return cardTypeDao.getPublishedProjectsCardType();
105     }
106 
107     /**
108      * Get card def identified by the given id
109      *
110      * @param id id of the card def to fetch
111      *
112      * @return the card def or null
113      */
114     @GET
115     @Path("{id}")
116     public CardType getCardType(@PathParam("id") Long id) {
117         logger.debug("get card def #{}", id);
118         return cardTypeDao.getCardType(id);
119     }
120 
121     /**
122      * Persist a card type.
123      *
124      * @param cardType the card definition to persist
125      *
126      * @return id of the persisted new card definition
127      */
128     @POST
129     public Long createCardType(CardType cardType) {
130         logger.debug("create card def {}", cardType);
131         return facade.createNewCardType(cardType).getId();
132     }
133 
134     /**
135      * Create and persist a new card type. The card type belongs to the given project.
136      *
137      * @param projectId the project the new card definition belongs to
138      *
139      * @return the persisted new card definition
140     @POST
141     @Path("create/{projectId}")
142     public CardType createNewCardType(@PathParam("projectId") Long projectId) {
143         logger.debug("create new card def for the project #{}", projectId);
144         return facade.createNewCardType(projectId);
145     }
146      */
147 
148     /**
149      * Save changes to database
150      *
151      * @param cardType card def to update
152      *
153      * @throws ColabMergeException if the merge is not possible
154      */
155     @PUT
156     public void updateCardType(CardType cardType) throws ColabMergeException {
157         logger.debug("update card def {}", cardType);
158         cardTypeDao.updateCardType(cardType);
159     }
160 
161     /**
162      * Permanently delete a card def
163      *
164      * @param id id of the card def to delete
165      */
166     @DELETE
167     @Path("{id}")
168     public void deleteCardType(@PathParam("id") Long id) {
169         logger.debug("delete card def #{}", id);
170         cardTypeDao.deleteCardType(id);
171     }
172 
173 }
