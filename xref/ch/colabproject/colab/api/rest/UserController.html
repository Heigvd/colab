7   package ch.colabproject.colab.api.rest;
8   
9   import ch.colabproject.colab.api.ejb.UserManagement;
10  import ch.colabproject.colab.api.exceptions.ColabErrorMessage;
11  import ch.colabproject.colab.api.model.user.AuthInfo;
12  import ch.colabproject.colab.api.model.user.AuthMethod;
13  import ch.colabproject.colab.api.model.user.SignUpInfo;
14  import javax.inject.Inject;
15  import javax.ws.rs.Consumes;
16  import javax.ws.rs.GET;
17  import javax.ws.rs.POST;
18  import javax.ws.rs.Path;
19  import javax.ws.rs.PathParam;
20  import javax.ws.rs.Produces;
21  import javax.ws.rs.core.MediaType;
22  
23  /**
24   * User controller
25   *
26   * @author maxence
27   */
28  @Path("users")
29  @Consumes(MediaType.APPLICATION_JSON)
30  @Produces(MediaType.APPLICATION_JSON)
31  public class UserController {
32  
33      /**
34       * Users related business logic
35       */
36      @Inject
37      private UserManagement userManagement;
38  
39      /**
40       * Get the authentication method and its parameters a user shall use to authenticate with the
41       * given email address
42       *
43       * @param email email address user want to sign in with
44       *
45       * @return Auth method to user
46       */
47      @GET
48      @Path("/AuthMethod/{email}")
49      public AuthMethod getAuthMethod(@PathParam("username") String email) {
50          return userManagement.getAuthenticationMethod(email);
51      }
52  
53      /**
54       * Create a new local account
55       *
56       * @param signup all data required to create a local account
57       *
58       * @throws ColabErrorMessage if creation fails for any reason
59       */
60      @POST
61      @Path("SignUp")
62      public void signUp(SignUpInfo signup) throws ColabErrorMessage {
63          userManagement.signup(signup);
64      }
65  
66      /**
67       * Authenticate with local-account credentials
68       *
69       * @param authInfo credentials
70       *
71       * @throws ColabErrorMessage if authentication fails for any reason
72       */
73      @POST
74      @Path("/SignIn")
75      public void signIn(AuthInfo authInfo) throws ColabErrorMessage {
76          userManagement.authenticate(authInfo);
77      }
78  
79      /**
80       * Sign out
81       */
82      @POST
83      public void signOut() {
84          userManagement.logout();
85      }
86  }
