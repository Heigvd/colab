7   package ch.colabproject.colab.api.rest;
8   
9   import ch.colabproject.colab.api.ejb.RequestManager;
10  import ch.colabproject.colab.api.ejb.SecurityFacade;
11  import ch.colabproject.colab.api.ejb.UserManagement;
12  import ch.colabproject.colab.api.exceptions.ColabMergeException;
13  import ch.colabproject.colab.api.model.user.Account;
14  import ch.colabproject.colab.api.model.user.AuthInfo;
15  import ch.colabproject.colab.api.model.user.AuthMethod;
16  import ch.colabproject.colab.api.model.user.SignUpInfo;
17  import ch.colabproject.colab.api.model.user.User;
18  import ch.colabproject.colab.generator.model.annotations.AdminResource;
19  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
20  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
21  import javax.inject.Inject;
22  import javax.ws.rs.Consumes;
23  import javax.ws.rs.GET;
24  import javax.ws.rs.POST;
25  import javax.ws.rs.PUT;
26  import javax.ws.rs.Path;
27  import javax.ws.rs.PathParam;
28  import javax.ws.rs.Produces;
29  import javax.ws.rs.core.MediaType;
30  
31  /**
32   * User controller
33   *
34   * @author maxence
35   */
36  @Path("users")
37  @Consumes(MediaType.APPLICATION_JSON)
38  @Produces(MediaType.APPLICATION_JSON)
39  public class UserController {
40  
41      /**
42       * Users related business logic
43       */
44      @Inject
45      private UserManagement userManagement;
46  
47      /**
48       * Request related logic
49       */
50      @Inject
51      private RequestManager requestManager;
52  
53      /**
54       * to control access
55       */
56      @Inject
57      private SecurityFacade securityFacade;
58  
59      /**
60       * Get the authentication method and its parameters a user shall use to authenticate with the
61       * given email address.
62       *
63       * @param identifier email address or username the user want to sign in with
64       *
65       * @return Auth method to user
66       */
67      @GET
68      @Path("AuthMethod/{identifier : [^/]*}")
69      public AuthMethod getAuthMethod(@PathParam("identifier") String identifier) {
70          return userManagement.getAuthenticationMethod(identifier);
71      }
72  
73      /**
74       * Return the current authenticated user.
75       *
76       * @return current user or null
77       */
78      @GET
79      @Path("CurrentUser")
80      public User getCurrentUser() {
81          return requestManager.getCurrentUser();
82      }
83  
84      /**
85       * Return the current authenticated account.
86       *
87       * @return current account or null
88       */
89      @GET
90      @Path("CurrentAccount")
91      public Account getCurrentAccount() {
92          return requestManager.getCurrentAccount();
93      }
94  
95      /**
96       * Create a new local account.
97       *
98       * @param signup all data required to create a local account
99       *
100      * @throws HttpErrorMessage if creation fails for any reason
101      */
102     @POST
103     @Path("SignUp")
104     public void signUp(SignUpInfo signup) {
105         userManagement.signup(signup);
106     }
107 
108     /**
109      * Authenticate with local-account credentials.
110      *
111      * @param authInfo credentials
112      *
113      * @throws HttpErrorMessage if authentication fails for any reason
114      */
115     @POST
116     @Path("SignIn")
117     public void signIn(AuthInfo authInfo) {
118         userManagement.authenticate(authInfo);
119     }
120 
121     /**
122      * Sign out.
123      */
124     @POST
125     @Path("SignOut")
126     public void signOut() {
127         userManagement.logout();
128     }
129 
130     /**
131      * Update user names.
132      *
133      * @param user user to update
134      *
135      * @throws ColabMergeException if update fails
136      */
137     @PUT
138     @AuthenticationRequired
139     public void updateUser(User user) throws ColabMergeException {
140         securityFacade.assertCanWrite(user);
141         userManagement.updateUser(user);
142     }
143 
144     /**
145      * Grant admin right to user identified by given id.
146      *
147      * @param id id of the user
148      */
149     @PUT
150     @Path("{id : [1-9][0-9]*}/GrantAdminRight")
151     @AdminResource
152     public void grantAdminRight(@PathParam("id") Long id) {
153         userManagement.grantAdminRight(id);
154     }
155 
156     /**
157      * ask the user to switch to new hash method on next sign-in.
158      *
159      * @param id id of the local account
160      */
161     @PUT
162     @Path("{id : [1-9][0-9]*}/SwitchClientHashMethod")
163     @AdminResource
164     public void switchClientHashMethod(@PathParam("id") Long id) {
165         userManagement.switchClientHashMethod(id);
166     }
167 
168     /**
169      * stage new server hash method.
170      *
171      * @param id id of the local account
172      */
173     @PUT
174     @Path("{id : [1-9][0-9]*}/SwitchServerHashMethod")
175     @AdminResource
176     public void switchServerHashMethod(@PathParam("id") Long id) {
177         userManagement.switchServerHashMethod(id);
178     }
179 
180 }
