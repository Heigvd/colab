7   package ch.colabproject.colab.api.rest.user;
8   
9   import ch.colabproject.colab.api.controller.RequestManager;
10  import ch.colabproject.colab.api.controller.user.UserManager;
11  import ch.colabproject.colab.api.exceptions.ColabMergeException;
12  import ch.colabproject.colab.api.model.user.Account;
13  import ch.colabproject.colab.api.model.user.AuthInfo;
14  import ch.colabproject.colab.api.model.user.AuthMethod;
15  import ch.colabproject.colab.api.model.user.HttpSession;
16  import ch.colabproject.colab.api.model.user.LocalAccount;
17  import ch.colabproject.colab.api.model.user.SignUpInfo;
18  import ch.colabproject.colab.api.model.user.User;
19  import ch.colabproject.colab.api.persistence.jpa.user.UserDao;
20  import ch.colabproject.colab.generator.model.annotations.AdminResource;
21  import ch.colabproject.colab.generator.model.annotations.AuthenticationRequired;
22  import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
23  import java.util.List;
24  import javax.inject.Inject;
25  import javax.ws.rs.Consumes;
26  import javax.ws.rs.DELETE;
27  import javax.ws.rs.GET;
28  import javax.ws.rs.POST;
29  import javax.ws.rs.PUT;
30  import javax.ws.rs.Path;
31  import javax.ws.rs.PathParam;
32  import javax.ws.rs.Produces;
33  import javax.ws.rs.core.MediaType;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  /**
38   * User controller
39   *
40   * @author maxence
41   */
42  @Path("users")
43  @Consumes(MediaType.APPLICATION_JSON)
44  @Produces(MediaType.APPLICATION_JSON)
45  public class UserRestEndpoint {
46  
47      /** logger */
48      private static final Logger logger = LoggerFactory.getLogger(UserRestEndpoint.class);
49  
50      /**
51       * Users related business logic
52       */
53      @Inject
54      private UserManager userManager;
55  
56      /**
57       * User persistence
58       */
59      @Inject
60      private UserDao userDao;
61  
62      /**
63       * Request related logic
64       */
65      @Inject
66      private RequestManager requestManager;
67  
68      /**
69       * Get the authentication method and its parameters a user shall use to authenticate with the
70       * given email address.
71       *
72       * @param identifier email address or username the user want to sign in with
73       *
74       * @return Authentication method to user
75       */
76      @GET
77      @Path("AuthMethod/{identifier : [^/]*}")
78      public AuthMethod getAuthMethod(@PathParam("identifier") String identifier) {
79          logger.debug("get auth method for {}", identifier);
80          return userManager.getAuthenticationMethod(identifier);
81      }
82  
83      /**
84       * Get all users
85       *
86       * @return list of all users
87       */
88      @GET
89      @AdminResource
90      public List<User> getAllUsers() {
91          logger.debug("get all users");
92          return userDao.findAllUsers();
93      }
94  
95      /**
96       * Return the current authenticated user.
97       *
98       * @param id id of the user
99       *
100      * @return user which match the given id or null
101      */
102     @GET
103     @Path("{id : [0-9]*}")
104     @AuthenticationRequired
105     public User getUserById(@PathParam("id") Long id) {
106         logger.debug("get user #{}", id);
107         return userManager.getUserById(id);
108     }
109 
110     /**
111      * Return the current authenticated user.
112      *
113      * @return current user or null
114      */
115     @GET
116     @Path("CurrentUser")
117     public User getCurrentUser() {
118         logger.debug("get current user");
119         return requestManager.getCurrentUser();
120     }
121 
122     /**
123      * Return the current authenticated account.
124      *
125      * @return current account or null
126      */
127     @GET
128     @Path("CurrentAccount")
129     public Account getCurrentAccount() {
130         logger.debug("get current account");
131         return requestManager.getCurrentAccount();
132     }
133 
134     /**
135      * Get all accounts owned by the current user
136      *
137      * @return list of accounts
138      */
139     @GET
140     @Path("AllCurrentUserAccount")
141     public List<Account> getAllCurrentUserAccounts() {
142         logger.debug("get all accounts of current user");
143         User user = requestManager.getCurrentUser();
144         if (user != null) {
145             return user.getAccounts();
146         } else {
147             return List.of();
148         }
149     }
150 
151     /**
152      * Create a new local account.
153      *
154      * @param signup all data required to create a local account
155      *
156      * @throws HttpErrorMessage if creation fails for any reason
157      */
158     @POST
159     @Path("SignUp")
160     public void signUp(SignUpInfo signup) {
161         logger.debug("sign-up {}", signup);
162         userManager.signup(signup);
163     }
164 
165     /**
166      * Authenticate with local-account credentials.
167      *
168      * @param authInfo credentials
169      *
170      * @throws HttpErrorMessage if authentication fails for any reason
171      */
172     @POST
173     @Path("SignIn")
174     public void signIn(AuthInfo authInfo) {
175         logger.debug("sign-in {}", authInfo);
176         userManager.authenticate(authInfo);
177     }
178 
179     /**
180      * Sign out.
181      */
182     @POST
183     @Path("SignOut")
184     public void signOut() {
185         logger.debug("sign-out");
186         userManager.logout();
187     }
188 
189     /**
190      * Get all active session for the current user
191      *
192      * @return list of active session linked to the current user
193      */
194     @GET
195     @Path("Sessions")
196     @AuthenticationRequired
197     public List<HttpSession> getActiveSessions() {
198         return userManager.getCurrentUserActiveSessions();
199     }
200 
201     /**
202      * Force session logout
203      *
204      * @param sessionId if of the HTTP session to delete
205      */
206     @DELETE
207     @Path("Session/{id}")
208     @AuthenticationRequired
209     public void forceLogout(@PathParam("id") Long sessionId) {
210         userManager.forceLogout(sessionId);
211     }
212 
213     /**
214      * Update user. Only fields which are editable by users will be impacted..
215      *
216      * @param user user to update
217      *
218      * @throws ColabMergeException if update fails
219      */
220     @PUT
221     @AuthenticationRequired
222     public void updateUser(User user) throws ColabMergeException {
223         logger.debug("update user profile: {}", user);
224         userDao.updateUser(user);
225     }
226 
227     /**
228      * Grant admin right to user identified by given id.
229      *
230      * @param id id of the user
231      */
232     @PUT
233     @Path("{id : [1-9][0-9]*}/GrantAdminRight")
234     @AdminResource
235     public void grantAdminRight(@PathParam("id") Long id) {
236         logger.debug("Grant admin right to user #{}", id);
237         userManager.grantAdminRight(id);
238     }
239 
240     /**
241      * remove admin right to user identified by given id.
242      *
243      * @param id id of the user
244      */
245     @DELETE
246     @Path("{id : [1-9][0-9]*}/GrantAdminRight")
247     @AdminResource
248     public void revokeAdminRight(@PathParam("id") Long id) {
249         logger.debug("Grant admin right to user #{}", id);
250         userManager.revokeAdminRight(id);
251     }
252 
253     /**
254      * Update password of a localAccount. The password MUST be hashed a first time by the client.
255      * <p>
256      * @param authInfo identifier and new password
257      *
258      * @throws HttpErrorMessage if currentUser is not allowed to update the given account
259      */
260     @PUT
261     @Path("/UpdatePassword")
262     public void updateLocalAccountPassword(AuthInfo authInfo) {
263         logger.debug("Update local account \"{}\" password", authInfo.getIdentifier());
264         userManager.updatePassword(authInfo);
265     }
266 
267     /**
268      * Request a local account password reset. This method always returns 204 no content
269      * <p>
270      * @param email email address of account
271      *
272      */
273     @PUT
274     @Path("/RequestPasswordReset/{email}")
275     public void requestPasswordReset(@PathParam("email") String email) {
276         logger.debug("Request password reset token {}", email);
277         userManager.requestPasswordReset(email);
278     }
279 
280     /**
281      * Update local account
282      *
283      * @param account local account with new email address inside
284      *
285      * @throws ColabMergeException if update fails
286      */
287     @PUT
288     @Path("/ChangeEmail")
289     @AuthenticationRequired
290     public void updateLocalAccountEmailAddress(LocalAccount account) throws ColabMergeException {
291         logger.debug("Update local account email address: {}", account);
292         userManager.updateLocalAccountEmailAddress(account);
293     }
294 
295     /**
296      * ask the user to switch to new hash method on next sign-in.
297      *
298      * @param id id of the local account
299      */
300     @PUT
301     @Path("{id : [1-9][0-9]*}/SwitchClientHashMethod")
302     @AdminResource
303     public void switchClientHashMethod(@PathParam("id") Long id) {
304         logger.debug("Switch client hash method for account #{}", id);
305         userManager.switchClientHashMethod(id);
306     }
307 
308     /**
309      * stage new server hash method.
310      *
311      * @param id id of the local account
312      */
313     @PUT
314     @Path("{id : [1-9][0-9]*}/SwitchServerHashMethod")
315     @AdminResource
316     public void switchServerHashMethod(@PathParam("id") Long id) {
317         logger.debug("Switch server hash method for account #{}", id);
318         userManager.switchServerHashMethod(id);
319     }

321 }
