40      /**
41       * Find a token by id
42       *
43       * @param id id of the tokenÂ§
44       *
45       * @return the token if found or null
46       */
47      public Token findToken(Long id) {
48          return em.find(Token.class, id);
49      }

51      /**
52       * Find any VerifyLocalAccountToken linked to the given account.
53       *
54       * @param account owner of the token
55       *
56       * @return the token if found. Null otherwise
57       */
58      public VerifyLocalAccountToken findVerifyTokenByAccount(LocalAccount account) {
59          try {
60              return em.createNamedQuery("VerifyLocalAccountToken.findByAccountId",
61                  VerifyLocalAccountToken.class)
62                  .setParameter("id", account.getId())
63                  .getSingleResult();
64          } catch (NoResultException ex) {
65              return null;
66          }
67      }

70       * Find any ResetLocalAccountPasswordToken linked to the given account.

87      /**
88       * Find if a pending invitation has already be sent to recipient to join the project
89       *
90       * @param project   the project
91       * @param recipient recipient
92       *
93       * @return invitation if there is a pending one, null otherwise
94       */
95      public InvitationToken findInvitationByProjectAndRecipient(Project project, String recipient) {
96          try {
97              return em.createNamedQuery("InvitationToken.findByProjectAndRecipient",
98                  InvitationToken.class)
99                  .setParameter("projectId", project.getId())
100                 .setParameter("recipient", recipient)
101                 .getSingleResult();
102         } catch (NoResultException ex) {
103             return null;
104         }
105     }

107     /**
108      * Find all pending invitation for a teamMember and a project

110      * @param teamMember the team member linked to the invitation token
111      *
112      * @return invitations for the team member

114     public List<InvitationToken> findInvitationByTeamMember(TeamMember teamMember) {
115         return em.createNamedQuery("InvitationToken.findByTeamMember",
116             InvitationToken.class)
117             .setParameter("teamMemberId", teamMember.getId())

126      *

129     public ModelSharingToken findModelShareByProjectAndRecipient(Project project, String recipient) {

134                 .setParameter("recipient", recipient)
135                 .getSingleResult();
136         } catch (NoResultException ex) {
137             return null;
138         }
139     }
140 
141     /**
142      * Persist the token
143      *
144      * @param token token to persist
145      */
146     public void persistToken(Token token) {
147         // set something to respect notNull constraints
148         // otherwise persist will fail
149         // These values will be reset when the e-mail is sent.
150         if (token.getHashMethod() == null) {
151             token.setHashMethod(Helper.getDefaultHashMethod());
152         }
153         if (token.getHashedToken() == null) {
154             token.setHashedToken(new byte[0]);
155         }
156 
157         em.persist(token);
158     }
159 
160     /**
161      * Delete the given token
162      *
163      * @param token token to remove
164      */
165     public void deleteToken(Token token) {
166         em.remove(token);
167     }
