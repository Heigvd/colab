7   package ch.colabproject.colab.api.persistence.jpa.team.acl;
8   
9   import ch.colabproject.colab.api.model.team.acl.AccessControl;
10  import javax.ejb.LocalBean;
11  import javax.ejb.Stateless;
12  import javax.persistence.EntityManager;
13  import javax.persistence.PersistenceContext;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  
17  /**
18   * Access control persistence
19   * <p>
20   * Note : Most of database operations are handled by managed entities and cascade.
21   *
22   * @author maxence
23   */
24  @Stateless
25  @LocalBean
26  public class AccessControlDao {
27  
28      /** logger */
29      private static final Logger logger = LoggerFactory.getLogger(AccessControlDao.class);
30  
31      /**
32       * Access to the persistence unit
33       */
34      @PersistenceContext(unitName = "COLAB_PU")
35      private EntityManager em;
36  
37      /**
38       * Delete the access control from database. This can't be undone
39       *
40       * @param ac access control to remove
41       */
42      public void deleteAccessControl(AccessControl ac) {
43          logger.trace("delete access control {}", ac);
44  
45          em.remove(ac);
46      }
47  
48  }
