7   package ch.colabproject.colab.api.persistence.jcr;
8   
9   import ch.colabproject.colab.generator.model.exceptions.HttpErrorMessage;
10  import javax.jcr.RepositoryException;
11  import javax.transaction.Status;
12  import javax.transaction.Synchronization;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  /**
17   * To synchronize JCR session with JTA transaction
18   *
19   * @author maxence
20   */
21  public class JcrSynchronizer implements Synchronization {
22  
23      /** Logger */
24      private static final Logger logger = LoggerFactory.getLogger(JcrSynchronizer.class);
25  
26      /** The synchronized session */
27      private final JcrSession session;
28  
29      /**
30       * Create JCR session synchronizer
31       *
32       * @param session the session to synchronize
33       */
34      public JcrSynchronizer(JcrSession session) {
35          logger.trace("Create synchronizer for {}", session);
36          this.session = session;
37      }
38  
39      @Override
40      public void beforeCompletion() {
41          try {
42              logger.trace("Prepare commit for {}", session);
43              session.prepareForCommit();
44          } catch (RepositoryException ex) {
45              // please rollback
46              logger.trace("Prepare commit failed: rollback");
47              throw HttpErrorMessage.dataIntegrityFailure();
48          }
49      }
50  
51      @Override
52      public void afterCompletion(int status) {
53          switch (status) {
54              case Status.STATUS_COMMITTED:
55                  logger.trace("JCR Sync afterCompletion: {} => COMMITED", status);
56                  session.saveAndClose();
57                  break;
58              case Status.STATUS_ROLLEDBACK:
59                  logger.trace("JCR Sync afterCompletion: {} => ROLLED_BACK", status);
60                  session.rollback();
61                  break;
62              default:
63                  logger.trace("JCR Sync afterCompletion: {} => UNKNOWN", status);
64                  break;
65          }
66  
67      }
68  }
