108     /**
109      * Process all classes annotated with {@link Path}. Generate a {@link RestEndpoint} instance for
110      * each class and store them in {@link #restEndpoints}
111      */
112     public void processPackages() {
113         Set<Class<?>> appConfig = reflections.getTypesAnnotatedWith(ApplicationPath.class);
114         if (!appConfig.isEmpty()) {
115             if (appConfig.size() > 1) {
116                 Logger.warn("Several ApplicationPath found");
117             }
118             Class<?> applicationConfig = appConfig.iterator().next();
119             ApplicationPath annotation = applicationConfig.getAnnotation(ApplicationPath.class);
120             if (annotation != null) {
121                 this.applicationPath = annotation.value();
122             }
123         }
124 
125         Set<Class<?>> restClasses = reflections.getTypesAnnotatedWith(Path.class);
126 
127         this.restEndpoints = restClasses.stream()
128             .map(klass -> RestEndpoint.build(klass, applicationPath))
129             .collect(Collectors.toSet());
130         /*
131          * .map(p -> p.generateJavaClient()) .innerClasses(Collectors.toList());
132          */
133     }
134 
135     /**
136      * One all classes have been processed with {@link #processPackages() }, this method will create
137      * subdirectories that match the <code>packageName</code>.Then the client will be generated in
138      * the <code>clientName</code>.java file.
139      *
140      * @param targetDir target directory
141      * @param dryRun    if true, do not generate files but print output to console
142      *
143      * @throws org.apache.maven.plugin.MojoFailureException if generation fails
144      */
145     public void generateJavaClient(String targetDir, boolean dryRun) throws MojoFailureException {
146         Map<String, String> imports = new HashMap<>();
147         imports.put("RestClient", "ch.colabproject.colab.generator.plugin.rest.RestClient");
148         imports.put("Jsonb", "javax.json.bind.Jsonb");
149         imports.put("GenericType", "javax.ws.rs.core.GenericType");
150         imports.put("PathPattern", "org.glassfish.jersey.uri.PathPattern");
151         imports.put("UriTemplate", "org.glassfish.jersey.uri.UriTemplate");
152         imports.put("void", null); // null means no import statement
153 
154         String innerClasses = this.restEndpoints.stream().map(controller -> {
155             String javaCode = controller.generateJavaClient(
156                 imports,
157                 clientName,
158                 javadoc,
159                 reflections
160             );
161             return javaCode;
162         }).collect(Collectors.joining(System.lineSeparator()));
163 
164         StringBuilder sb = new StringBuilder();
165 
166         sb.append("package ").append(packageName).append(";\n\n")
167             .append(
168                 imports.values().stream()
169                     .filter(pkg -> pkg != null)
170                     .filter(pkg -> !pkg.startsWith("java.lang")) // do not import java.lang
171                     .sorted()
172                     .map(pkg -> "import " + pkg + ";")
173                     .collect(Collectors.joining(System.lineSeparator()))
174             )
175             .append("\n"
176                 + "/**\n"
177                 + " * The ")
178             .append(clientName)
179             .append(" REST client"
180                 + " */\n"
181                 + "@SuppressWarnings(\"PMD.FieldDeclarationsShouldBeAtStartOfClass\")\n"
182                 + "public class ")
183             .append(clientName)
184             .append(" extends RestClient {"
185                 + "\n"
186                 + "\n"
187                 + "    /**\n"
188                 + "     * Get a REST client\n"
189                 + "     *\n"
190                 + "     * @param baseUri        base URI\n"
191                 + "     * @param cookieName     session cookie name\n"
192                 + "     * @param jsonb          jsonb\n"
193                 + "     * @param clientFeatures addition http client feature\n"
194                 + "     */\n"
195                 + "    public ")
196             .append(clientName)
197             .append("(String baseUri, String cookieName, Jsonb jsonb, Object... clientFeatures) {\n"
198                 + "        super(baseUri, cookieName, jsonb, clientFeatures);\n"
199                 + "    }")
200             .append(innerClasses)
201             .append("}");
202 
203         if (dryRun) {
204             Logger.debug(sb.toString());
205         } else {
206             String packagePath = targetDir + "/" + packageName.replaceAll("\\.", "/");
207             writeFile(sb.toString(), packagePath, clientName + ".java");
208         }
209     }

220     /**
221      * Generate typescript client in targetDir
222      *
223      * @param targetDir generate TS module in this directory
224      * @param dryRun    if true, do not generate any file but print output to console
225      *
226      * @throws org.apache.maven.plugin.MojoFailureException if generation fails
227      */
228     public void generateTypescriptClient(String targetDir, boolean dryRun)
229         throws MojoFailureException {
230         Map<String, Type> extraTypes = new HashMap<>();
231         StringBuilder sb = new StringBuilder();
232         sb.append(this.getTsClientTemplate());
233         extraTypes.put("WithJsonDiscriminator", WithJsonDiscriminator.class);
234         extraTypes.put("WithId", WithId.class);
235 
236         String modules = this.restEndpoints.stream().map(
237             controller -> controller.generateTypescriptClient(extraTypes, this.javadoc, reflections)
238         ).collect(Collectors.joining(System.lineSeparator()));
239 
240         // TS interface name => list of @class values
241         Map<String, List<String>> inheritance = new HashMap<>();
242 
243         List<Entry<String, Type>> queue = new ArrayList<>(extraTypes.entrySet());
244         while (!queue.isEmpty()) {
245             Map<String, Type> snowballedTypes = new HashMap<>();
246 
247             Entry<String, Type> entry = queue.remove(0);
248 
249             String tsInterface = TypeScriptHelper.generateInterface(
250                 entry.getValue(),
251                 snowballedTypes,
252                 inheritance,
253                 reflections,
254                 javadoc
255             );
256             sb.append(tsInterface);
257 
258             snowballedTypes.forEach((name, type) -> {
259                 if (!extraTypes.containsKey(name)) {
260                     extraTypes.put(name, type);
261                     queue.add(0, new SimpleEntry<>(name, type));
262                 }
263             });
264         }
265 
266         sb.append("/**\n"
267             + " * Some orthopedic tools\n"
268             + " */\n\n"
269             + "export interface TypeMap {\n  ")
270             .append(
271                 inheritance.keySet().stream().map((key) -> key + ": " + key + ";")
272                     .collect(Collectors.joining("\n  "))
273             )
274             .append("\n}\n\n")
275             .append("const inheritance : {[key: string]: string[]} = {\n")
276             .append(
277                 inheritance.entrySet().stream()
278                     .map((entry) -> entry.getKey() + ": [" + entry.getValue().stream()
279                         .map(v -> "'" + v + "'")
280                         .collect(Collectors.joining(", ")) + "]"
281                     ).collect(Collectors.joining(",\n  ")))
282             .append("\n}\n\n")
283             .append("export const entityIs = <T extends keyof TypeMap>(entity: unknown, klass: T)\n"
284                 + "    : entity is TypeMap[T] => {\n"
285                 + "\n"
286                 + "    if (typeof entity === 'object' && entity != null) {\n"
287                 + "        if (\"@class\" in entity) {\n"
288                 + "            const dis = entity[\"@class\"];\n"
289                 + "            if (typeof dis === 'string') {\n"
290                 + "                return inheritance[klass].includes(dis);\n"
291                 + "            }\n"
292                 + "        }\n"
293                 + "    }\n"
294                 + "    return false;\n"
295                 + "}")
296             .append("\n\n\n/**\n"
297                 + "* The ")
298             .append(clientName).append(" REST client\n"
299                 + " */\n"
300                 + "export const ")
301             .append(clientName)
302             .append(
303                 " = function (baseUrl: string, defaultErrorHandler: (error: unknown) => void) {")
304             .append("\n    return {")
305             .append(modules)
306             .append("    }\n")
307             .append("}");
308 
309         if (dryRun) {
310             Logger.debug(sb.toString());
311         } else {
312             String srcPath = targetDir + "/src";
313             writeFile(sb.toString(), srcPath, clientName + ".ts");
314 
315             writeFile("export * from './dist/" + clientName + "';", targetDir, "index.ts");
316             writeFile(generatePackageDotJson(), targetDir, "package.json");
317             writeFile(generateTsconfigDotJson(), targetDir, "tsconfig.json");
318         }
319     }

321     /**
322      * Write file to disk.
323      *
324      * @param content   file content
325      * @param directory directory, will be created if missing
326      * @param filename  filename
327      */
328     private void writeFile(String content, String directory, String filename)
329         throws MojoFailureException {
330         try {
331             Files.createDirectories(java.nio.file.Path.of(directory));
332 
333             try (BufferedWriter writer = Files.newBufferedWriter(
334                 java.nio.file.Path.of(directory, filename))) {
335                 writer.write(content);
336             } catch (IOException ex) {
337                 if (Logger.isInit()) {
338                     throw new MojoFailureException("Failed to write '"
339                         + filename + "' in '" + directory + "'", ex);
340                 }
341             }
342 
343         } catch (IOException ex) {
344             if (Logger.isInit()) {
345                 throw new MojoFailureException("Failed to create package directory "
346                     + directory, ex);
347             }
348         }
349     }

351     /**
352      * Convert Java className-like to dash-separated-lower-case version.
353      *
354      * @param name eg. MyAwesomeRestClient
355      *
356      * @return eg. my-awesome-rest-client
357      */
358     private String generateModuleName(String name) {
359         return name
360             .trim()
361             // prefix all uppercase char preceded by something with an dash
362             .replaceAll("(?<!^)[A-Z](?!$)", "-$0")
363             .toLowerCase();
364     }

372         String tsConfig = FileHelper.readFile("templates/client.ts");
