7   package ch.colabproject.colab.generator.model.exceptions;
8   
9   import ch.colabproject.colab.generator.model.interfaces.WithJsonDiscriminator;
10  import ch.colabproject.colab.generator.model.tools.PolymorphicDeserializer;
11  import javax.ejb.ApplicationException;
12  import javax.json.bind.annotation.JsonbTransient;
13  import javax.json.bind.annotation.JsonbTypeDeserializer;
14  import javax.ws.rs.core.Response;
15  
16  /**
17   * run-time exception which may be sent back to client.
18   *
19   * @author maxence
20   */
21  @ApplicationException(rollback = true)
22  @JsonbTypeDeserializer(PolymorphicDeserializer.class)
23  public abstract class HttpException extends RuntimeException implements WithJsonDiscriminator {
24  
25      private static final long serialVersionUID = 1L;
26  
27      /**
28       * HTTP status
29       */
30      private final Response.Status httpStatus;
31  
32      /**
33       * Create a default error with HTTP 400 bad request status
34       */
35      public HttpException() {
36          this(Response.Status.BAD_REQUEST);
37      }
38  
39      /**
40       * Create error with custom HTTP status
41       *
42       * @param httpStatus HTTP status code
43       */
44      public HttpException(Response.Status httpStatus) {
45          this.httpStatus = httpStatus;
46      }
47  
48      @JsonbTransient
49      @Override
50      public synchronized Throwable getCause() {
51          return super.getCause();
52      }
53  
54      @JsonbTransient
55      @Override
56      public StackTraceElement[] getStackTrace() {
57          return super.getStackTrace();
58      }
59  
60      /**
61       * Get the code to use to send HTTP response
62       *
63       * @return HTTP code, should be 400+
64       */
65      @JsonbTransient
66      public Response.Status getHttpStatus() {
67          return this.httpStatus;
68      }
69  
70      /*
71       * want to ignore getSuppressed() as well, but the method is final.
72       * Setting custom JsonbVisibility stategry on this class is useles as super properties are
73       * not filter by this strategy...
74       */
75  
76  }
