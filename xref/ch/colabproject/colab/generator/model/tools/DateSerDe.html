7   package ch.colabproject.colab.generator.model.tools;
8   
9   import java.lang.reflect.Type;
10  import java.time.Instant;
11  import java.time.OffsetDateTime;
12  import java.time.ZoneId;
13  import javax.json.bind.serializer.DeserializationContext;
14  import javax.json.bind.serializer.JsonbDeserializer;
15  import javax.json.bind.serializer.JsonbSerializer;
16  import javax.json.bind.serializer.SerializationContext;
17  import javax.json.stream.JsonGenerator;
18  import javax.json.stream.JsonParser;
19  
20  /**
21   * Date Serializer and Deserializer.
22   * <p>
23   * Serialize offsetDateTime as string-encoded UTC timestamp.
24   * <p>
25   * Deserialize back to OffsetDateTime using system timezone.
26   *
27   * @author Maxence
28   */
29  public class DateSerDe implements JsonbDeserializer<OffsetDateTime>, JsonbSerializer<OffsetDateTime> {
30  
31      @Override
32      public OffsetDateTime deserialize(JsonParser jp, DeserializationContext dc, Type type) {
33          String aString = jp.getString();
34          Long aLong = Long.parseLong(aString);
35          return OffsetDateTime.ofInstant(Instant.ofEpochMilli(aLong), ZoneId.systemDefault());
36      }
37  
38      @Override
39      public void serialize(OffsetDateTime t, JsonGenerator jg, SerializationContext sc) {
40          sc.serialize(Instant.from(t).toEpochMilli(), jg);
41      }
42  
43  }
