117     private void doTest(HierarchicalPosition hierarchicalPosition,
118         InvolvementLevel roleInvolvementLevel,
119         boolean expectedCanRead, boolean expectedCanWrite) throws Exception {
120         // creation of the context :
121         // project, global type, project type, card, sub card,
122         // role : designer
123         Project project = ColabFactory.createProject(client, "testResource");
124         Long projectId = project.getId();
125 
126         Long designerRoleId = ColabFactory.createRole(client, project, "designer").getId();
127         Long developerRoleId = ColabFactory.createRole(client, project, "developer").getId();
128 
129         Long rootCardContentId = ColabFactory.getRootContent(client, project).getId();
130 
131         Long globalCardTypeId = ColabFactory.createPublishedCardType(client, null).getId();
132 
133         Card card = ColabFactory.createNewCard(client, rootCardContentId, globalCardTypeId);
134         Long cardId = card.getId();
135 
136         Long cardContentId = ColabFactory.getCardContent(client, cardId).getId();
137 
138         Long subCardTypeId = ColabFactory.createCardType(client, projectId).getId();
139 
140         Card subCard = ColabFactory.createNewCard(client, cardContentId, subCardTypeId);
141         Long subCardId = subCard.getId();
142 
143         // new team member
144 
145         TestUser guestACL = this.signup(
146             "guestACL",
147             "guestacl@test.local",
148             "9u3stACL43v3r");
149         ColabClient guestHttpClient = this.createRestClient();
150         this.signIn(guestHttpClient, guestACL);
151         WebsocketClient guestWsClient = this.createWsClient();
152         guestHttpClient.websocketRestEndpoint
153             .subscribeToBroadcastChannel(guestWsClient.getSessionId());
154         guestHttpClient.websocketRestEndpoint.subscribeToUserChannel(guestWsClient.getSessionId());
155 
156         // invited to the project
157 
158         Long guestTeamMemberId = ColabFactory.inviteAndJoin(client, project,
159             "guestacl@test.local", guestHttpClient, mailClient).getId();
160 
161         guestHttpClient.websocketRestEndpoint.subscribeToProjectChannel(projectId,
162             guestWsClient.getSessionId());
163 
164         // access setup
165 
166         client.teamRestEndpoint.changeMemberPosition(guestTeamMemberId, hierarchicalPosition);
167 
168         client.teamRestEndpoint.giveRoleTo(designerRoleId, guestTeamMemberId);
169         client.teamRestEndpoint.giveRoleTo(developerRoleId, guestTeamMemberId);
170 
171         if (roleInvolvementLevel != null) {
172 //            client.teamRestEndpoint.setRoleInvolvement(cardId, designerRoleId,
173 //                roleInvolvementLevel);
174             client.teamRestEndpoint.setMemberInvolvement(cardId, guestTeamMemberId,
175                 roleInvolvementLevel);
176         }
177 
178         // access check
179 
180         checkCanRead(guestHttpClient, cardId, subCardId, expectedCanRead);
181         checkCanWrite(guestHttpClient, cardId, subCardId, expectedCanWrite);
182     }
183 
184     /**
185      * Verify if a card and its sub card can be read
186      *
187      * @param guestHttpClient Colab client
188      * @param card            the card
189      * @param subCard         the sub card
190      * @param expectedCanRead the expected result
191      */
192     private static void checkCanRead(ColabClient guestHttpClient, Long cardId, Long subCardId,
193         boolean expectedCanRead) {
194         boolean effectiveCanRead = false;
195         try {
196             checkReadAccess(guestHttpClient, cardId);
197             checkReadAccess(guestHttpClient, subCardId);
198             effectiveCanRead = true;
199 //        } catch (HttpErrorMessage hem) {
200 //            if (MessageCode.ACCESS_DENIED == hem.getMessageCode()) {
201 //                effectiveCanRead = false;
202 //            } else {
203 //                Assertions.fail(hem);
204 //            }
205         } catch (@SuppressWarnings("unused") ServerException se) {
206             effectiveCanRead = false;
207             // TODO see if can check it more accurately
208         } catch (Exception e) {
209             Assertions.fail(e);
210         }
211 
212         Assertions.assertTrue(expectedCanRead == effectiveCanRead, "read permission check failed");
213     }

215     /**
216      * A way to check the permission to read a card
217      *
218      * @param guestHttpClient the colab client of the user
219      * @param card            the card to check
220      */
221     private static void checkReadAccess(ColabClient guestHttpClient, Long cardId) {
222         guestHttpClient.cardRestEndpoint.getContentVariantsOfCard(cardId);
223     }

225     /**

258     /**
259      * A way to check the permission to read and write a card
260      *
261      * @param guestHttpClient the colab client of the user
262      * @param card            the card to check
263      */
264     private static void checkReadWriteAccess(ColabClient guestHttpClient, Long cardId) {
265         Card persistedCard = guestHttpClient.cardRestEndpoint.getCard(cardId);
266         persistedCard.setColor("lemongreen"); // any change on the card
267         guestHttpClient.cardRestEndpoint.updateCard(persistedCard);
268     }
