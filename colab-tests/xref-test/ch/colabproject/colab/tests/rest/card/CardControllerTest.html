7   package ch.colabproject.colab.tests.rest.card;
8   
9   import ch.colabproject.colab.api.model.card.Card;
10  import ch.colabproject.colab.tests.tests.AbstractArquillianTest;
11  import java.util.List;
12  import org.junit.jupiter.api.Assertions;
13  import org.junit.jupiter.api.Test;
14  
15  /**
16   * Unit testing of card controller
17   *
18   * @author sandra
19   */
20  public class CardControllerTest extends AbstractArquillianTest {
21  
22      @Test
23      public void testCreateCard() {
24          Card card = new Card();
25          card.setColor("purple");
26          card.setIndex(6);
27  
28          Long cardId = client.cardController.createCard(card);
29          Card persistedCard = client.cardController.getCard(cardId);
30  
31          Assertions.assertNotNull(persistedCard);
32          Assertions.assertNotNull(persistedCard.getId());
33  
34          Assertions.assertEquals(cardId, persistedCard.getId());
35          Assertions.assertEquals("purple", persistedCard.getColor());
36          Assertions.assertEquals(6, persistedCard.getIndex());
37      }
38  
39      @Test
40      public void testUpdateCard() {
41          Card card = new Card();
42          Long cardId = client.cardController.createCard(card);
43  
44          card = client.cardController.getCard(cardId);
45          card.setColor("blue");
46          card.setIndex(3);
47          client.cardController.updateCard(card);
48  
49          Card card2 = client.cardController.getCard(cardId);
50          Assertions.assertEquals("blue", card2.getColor());
51          Assertions.assertEquals(3, card2.getIndex());
52      }
53  
54      @Test
55      public void testGetAllCards() {
56          Card card = new Card();
57          card.setIndex(1337);
58          card.setColor("red");
59          client.cardController.createCard(card);
60  
61          card = new Card();
62          card.setIndex(42);
63          card.setColor("yellow");
64          client.cardController.createCard(card);
65  
66          List<Card> cards = client.cardController.getAllCards();
67          Assertions.assertEquals(2, cards.size());
68      }
69  
70      @Test
71      public void testDeleteCard() {
72          Card card = new Card();
73          Long cardId = client.cardController.createCard(card);
74  
75          Card persistedCard = client.cardController.getCard(cardId);
76          Assertions.assertNotNull(persistedCard);
77  
78          client.cardController.deleteCard(cardId);
79  
80          persistedCard = client.cardController.getCard(cardId);
81          Assertions.assertNull(persistedCard);
82      }
83  }
