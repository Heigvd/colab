39      public void testCreateProject() {
40          TestUser user = this.signup("goulashsensei", "goulash@test.local", "SoSecure");
41          this.signIn(user);
42  
43          User currentUser = client.userController.getCurrentUser();
44  
45          Project project = new Project();
46  
47          Long projectId = client.projectController.createProject(project);
48          Project persistedProject = client.projectController.getProject(projectId);
49  
50          Assertions.assertNotNull(persistedProject);
51          Assertions.assertNotNull(persistedProject.getId());
52  
53          Assertions.assertEquals(persistedProject.getId(), projectId);
54  
55          Assertions.assertNotNull(persistedProject.getRootCardId());
56          Card rootCard = client.cardController.getCard(persistedProject.getRootCardId());
57          Assertions.assertNotNull(rootCard);
58          List<CardContent> rootCardContents = client.cardController.getContentVariantsOfCard(rootCard.getId());
59          Assertions.assertNotNull(rootCardContents);
60          Assertions.assertEquals(1, rootCardContents.size());
61  
62          List<TeamMember> members = client.projectController.getMembers(projectId);
63          Assertions.assertEquals(1, members.size());
64  
65          TeamMember me = members.get(0);
66          Assertions.assertEquals(currentUser.getId(), me.getUserId());
67          Assertions.assertEquals(projectId, me.getProjectId());
68      }

115     public void testInvite() throws URISyntaxException, DeploymentException, IOException, InterruptedException {
116 //        TestHelper.setLoggerLevel(LoggerFactory.getLogger(WebsocketHelper.class), Level.TRACE);
117 //        TestHelper.setLoggerLevel(LoggerFactory.getLogger(WebsocketClient.class), Level.TRACE);
118 //        TestHelper.setLoggerLevel(LoggerFactory.getLogger(WebsocketFacade.class), Level.TRACE);

121         ////////////////////////////////////////////////////////////////////////////////////////////
122         // this.client sign up as GoulashSensei and subscribes to its channel with wsGoulashClient
123         ////////////////////////////////////////////////////////////////////////////////////////////
124         TestUser user = this.signup("goulashsensei", "goulash@test.local", "SoSecure");
125         String mateAddress = "georges@valgeorges.local";
126         this.signIn(user);
127 
128         WebsocketClient wsGoulashClient = this.createWsClient();
129         client.websocketController.subscribeToUserChannel(wsGoulashClient.getSessionId());
130 
131         ////////////////////////////////////////////////////////////////////////////////////////////
132         // Goulash creates a projects
133         ////////////////////////////////////////////////////////////////////////////////////////////
134         Project p = new Project();
135         p.setName("The Hitchhiker's Guide to the Serious-Game");
136 
137         Long projectId = client.projectController.createProject(p);
138 
139         // Goulash receives the project, the user (ie himself) and the teamMmeber linking the project to the user
140         WsUpdateMessage wsProjectUpdate = TestHelper.waitForMessagesAndAssert(wsGoulashClient, 1, 5, WsUpdateMessage.class).get(0);
141         Assertions.assertEquals(5, wsProjectUpdate.getUpdated().size(), "Got " + wsProjectUpdate.getUpdated());
142 
143         Project project = TestHelper.findFirst(wsProjectUpdate.getUpdated(), Project.class);
144         TeamMember member = TestHelper.findFirst(wsProjectUpdate.getUpdated(), TeamMember.class);
145         User wsUpdatedUser = TestHelper.findFirst(wsProjectUpdate.getUpdated(), User.class);
146         Card wsUpdatedRoot = TestHelper.findFirst(wsProjectUpdate.getUpdated(), Card.class);
147         CardContent wsUpdatedRootContent = TestHelper.findFirst(wsProjectUpdate.getUpdated(), CardContent.class);
148 
149         Assertions.assertEquals(projectId, project.getId());
150         Assertions.assertEquals(projectId, member.getProjectId());
151         Assertions.assertEquals(member.getUserId(), wsUpdatedUser.getId());
152         Assertions.assertNotNull(wsUpdatedRoot.getId());
153         Assertions.assertEquals(projectId, wsUpdatedRoot.getRootCardProjectId());
154         Assertions.assertEquals(wsUpdatedRoot.getId(), wsUpdatedRootContent.getCardId());
155 
156         // user is a lonely team member
157         List<TeamMember> members = client.projectController.getMembers(project.getId());
158         Assertions.assertEquals(1, members.size());
159 
160         ////////////////////////////////////////////////////////////////////////////////////////////
161         // Goulash invites Georges
162         ////////////////////////////////////////////////////////////////////////////////////////////
163         mailClient.deleteAllMessages();
164         client.projectController.inviteSomeone(projectId, mateAddress);
165 
166         // Invitation creates a TeamMember and touch the project;
167         // The two objects are propagated to goulashsensei
168         WsUpdateMessage wsInvitationUpdate = TestHelper.waitForMessagesAndAssert(
169             wsGoulashClient, 1, 5, WsUpdateMessage.class).get(0);
170         Assertions.assertEquals(2, wsInvitationUpdate.getUpdated().size());
171         TestHelper.findFirst(wsInvitationUpdate.getUpdated(), Project.class);
172         TestHelper.findFirst(wsInvitationUpdate.getUpdated(), TeamMember.class);
173 
174         ////////////////////////////////////////////////////////////////////////////////////////////
175         // Fetch e-maild and extract the token
176         ////////////////////////////////////////////////////////////////////////////////////////////
177         List<Message> messages = getMessageByRecipient(mateAddress);
178         Assertions.assertEquals(1, messages.size());
179         Message invitation = messages.get(0);
180 
181         Matcher matcher = TOKEN_EXTRACTOR.matcher(invitation.getContent().getBody());
182         Assertions.assertTrue(matcher.matches());
183 
184         Long tokenId = Long.parseLong(matcher.group(1));
185         String plainToken = matcher.group(2);
186         this.mailClient.deleteMessage(invitation.getId());
187 
188         ////////////////////////////////////////////////////////////////////////////////////////////
189         // With is own http client 'borschClient', Georges fetch the token
190         ////////////////////////////////////////////////////////////////////////////////////////////
191         ColabClient borschClient = this.createRestClient();
192         Token token = borschClient.tokenController.getToken(tokenId);
193 
194         Assertions.assertTrue(token instanceof InvitationToken);
195 
196         Assertions.assertTrue(token.isAuthenticationRequired());
197 
198         // an try to consume the token being unauthenticated
199         TestHelper.assertThrows(HttpErrorMessage.MessageCode.AUTHENTICATION_REQUIRED, () -> {
200             // consuming the token without being authenticated is not possible
201             borschClient.tokenController.consumeToken(tokenId, plainToken);
202         });
203 
204         ////////////////////////////////////////////////////////////////////////////////////////////
205         // Georges sign up as borschsensei and subscribes to its channel with wsBorschClient
206         ////////////////////////////////////////////////////////////////////////////////////////////
207         TestUser mateUser = this.signup("borschsensei", mateAddress, "SoSoSoSecure");
208         signIn(borschClient, mateUser);
209         WebsocketClient wsBorschClient = this.createWsClient();
210         borschClient.websocketController.subscribeToUserChannel(wsBorschClient.getSessionId());
211 
212         ////////////////////////////////////////////////////////////////////////////////////////////
213         // Borsch consumes the tokens amd, thus, join the team.
214         // Borsch and Goulash both receive the temmMember and borsch's user update.
215         ////////////////////////////////////////////////////////////////////////////////////////////
216         borschClient.tokenController.consumeToken(tokenId, plainToken);
217 
218         WsUpdateMessage goulashTeamUpdate = TestHelper.waitForMessagesAndAssert(
219             wsGoulashClient, 1, 5, WsUpdateMessage.class).get(0);
220         WsUpdateMessage borschTeamUpdate = TestHelper.waitForMessagesAndAssert(wsBorschClient, 1, 5, WsUpdateMessage.class).get(0);
221 
222         Assertions.assertEquals(2, goulashTeamUpdate.getUpdated().size());
223         Assertions.assertEquals(2, borschTeamUpdate.getUpdated().size());
224 
225         TeamMember gTeamMemeber = TestHelper.findFirst(goulashTeamUpdate.getUpdated(), TeamMember.class);
226         TeamMember bTeamMember = TestHelper.findFirst(borschTeamUpdate.getUpdated(), TeamMember.class);
227 
228         ////////////////////////////////////////////////////////////////////////////////////////////
229         // They both reload the team from REST method
230         ////////////////////////////////////////////////////////////////////////////////////////////
231         reloadTwoMembersTeam(client, projectId);
232         reloadTwoMembersTeam(borschClient, projectId);
233     }

235     private void reloadTwoMembersTeam(ColabClient client, Long projectId) {
236         List<TeamMember> members = client.projectController.getMembers(projectId);
237         Assertions.assertEquals(2, members.size());
238 
239         //assert currentUser can read user of teammate
240         members.forEach((TeamMember member) -> {
241             User u = client.userController.getUserById(member.getUserId());
242             Assertions.assertNotNull(u);
243         });
244 
245         // assert currentuser can not read other users
246         TestHelper.assertThrows(HttpErrorMessage.MessageCode.ACCESS_DENIED, () -> {
247             client.userController.getUserById(this.adminUserId);
248         });
249     }
