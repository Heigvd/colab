30      @Test
31      public void testCreateProject() {
32          TestUser user = this.signup("goulashsensei", "goulash@test.local", "SoSecure");

34  
35          User currentUser = client.userController.getCurrentUser();

37          Project project = new Project();
38  
39          Long projectId = client.projectController.createProject(project);
40          Project persistedProject = client.projectController.getProject(projectId);

42          Assertions.assertNotNull(persistedProject);
43          Assertions.assertNotNull(persistedProject.getId());
44  
45          Assertions.assertEquals(persistedProject.getId(), projectId);

47          Assertions.assertNotNull(persistedProject.getRootCard());
48          Assertions.assertNotNull(persistedProject.getRootCard().getId());
49  
50          List<TeamMember> members = client.projectController.getMembers(projectId);
51          Assertions.assertEquals(1, members.size());
52  
53          TeamMember me = members.get(0);
54          Assertions.assertEquals(currentUser.getId(), me.getUserId());
55          Assertions.assertEquals(projectId, me.getProjectId());
56      }

102     @Test
103     public void testInvite() {
104         TestUser user = this.signup("goulashsensei", "goulash@test.local", "SoSecure");
105         String mateAddress = "georges@valgeorges.local";
106 
107         this.signIn(user);
108 
109         Project p = new Project();
110         p.setName("The Hitchhiker's Guide to the Serious-Game");
111 
112         Long projectId = client.projectController.createProject(p);
113         Project project = client.projectController.getProject(projectId);
114 
115         // user is a lonely team member
116         List<TeamMember> members = client.projectController.getMembers(project.getId());
117         Assertions.assertEquals(1, members.size());
118 
119         mailClient.deleteAllMessages();
120         client.projectController.inviteSomeone(projectId, mateAddress);
121 
122         List<Message> messages = getMessageByRecipient(mateAddress);
123         Assertions.assertEquals(1, messages.size());
124         Message invitation = messages.get(0);
125 
126         Matcher matcher = TOKEN_EXTRACTOR.matcher(invitation.getContent().getBody());
127         Assertions.assertTrue(matcher.matches());
128 
129         Long tokenId = Long.parseLong(matcher.group(1));
130         String plainToken = matcher.group(2);
131 
132         this.signOut();
133 
134         Token token = client.tokenController.getToken(tokenId);
135 
136         Assertions.assertTrue(token instanceof InvitationToken);
137 
138         Assertions.assertTrue(token.isAuthenticationRequired());
139 
140         TestHelper.assertThrows(HttpErrorMessage.MessageCode.AUTHENTICATION_REQUIRED, () -> {
141             // consuming the token without being authenticated is not possible
142             client.tokenController.consumeToken(tokenId, plainToken);
143         });
144 
145         TestUser mateUser = this.signup("borschsensei", mateAddress, "SoSoSoSecure");
146         this.signIn(mateUser);
147         client.tokenController.consumeToken(tokenId, plainToken);
148         this.mailClient.deleteMessage(invitation.getId());
149 
150         // not lonely anylonger
151         members = client.projectController.getMembers(project.getId());
152         Assertions.assertEquals(2, members.size());
153     }
