40      @Test
41      public void testCreateResource() {
42          // creation of the context : global card type
43          Long globalCardTypeId = ColabFactory.createCardType(client, null).getId();
44  
45          String title = "the guide " + ((int) (Math.random() * 1000));
46          String teaser = "everything you need to know " + ((int) (Math.random() * 1000));
47          String category = "awesome resources #" + ((int) (Math.random() * 1000));
48          String url = "http://www.123soleil.chat/theGameEncyclopedia.pdf";
49  
50          ExternalLink doc = new ExternalLink();
51          doc.setUrl(url);
52  
53          TextDataBlock teaserBlock = TextDataBlock.initNewDefaultTextDataBlock();
54          teaserBlock.setTextData(teaser);
55  
56          ResourceCreationBean resourceCreationBean = new ResourceCreationBean();
57  
58          resourceCreationBean.setTitle(title);
59          resourceCreationBean.setTeaser(teaserBlock);
60          resourceCreationBean.setDocument(doc);
61          resourceCreationBean.setCategory(category);
62          resourceCreationBean.setAbstractCardTypeId(globalCardTypeId);
63  
64          Long resourceId = client.resourceRestEndpoint.createResource(resourceCreationBean);
65  
66          Resource persistedResource = (Resource) client.resourceRestEndpoint
67              .getAbstractResource(resourceId);
68          Assertions.assertNotNull(persistedResource);
69          Assertions.assertNotNull(persistedResource.getId());
70          Assertions.assertEquals(resourceId, persistedResource.getId());
71          Assertions.assertEquals(title, persistedResource.getTitle());
72          Assertions.assertFalse(persistedResource.isPublished());
73          Assertions.assertFalse(persistedResource.isRequestingForGlory());
74          Assertions.assertFalse(persistedResource.isDeprecated());
75  
76          Assertions.assertNotNull(persistedResource.getDocumentId());
77          Document persistedDocument = client.documentRestEndPoint
78              .getDocument(persistedResource.getDocumentId());
79          Assertions.assertNotNull(persistedDocument);
80          Assertions.assertEquals(persistedResource.getDocumentId(), persistedDocument.getId());
81          Assertions.assertTrue(persistedDocument instanceof ExternalLink);
82          ExternalLink persistedExtDoc = (ExternalLink) persistedDocument;
83          Assertions.assertEquals(url, persistedExtDoc.getUrl());
84          Assertions.assertNull(persistedExtDoc.getDeliverableCardContentId());
85          Assertions.assertEquals(resourceId, persistedExtDoc.getResourceId());
86  
87          Assertions.assertNotNull(persistedResource.getTeaserId());
88          Block persistedTeaserBlock = client.blockRestEndPoint
89              .getBlock(persistedResource.getTeaserId());
90          Assertions.assertNotNull(persistedTeaserBlock);
91          Assertions.assertEquals(persistedResource.getTeaserId(), persistedTeaserBlock.getId());
92          Assertions.assertTrue(persistedTeaserBlock instanceof TextDataBlock);
93          TextDataBlock persistedTeaserTextDataBlock = (TextDataBlock) persistedTeaserBlock;
94          Assertions.assertNull(persistedTeaserTextDataBlock.getDocumentId());
95          Assertions.assertEquals(persistedTeaserTextDataBlock.getMimeType(),
96              TextDataBlock.DEFAULT_MIME_TYPE);
97          Assertions.assertEquals(persistedTeaserTextDataBlock.getTextData(), teaser);
98      }
