24      @Test
25      public void testCreateTextDataBlock() {
26          int index = 3;
27          String mimeType = "text/plain";
28          String textData = "a valuable plain text";
29          String revision = "0.23.5";
30  
31          TextDataBlock doc = new TextDataBlock();
32          doc.setIndex(index);
33          doc.setMimeType(mimeType);
34          doc.setTextData(textData);
35          doc.setRevision(revision);
36  
37          Long docId = ColabFactory.createADocument(client, doc).getId();
38  
39          Document persistedDoc = client.documentRestEndpoint.getDocument(docId);
40          Assertions.assertNotNull(persistedDoc);
41          Assertions.assertEquals(docId, persistedDoc.getId());
42          Assertions.assertEquals(index, persistedDoc.getIndex());
43  
44          Assertions.assertTrue(persistedDoc instanceof TextDataBlock);
45          TextDataBlock persistedTextDataBlock = (TextDataBlock) persistedDoc;
46          Assertions.assertEquals(mimeType, persistedTextDataBlock.getMimeType());
47          Assertions.assertEquals(textData, persistedTextDataBlock.getTextData());
48          Assertions.assertEquals(revision, persistedTextDataBlock.getRevision());
49      }

72      @Test
73      public void testUpdateTextDataBlock() {
74          Long docId = ColabFactory.createADocument(client, new TextDataBlock()).getId();
75  
76          Document doc = client.documentRestEndpoint.getDocument(docId);
77          Assertions.assertNotNull(doc);
78          Assertions.assertTrue(doc instanceof TextDataBlock);
79          TextDataBlock textDataBlock = (TextDataBlock) doc;
80          Assertions.assertEquals(docId, textDataBlock.getId());
81          Assertions.assertEquals(1000, textDataBlock.getIndex());
82  
83          int index = 4;
84          String mimeType = "text/plain";
85          String textData = "a plain text";
86  
87          textDataBlock.setIndex(index);
88          textDataBlock.setMimeType(mimeType);
89          textDataBlock.setTextData(textData);
90          client.documentRestEndpoint.updateDocument(textDataBlock);
91  
92          Document persistedDoc = client.documentRestEndpoint.getDocument(docId);
93          Assertions.assertNotNull(persistedDoc);
94          Assertions.assertEquals(docId, persistedDoc.getId());
95          Assertions.assertEquals(index, persistedDoc.getIndex());
96          Assertions.assertTrue(persistedDoc instanceof TextDataBlock);
97          TextDataBlock persistedTextDataBlock = (TextDataBlock) persistedDoc;
98          Assertions.assertEquals(mimeType, persistedTextDataBlock.getMimeType());
99          Assertions.assertEquals(textData, persistedTextDataBlock.getTextData());
100     }

102     @Test
103     public void testUpdateExternalLink() {
104         Long docId = ColabFactory.createADocument(client, new ExternalLink()).getId();
105 
106         Document doc = client.documentRestEndpoint.getDocument(docId);
107         Assertions.assertNotNull(doc);
108         Assertions.assertTrue(doc instanceof ExternalLink);
109         ExternalLink extDocLink = (ExternalLink) doc;
110         Assertions.assertEquals(docId, extDocLink.getId());
111         Assertions.assertNull(extDocLink.getUrl());
112         Assertions.assertEquals(1000, extDocLink.getIndex());
113 
114         int index = 8;
115         String url = "https://www.colab-project.ch/sites/default/files/2021-03/WP1%20-%20Project%20Description%20Model_0.pdf";
116 
117         extDocLink.setIndex(index);
118         extDocLink.setUrl(url);
119         client.documentRestEndpoint.updateDocument(extDocLink);
120 
121         Document persistedDoc = client.documentRestEndpoint.getDocument(docId);
122         Assertions.assertTrue(persistedDoc instanceof ExternalLink);
123         ExternalLink persistedExtDocLink = (ExternalLink) persistedDoc;
124         Assertions.assertNotNull(persistedExtDocLink);
125         Assertions.assertEquals(docId, persistedExtDocLink.getId());
126         Assertions.assertEquals(url, persistedExtDocLink.getUrl());
127         Assertions.assertEquals(index, persistedExtDocLink.getIndex());
128     }
