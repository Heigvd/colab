23      /**
24       * Expected
25       *  |1|2|3|6|
26       *  |4|
27       */
28      @Test
29      public void testThreeBytwo() {
30          List<GridCellWithId> cells =new ArrayList<>();
31  
32          // stack five cells on (1,1)
33          cells.add(new GridCellTest(1l, 1, 1, 1, 1));
34          cells.add(new GridCellTest(2l, 1, 1, 1, 1));
35          cells.add(new GridCellTest(3l, 1, 1, 1, 1));
36          cells.add(new GridCellTest(4l, 1, 1, 1, 1));
37          cells.add(new GridCellTest(5l, 1, 1, 1, 1));
38  
39          Grid grid = Grid.resolveConflicts(cells);
40  
41  
42          // 1 row of three, 2d row of 1
43          Assertions.assertEquals(grid.getCellAt(1, 1).getId(), 1);
44          Assertions.assertEquals(1, cells.get(0).getX(), "1st cell.x not match");
45          Assertions.assertEquals(1, cells.get(0).getY(), "1st cell.y not match");
46  
47          Assertions.assertEquals(grid.getCellAt(2, 1).getId(), 2);
48          Assertions.assertEquals(2, cells.get(1).getX(), "2nd cell.x not match");
49          Assertions.assertEquals(1, cells.get(1).getY(), "2nd cell.y not match");
50  
51          Assertions.assertEquals(grid.getCellAt(3, 1).getId(), 3);
52          Assertions.assertEquals(3, cells.get(2).getX(), "3rd cell.x not match");
53          Assertions.assertEquals(1, cells.get(2).getY(), "3rd cell.y not match");
54  
55          Assertions.assertEquals(grid.getCellAt(1, 2).getId(), 4);
56          Assertions.assertEquals(1, cells.get(3).getX(), "4th cell.x not match");
57          Assertions.assertEquals(2, cells.get(3).getY(), "4th cell.y not match");
58  
59          Assertions.assertEquals(grid.getCellAt(2, 2).getId(), 5);
60          Assertions.assertEquals(2, cells.get(4).getX(), "5th cell.x not match");
61          Assertions.assertEquals(2, cells.get(4).getY(), "5th cell.y not match");
62      }
