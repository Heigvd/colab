36      /**
37       * Create a brand new card type in the given project.
38       *
39       * @param client    rest client to execute HTTP requests
40       * @param projectId id of the project the card type will belongs to. If this id is null the type
41       *                  will be a global type
42       *
43       * @return the CardType
44       */
45      public static CardType createCardType(ColabClient client, Long projectId) {
46          CardType cardType = new CardType();
47          cardType.setProjectId(projectId);
48  
49          Long cardTypeId = client.cardTypeRestEndpoint.createCardType(cardType);
50          return (CardType) client.cardTypeRestEndpoint.getCardType(cardTypeId);
51      }
52  
53      /**
54       * Create a brand new card in the given parent with the given type
55       *
56       * @param client     rest client to execute HTTP requests
57       * @param parentId   id of the card content the card will belong to
58       * @param cardTypeId id of the card type
59       *
60       * @return the newly created card
61       */
62      public static Card createNewCard(ColabClient client, Long parentId, Long cardTypeId) {
63          return client.cardRestEndpoint.createNewCard(parentId, cardTypeId);
64      }
65  
66      /**
67       * Create a brand new card in the given project just under the root card and with a new card
68       * type
69       *
70       * @param client  rest client to execute HTTP requests
71       * @param project project in which the card is
72       *
73       * @return the newly created card
74       */
75      public static Card createNewCard(ColabClient client, Project project) {
76          CardType cardType = ColabFactory.createCardType(client, project.getId());
77          Long cardTypeId = cardType.getId();
78  
79          Long parentId = ColabFactory.getRootContent(client, project).getId();
80  
81          return ColabFactory.createNewCard(client, parentId, cardTypeId);
82      }

84      /**
85       * Create a brand new project.

87       * @param client rest client to execute HTTP requests

89       *

91       */
92      public static Project createProject(ColabClient client, String name) {
93          Project p = new Project();
94          p.setName(name);

96          return client.projectRestEndpoint.getProject(id);

111     /**

123     /**
124      * <code>host</code> invites <code>guest</code> in the project.
125      *
126      * @param host         authenticated rest client with write-access on project
127      * @param project      the project
128      * @param emailAddress email address to send the message to
129      * @param guest        authenticated rest client, not yet member of the project
130      * @param mailClient   mailhost client to fetch the invitation
131      */
132     public static void inviteAndJoin(ColabClient host, Project project,
133         String emailAddress, ColabClient guest, MailhogClient mailClient
134     ) {
135         host.teamRestEndpoint.inviteSomeone(project.getId(), emailAddress);
136         Message invitation = TestHelper.getMessageByRecipient(mailClient, emailAddress).get(0);
137 
138         Matcher matcher = TestHelper.extractToken(invitation);
139 
140         if (matcher.matches()) {
141             Long tokenId = Long.parseLong(matcher.group(1));
142             String plainToken = matcher.group(2);
143 
144             mailClient.deleteMessage(invitation.getId());
145 
146             Token token = guest.tokenRestEndpoint.getToken(tokenId);
147             Assertions.assertTrue(token instanceof InvitationToken);
148             guest.tokenRestEndpoint.consumeToken(tokenId, plainToken);
149         } else {
150             Assertions.fail("Failed to parse token");
151         }
152     }

154     public static Role createRole(ColabClient client, Project project, String name) {
155         Role role = new Role();
156         role.setProjectId(project.getId());
157         role.setName(name);
158 
159         Long roleId = client.teamRestEndpoint.createRole(role);
160 
161         return client.teamRestEndpoint.getRole(roleId);
162     }

166      *
167      * @param client rest client to execute HTTP requests
168      * @param cardId the id of the card the resource belongs to
169      * @param title  title of the document
170      *
171      * @return the freshly created document
172      */
