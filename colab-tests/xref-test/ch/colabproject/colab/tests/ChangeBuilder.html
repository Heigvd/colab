7   package ch.colabproject.colab.tests;
8   
9   import ch.colabproject.colab.api.microchanges.model.Change;
10  import ch.colabproject.colab.api.microchanges.model.MicroChange;
11  import java.util.Set;
12  
13  /**
14   *
15   * @author maxence
16   */
17  public class ChangeBuilder {
18  
19      private Change change;
20  
21      public static ChangeBuilder create(String liveSession, String newRevision, String... basedOn) {
22          ChangeBuilder cb = new ChangeBuilder();
23          cb.change = new Change();
24          cb.change.setBasedOn(Set.of((String[]) basedOn));
25          cb.change.setLiveSession(liveSession);
26          cb.change.setRevision(liveSession + "::" + newRevision);
27  
28          return cb;
29      }
30  
31      public ChangeBuilder ins(int offset, String data) {
32          MicroChange mu = new MicroChange();
33          mu.setT(MicroChange.Type.I);
34          mu.setO(offset);
35          mu.setV(data);
36          this.change.getMicrochanges().add(mu);
37  
38          return this;
39      }
40  
41      public ChangeBuilder del(int offset, int length) {
42          MicroChange mu = new MicroChange();
43          mu.setT(MicroChange.Type.D);
44          mu.setO(offset);
45          mu.setL(length);
46          this.change.getMicrochanges().add(mu);
47  
48          return this;
49      }
50  
51      public Change build() {
52          return this.change;
53      }
54  }
