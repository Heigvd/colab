7   package ch.colabproject.colab.tests.mock;
8   
9   import ch.colabproject.colab.api.setup.LiquibaseProducer;
10  import javax.annotation.Resource;
11  import javax.enterprise.inject.Produces;
12  import javax.enterprise.inject.Specializes;
13  import javax.sql.DataSource;
14  import liquibase.integration.cdi.CDILiquibaseConfig;
15  import liquibase.integration.cdi.annotations.LiquibaseType;
16  import liquibase.resource.ClassLoaderResourceAccessor;
17  import liquibase.resource.ResourceAccessor;
18  
19  /**
20   * Override default Producer as we do not want to use liquibase in test env
21   *
22   * @author maxence
23   */
24  public class LiquibaseTestProducer extends LiquibaseProducer {
25  
26      /**
27       * SQL Datasource
28       */
29      @Resource(lookup = "java:global/colabDS")
30      private DataSource myDataSource;
31  
32      /**
33       * Liquibase configuration which indicates liquibase should not run
34       *
35       * @return the config
36       */
37      @Produces
38      @LiquibaseType
39      @Specializes
40      public CDILiquibaseConfig createConfig() {
41          CDILiquibaseConfig config = new CDILiquibaseConfig();
42          config.setShouldRun(false);
43          return config;
44      }
45  
46      /**
47       * Give liquibase access to the datasource
48       *
49       * @return the datasource
50       *
51       */
52      @Produces
53      @LiquibaseType
54      @Specializes
55      public DataSource createDataSource() {
56          return myDataSource;
57      }
58  
59      /**
60       * Give liquibase access to our classloader
61       *
62       * @return the classloader
63       */
64      @Produces
65      @LiquibaseType
66      @Specializes
67      public ResourceAccessor create() {
68          return new ClassLoaderResourceAccessor(getClass().getClassLoader());
69      }
70  
71  }
