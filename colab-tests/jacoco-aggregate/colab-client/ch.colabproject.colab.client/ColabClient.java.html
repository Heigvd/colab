        /**
         * GET api/tokens/{id} calls {@link ch.colabproject.colab.api.rest.TokenRestEndpoint#getToken}
         *
         * Fetch a token by id. Once fetched, the client should decide what to do. If the token does not
         * require authentication, client may consume the token directly. If the token require
         * authentication, the client should ask the user to sing in/up and consume the token
         * eventually.
         *
         * @param id  of the token to fetch.
         *
         * @return the token
         *
         */
        public Token getToken(Long id) {
            UriTemplate pathTemplate = new PathPattern("api/tokens/{id}").getTemplate();
            String path = pathTemplate.createURI(id.toString());
            return ColabClient.this.get(path, new GenericType<>(Token.class));
        }
        
        /**
         * PUT api/tokens/{id}/{token: [a-fA-F0-9]+} calls {@link ch.colabproject.colab.api.rest.TokenRestEndpoint#consumeToken}
         *
         * Consume and destroy the token
         *
         * @param id          if of the token to consume
         * @param plainToken  plain token as receive by e-mail
         *
         * @return the consumed token
         *
         * .colabproject.colab.generator.model.exceptions.HttpErrorMessage <ul>
         * <li>notFound if the token does not exists;
         * <li>bad request if token does not match;
         * <li>authenticationRequired if token requires authentication but current user id not
         * </ul>
         *
         */
        public Token consumeToken(Long id, String plainToken) {
            UriTemplate pathTemplate = new PathPattern("api/tokens/{id}/{token: [a-fA-F0-9]+}").getTemplate();
            String path = pathTemplate.createURI(id.toString(),plainToken.toString());
            return ColabClient.this.put(path, new GenericType<>(Token.class));
        }
    }
