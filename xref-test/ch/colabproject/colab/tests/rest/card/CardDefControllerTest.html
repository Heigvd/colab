25      @Test
26      public void testCreateCardDef() {
27          String uniqueId = String.valueOf(new Date().getTime() + ((long) (Math.random() * 1000)));
28          String title = " " + ((int) (Math.random() * 1000));
29          String purpose = " " + ((int) (Math.random() * 1000));
30          ConcretizationCategory authorityHolder = ConcretizationCategory.PROJECT;
31  
32          CardDef cardDef = new CardDef();
33          cardDef.setUniqueId(uniqueId);
34          cardDef.setTitle(title);
35          cardDef.setPurpose(purpose);
36          cardDef.setAuthorityHolder(authorityHolder);
37          Long cardDefId = client.cardDefController.createCardDef(cardDef);
38  
39          CardDef persistedCardDef = client.cardDefController.getCardDef(cardDefId);
40  
41          Assertions.assertNotNull(persistedCardDef);
42          Assertions.assertNotNull(persistedCardDef.getId());
43  
44          Assertions.assertEquals(persistedCardDef.getId(), cardDefId);
45          Assertions.assertEquals(persistedCardDef.getUniqueId(), uniqueId);
46          Assertions.assertEquals(persistedCardDef.getTitle(), title);
47          Assertions.assertEquals(persistedCardDef.getPurpose(), purpose);
48          Assertions.assertEquals(persistedCardDef.getAuthorityHolder(), authorityHolder);
49      }

117         String cardDefTitle = "design " + ((int) (Math.random() * 1000));
118 
119         Project project = new Project();

121         Long projectId = client.projectController.createProject(project);
122 
123         Project persistedProject = client.projectController.getProject(projectId);
124 
125         CardDef cardDef = new CardDef();
126         cardDef.setTitle(cardDefTitle);
127         cardDef.setProject(persistedProject);
128         Long cardDefId = client.cardDefController.createCardDef(cardDef);
