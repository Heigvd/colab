28      @Test
29      public void testCreateCard() {
30          String color = "purple " + ((int) (Math.random() * 1000));
31          int index = (int) (Math.random() * 100);
32  
33          Card card = new Card();
34          card.setColor(color);
35          card.setIndex(index);
36          Long cardId = client.cardController.createCard(card);
37  
38          Card persistedCard = client.cardController.getCard(cardId);
39  
40          Assertions.assertNotNull(persistedCard);
41          Assertions.assertNotNull(persistedCard.getId());
42  
43          Assertions.assertEquals(cardId, persistedCard.getId());
44          Assertions.assertEquals(color, persistedCard.getColor());
45          Assertions.assertEquals(index, persistedCard.getIndex());
46      }

48      @Test
49      public void testUpdateCard() {
50          String color = "blue " + ((int) (Math.random() * 1000));
51          int index = (int) (Math.random() * 100);
52  
53          Card card = new Card();
54          Long cardId = client.cardController.createCard(card);
55  
56          Card persistedCard1 = client.cardController.getCard(cardId);
57          Assertions.assertNull(persistedCard1.getColor());
58          Assertions.assertEquals(0, persistedCard1.getIndex());
59  
60          persistedCard1.setColor(color);
61          persistedCard1.setIndex(index);
62          client.cardController.updateCard(persistedCard1);
63  
64          Card persistedCard2 = client.cardController.getCard(cardId);
65          Assertions.assertEquals(color, persistedCard2.getColor());
66          Assertions.assertEquals(index, persistedCard2.getIndex());
67      }
