43      @Test
44      public void testCreateCardType() {
45          Long projectId = ColabFactory.createProject(client, "testCreateCardType").getId();
46  
47          String title = "Context " + ((int) (Math.random() * 1000));
48          String purpose = "Define the game life context " + ((int) (Math.random() * 1000));
49          String aTag = "life hack";
50          String otherTag = "Contextualization";
51          Set<String> tags = Set.of(aTag, otherTag);
52  
53          TextDataBlock purposeBlock = new TextDataBlock();
54          purposeBlock.setMimeType(DEFAULT_MIME_TYPE);
55          purposeBlock.setTextData(purpose);
56  
57          CardTypeCreationBean cardTypeToCreate = new CardTypeCreationBean();
58          cardTypeToCreate.setProjectId(projectId);
59          cardTypeToCreate.setTitle(title);
60          cardTypeToCreate.setPurpose(purposeBlock);
61          cardTypeToCreate.setTags(tags);
62  
63          Long cardTypeId = client.cardTypeRestEndpoint.createCardType(cardTypeToCreate);
64  
65          CardType cardType = (CardType) client.cardTypeRestEndpoint.getCardType(cardTypeId);
66  
67          Assertions.assertNotNull(cardType);
68          Assertions.assertNotNull(cardType.getId());
69          Assertions.assertEquals(projectId, cardType.getProjectId());
70          Assertions.assertEquals(title, cardType.getTitle());
71          Assertions.assertNotNull(cardType.getPurposeId());
72          Assertions.assertNotNull(cardType.getTags());
73  
74          TextDataBlock persistedPurpose = (TextDataBlock) client.documentRestEndpoint
75              .getDocument(cardType.getPurposeId());
76          Assertions.assertNotNull(persistedPurpose);
77          Assertions.assertEquals(DEFAULT_MIME_TYPE, persistedPurpose.getMimeType());
78          Assertions.assertEquals(purpose, persistedPurpose.getTextData());
79          Assertions.assertNull(persistedPurpose.getOwningCardContentId());
80          Assertions.assertNull(persistedPurpose.getOwningResourceId());
81  
82          Assertions.assertEquals(cardType.getTags().size(), tags.size());
83          Assertions.assertTrue(cardType.getTags().contains(aTag));
84          Assertions.assertTrue(cardType.getTags().contains(otherTag));
85      }

87      @Test
88      public void testCreateAndUseGlobalCardType()
89          throws DeploymentException, IOException, URISyntaxException, InterruptedException {
90          // create some goulash user with its own clients
91          TestUser goulash = this.signup(
92              "goulash",
93              "goulash@test.local",
94              "MyPassword");
95          ColabClient goulashClient = this.createRestClient();
96          this.signIn(goulashClient, goulash);
97          WebsocketClient wsGoulashClient = this.createWsClient();
98          goulashClient.websocketRestEndpoint
99              .subscribeToBroadcastChannel(wsGoulashClient.getSessionId());
100         goulashClient.websocketRestEndpoint.subscribeToUserChannel(wsGoulashClient.getSessionId());
101 
102         // Sign in as admin and initialize websocket connection
103         this.signIn(admin);
104         WebsocketClient adminWsClient = this.createWsClient();
105         client.websocketRestEndpoint.subscribeToBroadcastChannel(adminWsClient.getSessionId());
106         client.websocketRestEndpoint.subscribeToUserChannel(adminWsClient.getSessionId());
107 
108         // Wait for ChannelUpdateMessage
109         TestHelper.waitForMessagesAndAssert(adminWsClient, 1, 5, WsChannelUpdate.class);
110 
111         // -----
112         // Global type
113         // -----
114         adminWsClient.clearMessages();
115         wsGoulashClient.clearMessages();
116         CardType globalType = ColabFactory.createGlobalCardType(client);
117 
118         // admin should have receive the type by websocket
119         WsUpdateMessage adminWsType = TestHelper.waitForMessagesAndAssert(
120             adminWsClient, 1, 5, WsUpdateMessage.class).get(0);
121         Assertions.assertEquals(2, adminWsType.getUpdated().size());
122         CardType aWsCardType = TestHelper.findFirst(adminWsType.getUpdated(), CardType.class);
123 
124         // Publish the projectOneType
125         globalType.setPublished(true);
126         client.cardTypeRestEndpoint.updateCardType(globalType);
127 
128         // Assert the published type is sent to all users
129         WsUpdateMessage goulashGlobalTypeUpdate = TestHelper.waitForMessagesAndAssert(
130             wsGoulashClient, 1, 5, WsUpdateMessage.class).get(0);
131         Assertions.assertEquals(1, goulashGlobalTypeUpdate.getUpdated().size());
132         aWsCardType = TestHelper.findFirst(goulashGlobalTypeUpdate.getUpdated(), CardType.class);
133         Assertions.assertEquals(globalType, aWsCardType);
134         Assertions.assertTrue(aWsCardType.isPublished());
135 
136         // -----
137         // Global type is used by projectOne
138         // -----
139         Project projectOne = ColabFactory.createProject(client, "Project One");
140         Set<AbstractCardType> types = client.projectRestEndpoint
141             .getCardTypesOfProject(projectOne.getId());
142         Assertions.assertEquals(0, types.size());
143 
144         Long parentId = ColabFactory.getRootContent(client, projectOne).getId();
145 
146         // create a card based on a global type
147         ColabFactory.createNewCard(client, parentId, globalType.getId());
148 
149         // assert the project now contains a CardTypeRef to the global type
150         types = client.projectRestEndpoint.getCardTypesOfProject(projectOne.getId());
151         // One concrete type
152         List<CardType> concrete = TestHelper.filterAndAssert(types, 1, CardType.class);
153 
154         // One type reference
155         List<CardTypeRef> refs = TestHelper.filterAndAssert(types, 1, CardTypeRef.class);
156 
157         CardType theType = concrete.get(0);
158         CardTypeRef projectOneRef = refs.get(0);
159 
160         Assertions.assertEquals(projectOne.getId(), projectOneRef.getProjectId());
161         Assertions.assertEquals(globalType.getId(), projectOneRef.getTargetId());
162         Assertions.assertEquals(globalType, theType);
163 
164         // -----
165         // ProjectOne type is used by projectTwo
166         // -----
167         Project projectTwo = ColabFactory.createProject(client, "Project Two");
168         parentId = ColabFactory.getRootContent(client, projectTwo).getId();
169 
170         // create a card based on projectOne type
171         ColabFactory.createNewCard(client, parentId, projectOneRef.getId());
172 
173         // assert the project now contains a CardTypeRef to the project one type
174         types = client.projectRestEndpoint.getCardTypesOfProject(projectTwo.getId());
175         // One concrete type
176         concrete = TestHelper.filterAndAssert(types, 1, CardType.class);
177 
178         // One type reference
179         refs = TestHelper.filterAndAssert(types, 2, CardTypeRef.class);
180 
181         theType = concrete.get(0);
182 
183         Optional<CardTypeRef> optRef = refs.stream()
184             .filter(ref -> projectTwo.getId().equals(ref.getProjectId())).findFirst();
185         Assertions.assertTrue(optRef.isPresent());
186         CardTypeRef projectTwoRef = optRef.get();
187 
188         Assertions.assertEquals(projectTwo.getId(), projectTwoRef.getProjectId());
189         Assertions.assertEquals(projectOneRef.getId(), projectTwoRef.getTargetId());
190         Assertions.assertEquals(globalType, theType);
191     }

195         throws DeploymentException, IOException, URISyntaxException, InterruptedException {
196 //        TestHelper.setLoggerLevel(LoggerFactory.getLogger(WebsocketHelper.class), Level.TRACE);
197 //        TestHelper.setLoggerLevel(LoggerFactory.getLogger(WebsocketClient.class), Level.TRACE);

207         this.signIn(goulash);
208         WebsocketClient wsClient = this.createWsClient();
209         client.websocketRestEndpoint.subscribeToBroadcastChannel(wsClient.getSessionId());
210         client.websocketRestEndpoint.subscribeToUserChannel(wsClient.getSessionId());
211 
212         TestUser pizzaiolo = this.signup(
213             "pizzaiolo",
214             "pizza@test.local",
215             "qu3stap1zzaèlamI9lior3d3l0OndO");
216         // Sign in as goulash and initialize websocket connection
217         ColabClient pizzaHttpClient = this.createRestClient();
218         this.signIn(pizzaHttpClient, pizzaiolo);
219         WebsocketClient pizzaWsClient = this.createWsClient();
220         pizzaHttpClient.websocketRestEndpoint
221             .subscribeToBroadcastChannel(pizzaWsClient.getSessionId());
222         pizzaHttpClient.websocketRestEndpoint.subscribeToUserChannel(pizzaWsClient.getSessionId());
223 
224         // -----
225         // Goulash creates a project
226         // -----
227         Project projectOne = ColabFactory.createProject(client, "Project One");
228         Assertions.assertEquals(0,
229             client.projectRestEndpoint.getCardTypesOfProject(projectOne.getId()).size());
230         client.websocketRestEndpoint.subscribeToProjectChannel(projectOne.getId(),
231             wsClient.getSessionId());
232 
233         // wait for propagation
234         TestHelper.waitForMessagesAndAssert(wsClient, 1, 5, WsUpdateMessage.class).get(0);
235 
236         // -----
237         // Create type in project one
238         // -----
239         CardType projectOneType = ColabFactory.createCardType(client, projectOne.getId());
240 
241         // user should have receive two message. One contains the project and has been sent through
242         // the user own channel. Second message contains the type and has been sent through the
243         // project channel
244         List<WsUpdateMessage> wsUpdate = TestHelper.waitForMessagesAndAssert(wsClient, 2, 5,
245             WsUpdateMessage.class);
246         // combine messages
247         Set<WithWebsocketChannels> updated = new HashSet<>();
248         updated.addAll(wsUpdate.get(0).getUpdated());
249         updated.addAll(wsUpdate.get(1).getUpdated());
250 
251         // Assertions.assertEquals(1, wsUpdate.getUpdated().size());
252         CardType wsProjectOneType = TestHelper.findFirst(updated, CardType.class);
253         Assertions.assertEquals(projectOneType, wsProjectOneType);
254 
255         // create a card based on the type
256         Long projectOneRootContentId = ColabFactory.getRootContent(client, projectOne).getId();
257         ColabFactory.createNewCard(client, projectOneRootContentId, projectOneType.getId());
258 
259         // consume websocket message
260         TestHelper.waitForMessagesAndAssert(wsClient, 1, 5, WsUpdateMessage.class).get(0);
261 
262         // As the type is not published, Goulash does not read it from the published list
263         Assertions.assertTrue(client.cardTypeRestEndpoint.getPublishedCardTypes().isEmpty());
264 
265         // but can read it from the project list
266         Assertions.assertEquals(1,
267             client.projectRestEndpoint.getCardTypesOfProject(projectOne.getId()).size());
268 
269         // Goulash creates projectTwo
270         Project projectTwo = ColabFactory.createProject(client, "Project Two");
271 
272         // consume websocket message (new project)
273         TestHelper.waitForMessagesAndAssert(wsClient, 1, 5, WsUpdateMessage.class).get(0);
274 
275         // Goulash invites pizzaiolo
276         ColabFactory.inviteAndJoin(client, projectTwo, "pizza@pizza.local", pizzaHttpClient,
277             mailClient);
278 
279         // consume 2 websocket messages (new invitation; new team member)
280         TestHelper.waitForMessagesAndAssert(wsClient, 2, 5, WsUpdateMessage.class).get(0);
281         this.client.websocketRestEndpoint.subscribeToProjectChannel(projectTwo.getId(),
282             wsClient.getSessionId());
283 
284         // pizza consume 1 websocket message (new team member)
285         TestHelper.waitForMessagesAndAssert(pizzaWsClient, 1, 5, WsUpdateMessage.class).get(0);
286 
287         pizzaHttpClient.websocketRestEndpoint.subscribeToProjectChannel(projectTwo.getId(),
288             pizzaWsClient.getSessionId());
289 
290         // publish the projectOneType
291         projectOneType.setPublished(true);
292         client.cardTypeRestEndpoint.updateCardType(projectOneType);
293 
294         // it is visible for goulash
295         Assertions.assertEquals(1l, client.cardTypeRestEndpoint.getPublishedCardTypes().size());
296         // consume websocket messages (one through project channel, other through user channel)
297         TestHelper.waitForMessagesAndAssert(wsClient, 2, 5, WsUpdateMessage.class).get(0);
298 
299         // but not for pizzaiolo
300         Assertions.assertEquals(0,
301             pizzaHttpClient.cardTypeRestEndpoint.getPublishedCardTypes().size());
302 
303         // goulash create a card in projectTwo based on the projectOne type
304         Long projectTwoRootContentId = ColabFactory.getRootContent(client, projectTwo).getId();
305         ColabFactory.createNewCard(client, projectTwoRootContentId, projectOneType.getId());
306         // consume websocket messages (overview update; project 1 update; project 2 update)
307         TestHelper.waitForMessagesAndAssert(wsClient, 3, 5, WsUpdateMessage.class).get(0);
308         // consume websocket messages (overview update; project 2 update)
309         TestHelper.waitForMessagesAndAssert(pizzaWsClient, 2, 5, WsUpdateMessage.class).get(0);
310 
311         // goulash and pizzaiolo can now access the type through the reference from projectTwo
312         Set<AbstractCardType> goulashTypes = client.projectRestEndpoint
313             .getCardTypesOfProject(projectTwo.getId());
314         Set<AbstractCardType> pizzaTypes = pizzaHttpClient.projectRestEndpoint
315             .getCardTypesOfProject(projectTwo.getId());
316 
317         TestHelper.assertEquals(goulashTypes, pizzaTypes);
318 
319         // Update the projectOneType
320         projectOneType.setTitle("My Favourite Recipes");
321         client.cardTypeRestEndpoint.updateCardType(projectOneType);
322 
323         // both goulash and pizzaiolo should receive the update through websocket
324         // goulash consume throw websocket messages: (overview update; project 1 update; project 2
325         // update)
326         WsUpdateMessage goulashMessage = TestHelper
327             .waitForMessagesAndAssert(wsClient, 3, 5, WsUpdateMessage.class).get(0);
328         // (as type in project is not published, it's not propagated to Pizza, this only one message
329         // is sent)
330         WsUpdateMessage pizzaMessage = TestHelper
331             .waitForMessagesAndAssert(pizzaWsClient, 1, 5, WsUpdateMessage.class).get(0);
332 
333         List<CardType> pizzaList = TestHelper.filterAndAssert(pizzaMessage.getUpdated(), 1,
334             CardType.class);
335         CardType pizzaType = pizzaList.get(0);
336 
337         List<CardType> goulashList = TestHelper.filterAndAssert(goulashMessage.getUpdated(), 1,
338             CardType.class);
339         CardType goulashType = goulashList.get(0);
340 
341         Assertions.assertEquals(pizzaType, goulashType);
342         Assertions.assertEquals(projectOneType.getTitle(), goulashType.getTitle());
343         Assertions.assertEquals(projectOneType.getTitle(), pizzaType.getTitle());
344     }
