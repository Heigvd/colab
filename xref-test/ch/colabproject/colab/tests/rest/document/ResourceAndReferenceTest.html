34      @Test
35      public void testOnGlobalCardType() {
36          doTest(Access.GLOBAL_CARD_TYPE,
37              Set.of(Access.CARD_TYPE_REF, Access.CARD, Access.CARD_CONTENT));
38          // the resources on a card type are not accessible from the sub cards
39      }

41      @Test
42      public void testOnCardTypeRef() {
43          doTest(Access.CARD_TYPE_REF,
44              Set.of(Access.CARD, Access.CARD_CONTENT));
45          // the resources on a card type reference are not accessible from the sub cards
46      }

79      private enum Access {
80          GLOBAL_CARD_TYPE,
81          CARD_TYPE_REF,
82          CARD,
83          CARD_CONTENT,
84          SUB_CARD_TYPE,
85          SUB_CARD,
86          SUB_CARD_CONTENT;
87      }
88  
89      private void doTest(Access resourceLocation, Set<Access> accessToResourceRef) {
90          // creation of the context : project, global card type, card type, card, sub card
91          Project project = ColabFactory.createProject(client, "testResource");
92          Long projectId = project.getId();
93  
94          Long rootCardContentId = ColabFactory.getRootContent(client, project).getId();
95  
96          Long globalCardTypeId = ColabFactory.createCardType(client, null).getId();
97  
98          Long cardId = ColabFactory.createNewCard(client, rootCardContentId, globalCardTypeId)
99              .getId();
100 
101         Long cardTypeRefId = client.cardRestEndpoint.getCard(cardId).getCardTypeId();
102 
103         Long cardContentId = ColabFactory.getCardContent(client, cardId).getId();
104 
105         Long subCardTypeId = ColabFactory.createCardType(client, projectId).getId();
106 
107         Long subCardId = ColabFactory.createNewCard(client, cardContentId, subCardTypeId).getId();
108 
109         Long subCardContentId = ColabFactory.getCardContent(client, subCardId).getId();
110 
111         // creation of a resource
112 
113         String category = "awesome resources #" + ((int) (Math.random() * 1000));
114         String title = "The game encyclopedia #" + ((int) (Math.random() * 1000));
115         String mimeType = "text/markdown";
116         String teaser = "Word by word descriptions #" + ((int) (Math.random() * 1000));
117         String url = "http://www.123soleil.chat/theGameEncyclopedia.pdf";
118 
119         ExternalDocLink doc = new ExternalDocLink();
120         doc.setUrl(url);
121 
122         Long persistedResourceId;
123         Resource persistedResource;
124 
125         TextDataBlock tea = new TextDataBlock();
126         tea.setMimeType(mimeType);
127         tea.setTextData(teaser);
128 
129         ResourceCreationBean toCreate = new ResourceCreationBean();
130         toCreate.setTitle(title);
131         toCreate.setTeaser(tea);
132         toCreate.setDocument(doc);
133         toCreate.setCategory(category);
134 
135         switch (resourceLocation) {
136             case GLOBAL_CARD_TYPE:
137                 toCreate.setAbstractCardTypeId(globalCardTypeId);
138                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
139                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
140                 break;
141             case CARD_TYPE_REF:
142                 toCreate.setAbstractCardTypeId(cardTypeRefId);
143                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
144                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
145                 break;
146             case CARD:
147                 toCreate.setCardId(cardId);
148                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
149                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
150                 break;
151             case CARD_CONTENT:
152                 toCreate.setCardContentId(cardContentId);
153                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
154                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
155                 break;
156             case SUB_CARD_TYPE:
157                 toCreate.setAbstractCardTypeId(subCardTypeId);
158                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
159                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
160                 break;
161             case SUB_CARD:
162                 toCreate.setCardId(subCardId);
163                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
164                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
165                 break;
166             case SUB_CARD_CONTENT:
167                 toCreate.setCardContentId(subCardContentId);
168                 persistedResourceId = client.resourceRestEndpoint.createResource(toCreate);
169                 persistedResource = (Resource) client.resourceRestEndpoint.getAbstractResource(persistedResourceId);
170                 break;
171             default:
172                 Assertions.fail();
173                 persistedResource = new Resource(); // just for compilation / code checks
174                 break;
175         }
176 
177         // check the abstract card type / card / card content link with the resource
178 
179         Assertions.assertNotNull(persistedResource);
180         Assertions.assertNotNull(persistedResource.getId());
181         Assertions.assertEquals(title, persistedResource.getTitle());
182         switch (resourceLocation) {
183             case GLOBAL_CARD_TYPE:
184                 Assertions.assertEquals(globalCardTypeId,
185                     persistedResource.getAbstractCardTypeId());
186                 Assertions.assertNull(persistedResource.getCardId());
187                 Assertions.assertNull(persistedResource.getCardContentId());
188                 break;
189             case CARD_TYPE_REF:
190                 Assertions.assertEquals(cardTypeRefId, persistedResource.getAbstractCardTypeId());
191                 Assertions.assertNull(persistedResource.getCardId());
192                 Assertions.assertNull(persistedResource.getCardContentId());
193                 break;
194             case CARD:
195                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
196                 Assertions.assertEquals(cardId, persistedResource.getCardId());
197                 Assertions.assertNull(persistedResource.getCardContentId());
198                 break;
199             case CARD_CONTENT:
200                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
201                 Assertions.assertNull(persistedResource.getCardId());
202                 Assertions.assertEquals(cardContentId, persistedResource.getCardContentId());
203                 break;
204             case SUB_CARD_TYPE:
205                 Assertions.assertEquals(subCardTypeId, persistedResource.getAbstractCardTypeId());
206                 Assertions.assertNull(persistedResource.getCardId());
207                 Assertions.assertNull(persistedResource.getCardContentId());
208                 break;
209             case SUB_CARD:
210                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
211                 Assertions.assertEquals(subCardId, persistedResource.getCardId());
212                 Assertions.assertNull(persistedResource.getCardContentId());
213                 break;
214             case SUB_CARD_CONTENT:
215                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
216                 Assertions.assertNull(persistedResource.getCardId());
217                 Assertions.assertEquals(subCardContentId, persistedResource.getCardContentId());
218                 break;
219             default:
220                 Assertions.fail();
221         }
222 
223         // check the teaser
224 
225         Long teaserId = persistedResource.getTeaserId();
226         Block persistedTeaserBlock  = client.blockRestEndPoint.getBlock(teaserId);
227         Assertions.assertNotNull(persistedTeaserBlock);
228         Assertions.assertEquals(teaserId, persistedTeaserBlock.getId());
229         Assertions.assertTrue(persistedTeaserBlock instanceof TextDataBlock);
230         TextDataBlock persistedTeaserTextDataBlock = (TextDataBlock) persistedTeaserBlock;
231         Assertions.assertEquals(mimeType, persistedTeaserTextDataBlock.getMimeType());
232         Assertions.assertEquals(teaser, persistedTeaserTextDataBlock.getTextData());
233 
234         // check the document
235 
236         Long documentId = persistedResource.getDocumentId();
237         Document persistedDocument = client.documentRestEndPoint.getDocument(documentId);
238         Assertions.assertNotNull(persistedDocument);
239         Assertions.assertEquals(documentId, persistedDocument.getId());
240         Assertions.assertTrue(persistedDocument instanceof ExternalDocLink);
241         ExternalDocLink persistedExtDocLink = (ExternalDocLink) persistedDocument;
242         Assertions.assertEquals(url, persistedExtDocLink.getUrl());
243         Assertions.assertEquals(persistedResource.getId(), persistedDocument.getResourceId());
244 
245         // check the resource references
246 
247         AbstractResource persistedGlobalCardTypeAbstractResource = null;
248         List<List<AbstractResource>> persistedGlobalCardTypeResources = client.resourceRestEndpoint
249             .getResourceChainForAbstractCardType(globalCardTypeId);
250         if (resourceLocation == Access.GLOBAL_CARD_TYPE) {
251             Assertions.assertNotNull(persistedGlobalCardTypeResources);
252             Assertions.assertEquals(1, persistedGlobalCardTypeResources.size());
253             Assertions.assertEquals(1, persistedGlobalCardTypeResources.get(0).size());
254             Assertions.assertNotNull(persistedGlobalCardTypeResources.get(0).get(0));
255             Assertions
256                 .assertTrue(persistedGlobalCardTypeResources.get(0).get(0) instanceof Resource);
257             persistedGlobalCardTypeAbstractResource = persistedGlobalCardTypeResources.get(0)
258                 .get(0);
259             Assertions.assertEquals(persistedResource, persistedGlobalCardTypeAbstractResource);
260         } else {
261             Assertions.assertNotNull(persistedGlobalCardTypeResources);
262             Assertions.assertEquals(0, persistedGlobalCardTypeResources.size());
263         }
264 
265         AbstractResource persistedCardTypeRefAbstractResource = null;
266         List<List<AbstractResource>> persistedCardTypeRefResources = client.resourceRestEndpoint
267             .getResourceChainForAbstractCardType(cardTypeRefId);
268         if (resourceLocation == Access.CARD_TYPE_REF) {
269             Assertions.assertNotNull(persistedCardTypeRefResources);
270             Assertions.assertEquals(1, persistedCardTypeRefResources.size());
271             Assertions.assertEquals(1, persistedCardTypeRefResources.get(0).size());
272             Assertions.assertNotNull(persistedCardTypeRefResources.get(0).get(0));
273             Assertions.assertTrue(persistedCardTypeRefResources.get(0).get(0) instanceof Resource);
274             persistedCardTypeRefAbstractResource = persistedCardTypeRefResources.get(0).get(0);
275             Assertions.assertEquals(persistedResource, persistedCardTypeRefAbstractResource);
276         } else if (accessToResourceRef.contains(Access.CARD_TYPE_REF)) {
277             Assertions.assertNotNull(persistedCardTypeRefResources);
278             Assertions.assertEquals(1, persistedCardTypeRefResources.size());
279             Assertions.assertTrue(persistedCardTypeRefResources.get(0).size() > 1);
280             Assertions
281                 .assertTrue(persistedCardTypeRefResources.get(0).get(0) instanceof ResourceRef);
282             persistedCardTypeRefAbstractResource = persistedCardTypeRefResources
283                 .get(0).get(0);
284             Assertions.assertNotNull(persistedGlobalCardTypeAbstractResource);
285             Assertions.assertEquals(persistedGlobalCardTypeAbstractResource.getId(),
286                 ((ResourceRef) persistedCardTypeRefAbstractResource).getTargetId());
287         } else {
288             Assertions.assertNotNull(persistedCardTypeRefResources);
289             Assertions.assertEquals(0, persistedCardTypeRefResources.size());
290         }
291 
292         AbstractResource persistedCardAbstractResource = null;
293         List<List<AbstractResource>> persistedCardResources = client.resourceRestEndpoint
294             .getResourceChainForCard(cardId);
295         if (resourceLocation == Access.CARD) {
296             Assertions.assertNotNull(persistedCardResources);
297             Assertions.assertEquals(1, persistedCardResources.size());
298             Assertions.assertEquals(1, persistedCardResources.get(0).size());
299             Assertions.assertNotNull(persistedCardResources.get(0).get(0));
300             Assertions.assertTrue(persistedCardResources.get(0).get(0) instanceof Resource);
301             persistedCardAbstractResource = persistedCardResources.get(0).get(0);
302             Assertions.assertEquals(persistedResource, persistedCardAbstractResource);
303         } else if (accessToResourceRef.contains(Access.CARD)) {
304             Assertions.assertNotNull(persistedCardResources);
305             Assertions.assertEquals(1, persistedCardResources.size());
306             Assertions.assertTrue(persistedCardResources.get(0).size() > 1);
307             Assertions.assertTrue(persistedCardResources.get(0).get(0) instanceof ResourceRef);
308             persistedCardAbstractResource = persistedCardResources.get(0).get(0);
309             Assertions.assertNotNull(persistedCardTypeRefAbstractResource);
310             Assertions.assertEquals(persistedCardTypeRefAbstractResource.getId(),
311                 ((ResourceRef) persistedCardAbstractResource).getTargetId());
312         } else {
313             Assertions.assertNotNull(persistedCardResources);
314             Assertions.assertEquals(0, persistedCardResources.size());
315         }
316 
317         AbstractResource persistedCardContentAbstractResource = null;
318         List<List<AbstractResource>> persistedCardContentResources = client.resourceRestEndpoint
319             .getResourceChainForCardContent(cardContentId);
320         if (resourceLocation == Access.CARD_CONTENT) {
321             Assertions.assertNotNull(persistedCardContentResources);
322             Assertions.assertEquals(1, persistedCardContentResources.size());
323             Assertions.assertEquals(1, persistedCardContentResources.get(0).size());
324             Assertions.assertNotNull(persistedCardContentResources.get(0).get(0));
325             Assertions.assertTrue(persistedCardContentResources.get(0).get(0) instanceof Resource);
326             persistedCardContentAbstractResource = persistedCardContentResources.get(0).get(0);
327             Assertions.assertEquals(persistedResource, persistedCardContentAbstractResource);
328         } else if (accessToResourceRef.contains(Access.CARD_CONTENT)) {
329             Assertions.assertNotNull(persistedCardContentResources);
330             Assertions.assertEquals(1, persistedCardContentResources.size());
331             Assertions.assertTrue(persistedCardContentResources.get(0).size() > 1);
332             Assertions
333                 .assertTrue(persistedCardContentResources.get(0).get(0) instanceof ResourceRef);
334             persistedCardContentAbstractResource = persistedCardContentResources
335                 .get(0).get(0);
336             Assertions.assertNotNull(persistedCardAbstractResource);
337             Assertions.assertEquals(persistedCardAbstractResource.getId(),
338                 ((ResourceRef) persistedCardContentAbstractResource).getTargetId());
339         } else {
340             Assertions.assertNotNull(persistedCardContentResources);
341             Assertions.assertEquals(0, persistedCardContentResources.size());
342         }
343 
344         AbstractResource persistedSubCardTypeAbstractResource = null;
345         List<List<AbstractResource>> persistedSubCardTypeResources = client.resourceRestEndpoint
346             .getResourceChainForAbstractCardType(subCardTypeId);
347         if (resourceLocation == Access.SUB_CARD_TYPE) {
348             Assertions.assertNotNull(persistedSubCardTypeResources);
349             Assertions.assertEquals(1, persistedSubCardTypeResources.size());
350             Assertions.assertEquals(1, persistedSubCardTypeResources.get(0).size());
351             Assertions.assertNotNull(persistedSubCardTypeResources.get(0).get(0));
352             Assertions.assertTrue(persistedSubCardTypeResources.get(0).get(0) instanceof Resource);
353             persistedSubCardTypeAbstractResource = persistedSubCardTypeResources.get(0).get(0);
354             Assertions.assertEquals(persistedResource, persistedSubCardTypeAbstractResource);
355         } else if (accessToResourceRef.contains(Access.SUB_CARD_TYPE)) {
356             Assertions.fail("not handled in current example");
357         } else {
358             Assertions.assertNotNull(persistedSubCardTypeResources);
359             Assertions.assertEquals(0, persistedSubCardTypeResources.size());
360         }
361 
362         AbstractResource persistedSubCardResourceRef = null;
363         List<List<AbstractResource>> persistedSubCardResources = client.resourceRestEndpoint
364             .getResourceChainForCard(subCardId);
365         if (resourceLocation == Access.SUB_CARD) {
366             Assertions.assertNotNull(persistedSubCardResources);
367             Assertions.assertEquals(1, persistedSubCardResources.size());
368             Assertions.assertEquals(1, persistedSubCardResources.get(0).size());
369             Assertions.assertNotNull(persistedSubCardResources.get(0).get(0));
370             Assertions.assertTrue(persistedSubCardResources.get(0).get(0) instanceof Resource);
371             persistedSubCardResourceRef = persistedSubCardResources.get(0).get(0);
372             Assertions.assertEquals(persistedResource, persistedSubCardResourceRef);
373         } else if (accessToResourceRef.contains(Access.SUB_CARD)) {
374             Assertions.assertNotNull(persistedSubCardResources);
375             Assertions.assertEquals(1, persistedSubCardResources.size());
376             Assertions.assertTrue(persistedSubCardResources.get(0).size() > 1);
377             Assertions.assertTrue(persistedSubCardResources.get(0).get(0) instanceof ResourceRef);
378             persistedSubCardResourceRef = persistedSubCardResources.get(0).get(0);
379             Assertions.assertTrue(persistedCardContentAbstractResource != null
380                 || persistedSubCardTypeAbstractResource != null);
381             if (persistedCardContentAbstractResource != null) {
382                 Assertions.assertEquals(persistedCardContentAbstractResource.getId(),
383                     ((ResourceRef) persistedSubCardResourceRef).getTargetId());
384             } else if (persistedSubCardTypeAbstractResource != null) {
385                 Assertions.assertEquals(persistedSubCardTypeAbstractResource.getId(),
386                     ((ResourceRef) persistedSubCardResourceRef).getTargetId());
387             } else {
388                 Assertions.fail(
389                     "a sub card resource reference must be linked to its card type reference or to its parent card content reference ");
390             }
391         } else {
392             Assertions.assertNotNull(persistedSubCardResources);
393             Assertions.assertEquals(0, persistedSubCardResources.size());
394         }
395 
396         AbstractResource persistedSubCardContentResourceRef = null;
397         List<List<AbstractResource>> persistedSubCardContentResources = client.resourceRestEndpoint
398             .getResourceChainForCardContent(subCardContentId);
399         if (resourceLocation == Access.SUB_CARD_CONTENT) {
400             Assertions.assertNotNull(persistedSubCardContentResources);
401             Assertions.assertEquals(1, persistedSubCardContentResources.size());
402             Assertions.assertEquals(1, persistedSubCardContentResources.get(0).size());
403             Assertions.assertNotNull(persistedSubCardContentResources.get(0).get(0));
404             Assertions
405                 .assertTrue(persistedSubCardContentResources.get(0).get(0) instanceof Resource);
406             Assertions.assertEquals(persistedResource,
407                 persistedSubCardContentResources.get(0).get(0));
408             persistedSubCardContentResourceRef = persistedSubCardContentResources.get(0).get(0);
409             Assertions.assertEquals(persistedResource, persistedSubCardContentResourceRef);
410         } else if (accessToResourceRef.contains(Access.SUB_CARD_CONTENT)) {
411             Assertions.assertNotNull(persistedSubCardContentResources);
412             Assertions.assertEquals(1, persistedSubCardContentResources.size());
413             Assertions.assertTrue(persistedSubCardContentResources.get(0).size() > 1);
414             Assertions
415                 .assertTrue(persistedSubCardContentResources.get(0).get(0) instanceof ResourceRef);
416             persistedSubCardContentResourceRef = persistedSubCardContentResources
417                 .get(0).get(0);
418             Assertions.assertNotNull(persistedSubCardResourceRef);
419             Assertions.assertEquals(persistedSubCardResourceRef.getId(),
420                 ((ResourceRef) persistedSubCardContentResourceRef).getTargetId());
421         } else {
422             Assertions.assertNotNull(persistedSubCardContentResources);
423             Assertions.assertEquals(0, persistedSubCardContentResources.size());
424         }
425 
426         // And now, delete the resource
427 
428         Long resourceId = persistedResource.getId();
429 
430         client.resourceRestEndpoint.deleteResource(resourceId);
431 
432         Assertions.assertNull(client.resourceRestEndpoint.getAbstractResource(resourceId));
433 
434         persistedGlobalCardTypeResources = client.resourceRestEndpoint
435             .getResourceChainForAbstractCardType(globalCardTypeId);
436         Assertions.assertNotNull(persistedGlobalCardTypeResources);
437         Assertions.assertEquals(0, persistedGlobalCardTypeResources.size());
438 
439         persistedCardTypeRefResources = client.resourceRestEndpoint
440             .getResourceChainForAbstractCardType(cardTypeRefId);
441         Assertions.assertNotNull(persistedCardTypeRefResources);
442         Assertions.assertEquals(0, persistedCardTypeRefResources.size());
443 
444        persistedCardResources = client.resourceRestEndpoint
445             .getResourceChainForCard(cardId);
446        Assertions.assertNotNull(persistedCardResources);
447        Assertions.assertEquals(0, persistedCardResources.size());
448 
449         persistedCardContentResources = client.resourceRestEndpoint
450             .getResourceChainForCardContent(cardContentId);
451         Assertions.assertNotNull(persistedCardContentResources);
452         Assertions.assertEquals(0, persistedCardContentResources.size());
453 
454         persistedSubCardTypeResources = client.resourceRestEndpoint
455             .getResourceChainForAbstractCardType(subCardTypeId);
456         Assertions.assertNotNull(persistedSubCardTypeResources);
457         Assertions.assertEquals(0, persistedSubCardTypeResources.size());
458 
459 
460         persistedSubCardResources = client.resourceRestEndpoint
461             .getResourceChainForCard(subCardId);
462         Assertions.assertNotNull(persistedSubCardResources);
463         Assertions.assertEquals(0, persistedSubCardResources.size());
464 
465         persistedSubCardContentResources = client.resourceRestEndpoint
466             .getResourceChainForCardContent(subCardContentId);
467         Assertions.assertNotNull(persistedSubCardContentResources);
468         Assertions.assertEquals(0, persistedSubCardContentResources.size());
469     }
