25      @Test
26      public void testCreateTextDataBlock() {
27          Long documentId = client.documentRestEndPoint.createDocument(new BlockDocument());
28  
29          String mimeType = "text/markdown";
30          String content = "Here is some texte to explain how it is important to say things "
31                  + ((int) (Math.random() * 1000));
32  
33          TextDataBlock block = new TextDataBlock();
34          block.setDocumentId(documentId);
35          block.setMimeType(mimeType);
36          block.setTextData(content);
37  
38          Long blockId = client.blockRestEndPoint.createBlock(block);
39  
40          Block persistedBlock = client.blockRestEndPoint.getBlock(blockId);
41          Assertions.assertNotNull(persistedBlock);
42          Assertions.assertNotNull(persistedBlock.getId());
43          Assertions.assertEquals(blockId, persistedBlock.getId());
44          Assertions.assertEquals(0, persistedBlock.getIndex());
45  
46          Assertions.assertTrue(persistedBlock instanceof TextDataBlock);
47          TextDataBlock persistedTextDataBlock = (TextDataBlock) persistedBlock;
48  
49          Assertions.assertEquals(mimeType, persistedTextDataBlock.getMimeType());
50          Assertions.assertEquals(content, persistedTextDataBlock.getTextData());
51      }

74      @Test

76          Long documentId = client.documentRestEndPoint.createDocument(new BlockDocument());

78          TextDataBlock textDataBlock = new TextDataBlock();
79          textDataBlock.setDocumentId(documentId);
80  
81          Long blockId = client.blockRestEndPoint.createBlock(textDataBlock);
82  
83          Block block = client.blockRestEndPoint.getBlock(blockId);

89          Assertions.assertNull(textDataBlock.getMimeType());
90          Assertions.assertNull(textDataBlock.getTextData());
91  
92          int index = (int) (Math.random() * 100);
93          String mimeType = "text/plain";
94          String content = "Just a placeholder text # " + ((int) (Math.random() * 1000));
95  
96          textDataBlock.setIndex(index);
97          textDataBlock.setMimeType(mimeType);
98          textDataBlock.setTextData(content);
99          client.blockRestEndPoint.updateBlock(textDataBlock);

101         Block persistedBlock = client.blockRestEndPoint.getBlock(blockId);
102         Assertions.assertNotNull(persistedBlock);

104         Assertions.assertEquals(index, persistedBlock.getIndex());

131   @Test
132   public void testDocumentAccess() {
133       String title = "Random ideas #" + ((int) (Math.random() * 1000));
134 
135       BlockDocument doc = new BlockDocument();
136       doc.setTitle(title);
137       Long documentId = client.documentRestEndPoint.createDocument(doc);
138 
139       Document persistedDoc = client.documentRestEndPoint.getDocument(documentId);
140       Assertions.assertTrue(persistedDoc instanceof BlockDocument);
141 
142       int index = (int) (Math.random() * 100);
143       String mimeType = "text/plain";
144       String content = "The dinosaurs history is absolutely amazing # "
145               + ((int) (Math.random() * 1000));
146 
147       TextDataBlock block1 = new TextDataBlock();
148       block1.setDocumentId(documentId);
149       block1.setIndex(index);
150       block1.setMimeType(mimeType);
151       block1.setTextData(content);
152       Long block1Id = client.blockRestEndPoint.createBlock(block1);
153 
154       Block persistedBlock = client.blockRestEndPoint.getBlock(block1Id);
155       Assertions.assertNotNull(persistedBlock);
156       Assertions.assertEquals(documentId, persistedBlock.getDocumentId());
157 
158       Block block2 = client.blockRestEndPoint.createNewTextDataBlock(documentId);
159       Assertions.assertNotNull(block2);
160       Assertions.assertEquals(documentId, block2.getDocumentId());
161       Long block2Id = block2.getId();
162 
163       List<Block> blocksOfDocument = client.documentRestEndPoint.getBlocksOfDocument(documentId);
164       Assertions.assertNotNull(blocksOfDocument);
165       Assertions.assertEquals(2, blocksOfDocument.size());
166       Assertions.assertTrue(block1Id.equals(blocksOfDocument.get(0).getId()) || block1Id.equals(blocksOfDocument.get(1).getId()));
167       Assertions.assertTrue(block2Id.equals(blocksOfDocument.get(0).getId()) || block2Id.equals(blocksOfDocument.get(1).getId()));
168   }
