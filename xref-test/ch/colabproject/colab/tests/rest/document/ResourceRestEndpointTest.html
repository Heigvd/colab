86      private enum Access {
87          CARD_TYPE,
88          CARD,
89          CARD_CONTENT,
90          SUB_CARD_TYPE,
91          SUB_CARD,
92          SUB_CARD_CONTENT;
93      }
94  
95      private enum ResourceState {
96          PUBLISHED,
97          REQUESTING_FOR_GLORY,
98          DEPRECATED;
99      }
100 
101     private void helper(Access resourceSpace, Set<ResourceState> resourceState,
102             Set<Access> accessToResource) {
103         // creation of the context : project, 2 card types, card, sub card
104         Project project = ColabFactory.createProject(client, "testResource");
105         Long projectId = project.getId();
106 
107         Long rootCardContentId = ColabFactory.getRootContent(client, project).getId();
108 
109         Long cardTypeId = ColabFactory.createCardType(client, projectId).getId();
110 
111         Long cardId = ColabFactory.createNewCard(client, rootCardContentId, cardTypeId).getId();
112 
113         Long cardContentId = client.cardRestEndpoint.getContentVariantsOfCard(cardId).get(0).getId();
114 
115         Long subCardTypeId = ColabFactory.createCardType(client, projectId).getId();
116 
117         Long subCardId = ColabFactory.createNewCard(client, cardContentId, subCardTypeId).getId();
118 
119         Long subCardContentId  = client.cardRestEndpoint.getContentVariantsOfCard(subCardId).get(0).getId();
120 
121         // creation of a resource
122 
123         String title = "The game encyclopedia #" + ((int) (Math.random() * 1000));
124         String teaser = "Word by word descriptions #" + ((int) (Math.random() * 1000));
125         String url = "http://www.123soleil.chat/theGameEncyclopedia.pdf";
126 
127         ExternalDocLink doc = new ExternalDocLink();
128         doc.setTitle(title);
129         doc.setTeaser(teaser);
130         doc.setUrl(url);
131 
132         Resource persistedResource;
133 
134         switch (resourceSpace) {
135             case CARD_TYPE:
136                 persistedResource = client.resourceRestEndpoint
137                         .createResourceForCardType(cardTypeId, doc);
138                 break;
139             case CARD:
140                 persistedResource = client.resourceRestEndpoint.createResourceForCard(cardId, doc);
141                 break;
142             case CARD_CONTENT:
143                 persistedResource = client.resourceRestEndpoint
144                         .createResourceForCardContent(cardContentId, doc);
145                 break;
146             case SUB_CARD_TYPE:
147                 persistedResource = client.resourceRestEndpoint
148                         .createResourceForCardType(subCardTypeId, doc);
149                 break;
150             case SUB_CARD:
151                 persistedResource = client.resourceRestEndpoint.createResourceForCard(subCardId,
152                     doc);
153                 break;
154             case SUB_CARD_CONTENT:
155                 persistedResource = client.resourceRestEndpoint
156                         .createResourceForCardContent(subCardContentId, doc);
157                 break;
158             default:
159                 Assertions.fail();
160                 persistedResource = new Resource(); // just for compilation / code checks
161                 break;
162         }
163 
164         if (!resourceState.isEmpty()) {
165             if (resourceState.contains(ResourceState.PUBLISHED)) {
166                 persistedResource.setPublished(true);
167             }
168 
169             if (resourceState.contains(ResourceState.REQUESTING_FOR_GLORY)) {
170                 persistedResource.setRequestingForGlory(true);
171             }
172 
173             if (resourceState.contains(ResourceState.DEPRECATED)) {
174                 persistedResource.setDeprecated(true);
175             }
176 
177             client.resourceRestEndpoint.updateResource(persistedResource);
178 
179             persistedResource = client.resourceRestEndpoint.getResource(persistedResource.getId());
180         }
181 
182         Assertions.assertNotNull(persistedResource);
183         Assertions.assertNotNull(persistedResource.getId());
184         switch (resourceSpace) {
185             case CARD_TYPE:
186                 Assertions.assertEquals(cardTypeId, persistedResource.getAbstractCardTypeId());
187                 Assertions.assertNull(persistedResource.getCardId());
188                 Assertions.assertNull(persistedResource.getCardContentId());
189                 break;
190             case CARD:
191                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
192                 Assertions.assertEquals(cardId, persistedResource.getCardId());
193                 Assertions.assertNull(persistedResource.getCardContentId());
194                 break;
195             case CARD_CONTENT:
196                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
197                 Assertions.assertNull(persistedResource.getCardId());
198                 Assertions.assertEquals(cardContentId, persistedResource.getCardContentId());
199                 break;
200             case SUB_CARD_TYPE:
201                 Assertions.assertEquals(subCardTypeId, persistedResource.getAbstractCardTypeId());
202                 Assertions.assertNull(persistedResource.getCardId());
203                 Assertions.assertNull(persistedResource.getCardContentId());
204                 break;
205             case SUB_CARD:
206                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
207                 Assertions.assertEquals(subCardId, persistedResource.getCardId());
208                 Assertions.assertNull(persistedResource.getCardContentId());
209                 break;
210             case SUB_CARD_CONTENT:
211                 Assertions.assertNull(persistedResource.getAbstractCardTypeId());
212                 Assertions.assertNull(persistedResource.getCardId());
213                 Assertions.assertEquals(subCardContentId, persistedResource.getCardContentId());
214                 break;
215             default:
216                 Assertions.fail();
217         }
218         Assertions.assertTrue(
219             resourceState.contains(ResourceState.PUBLISHED) == persistedResource.isPublished());
220         Assertions.assertTrue(resourceState.contains(
221             ResourceState.REQUESTING_FOR_GLORY) == persistedResource.isRequestingForGlory());
222         Assertions.assertTrue(
223             resourceState.contains(ResourceState.DEPRECATED) == persistedResource.isDeprecated());
224         Assertions.assertNotNull(persistedResource.getDocumentId());
225 
226         Long documentId = persistedResource.getDocumentId();
227         Document persistedDocument = client.documentRestEndPoint.getDocument(documentId);
228         Assertions.assertNotNull(persistedDocument);
229         Assertions.assertEquals(documentId, persistedDocument.getId());
230         Assertions.assertEquals(title, persistedDocument.getTitle());
231         Assertions.assertEquals(teaser, persistedDocument.getTeaser());
232         Assertions.assertTrue(persistedDocument instanceof ExternalDocLink);
233         ExternalDocLink persistedExtDocLink = (ExternalDocLink) persistedDocument;
234         Assertions.assertEquals(url, persistedExtDocLink.getUrl());
235 
236         if (accessToResource.contains(Access.CARD_TYPE)) {
237             List<Resource> persistedCardTypeResources = client.cardTypeRestEndpoint
238                     .getAvailableActiveLinkedResources(cardTypeId);
239             Assertions.assertNotNull(persistedCardTypeResources);
240             Assertions.assertEquals(1, persistedCardTypeResources.size());
241             Assertions.assertEquals(persistedResource, persistedCardTypeResources.get(0));
242         } else {
243             List<Resource> persistedCardTypeResources = client.cardTypeRestEndpoint
244                     .getAvailableActiveLinkedResources(cardTypeId);
245             Assertions.assertNotNull(persistedCardTypeResources);
246             Assertions.assertEquals(0, persistedCardTypeResources.size());
247         }
248 
249         if (accessToResource.contains(Access.CARD)) {
250             List<Resource> persistedCardResources = client.cardRestEndpoint
251                     .getAvailableActiveLinkedResources(cardId);
252             Assertions.assertNotNull(persistedCardResources);
253             Assertions.assertEquals(1, persistedCardResources.size());
254             Assertions.assertEquals(persistedResource, persistedCardResources.get(0));
255         } else {
256             List<Resource> persistedCardResources = client.cardRestEndpoint
257                     .getAvailableActiveLinkedResources(cardId);
258             Assertions.assertNotNull(persistedCardResources);
259             Assertions.assertEquals(0, persistedCardResources.size());
260         }
261 
262         if (accessToResource.contains(Access.CARD_CONTENT)) {
263             List<Resource> persistedCardContentResources = client.cardContentRestEndpoint
264                     .getAvailableActiveLinkedResources(cardContentId);
265             Assertions.assertNotNull(persistedCardContentResources);
266             Assertions.assertEquals(1, persistedCardContentResources.size());
267             Assertions.assertEquals(persistedResource, persistedCardContentResources.get(0));
268         } else {
269             List<Resource> persistedCardContentResources = client.cardContentRestEndpoint
270                     .getAvailableActiveLinkedResources(cardContentId);
271             Assertions.assertNotNull(persistedCardContentResources);
272             Assertions.assertEquals(0, persistedCardContentResources.size());
273         }
274 
275         if (accessToResource.contains(Access.SUB_CARD_TYPE)) {
276             List<Resource> persistedCardTypeResources = client.cardTypeRestEndpoint
277                     .getAvailableActiveLinkedResources(subCardTypeId);
278             Assertions.assertNotNull(persistedCardTypeResources);
279             Assertions.assertEquals(1, persistedCardTypeResources.size());
280             Assertions.assertEquals(persistedResource, persistedCardTypeResources.get(0));
281         } else {
282             List<Resource> persistedCardTypeResources = client.cardTypeRestEndpoint
283                     .getAvailableActiveLinkedResources(subCardTypeId);
284             Assertions.assertNotNull(persistedCardTypeResources);
285             Assertions.assertEquals(0, persistedCardTypeResources.size());
286         }
287 
288         if (accessToResource.contains(Access.SUB_CARD)) {
289             List<Resource> persistedSubCardResources = client.cardRestEndpoint
290                     .getAvailableActiveLinkedResources(subCardId);
291             Assertions.assertNotNull(persistedSubCardResources);
292             Assertions.assertEquals(1, persistedSubCardResources.size());
293             Assertions.assertEquals(persistedResource, persistedSubCardResources.get(0));
294         } else {
295             List<Resource> persistedSubCardResources = client.cardRestEndpoint
296                     .getAvailableActiveLinkedResources(subCardId);
297             Assertions.assertNotNull(persistedSubCardResources);
298             Assertions.assertEquals(0, persistedSubCardResources.size());
299         }
300 
301         if (accessToResource.contains(Access.SUB_CARD_CONTENT)) {
302             List<Resource> persistedSubCardContentResources = client.cardContentRestEndpoint
303                     .getAvailableActiveLinkedResources(subCardContentId);
304             Assertions.assertNotNull(persistedSubCardContentResources);
305             Assertions.assertEquals(1, persistedSubCardContentResources.size());
306             Assertions.assertEquals(persistedResource, persistedSubCardContentResources.get(0));
307         } else {
308             List<Resource> persistedSubCardContentResources = client.cardContentRestEndpoint
309                     .getAvailableActiveLinkedResources(subCardContentId);
310             Assertions.assertNotNull(persistedSubCardContentResources);
311             Assertions.assertEquals(0, persistedSubCardContentResources.size());
312         }
313     }
