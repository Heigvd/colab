397     private void checkRelevantResources(String context, CardContent cardContent,
398         List<Resource> expectedResources) {
399         List<List<AbstractResource>> persistedResourcesAndRefs = client.resourceRestEndpoint
400             .getResourceChainForCardContent(cardContent.getId());
401 
402         List<Long> relevant = persistedResourcesAndRefs.stream()
403             .filter(list -> !isResidual(list))
404             .map(list -> list.get(list.size() - 1).getId())
405             .collect(Collectors.toList());
406 
407         for (Resource resource : expectedResources) {
408             Assertions.assertTrue(relevant.contains(resource.getId()),
409                 MessageFormat.format("For {0}, the wandering card miss the relevant resource \"{1}\"",
410                     context, resource.getTitle()));
411         }
412 
413         Assertions.assertEquals(expectedResources.size(), relevant.size(),
414             MessageFormat.format("For {0}, the wandering card must have {1} relevant resources",
415                 context, Integer.toString(expectedResources.size())));
416     }
417 
418     private void checkResidualResources(String context, CardContent cardContent,
419         List<Resource> expectedResources) {
420         List<List<AbstractResource>> persistedResourcesAndRefs = client.resourceRestEndpoint
421             .getResourceChainForCardContent(cardContent.getId());
422 
423         List<Long> residues = persistedResourcesAndRefs.stream()
424             .filter(list -> isResidual(list))
425             .map(list -> list.get(list.size() - 1).getId())
426             .collect(Collectors.toList());
427 
428         for (Resource resource : expectedResources) {
429             Assertions.assertTrue(residues.contains(resource.getId()),
430                 MessageFormat.format("For {0}, the wandering card miss the residual resource \"{1}\"",
431                     context, resource.getTitle()));
432         }
433 
434         Assertions.assertEquals(expectedResources.size(), residues.size(),
435             MessageFormat.format("For {0}, the wandering card must have {1} residual resources",
436                 context, Integer.toString(expectedResources.size())));
437     }
438 
439     private static boolean isResidual(List<AbstractResource> list) {
440         return (list.get(0) instanceof ResourceRef) && ((ResourceRef) list.get(0)).isResidual();
441     }
