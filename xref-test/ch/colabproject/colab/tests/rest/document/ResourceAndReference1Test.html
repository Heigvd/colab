539         if (accessToResourceRef.contains(Access.SUB_CARD_CONTENT)
540             && resourceLocation != Access.SUB_CARD) {
541             Assertions.assertNotNull(persistedSisterSubCardContentResources);
542             Assertions.assertEquals(1, persistedSisterSubCardContentResources.size());
543             Assertions.assertTrue(persistedSisterSubCardContentResources.get(0).size() > 1,
544                 "There should be resource reference on the sister sub card content");
545             Assertions
546                 .assertTrue(
547                     persistedSisterSubCardContentResources.get(0).get(0) instanceof ResourceRef);
548             persistedSisterSubCardContentResourceRef = persistedSisterSubCardContentResources
549                 .get(0).get(0);
550             Assertions.assertNotNull(persistedSisterSubCardResourceRef);
551             Assertions.assertEquals(persistedSisterSubCardResourceRef.getId(),
552                 ((ResourceRef) persistedSisterSubCardContentResourceRef).getTargetId());
553         } else {
554             Assertions.assertNotNull(persistedSisterSubCardContentResources);
555             Assertions.assertEquals(0, persistedSisterSubCardContentResources.size(),
556                 "There should be no resource on the sister sub card content");
557         }
558

560
