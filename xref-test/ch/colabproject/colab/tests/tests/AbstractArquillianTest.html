160     private static void addFiles(WebArchive war, File file, String to) {
161 
162         if (file.isDirectory()) {
163 
164             for (File child : file.listFiles()) {
165                 if (child.isFile()) {
166                     war.addAsWebResource(child, to + "/" + child.getName());
167                 } else {
168                     addFiles(war, child, to + "/" + child.getName());
169                 }
170             }
171         } else {
172             war.addAsWebResource(file, to);
173         }
174     }

223     /**
224      * Create a new localAccount with given identifier and credential
225      *
226      * @param username username
227      * @param email    address to use as identifier
228      * @param password plain text password
229      *
230      * @return a user who can login with {@link #signIn(TestUser) signIn}
231      *
232      * @throws HttpErrorMessage if something went wrong
233      */
234     protected TestUser signup(String username, String email, String password)
235         throws ClientErrorException {
236         // Create a brand new user with a local account
237         AuthMethod authMethod = client.userRestEndpoint.getAuthMethod(email);
238 
239         SignUpInfo signup = new SignUpInfo();
240         signup.setUsername(username);
241         signup.setFirstname(username); // hack to have something in firstname
242         signup.setEmail(email);
243         signup.setSalt(authMethod.getSalt());
244         signup.setHashMethod(authMethod.getMandatoryMethod());
245 
246         String hash = Helper.bytesToHex(
247             authMethod.getMandatoryMethod().hash(
248                 password, signup.getSalt()
249             )
250         );
251 
252         signup.setHash(hash);
253 
254         client.userRestEndpoint.signUp(signup);
255 
256         return new TestUser(username, email, password);
257     }

275     /**
276      * hash password
277      *
278      * @param identifier    email address or username
279      * @param plainPassword plain-text password
280      *
281      * @return authinfo with hashed password
282      */
283     protected AuthInfo getAuthInfo(String identifier, String plainPassword) {
284         AuthMethod authMethod = client.userRestEndpoint.getAuthMethod(identifier);
285         AuthInfo authInfo = new AuthInfo();
286         authInfo.setIdentifier(identifier);
287 
288         // compute mandatory hash
289         String hash = Helper.bytesToHex(
290             authMethod.getMandatoryMethod().hash(
291                 plainPassword,
292                 authMethod.getSalt()
293             )
294         );
295         authInfo.setMandatoryHash(hash);
296 
297         if (authMethod.getOptionalMethod() != null) {
298             // method rotation is requested : computes hash with new method too
299             authInfo.setOptionalHash(
300                 Helper.bytesToHex(
301                     authMethod.getOptionalMethod().hash(
302                         plainPassword,
303                         authMethod.getNewSalt()
304                     )
305                 )
306             );
307         }
308 
309         return authInfo;
310     }

312     /**
313      * Connect to mailhog and process all pending "Please verify your account" messages. Once
314      * process, messages are deletes from mailhog.
315      */
316     protected void verifyAccounts() {
317         mailClient.getMessages().forEach(message -> {
318             List<String> headers = message.getContent().getHeaders().get("Subject");
319 
320             if (!headers.isEmpty()
321                 && VerifyLocalAccountToken.EMAIL_SUBJECT.equals(headers.get(0))) {
322                 Matcher matcher = TestHelper.extractToken(message);
323                 if (matcher.matches()) {
324                     Long tokenId = Long.parseLong(matcher.group(1));
325                     String plainToken = matcher.group(2);
326 
327                     Token token = client.tokenRestEndpoint.getToken(tokenId);
328                     if (token instanceof VerifyLocalAccountToken) {
329                         client.tokenRestEndpoint.consumeToken(tokenId, plainToken);
330                         this.mailClient.deleteMessage(message.getId());
331                     }
332                 }
333             }
334         });
335     }

392     protected WebsocketClient createWsClient() throws DeploymentException, IOException, InterruptedException, URISyntaxException {
393         String httpProtocol = deploymentURL.getProtocol();
394         String host = deploymentURL.getHost();
395         int port = deploymentURL.getPort();
396         String path = deploymentURL.getPath();
397         String endpoint = (httpProtocol.equals("http") ? "ws" : "wss")
398             + "://" + host + ":" + port + path + "ws";
399 
400         WebsocketClient wsClient = WebsocketClient.build(endpoint);
401         this.wsClients.add(wsClient);
402         return wsClient;
403     }

405     /**
406      * Before each step: reset db state
407      *
408      * @param info Jupiter Test info
409      *
410      * @throws java.sql.SQLException if db initialization failed
411      */
412     @BeforeEach
413     public void init(TestInfo info) throws SQLException {
414         if (userManager != null) {
415             this.mailClient = new MailhogClient();
416             try {
417                 mailClient.deleteAllMessages();
418             } catch (Throwable ex) {
419                 logger.error("NO MAILHOG");
420                 Assertions.fail("No Mailhog ! did you \"docker run -d --restart always -p 8025:8025 -p 1025:1025 mailhog/mailhog\"?");
421             }
422 
423             this.client = createRestClient();
424             logger.info("Start TEST {}", info.getDisplayName());
425             this.startTime = System.currentTimeMillis();
426 
427             // start each test with a clean database
428             resetDatabase();
429 
430             // create one admin
431             this.admin = this.signup(ADMIN_USERNAME, ADMIN_EMAIL, ADMIN_PASSWORD);
432 
433             verifyAccounts();
434 
435             User adminUser = userDao.findUserByUsername("admin");
436             this.adminUserId = adminUser.getId();
437 
438             testFacade.grantAdminRight(adminUser.getId());
439 
440             signIn(admin);
441 
442             this.initTime = System.currentTimeMillis();
443         }
444     }

446     /**
447      * Perform some cleanup and log duration. Note that db is clear before each test, thus, it's not
448      * necessary to clear data here
449      *
450      * @param info jupiter test info
451      */
452     @AfterEach
453     public void clean(TestInfo info) {
454         if (userManager != null) {
455             if (wsClients != null) {
456                 wsClients.forEach(wsClient -> {
457                     try {
458                         logger.info("Close websocket client {}", wsClient);
459                         wsClient.close();
460                     } catch (IOException ex) {
461                         logger.warn("Failed to close websocket client");
462                     }
463                 });
464                 wsClients.clear();
465             }
466             long now = System.currentTimeMillis();
467             logger.info("TEST {} DURATION: total: {} ms; init: {} ms; test: {} ms",
468                 info.getDisplayName(),
469                 now - this.startTime,
470                 this.initTime - this.startTime,
471                 now - this.initTime);
472         }
