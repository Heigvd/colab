18  import java.util.List;
19  import java.util.regex.Matcher;
20  import org.junit.jupiter.api.Assertions;
21  
22  /**
23   * Some method to create item easily
24   *
25   * @author maxence
26   */
27  public class ColabFactory {
28  
29      private ColabFactory() {
30          throw new UnsupportedOperationException(
31              "This is a utility class and cannot be instantiated");
32      }
33  
34      /**
35       * Create a brand new card type in the given project.
36       *
37       * @param client    rest client to execute HTTP requests
38       * @param projectId id of the project the card type will belongs to. If this id is null the type
39       *                  will be a global type
40       *
41       * @return the CardType
42       */
43      public static CardType createCardType(ColabClient client, Long projectId) {
44          CardType cardType = new CardType();
45          cardType.setProjectId(projectId);
46  
47          Long cardTypeId = client.cardTypeRestEndpoint.createCardType(cardType);
48          return (CardType) client.cardTypeRestEndpoint.getCardType(cardTypeId);
49      }
50  
51      /**
52       * Create a brand new project.
53       *
54       * @param client rest client to execute HTTP requests
55       * @param name   name of the new project
56       *
57       * @return the new project
58       */
59      public static Project createProject(ColabClient client, String name) {
60          Project p = new Project();
61          p.setName(name);
62          Long id = client.projectRestEndpoint.createProject(p);
63          return client.projectRestEndpoint.getProject(id);
64      }
65  
66      /**
67       * Retrieve the root cardContent of the given project.
68       *
69       * @param client  rest client to execute HTTP requests
70       * @param project the project to fetch the root content in
71       *
72       * @return the root card content of project
73       */
74      public static CardContent getRootContent(ColabClient client, Project project) {
75          Card card = client.cardRestEndpoint.getCard(project.getRootCardId());
76          List<CardContent> rootCardContents = client.cardRestEndpoint
77              .getContentVariantsOfCard(card.getId());
78  
79          return rootCardContents.get(0);
80      }
81  
82      /**
83       * <code>host</code> invites <code>guest</code> in the project.
84       *
85       * @param host         authenticated rest client with write-access on project
86       * @param project      the project
87       * @param emailAddress email address to send the message to
88       * @param guest        authenticated rest client, not yet member of the project
89       * @param mailClient   mailhost client to fetch the invitation
90       */
91      public static void inviteAndJoin(ColabClient host, Project project,
92          String emailAddress, ColabClient guest, MailhogClient mailClient
93      ) {
94          host.projectRestEndpoint.inviteSomeone(project.getId(), emailAddress);
95          Message invitation = TestHelper.getMessageByRecipient(mailClient, emailAddress).get(0);
96  
97          Matcher matcher = TestHelper.extractToken(invitation);
98  
99          if (matcher.matches()) {
100             Long tokenId = Long.parseLong(matcher.group(1));
101             String plainToken = matcher.group(2);
102 
103             mailClient.deleteMessage(invitation.getId());
104 
105             Token token = guest.tokenRestEndpoint.getToken(tokenId);
106             Assertions.assertTrue(token instanceof InvitationToken);
107             guest.tokenRestEndpoint.consumeToken(tokenId, plainToken);
108         } else {
109             Assertions.fail("Failed to parse token");
110         }
111     }
112 }
