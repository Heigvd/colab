7   package ch.colabproject.colab.generator.model.interfaces;
8   
9   import ch.colabproject.colab.generator.model.annotations.JsonClassName;
10  import ch.colabproject.colab.generator.model.tools.PolymorphicDeserializer;
11  import java.io.Serializable;
12  import javax.json.bind.annotation.JsonbProperty;
13  import javax.json.bind.annotation.JsonbTypeDeserializer;
14  
15  /**
16   * Represent an object which can be serialize to JSON. Implementing this interface force
17   * serialization of class discriminator in "@class" property.
18   * <p>
19   * Default @class value is the concrete class simple name but it can be overriden with
20   * {@link JsonClassName} annotation
21   *
22   * @author maxence
23   */
24  @JsonbTypeDeserializer(PolymorphicDeserializer.class)
25  public interface WithJsonDiscriminator extends Serializable {
26  
27      /**
28       * Get the JSON discriminator for the given class.
29       *
30       * @param klass the class to identify
31       *
32       * @return name to be used as discriminant
33       */
34      static String getJsonDiscriminator(Class<?> klass) {
35          JsonClassName annotation = klass.getAnnotation(JsonClassName.class);
36  
37          if (annotation != null) {
38              return annotation.value();
39          } else {
40              return klass.getSimpleName();
41          }
42      }
43  
44      /**
45       * Get a unique name which identify the type of the JSON object
46       *
47       * @return discriminator
48       */
49      @JsonbProperty("@class")
50      default String getJsonDiscriminator() {
51          return WithJsonDiscriminator.getJsonDiscriminator(this.getClass());
52      }
53  }
